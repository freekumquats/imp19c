########################################################################
# qing_mexico_adventure_events.txt — THE QING AND THE MEXICAN ADVENTURE
# (墨西哥帝業, task #96 consumer side).
#
# The Mexico subsystem (se_MEXICO.txt / mex_instability_events.txt) OWNS the arc
# and its climax: mex_instability.12 has already crowned Maximilian and set
# global_var:mex_empire_country. THIS event is how a human Qing court REACTS to a
# Second Empire that arc set in motion — the exact usa_section.12 -> qing_uscw.1
# inversion pattern (the new subsystem notifies the Qing; the Qing consumes).
# mex_instability.12 fires this (human CHI only) once the Empire exists.
#
# Machinery: se_QING_MEXICO.txt (QING_mexico_prop_empire / QING_mexico_back_juarez)
# + the #69 modifiers already shipped (qing_col_mexican_crown). CHI-only,
# player-only — the AI Qing's stance is expressed through the #69 mission tree, not
# this pop-up.
#
#   ROOT / this = the Qing overlord (CHI, player).
#
# Namespace = qing_mexico_adventure
#   .1  墨西哥帝業 — news of Maximilian's throne reaches Beijing; prop it or back Juarez
#
# Sources: Cunningham, Mexico and the Foreign Policy of Napoleon III; Schoonover,
#   Dollars over Dominion; Hamnett, Juárez (1994).
########################################################################

namespace = qing_mexico_adventure

########################################################################
# .1  墨西哥帝業 — THE THRONE ACROSS THE OCEAN
# Word crosses the Pacific: a Habsburg prince has been crowned Emperor of Mexico
# on French bayonets. A Qing that has reached the far shore must decide whether to
# make the enterprise its own — prop the throne against the day Washington and the
# Republic come for it — or throw in with Juarez and bet on its fall.
########################################################################
qing_mexico_adventure.1 = {
	type = country_event
	title = "qing_mexico_adventure.1.t"
	desc = "qing_mexico_adventure.1.desc"
	picture = war_council

	left_portrait = root.current_ruler

	is_triggered_only = yes

	# Re-check the notifier's guards defensively: CHI, the Empire actually on the
	# map (the Mexico arc's concrete object), and not already decided here.
	trigger = {
		tag = CHI
		exists = global_var:mex_empire_country
		NOT = { has_variable = qing_mexico_decided }
	}

	immediate = {
		set_variable = qing_mexico_decided
		LOG_line = { sys = QING  msg = "EVENT qing_mexico_adventure.1 (Maximilian's throne reaches the court) for [ROOT.GetTag]" }
	}

	# --- Prop the throne — a Franco-Qing patronage of the Second Empire --------
	# Sets mex_foreign_patron: the Mexico arc keeps the Empire alive past 1867.
	option = {
		name = "qing_mexico_adventure.1.a"
		custom_tooltip = "qing_mexico_adventure.1.a.tt"
		QING_mexico_prop_empire = yes
	}

	# --- Back Juarez — court the Republic and bet on the Empire's fall ---------
	option = {
		name = "qing_mexico_adventure.1.b"
		custom_tooltip = "qing_mexico_adventure.1.b.tt"
		QING_mexico_back_juarez = yes
	}
}

# qing_force_setup_events.txt — [#223/#224-fix] STAGGERED 1815 FORCE SETUP.
#
# WHY THIS FILE EXISTS: the Qing starting army + navy were both set up INLINE at
# on_game_initialized (oa_economy_setup.txt). That single tick is too early and too
# crowded, producing two boot-test bugs the user reported:
#
#   #224 ARMY: all ~27 garrisons mustered in Beijing instead of dispersing. At
#     on_game_initialized the per-province state.governorships are not yet resolved,
#     so every SE_qing_raise_garrison[_cmd] call fell through its
#     `$prov$ = { exists = state.governorship }` guard to the c:CHI.capital_scope
#     fallback — i.e. Beijing — stacking the entire order of battle in one province.
#     (The garrison provinces span 8 distinct regions — Zhili/Shaanxi/Gansu/Liaoning/
#     Jiangsu/Sichuan/Guangdong/Fujian — so they are NOT one shared governorship;
#     the all-Beijing result can only be the fallback path firing every time.)
#
#   #223 NAVY: only the Fujian squadron survived. Three create_unit navy=yes calls
#     issued sequentially in ONE scope on ONE tick collide; only one persists.
#
# THE FIX (both): defer force setup off the on_game_initialized tick and STAGGER it
# across distinct days, mirroring the only PROVEN-WORKING multi-unit setups in this
# mod (the ILI Chu Army and the four Self-Strengthening fleets, each raised on its
# own mid-game tick). By the time these fire, governorships are resolved (army
# disperses) and each navy create lands on its own tick (all three fleets survive).
#
# All events: country_event, hidden, is_triggered_only, ROOT = c:CHI. Scheduling is
# guarded once by qing_force_setup_scheduled at the call site (oa_economy_setup.txt);
# scheduled events persist across save/reload, so they are queued exactly once.
# Individual effects keep their own idempotency: SE_qing_armies self-guards on
# qing_armies_setup_done, SE_qing_navy_disband on qing_navy_setup_done.
#
#   Namespace = qing_force_setup
#     .1   raise the dispersed land order of battle  (SE_qing_armies)
#     .10  disband the ahistorical auto-generated fleets  (SE_qing_navy_disband)
#     .11  raise all three 水師 squadrons (Guangdong/Fujian/Zhejiang) on ONE tick [#242-fix]
########################################################################

namespace = qing_force_setup

########################################################################
# .1  LAND ORDER OF BATTLE — raised on a settled tick so each garrison musters at
# its own province's governorship (dispersed) rather than the capital fallback.
########################################################################
qing_force_setup.1 = {
	type = country_event
	hidden = yes
	is_triggered_only = yes
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_force_setup.1 (deferred land OOB)" }
		SE_qing_armies = yes
		# [BT-50 / #396] the OOB attach just made commanders — some of whom the game-start council
		# autofill (on_game_initialized, day 0) already seated as ministers, producing the reported
		# "councillors who are also commanders". Reconcile now: prune relieves any seat whose holder
		# is now a commander/governor, then autofill re-fills the vacated seats with eligible men
		# (the eligibility gate excludes commanders/governors/ruler/heir), and recompute refreshes
		# the effectiveness read. CHI-only (this event is ROOT = c:CHI).
		if = {
			limit = { tag = CHI }
			QING_council_prune_all_seats = yes
			QING_council_autofill = yes
			QING_council_recompute = yes
			# [#5 fix 2026-07-17; BOOT-CRASHFIX 2026-07-18] MARTIAL EXAM CONGRUENCE — granted HERE, at
			# RUNTIME (day 30, is_triggered_only), NOT inside QING_office_appoint. The appoint verb is
			# reached at gamestate CONSTRUCTION via the game-start council autofill, and an add_trait
			# there (opposites recomputation on a char holding the civil jinshi) crashed the boot with
			# no log — the class of the two prior boot crashfixes. Doing it in this deferred event keeps
			# it entirely off the construction path.
			# Covers BOTH populations that should read as martial-degree holders:
			#   (a) serving army/navy commanders (garrison/fleet officers rose through the 武科), AND
			#   (b) the two MARTIAL council offices — War Minister (兵部, qing_office_held=war) and
			#       Commandant of the Imperial Guard (領侍衛內大臣, guard_commandant) — which the general
			#       autofill pool seats with civil-degree courtiers, so they showed the wrong (civil)
			#       Presented Scholar. Grant 武進士 (wu_jinshi) to any such man lacking a wuju degree.
			# Does NOT strip a civil degree beyond the trait's own opposite-resolution; skips men who
			# already hold a wuju rank. is_general/is_admiral = proven commander triggers.
			every_character = {
				limit = {
					employer = ROOT
					is_alive = yes
					OR = {
						is_general = yes
						is_admiral = yes
						AND = {
							has_variable = qing_office_held
							OR = { var:qing_office_held = flag:war  var:qing_office_held = flag:guard_commandant }
						}
					}
					NOT = { has_trait = wu_shengyuan }
					NOT = { has_trait = wu_juren }
					NOT = { has_trait = wu_jinshi }
					NOT = { has_trait = wu_zhuangyuan }
				}
				# [BT-N fix 2026-07-19] REMOVE any held civil exam degree FIRST, then grant the martial one.
				# wu_jinshi lists the civil degrees as opposites, but relying on the engine's opposite-
				# resolution to strip the held civil trait on add_trait proved unreliable here (the War
				# Minister / Guard Commandant kept showing the civil Presented Scholar). Proven idiom
				# (Invictus 00_event_effects_inv_1_0.txt:4312+ swaps mutually-exclusive status traits via
				# remove_trait-then-add_trait) is to strip explicitly. A martial official should read as a
				# 武科 laureate, so dropping his civil degree here is intended.
				if = { limit = { has_trait = shengyuan }     remove_trait = shengyuan }
				if = { limit = { has_trait = juren }         remove_trait = juren }
				if = { limit = { has_trait = gongshi }       remove_trait = gongshi }
				if = { limit = { has_trait = jinshi }        remove_trait = jinshi }
				if = { limit = { has_trait = hanlin }        remove_trait = hanlin }
				if = { limit = { has_trait = fanyi_jinshi }  remove_trait = fanyi_jinshi }
				if = { limit = { has_trait = jiansheng }     remove_trait = jiansheng }
				add_trait = wu_jinshi
			}
		}
	}
}

########################################################################
# .10  DISBAND the engine's auto-generated blue-water fleets (own tick, before the
# squadron raises). Self-guarded (destroy_unit is destructive) via qing_navy_setup_done.
########################################################################
qing_force_setup.10 = {
	type = country_event
	hidden = yes
	is_triggered_only = yes
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_force_setup.10 (disband auto-gen fleets)" }
		SE_qing_navy_disband = yes
	}
}

########################################################################
# .11  RAISE all three 水師 squadrons on ONE tick.
#
# [#242-fix develop] The previous design raised each squadron on its OWN staggered
# tick (.11/.12/.13 on days 4/5/6) on the theory that same-tick create_unit navy=yes
# calls collide. The debug.log DISPROVED that: create_unit ran for all three, yet only
# Fujian survived in-game. Oracle consult (2026-07-08) identified the real culprit as
# the day-3 disband's `every_navy = { destroy_unit }` interacting with the staggered
# raises, and found the PROVEN multi-navy idiom to be several create_unit navy=yes
# blocks in ONE event on ONE tick (Invictus events/flavor_gaul.txt:590-632). So all
# three squadrons now raise together here, AFTER the .10 disband. ROOT = c:CHI.
########################################################################
qing_force_setup.11 = {
	type = country_event
	hidden = yes
	is_triggered_only = yes
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_force_setup.11 (all three water-forces, one tick)" }
		SE_qing_navy_guangdong = yes
		SE_qing_navy_fujian = yes
		SE_qing_navy_zhejiang = yes
		# [BT-P fix 2026-07-19] grant the martial exam degree (武進士) to the NAVAL commanders the three
		# squadrons above just created. This MUST be here, not in qing_force_setup.1: the land grant runs
		# at day 30 but the navies raise at day 31 (this event), so at day 30 no admiral exists yet and
		# is_admiral matched nobody — naval commanders were left with no martial degree (BT-P). Same
		# remove-civil-first idiom as the day-30 land grant (BT-N). CHI-only (ROOT = c:CHI here).
		every_character = {
			limit = {
				employer = ROOT
				is_alive = yes
				is_admiral = yes
				NOT = { has_trait = wu_shengyuan }
				NOT = { has_trait = wu_juren }
				NOT = { has_trait = wu_jinshi }
				NOT = { has_trait = wu_zhuangyuan }
			}
			if = { limit = { has_trait = shengyuan }     remove_trait = shengyuan }
			if = { limit = { has_trait = juren }         remove_trait = juren }
			if = { limit = { has_trait = gongshi }       remove_trait = gongshi }
			if = { limit = { has_trait = jinshi }        remove_trait = jinshi }
			if = { limit = { has_trait = hanlin }        remove_trait = hanlin }
			if = { limit = { has_trait = fanyi_jinshi }  remove_trait = fanyi_jinshi }
			if = { limit = { has_trait = jiansheng }     remove_trait = jiansheng }
			add_trait = wu_jinshi
		}
	}
}

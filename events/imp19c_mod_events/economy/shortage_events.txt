namespace = shortage

shortage.1 = { # General food shortage
	type = country_event
	title = "shortage.1.t"
    desc = "shortage.1.desc"
    picture = looting
	hidden = no

	left_portrait = scope:famine_province.governorship.governor

	goto_location = scope:famine_province

	trigger = {
		# [logfix] require a governorship that is BOTH short of food AND still has an
		# eligible province (populated + not already under recent_famine). The immediate
		# block saves scope:famine_province only from such a province; without this guard
		# the event could fire with every candidate already famine-flagged, leaving the
		# scope unsaved and every scope:famine_province read below throwing "Undefined
		# event target 'famine_province'" (245/session).
		any_governorships = {
			ECON_governorship_food_shortage_physical > 0.3
			any_governorship_state = {
				# [#1 boot] the owning STATE must ACTUALLY be short of STORED food before its pops
				# starve. The governorship trade-sim shortage (ECON_governorship_food_shortage_physical,
				# a per-good demand-vs-stockpile snapshot) is a SEPARATE accounting from the engine's
				# State food pool (has_state_food) shown in the UI as "Amenities" and topped up by the
				# ever-normal granary (QING_DECLINE_granary_pool). Without this gate a State sitting on a
				# large food surplus still starved pops whenever the trade-sim tripped — the reported bug.
				# "Real shortage" = below 1/5 of capacity, byte-identical to the granary SHIP-leg test
				# (se_QING_DECLINE.txt:2018-2019); capacity>0 guards the degenerate no-storage state.
				has_state_food_capacity > 0
				has_state_food < { value = has_state_food_capacity  multiply = 0.2 }
				any_state_province = {
					total_population > 1
					NOT = { has_province_modifier = recent_famine }
				}
			}
		}
	}

	immediate = {
		add_stability = -2
		random_governorships = {
			# [logfix] the limits must guarantee a save-able province at EVERY level, or the
			# random pick can land on a shortage-governorship/state with no eligible province
			# and famine_province is never saved (Undefined event target below). Each level now
			# requires a descendant province that is populated AND not already famine-flagged.
			limit = {
				ECON_governorship_food_shortage_physical > 0.3
				any_governorship_state = {
					# [#1 boot] mirror the trigger: the state must genuinely lack STORED food (below 1/5
					# capacity) AND still have a save-able province, or the random pick could land on a
					# surplus/no-eligible-province state and famine_province is never saved.
					has_state_food_capacity > 0
					has_state_food < { value = has_state_food_capacity  multiply = 0.2 }
					any_state_province = {
						total_population > 1
						NOT = { has_province_modifier = recent_famine }
					}
				}
			}
			random_governorship_state = {
				limit = {
					# [#1 boot] only a state actually short of stored food starves its pops (see trigger).
					has_state_food_capacity > 0
					has_state_food < { value = has_state_food_capacity  multiply = 0.2 }
					any_state_province = {
						total_population > 1
						NOT = { has_province_modifier = recent_famine }
					}
				}
				random_state_province = {
					limit = {
						total_population > 1
						NOT = {
							has_province_modifier = recent_famine
						}
					}
					save_scope_as = famine_province
					add_province_modifier = {
						name = recent_famine
						duration = 1095
					}
					random_pops_in_province = {
						weight = {
							modifier = {
								factor = 100
								NOT = {
									pop_type = upper_strata
								}
							}
							modifier = {
								factor = 75
								NOT = {
									pop_type = middle_strata
								}
							}
						}
						kill_pop = yes
					}
				}
			}
			change_variable = {
				name = upper_strata_wealth
				subtract = {
					value = WEALTH_governorship_per_capita
					divide = 7
				}
			}
			change_variable = {
				name = middle_strata_wealth
				subtract = {
					value = WEALTH_governorship_per_capita
					divide = 7
				}
			}
			change_variable = {
				name = lower_strata_wealth
				subtract = {
					value = WEALTH_governorship_per_capita
					divide = 7
				}
			}
			change_variable = {
				name = proletariat_wealth
				subtract = {
					value = WEALTH_governorship_per_capita
					divide = 7
				}
			}
			change_variable = {
				name = tribesmen_wealth
				subtract = {
					value = WEALTH_governorship_per_capita
					divide = 7
				}
			}
		}
	}

	option = {
		name = "shortage.1.a"
		add_treasury = {
			value = var:country_unit_price_grain
			add = var:country_unit_price_livestock
			add = var:country_unit_price_vegetables
			add = var:country_unit_price_temperate_fruit
			add = var:country_unit_price_fish
			multiply = scope:famine_province.total_population
			multiply = -1
		}
		add_political_influence = -20
	}

	option = {
		name = "shortage.1.b"
		scope:famine_province = {
			add_province_modifier = {
				name = recent_famine_ignored
				duration = 1095
			}
			governorship.governor ?= {
				add_loyalty = loyalty_governor_let_down
				add_popularity = -10
			}
		}
		add_stability = -5
		current_ruler = {
			add_popularity = -5
		}
	}

}

########################################################################
# asia_napoleonic_events.txt — [scripted-revolutions] the RIPPLE of the French
# Revolutionary / Napoleonic Wars onto the EUROPEAN COLONIES IN ASIA, and the
# OPENING it creates for the Qing (南洋 Nanyang / South Seas).
#
# When the wars in Europe cut the Asian colonies off from their metropoles, they are
# left isolated and undermanned — the historical window (~1795-1815) in which a
# stronger Qing could have pressed into the archipelago and the subcontinent. These
# events STAMP the "cut off" state (a country modifier + a global flag) on the
# affected colonies; the Qing NANYANG mission tree + the India tree read those flags
# as the opening. They do NOT force the Qing to act (the Qing is the player — D0:
# offer, never force); they set the stage and notify.
#
# Non-Qing colonies (DEI, PHI, EIC) are AI and may be railroaded — so where history
# had the metropole actually lose control (Batavian Republic 1795; British Java 1811),
# the event applies the concrete modifier outright.
#
#   .1  1795  THE BATAVIAN REPUBLIC — the Dutch Republic falls to France; the Dutch
#             East Indies (DEI) are cut off from Amsterdam. (fires when NED is a French
#             satellite / at the 1795 date.) Opens the DEI window.
#   .2  1808  THE PENINSULAR WAR — Spain is overrun and Manila (PHI) is isolated from
#             a Spain fighting for its life. Opens the PHI window.
#   .3  1798  THE ORIENT THREATENED — Bonaparte in Egypt + the Mysore crisis stretch
#             the East India Company to its limit. Reinforces the India (#449) opening.
#   .4  1816  THE EUROPEANS RETURN — with Napoleon beaten, the Dutch and the Company
#             come back to the East (Java restored 1816). The window begins to close;
#             a Qing that did not act loses the chance. (Cleanup / closing beat.)
#
# All hidden one-shot, seeded from asia_napoleonic_on_actions.txt at §4 offsets.
# ROOT carrier varies; effects are self-scoping via absolute c:TAG. sys = ASIA.
#
# Sources: Leonard Blussé, Strange Company (1986) & Visible Cities (2008); M.C.
#   Ricklefs, A History of Modern Indonesia (2001); Tim Hannigan, Raffles and the
#   British Invasion of Java (2012); William Dalrymple, The Anarchy (2019); C.A.
#   Bayly, Imperial Meridian (1989).
########################################################################

# ======================================================================
# .1  1795.01.19  THE BATAVIAN REPUBLIC. The French overrun the Dutch Republic;
# the Stadtholder flees to England and issues the Kew Letters, and the Dutch East
# Indies are severed from a metropole that is now a French satellite. Batavia is on
# its own — the VOC is bankrupt (dissolved 1799) and no help will come from Europe.
# Stamp the "cut off" state on DEI and open the Nanyang window (global flag the Qing
# missions read). ROOT carrier = c:CHI (the player, so the notify is player-facing).
# ======================================================================
asia_napoleonic.1 = {
	type = country_event
	title = "asia_napoleonic.1.t"
	desc = "asia_napoleonic.1.desc"
	picture = senate
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_variable = asia_dei_cutoff }
		exists = c:DEI
	}
	immediate = {
		set_global_variable = asia_dei_cutoff
		# the colony is cut off from Amsterdam: a lasting weakness the Qing can exploit.
		c:DEI = {
			if = {
				limit = { NOT = { has_country_modifier = asia_colony_cut_off } }
				add_country_modifier = { name = asia_colony_cut_off  duration = -1 }
			}
		}
		LOG_line = { sys = ASIA  msg = "EVENT asia_napoleonic.1 (Batavian Republic — Dutch East Indies cut off, 1795)" }
	}
	# player-facing notify ONLY if the Qing is the one seeing it (CHI); AI just gets the modifier.
	option = {
		name = "asia_napoleonic.1.a"
		trigger = { tag = CHI }
	}
	option = {
		name = "asia_napoleonic.1.b"
		trigger = { NOT = { tag = CHI } }
	}
}

# ======================================================================

# his brother Joseph on the throne; Spain rises in revolt and is plunged into a war

# ======================================================================

# road to India, and in Mysore, Tipu Sultan — the Company's most dangerous foe —

# East India Company is stretched to its limit fighting for its life in the south. Open
# (reinforce) the India window that the #449 tree already reads. ROOT carrier = c:CHI.

			}

# ======================================================================
# .4  1816.08.19  THE EUROPEANS RETURN. With Napoleon beaten and the peace signed at
# Vienna, the powers of Europe turn back to the East: the British hand Java back to a
# restored Netherlands (Anglo-Dutch Treaty), the Company resumes its advance, and Spain
# reasserts Manila. The window begins to close — a Qing that seized the moment holds
# what it took; one that did not has let the chance slip. Clears the cut-off modifiers
# (the metropoles are back), but leaves any Qing gains standing. ROOT carrier = c:CHI.
# ======================================================================
asia_napoleonic.4 = {
	type = country_event
	title = "asia_napoleonic.4.t"
	desc = "asia_napoleonic.4.desc"
	picture = throneroom
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_variable = asia_window_closed }
		OR = {
			has_global_variable = asia_dei_cutoff
			has_global_variable = asia_phi_cutoff
		}
	}
	immediate = {
		set_global_variable = asia_window_closed
		if = { limit = { exists = c:DEI  c:DEI = { has_country_modifier = asia_colony_cut_off } }  c:DEI = { remove_country_modifier = asia_colony_cut_off } }
		if = { limit = { exists = c:PHI  c:PHI = { has_country_modifier = asia_colony_cut_off } }  c:PHI = { remove_country_modifier = asia_colony_cut_off } }
		LOG_line = { sys = ASIA  msg = "EVENT asia_napoleonic.4 (The Europeans return — the Nanyang window closes, 1816)" }
	}
	option = { name = "asia_napoleonic.4.a"  trigger = { tag = CHI } }
	option = { name = "asia_napoleonic.4.b"  trigger = { NOT = { tag = CHI } } }
}

# ======================================================================
# .5  1812.06.24  THE TSAR TURNS WEST. Napoleon's six hundred thousand cross the Niemen,
# and every regiment, every rouble, every ounce of the Tsar's attention is drawn to the
# fight for the life of Russia. The Russian Far East — always thin, garrisoned by a
# handful of Cossacks and cut off from Moscow by five thousand versts of taiga — is
# stripped barer still. The Amur frontier the Romanovs won at Nerchinsk (1689) lies all
# but undefended. For a High Qing with the will, the northern rivers are open as they
# have not been in a century. Stamps the "distracted" state on RUS and opens the Amur
# window (a global flag the colonization tree's Amur branch reads). ROOT carrier = c:CHI.
# Fires in step with fra_revolution.14 (the invasion of Russia).
# ======================================================================
asia_napoleonic.5 = {
	type = country_event
	title = "asia_napoleonic.5.t"
	desc = "asia_napoleonic.5.desc"
	picture = war_council
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_variable = asia_russia_distracted }
		exists = c:RUS
	}
	immediate = {
		set_global_variable = asia_russia_distracted
		# Russia's Far East is stripped bare while the Tsar fights for his life in the west.
		c:RUS = {
			if = {
				limit = { NOT = { has_country_modifier = asia_russia_pinned_west } }
				add_country_modifier = { name = asia_russia_pinned_west  duration = 1095 }   # ~3 yrs (1812-1814/15)
			}
		}
		LOG_line = { sys = ASIA  msg = "EVENT asia_napoleonic.5 (Napoleon invades Russia — the Far East stripped bare, 1812)" }
	}
	option = { name = "asia_napoleonic.5.a"  trigger = { tag = CHI } }
	option = { name = "asia_napoleonic.5.b"  trigger = { NOT = { tag = CHI } } }
}

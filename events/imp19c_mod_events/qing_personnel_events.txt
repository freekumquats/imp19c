########################################################################
# qing_personnel_events.txt — Personnel Ministry (吏部) governance events.
# The Grand Minister of Personnel oversees the provincial governorate; these
# events fire when the minister and his charges cooperate or clash, when the
# triennial 大計 reviews the corps, and when corruption infiltrates a
# governorship.
#
# Namespace: qing_personnel
# All country_event, CHI only, is_triggered_only = yes.
########################################################################

namespace = qing_personnel

# ======================================================================
# qing_personnel.1 — the 大計 (Great Reckoning) triennial evaluation.
# Fired from the flavour roll (see RETURN PAYLOAD dispatcher splice) once
# every three years. Reviews every governor, rewards the able, cashiers
# the venal. Gates outcomes on the Personnel holder's finesse + loyalty.
# ======================================================================
qing_personnel.1 = {
	type = country_event
	title = "qing_personnel.1.t"
	desc = "qing_personnel.1.desc"
	picture = chinese_throne_room
	left_portrait = root.current_ruler
	is_triggered_only = yes
	
	trigger = {
		tag = CHI
		# must have governors to review
		any_character = {
			count >= 1
			employer = ROOT
			is_governor = yes
			is_alive = yes
		}
	}
	
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_personnel.1 (大計 triennial review) [ROOT.GetTag]" }
		# [#117 integ-fix] triennial spacing. The original splice tried to mutate a
		# year-stamp INSIDE the roll trigger (invalid — a trigger cannot mutate). Instead
		# stamp a day-expiry cooldown here in the immediate; the dispatcher gate is a pure
		# read (NOT has_variable), so the 大計 comes at most ~once per 3 years.
		set_variable = { name = qing_personnel_daji_cooldown  days = 1095  value = 1 }
		# if Personnel office filled, save the minister for portrait + text
		if = {
			limit = {
				has_variable = qing_office_personnel_holder
				var:qing_office_personnel_holder = {
					is_alive = yes
					employer = ROOT
				}
			}
			var:qing_office_personnel_holder = { save_scope_as = personnel_minister }
		}
	}
	
	# OPTION A: conduct the 大計 under a strong Personnel minister (finesse>=9)
	option = {
		name = "qing_personnel.1.a"
		custom_tooltip = "qing_personnel.1.a.tt"
		trigger = {
			has_variable = qing_office_personnel_holder
			var:qing_office_personnel_holder = {
				is_alive = yes
				employer = ROOT
				finesse >= 9
			}
		}
		QING_personnel_daji_review = yes
		# the realm benefits from rigorous oversight
		add_country_modifier = { name = qing_personnel_daji_rigorous  duration = 1095 }
		add_political_influence = 10
	}
	
	# OPTION B: conduct the 大計 with a weaker Personnel minister (finesse<9)
	option = {
		name = "qing_personnel.1.b"
		custom_tooltip = "qing_personnel.1.b.tt"
		trigger = {
			has_variable = qing_office_personnel_holder
			var:qing_office_personnel_holder = {
				is_alive = yes
				employer = ROOT
				finesse < 9
			}
		}
		QING_personnel_daji_review = yes
		# the review is conducted, but outcomes are modest
		LOG_line = { sys = QING  msg = "大計 conducted under weaker Personnel minister for [ROOT.GetTag]" }
	}
	
	# OPTION C: the Personnel office is VACANT — the review is perfunctory
	option = {
		name = "qing_personnel.1.c"
		custom_tooltip = "qing_personnel.1.c.tt"
		trigger = {
			NOT = {
				has_variable = qing_office_personnel_holder
				var:qing_office_personnel_holder = {
					is_alive = yes
					employer = ROOT
				}
			}
		}
		# no minister = drift; governors are merely noted, no rewards/cashiering
		every_character = {
			limit = {
				employer = ROOT
				is_governor = yes
				is_alive = yes
			}
			add_loyalty = loyalty_qing_delta_n1
		}
		add_country_modifier = { name = qing_personnel_daji_perfunctory  duration = 1095 }
		LOG_line = { sys = QING  msg = "大計 perfunctory (no Personnel minister) for [ROOT.GetTag]" }
	}
}

# ======================================================================
# qing_personnel.2 — clash between Personnel minister and a proud governor.
# Queued from QING_personnel_evaluate_governors when high friction + low
# throne-affinity. The governor and minister grate; the throne must choose:
# recall the governor (lose his service but appease the minister), indulge
# the governor (strain the minister), or balance them (both discontent).
# ======================================================================
qing_personnel.2 = {
	type = country_event
	title = "qing_personnel.2.t"
	desc = "qing_personnel.2.desc"
	picture = senate_debate
	left_portrait = scope:personnel_minister
	right_portrait = scope:clashing_governor
	is_triggered_only = yes
	
	trigger = {
		tag = CHI
		# Personnel office must be filled
		has_variable = qing_office_personnel_holder
		var:qing_office_personnel_holder = {
			is_alive = yes
			employer = ROOT
		}
		# must have a governor who clashes with the minister
		any_character = {
			count >= 1
			employer = ROOT
			is_governor = yes
			is_alive = yes
		}
	}
	
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_personnel.2 (minister-governor clash) [ROOT.GetTag]" }
		var:qing_office_personnel_holder = { save_scope_as = personnel_minister }
		# pick a governor who is likely to clash (low loyalty or high prominence)
		random_character = {
			limit = {
				employer = ROOT
				is_governor = yes
				is_alive = yes
				OR = {
					loyalty < 50
					prominence >= 50
				}
			}
			save_scope_as = clashing_governor
		}
		# if no such governor, pick any governor
		if = {
			limit = { NOT = { exists = scope:clashing_governor } }
			random_character = {
				limit = {
					employer = ROOT
					is_governor = yes
					is_alive = yes
				}
				save_scope_as = clashing_governor
			}
		}
	}
	
	# OPTION A: recall the governor (appease the minister, lose the governor's service)
	option = {
		name = "qing_personnel.2.a"
		custom_tooltip = "qing_personnel.2.a.tt"
		scope:clashing_governor = {
			add_loyalty = loyalty_qing_delta_n15
			add_prominence = -5
			# in Imperator we cannot dismiss a governor directly; instead penalize heavily
		}
		scope:personnel_minister = {
			add_loyalty = loyalty_qing_delta_p5
			QING_char_promote_standing = { pop = 2  prom = 1 }
		}
		# the realm sees the minister's authority upheld
		add_political_influence = 5
	}
	
	# OPTION B: indulge the governor (preserve his service, strain the minister)
	option = {
		name = "qing_personnel.2.b"
		custom_tooltip = "qing_personnel.2.b.tt"
		scope:clashing_governor = {
			add_loyalty = loyalty_qing_delta_p3
			QING_char_promote_standing = { pop = 1  prom = 1 }
		}
		scope:personnel_minister = {
			add_loyalty = loyalty_qing_delta_n5
			# if the minister is corrupt or weak-willed, indulging the governor worsens governance
			if = {
				limit = { corruption >= 20 }
				ROOT = {
					QING_DECLINE_nudge = { var = qing_corruption_level  amount = 2 }
				}
			}
		}
	}
	
	# OPTION C: balance them (both discontent, but manageable)
	option = {
		name = "qing_personnel.2.c"
		custom_tooltip = "qing_personnel.2.c.tt"
		scope:clashing_governor = {
			add_loyalty = loyalty_qing_delta_n2
		}
		scope:personnel_minister = {
			add_loyalty = loyalty_qing_delta_n2
		}
		# cost in political capital
		add_political_influence = -5
	}
}

# ======================================================================
# qing_personnel.3 — talented provincial official flagged for promotion.
# Queued from QING_personnel_evaluate_governors when a governor is congenial
# with the throne AND Personnel office is strong. A capable governor is
# identified for advancement; the throne can promote him to a great office
# (if vacancy exists), reward him, or decline.
# ======================================================================
qing_personnel.3 = {
	type = country_event
	title = "qing_personnel.3.t"
	desc = "qing_personnel.3.desc"
	picture = chinese_throne_room
	left_portrait = root.current_ruler
	right_portrait = scope:talented_governor
	is_triggered_only = yes
	
	trigger = {
		tag = CHI
		# Personnel office must be filled with a capable minister
		has_variable = qing_office_personnel_holder
		var:qing_office_personnel_holder = {
			is_alive = yes
			employer = ROOT
			finesse >= 9
		}
		# must have a talented governor (finesse>=9 OR martial>=9)
		any_character = {
			count >= 1
			employer = ROOT
			is_governor = yes
			is_alive = yes
			OR = {
				finesse >= 9
				martial >= 9
			}
			loyalty >= 60
		}
	}
	
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_personnel.3 (talented governor promotion) [ROOT.GetTag]" }
		var:qing_office_personnel_holder = { save_scope_as = personnel_minister }
		# pick a talented, loyal governor
		random_character = {
			limit = {
				employer = ROOT
				is_governor = yes
				is_alive = yes
				OR = {
					finesse >= 9
					martial >= 9
				}
				loyalty >= 60
			}
			save_scope_as = talented_governor
		}
		# check for vacant great offices
		set_variable = { name = qing_personnel_vacant_offices  value = 0 }
		if = { limit = { NOT = { has_variable = qing_office_personnel_holder } } change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_revenue_holder } }   change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_rites_holder } }     change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_war_holder } }       change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_justice_holder } }   change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_works_holder } }     change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_censor_holder } }    change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_lifanyuan_holder } } change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_chamberlain_holder } }     change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_zongli_holder } }    change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
		if = { limit = { NOT = { has_variable = qing_office_chancellor_holder } } change_variable = { name = qing_personnel_vacant_offices  add = 1 } }
	}
	
	# OPTION A: promote him to a vacant great office (if any exist)
	option = {
		name = "qing_personnel.3.a"
		custom_tooltip = "qing_personnel.3.a.tt"
		trigger = { var:qing_personnel_vacant_offices >= 1 }
		scope:talented_governor = {
			QING_office_appoint_first_vacant = yes
			add_loyalty = loyalty_qing_delta_p10
		}
		scope:personnel_minister = {
			add_loyalty = loyalty_qing_delta_p3
			QING_char_promote_standing = { pop = 1  prom = 1 }
		}
		add_political_influence = 10
	}
	
	# OPTION B: reward him richly but leave him in his governorship
	option = {
		name = "qing_personnel.3.b"
		custom_tooltip = "qing_personnel.3.b.tt"
		scope:talented_governor = {
			add_loyalty = loyalty_qing_delta_p5
			QING_char_endow = { amount = 50 }
			QING_char_honour_family = { amount = 15 }
		}
		scope:personnel_minister = {
			add_loyalty = loyalty_qing_delta_p1
		}
		add_treasury = -50
	}
	
	# OPTION C: decline (no cost, but morale suffers)
	option = {
		name = "qing_personnel.3.c"
		custom_tooltip = "qing_personnel.3.c.tt"
		scope:talented_governor = {
			add_loyalty = loyalty_qing_delta_n3
		}
		scope:personnel_minister = {
			add_loyalty = loyalty_qing_delta_n2
		}
	}
}

# ======================================================================
# qing_personnel.4 — 捐納 (purchased office) scandal in a governorship.
# Queued from the flavour roll when corruption is high. A governor is
# discovered to have bought his post; the throne must investigate (cost
# in political capital but purges corruption), ignore it (corruption
# worsens), or make an example (harsh but effective).
# ======================================================================
qing_personnel.4 = {
	type = country_event
	title = "qing_personnel.4.t"
	desc = "qing_personnel.4.desc"
	picture = senate_debate
	left_portrait = root.current_ruler
	right_portrait = scope:venal_governor
	is_triggered_only = yes
	
	trigger = {
		tag = CHI
		# high corruption
		has_variable = qing_corruption_level
		var:qing_corruption_level >= 50
		# must have a corrupt governor
		any_character = {
			count >= 1
			employer = ROOT
			is_governor = yes
			is_alive = yes
			OR = {
				has_trait = corrupt
				corruption >= 40
			}
		}
	}
	
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_personnel.4 (捐納 scandal) [ROOT.GetTag]" }
		# pick a corrupt governor
		random_character = {
			limit = {
				employer = ROOT
				is_governor = yes
				is_alive = yes
				OR = {
					has_trait = corrupt
					corruption >= 40
				}
			}
			save_scope_as = venal_governor
		}
		# if Personnel office filled, save the minister
		if = {
			limit = {
				has_variable = qing_office_personnel_holder
				var:qing_office_personnel_holder = {
					is_alive = yes
					employer = ROOT
				}
			}
			var:qing_office_personnel_holder = { save_scope_as = personnel_minister }
		}
	}
	
	# OPTION A: investigate rigorously (purges corruption, costs political capital)
	option = {
		name = "qing_personnel.4.a"
		custom_tooltip = "qing_personnel.4.a.tt"
		scope:venal_governor = {
			QING_char_cleanse = yes
			add_loyalty = loyalty_qing_delta_n10
			add_prominence = -5
		}
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -5 }
		add_political_influence = -10
		# if Personnel minister exists and is capable, the purge is more effective
		if = {
			limit = {
				exists = scope:personnel_minister
				scope:personnel_minister = { finesse >= 9 }
			}
			QING_DECLINE_nudge = { var = qing_corruption_level  amount = -5 }
			scope:personnel_minister = {
				add_loyalty = loyalty_qing_delta_p3
			}
		}
	}
	
	# OPTION B: ignore the scandal (corruption worsens, but no immediate cost)
	option = {
		name = "qing_personnel.4.b"
		custom_tooltip = "qing_personnel.4.b.tt"
		scope:venal_governor = {
			add_loyalty = loyalty_qing_delta_p2
		}
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 5 }
		# if Personnel minister exists, he is demoralized by inaction
		if = {
			limit = { exists = scope:personnel_minister }
			scope:personnel_minister = {
				add_loyalty = loyalty_qing_delta_n3
			}
		}
	}
	
	# OPTION C: make an example (execute or exile) — harsh but effective
	option = {
		name = "qing_personnel.4.c"
		custom_tooltip = "qing_personnel.4.c.tt"
		scope:venal_governor = {
			death = {
				death_reason = death_execution
			}
		}
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -8 }
		# harsh action breeds fear, but also resentment among other corrupt officials
		every_character = {
			limit = {
				employer = ROOT
				OR = {
					has_trait = corrupt
					corruption >= 30
				}
			}
			add_loyalty = loyalty_qing_delta_n3
		}
		add_tyranny = 5
	}
}

# .5  CONVENE-THE-大計 TRAMPOLINE  (hidden)  [#443 boot-crash fix]
# The panel lever qing_personnel_convene_daji (QING_personnel_panel.txt) used to call
# QING_personnel_daji_review directly in its scripted_gui effect. A scripted_gui compile-inlines
# its button's entire effect chain at PARSE time, and QING_personnel_daji_review's body is an
# `every_character { ... }` CHARACTER ITERATOR whose loop body in turn inlines
# QING_char_promote_standing / QING_char_cleanse / QING_char_honour_family — each expanding the
# ~80-line QING_char_affinity THREE times inside the iterator. A heavy-bodied character iterator
# inlined into a compiled button access-violates the loader (EXCEPTION_ACCESS_VIOLATION, zero
# PostValidate) — the same crash class as the guard legion-iterator (bf5c46015) and the
# censorate/harem/justice sorting iterators. FIX: the button now trigger_events this hidden event;
# trigger_event is a RUNTIME ref, never compile-inlined, so the reckoning lives in an event unit
# the parser does not expand. The event owns the review + cooldown stamp + recompute, so the panel
# lever and the qing_personnel.1 flavour event still share one throttle. Behaviour is identical.
########################################################################
qing_personnel.5 = {
	type = country_event
	hidden = yes
	is_triggered_only = yes
	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_personnel.5 (大計 Great Reckoning trampoline) [ROOT.GetTag]" }
		QING_personnel_daji_review = yes
		# stamp the ~3-year throttle (shared with the qing_personnel.1 flavour event).
		set_variable = { name = qing_personnel_daji_cooldown  days = 1095  value = 1 }
		# refresh so the meter + corps integrity reflect the reckoning's outcomes.
		QING_ministry_recompute_perf_personnel = yes
	}
}

# After the convene-大計 trampoline (.5), panel bisection proved personnel was STILL the sole
# crasher: the user booted 22/23 panels clean with ONLY personnel disabled. Static discriminators
# (expanded-size, iterator-nest-depth) could NOT explain it — justice's structurally-identical
# recompute inlines the same every_character and boots. So, per the proven Study-panel cure, EVERY
# effect-bearing personnel button is now routed off the compiled parse path. These three events own
# the effects the open / cultivate / discipline buttons used to inline. Behaviour is identical.

#   .7  CULTIVATE A GOVERNOR (培養)  (character; ROOT = governor, scope:player carries — cf. justice.6)

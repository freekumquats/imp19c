# qing_nanyang_events.txt — [scripted-revolutions 2nd pass] Qing player-facing
# HISTORICAL-CHOICE events surfaced by the research agents (EN/FR/RU/CJK).
#
# These are the real, datable Qing engagements with the Nanyang (南洋) and the wider
# world in 1763-1815 that the mission trees did not yet capture. Because the QING is
# the PLAYER (D0), every one OFFERS a choice — it never forces. Each is a one-shot,
# CHI-gated, done-flag-guarded country_event with historically-weighted options that
# deliver real mechanical outcomes. Seeded from qing_nanyang_on_actions.txt at the
# §4 / 2nd-pass offsets (scripted_revolutions.md). sys = QING.
#
#   .1  1772  THE CHINESE KING OF SIAM (暹羅華王) — recognise Taksin, the half-Chinese
#             king backed by the Teochew merchants.
#   .2  1788  THE VIETNAM INTERVENTION (安南之役) — Lê Chiêu Thống begs aid against the
#             Tây Sơn; refuse / historical (the Ngọc Hồi rout) / commit decisively.
#   .3  1795  THE LANFANG SUCCESSION (蘭芳承統) — Luo Fangbo dies; back the elected
#             successors or let the kongsi drift toward the Dutch.
#   .4  1799  AVENGE THE HONGXI MASSACRE (紅溪之恨) — the VOC's fall + the old Batavia
#             pogrom (1740); protest, retaliate on trade, or let it lie.
#   .5  1786  THE MATTER OF PENANG (檳城之議) — the British plant a free port in the
#             Siamese sphere; press the tributary, compete, or ignore.
#
# [scripted-revolutions 2nd pass] The North American arc — the Louisiana claim, the War
# of 1812 opening, and Russian America/Alaska — lives in its own namespace, qing_americas
# (events/imp19c_mod_events/qing_americas_events.txt), NOT here (nanyang = the South Seas).
#
# Sources (research agents): Wang Wensheng, White Lotus Rebels & South China Pirates
#   (2014); Leonard Blussé; Carl Trocki; Yuan Bingling on the kongsi; Wei Yuan on the
#   tributaries; standard Tây Sơn / Taksin / Louisiana Purchase scholarship.
########################################################################

# ======================================================================
# .1  1772.01.01  THE CHINESE KING OF SIAM (暹羅華王). Taksin (鄭昭/鄭信) — son of a
# Teochew tax-farmer from Chenghai, Guangdong — has reconsolidated Siam from Thonburi
# and seeks the Qing's recognition. Historically Qianlong hesitated, then recognised him
# in 1772. Offer: embrace the special relationship, recognise coolly, or reject him as a
# usurper. Gated on Siam (SIA) existing.
# ======================================================================
qing_nanyang.1 = {
	type = country_event
	title = "qing_nanyang.1.t"
	desc = "qing_nanyang.1.desc"
	picture = chinese_throne_room
	is_triggered_only = yes
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_nanyang_beat_taksin }
		exists = c:SIA
	}
	immediate = {
		set_variable = qing_nanyang_beat_taksin
		LOG_line = { sys = QING  msg = "EVENT qing_nanyang.1 (The Chinese King of Siam, Taksin, 1772) for [ROOT.GetTag]" }
	}
	# EMBRACE — honour his heritage, take Siam warmly under the wing.
	option = {
		name = "qing_nanyang.1.a"
		if = {
			limit = { exists = c:SIA  NOT = { c:SIA = { is_subject_of = ROOT } } }
			FUNC_make_subject = { overlord = ROOT  target = c:SIA  type = sinosphere_tributary }
		}
		add_prestige = 10
		current_ruler = { add_popularity = 8 }
		ai_chance = { factor = 5 }
	}
	# RECOGNISE COOLLY — the historical middle path: acknowledge, no special tie.
	option = {
		name = "qing_nanyang.1.b"
		add_prestige = 5
		ai_chance = { factor = 10 }
	}
	# REJECT AS USURPER — withhold recognition; a chill on the southern trade.
	option = {
		name = "qing_nanyang.1.c"
		add_prestige = -5
		if = {
			limit = { exists = c:SIA }
			c:SIA = { add_opinion = { modifier = qing_gp_rivalry_opinion  target = ROOT } }
		}
		ai_chance = { factor = 2 }
	}
}

# ======================================================================
# .2  1788.11.01  THE VIETNAM INTERVENTION (安南之役). Lê Chiêu Thống, last of the Lê,
# has fled the Tây Sơn and begs the Son of Heaven to restore him. Historically Qianlong
# sent an army that took Thăng Long, then was routed by Nguyễn Huệ at Ngọc Hồi-Đống Đa
# in the 1789 New Year. Offer: refuse (spare the disaster), the historical intervention
# (glory then rout), or commit decisively (a winter-ready host — win but bleed).
# ROOT = CHI. TRH (Trịnh/Tonkin) stands in for the northern Vietnamese theatre.
# ======================================================================
qing_nanyang.2 = {
	type = country_event
	title = "qing_nanyang.2.t"
	desc = "qing_nanyang.2.desc"
	picture = war_council
	is_triggered_only = yes
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_nanyang_beat_vietnam }
	}
	immediate = {
		set_variable = qing_nanyang_beat_vietnam
		LOG_line = { sys = QING  msg = "EVENT qing_nanyang.2 (The Vietnam Intervention, 1788) for [ROOT.GetTag]" }
	}
	# REFUSE — no army marches south; spare the realm the disaster, at a cost in face.
	option = {
		name = "qing_nanyang.2.a"
		add_prestige = -5
		current_ruler = { add_popularity = -3 }
		ai_chance = { factor = 4 }
	}
	# THE HISTORICAL INTERVENTION — march in, install the Lê, then suffer the New Year rout.
	option = {
		name = "qing_nanyang.2.b"
		custom_tooltip = "qing_nanyang.2.b.tt"
		add_manpower = -12
		add_war_exhaustion = 8
		add_prestige = -8
		add_treasury = -60
		# a tributary tie is patched up afterward (the historical 1789 enfeoffment of Nguyễn Huệ).
		if = {
			limit = { exists = c:VIE  NOT = { c:VIE = { is_subject_of = ROOT } } }
			FUNC_make_subject = { overlord = ROOT  target = c:VIE  type = sinosphere_tributary }
		}
		ai_chance = { factor = 10 }
	}
	# COMMIT DECISIVELY — a winter-ready host, no New Year surprise: the throne prevails,
	# but at heavy cost. Take the northern Vietnamese theatre (TRH) firmly under the wing.
	option = {
		name = "qing_nanyang.2.c"
		custom_tooltip = "qing_nanyang.2.c.tt"
		add_manpower = -20
		add_war_exhaustion = 12
		add_treasury = -120
		add_prestige = 12
		current_ruler = { add_popularity = 10 }
		if = {
			limit = { exists = c:TRH  NOT = { c:TRH = { is_subject_of = ROOT } } }
			FUNC_make_subject = { overlord = ROOT  target = c:TRH  type = sinosphere_tributary }
		}
		if = {
			limit = { exists = c:VIE  NOT = { c:VIE = { is_subject_of = ROOT } } }
			FUNC_make_subject = { overlord = ROOT  target = c:VIE  type = sinosphere_tributary }
		}
		ai_chance = { factor = 3 }
	}
}

# ======================================================================
# .3  1795.01.01  THE LANFANG SUCCESSION (蘭芳承統). Luo Fangbo, founder of the Kongsi,
# has died; the elected successors look to Peking. Offer: affirm the tie (raise Lanfang
# to a firm tributary + a durable modifier), keep the loose trade tie (historical), or
# turn away (Lanfang drifts, easing into the Dutch grip). Gated on Lanfang (LAF) existing.
# ======================================================================
qing_nanyang.3 = {
	type = country_event
	title = "qing_nanyang.3.t"
	desc = "qing_nanyang.3.desc"
	picture = senate
	is_triggered_only = yes
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_nanyang_beat_lanfang_succession }
		exists = c:LAF
	}
	immediate = {
		set_variable = qing_nanyang_beat_lanfang_succession
		LOG_line = { sys = QING  msg = "EVENT qing_nanyang.3 (The Lanfang Succession, Luo Fangbo dies, 1795) for [ROOT.GetTag]" }
	}
	# AFFIRM — raise Lanfang to a firm tributary under the wing.
	option = {
		name = "qing_nanyang.3.a"
		if = {
			limit = { exists = c:LAF }
			FUNC_make_subject = { overlord = ROOT  target = c:LAF  type = sinosphere_tributary }
		}
		if = {
			limit = { NOT = { has_country_modifier = qing_nanyang_protector } }
			add_country_modifier = { name = qing_nanyang_protector  duration = -1 }
		}
		current_ruler = { add_popularity = 8 }
		ai_chance = { factor = 6 }
	}
	# LOOSE TRADE TIE — the historical arm's-length arrangement.
	option = {
		name = "qing_nanyang.3.b"
		add_prestige = 3
		ai_chance = { factor = 10 }
	}
	# TURN AWAY — the kongsi drifts; the Dutch will have their opening.
	option = {
		name = "qing_nanyang.3.c"
		add_prestige = -3
		ai_chance = { factor = 3 }
	}
}

# ======================================================================
# .4  1799.12.31  AVENGE THE HONGXI MASSACRE (紅溪之恨). The VOC is dissolved and its
# rule over the Chinese of Batavia passes to a cut-off colonial state — and the old
# wound of the 1740 Hongxi massacre (紅溪慘案, ~10,000 Chinese killed) is unhealed. Offer:
# a formal protest + protector stance, an embargo on Dutch trade at Canton, or let it lie.
# ======================================================================
qing_nanyang.4 = {
	type = country_event
	title = "qing_nanyang.4.t"
	desc = "qing_nanyang.4.desc"
	picture = looting
	is_triggered_only = yes
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_nanyang_beat_hongxi }
	}
	immediate = {
		set_variable = qing_nanyang_beat_hongxi
		LOG_line = { sys = QING  msg = "EVENT qing_nanyang.4 (Avenge the Hongxi Massacre, 1799) for [ROOT.GetTag]" }
	}
	# PROTEST + PROTECT — break with the old disdain; stand as protector of the 華僑.
	option = {
		name = "qing_nanyang.4.a"
		if = {
			limit = { NOT = { has_country_modifier = qing_nanyang_protector } }
			add_country_modifier = { name = qing_nanyang_protector  duration = -1 }
		}
		add_prestige = 6
		current_ruler = { add_popularity = 6 }
		if = {
			limit = { exists = c:DEI }
			c:DEI = { add_opinion = { modifier = qing_gp_rivalry_opinion  target = ROOT } }
		}
		ai_chance = { factor = 5 }
	}
	# EMBARGO — squeeze the Dutch at Canton; signals strength, costs some revenue.
	option = {
		name = "qing_nanyang.4.b"
		add_treasury = -20
		add_prestige = 3
		if = {
			limit = { exists = c:DEI }
			c:DEI = { add_opinion = { modifier = qing_gp_rivalry_opinion  target = ROOT } }
		}
		ai_chance = { factor = 4 }
	}
	# LET IT LIE — the court's traditional indifference to the men who sailed south.
	option = {
		name = "qing_nanyang.4.c"
		ai_chance = { factor = 10 }
	}
}

# ======================================================================
# .5  1786.08.11  THE MATTER OF PENANG (檳城之議). Francis Light has taken Penang from the
# Sultan of Kedah — a British free port planted at the mouth of the Straits, in the shadow
# of the Siamese tributary sphere. Offer: lean on the tributary (Siam/Kedah) to disown the
# deal, plant a rival Qing-backed free port, or let the barbarians have their island.
# ======================================================================
qing_nanyang.5 = {
	type = country_event
	title = "qing_nanyang.5.t"
	desc = "qing_nanyang.5.desc"
	picture = harbour
	is_triggered_only = yes
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_nanyang_beat_penang }
	}
	immediate = {
		set_variable = qing_nanyang_beat_penang
		LOG_line = { sys = QING  msg = "EVENT qing_nanyang.5 (The Matter of Penang, 1786) for [ROOT.GetTag]" }
	}
	# PRESS THE TRIBUTARY — lean on Siam to disown the British lease; provokes Britain.
	option = {
		name = "qing_nanyang.5.a"
		add_political_influence = -15
		QING_gp_provoke_britain = { severity = 6 }
		ai_chance = { factor = 4 }
	}
	# A RIVAL PORT — charter a Qing-backed free port for the kongsi traders; commerce lift.
	option = {
		name = "qing_nanyang.5.b"
		add_treasury = -40
		if = {
			limit = { NOT = { has_country_modifier = qing_nanyang_protector } }
			add_country_modifier = { name = qing_nanyang_protector  duration = -1 }
		}
		ai_chance = { factor = 5 }
	}
	# LET IT PASS — the historical indifference; the Straits slip toward Britain.
	option = {
		name = "qing_nanyang.5.c"
		ai_chance = { factor = 10 }
	}
}

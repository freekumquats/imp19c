########################################################################
# qing_embassy_events.txt — WESTERN EMBASSIES TO THE COURT (西使覲見 / 貢使問題).
# The INBOUND diplomatic crises of the engine (se_QING_EMBASSY.txt): a Western
# power arrives seeking audience and trade, and the court must choose between the
# tribute ceremony (rebuff / demand the kowtow) and a modern reception (receive /
# open limited trade). CHI-only, fired from the Qing flavour roll.
#
# Each embassy re-checks its own trigger (power on the map + not yet resolved),
# so a flavour-roll pick that no longer applies is a silent no-op — the same
# convention the vassal/treaty/missionary crises use.
#
#   .1  Macartney (馬戛爾尼, 1793)  — Britain, first embassy   → britain verbs
#   .2  Amherst   (阿美士德, 1816)  — Britain, second embassy  → britain verbs
#   .3  Titsingh  (德勝, 1795)      — the Dutch VOC embassy     → netherlands verbs
#   .4  Golovkin  (戈洛夫金, 1805)  — Russia, Kyakhta frontier  → russia verbs
#   .5  a French embassy                                        → france verbs
#   .6  Cushing / Wanghia (顧盛, 1844) — the American mission   → america verbs
#
# The two British events share the britain verbs (guarded once per power in the
# effect): whichever fires first resolves Britain; the other is then a no-op.
#
# Sources: se_QING_EMBASSY.txt header.
########################################################################

namespace = qing_embassy

# ======================================================================
# .1  THE MACARTNEY EMBASSY (馬戛爾尼使團, 1793). Britain's first great embassy —
# gifts, a telescope, a hot-air balloon, and demands for a resident minister and
# open ports. Historically met with Qianlong's edict of self-sufficiency.
#
# NOTE: Macartney (1793) predates the 1815 start by a generation, so this event is
# NO LONGER FIRED (the flavour roll offers only the Amherst embassy, qing_embassy.2).
# It is REFERENCED narratively in the Amherst text ("the demands Macartney could not
# win a generation ago"). The definition is retained (loc + a valid britain receive/
# rebuff path) as safe legacy in case a scenario ever wants it, but in normal play it
# does not occur. The Emperor Emeritus kickoff (#65) hooks ONLY the Amherst reception,
# which is also the historically-correct St Helena caller.
# ======================================================================
qing_embassy.1 = {
	type = country_event
	title = "qing_embassy.1.t"
	desc = "qing_embassy.1.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	# [bookmark-1763 #304-fix] Re-entry guard is Macartney's OWN occurrence flag, NOT the
	# shared qing_embassy_britain flag. At a 1763 start Macartney (1793) and Amherst (1816)
	# are BOTH scheduled; both used to guard on qing_embassy_britain, so Macartney's
	# resolution (which sets qing_embassy_britain via the britain triangle wrapper) silently
	# blocked Amherst — killing the only path into qing_napoleon.5 (the #65 Emeritus/Napoleon
	# chain). Per-embassy flags let each fire once, independently. qing_embassy.1 is fired ONLY
	# by the 1763 dated scheduler (the flavour roll offers only Amherst), so this is 1763-only.
	trigger = {
		tag = CHI
		exists = c:GBR
		NOT = { has_variable = qing_embassy_macartney_done }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_embassy.1 (Macartney) [ROOT.GetTag]" }
		set_variable = { name = qing_embassy_macartney_done  value = 1 }
		QING_embassy_init = yes
	}

	# --- Stand on the ceremony — the edict of self-sufficiency ------------------
	option = {
		name = "qing_embassy.1.rebuff"
		custom_tooltip = "qing_embassy.1.rebuff.tt"
		QING_embassy_britain_rebuff = yes
	}

	# --- Receive the embassy — open a channel of trade -------------------------
	option = {
		name = "qing_embassy.1.receive"
		custom_tooltip = "qing_embassy.1.receive.tt"
		QING_embassy_britain_receive = yes
	}

	# --- EMBRACE the barbarians' arts — the great divergence -------------------
	#     The counterfactual heart of the 1763 start: instead of "the Celestial Empire
	#     possesses all things", Qianlong resolves to LEARN what the English brought.
	#     Launches early industrialization ~67 years before the historical Self-
	#     Strengthening Movement (se_QING_EARLYINDUS.txt), reusing the SS engine. This is
	#     the strong-receive: it opens trade AND commits the dynasty to modernization —
	#     at the price of a fierce conservative-court reaction. Shown only at the 1763
	#     start (the divergence flag is unset; the era is early enough to matter).
	option = {
		name = "qing_embassy.1.embrace"
		custom_tooltip = "qing_embassy.1.embrace.tt"
		trigger = { NOT = { has_variable = qing_early_industrialization } }
		# the audience is still received (goodwill + trade), then the divergence launches.
		QING_embassy_britain_receive = yes
		QING_earlyindus_launch = yes
		# open the program hub so the player can drive the remaining foundations.
		trigger_event = { id = qing_embassy.10  days = { 20 40 } }
	}
}

# ======================================================================
# .2  THE AMHERST EMBASSY (阿美士德使團, 1816). Britain's second embassy — the year
# after this mod's start. Historically expelled from Peking over the kowtow
# without ever seeing the Jiaqing Emperor. The sharper snub.
# ======================================================================
qing_embassy.2 = {
	type = country_event
	title = "qing_embassy.2.t"
	desc = "qing_embassy.2.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	# [bookmark-1763 #304-fix] Re-entry guard is Amherst's OWN occurrence flag, NOT the shared
	# qing_embassy_britain flag. Formerly this guarded on qing_embassy_britain, which Macartney's
	# resolution (1763 start) also sets — silently killing this event and, with it, the sole path
	# into qing_napoleon.5. The resolution wrappers STILL set qing_embassy_britain (so the 1815
	# decline-roll britain branch stands down as before, se_QING_DECLINE.txt:1026); only this
	# re-entry guard moves to a per-embassy flag. At 1815 the #80 pin fires Amherst exactly once
	# and the flavour-roll fallback still no-ops once it has fired — behaviour preserved.
	trigger = {
		tag = CHI
		exists = c:GBR
		NOT = { has_variable = qing_embassy_amherst_done }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_embassy.2 (Amherst) [ROOT.GetTag]" }
		set_variable = { name = qing_embassy_amherst_done  value = 1 }
		QING_embassy_init = yes
	}

	# --- Expel the embassy over the kowtow -------------------------------------
	option = {
		name = "qing_embassy.2.rebuff"
		custom_tooltip = "qing_embassy.2.rebuff.tt"
		QING_embassy_britain_rebuff_sharp = yes
	}

	# --- Grant the audience — receive Lord Amherst -----------------------------
	option = {
		name = "qing_embassy.2.receive"
		custom_tooltip = "qing_embassy.2.receive.tt"
		QING_embassy_britain_receive = yes
		# #65 kickoff: Amherst, received, speaks of Waterloo — the historically-anchored
		# opening of the Emperor Emeritus chain (Amherst really did call at St Helena on
		# his homeward voyage). The overture self-guards (early window, once).
		trigger_event = { id = qing_napoleon.5  days = { 20 60 } }
	}
}

# ======================================================================
# .10  THE EARLY-INDUSTRIALIZATION PROGRAM HUB (洋務先聲). Opened once the Macartney
# divergence is launched (qing_embassy.1 EMBRACE → QING_earlyindus_launch). The court-
# science bureau, the steam yard, and the machine-works are the remaining foundations
# the player drives; the fourth option consolidates the program once all four stand and
# the movement has momentum. Each option gates on what has not yet been done, so the hub
# reads as a live checklist. Re-openable (see the "later" option) so the player is never
# locked out mid-program. See se_QING_EARLYINDUS.txt.
# ======================================================================
qing_embassy.10 = {
	type = country_event
	title = "qing_embassy.10.t"
	desc = "qing_embassy.10.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_early_industrialization
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_embassy.10 (early-industrialization hub) [ROOT.GetTag]" }
	}

	# --- Found the Translation & Science Bureau (譯館) -------------------------
	option = {
		name = "qing_embassy.10.translation"
		custom_tooltip = "qing_embassy.10.translation.tt"
		trigger = { NOT = { has_country_modifier = qing_earlyindus_translation } }
		QING_earlyindus_found_translation = yes
		trigger_event = { id = qing_embassy.10  days = { 200 400 } }
	}

	# --- Found the Steam-Experiment Yard (機器船廠) ---------------------------
	option = {
		name = "qing_embassy.10.yard"
		custom_tooltip = "qing_embassy.10.yard.tt"
		trigger = { NOT = { has_country_modifier = qing_earlyindus_yard } }
		QING_earlyindus_found_yard = yes
		trigger_event = { id = qing_embassy.10  days = { 200 400 } }
	}

	# --- Found the Imperial Machine-Works (機器製造局) ------------------------
	option = {
		name = "qing_embassy.10.manufactory"
		custom_tooltip = "qing_embassy.10.manufactory.tt"
		trigger = { NOT = { has_country_modifier = qing_earlyindus_manufactory } }
		QING_earlyindus_found_manufactory = yes
		trigger_event = { id = qing_embassy.10  days = { 200 400 } }
	}

	# --- Consolidate the program — the court accepts the new order -------------
	#     Shown only once all four foundations stand and the shared movement counter has
	#     real momentum (>=40). Lifts the conservative-reaction drag for good.
	option = {
		name = "qing_embassy.10.consolidate"
		custom_tooltip = "qing_embassy.10.consolidate.tt"
		trigger = {
			NOT = { has_variable = qing_earlyindus_consolidated }
			has_country_modifier = qing_earlyindus_foundry
			has_country_modifier = qing_earlyindus_translation
			has_country_modifier = qing_earlyindus_yard
			has_country_modifier = qing_earlyindus_manufactory
			# [#304 review-fix] gate 40 -> 20, matching QING_earlyindus_consolidate — the arc's
			# own progress (halved by weak 1790s machinery) cannot reliably reach 40.
			var:qing_selfstr_progress >= 20
		}
		QING_earlyindus_consolidate = yes
	}

	# --- Attend to it later — keep the program open ----------------------------
	option = {
		name = "qing_embassy.10.later"
		custom_tooltip = "qing_embassy.10.later.tt"
		# [#304 review-fix] the hub must actually STAY open. The found-* options each
		# re-queue the hub, but "later" did nothing — so picking it (the natural choice when
		# not ready to fund the next works, or when consolidate is not yet reachable) closed
		# the hub forever and stranded the conservative-reaction malus. Re-queue it here too,
		# unless the program is already consolidated (then there is nothing left to open for).
		if = {
			limit = { NOT = { has_variable = qing_earlyindus_consolidated } }
			trigger_event = { id = qing_embassy.10  days = { 200 400 } }
		}
	}
}

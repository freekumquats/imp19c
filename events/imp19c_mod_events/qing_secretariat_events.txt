########################################################################
# qing_secretariat_events.txt — Central Secretariat (內閣) governance events (#273).
#
# The 內閣大學士 (Grand Secretary of the Central Secretariat) heads the chancery
# that DRAFTS the empire's edicts and rescripts and marshals its political
# influence. These beats fire when the secretariat runs smoothly (a torrent of
# well-drafted edicts) or grinds (a backlog of rescripts, a drafting scandal).
# The office generates political influence (its country modifier), so its events
# turn on the influence pool the appoint verbs spend.
#
# Fired quarterly from QING_frontier_flavour_roll (se_QING_DECLINE.txt), gated on
# the Grand Secretariat office being filled by a living holder in our service.
#
# CHI player-only. All logged sys = QING.
# Namespace = qing_secretariat
########################################################################

namespace = qing_secretariat

# ======================================================================
# qing_secretariat.1 — THE EDICT MILL (票擬): the chancery drafts a great mass of
# rescripts. A capable Grand Secretary (high charisma) turns the throne's will
# into crisp, enforceable orders (a burst of political influence + a touch of
# legitimacy); a weaker one merely clears the desk.
# ======================================================================
qing_secretariat.1 = {
	type = country_event
	title = "qing_secretariat.1.t"
	desc = "qing_secretariat.1.desc"
	picture = chinese_throne_room
	left_portrait = root.current_ruler
	right_portrait = scope:secretary
	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_office_grand_secretariat_holder
		var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_secretariat.1 (the edict mill)" }
		var:qing_office_grand_secretariat_holder = { save_scope_as = secretary }
	}

	# A rigorous chancery under an able secretary (charisma>=9) — a burst of influence
	option = {
		name = "qing_secretariat.1.a"
		custom_tooltip = "qing_secretariat.1.a.tt"
		trigger = { scope:secretary = { charisma >= 9 } }
		add_political_influence = 15
		add_legitimacy = 3
		scope:secretary = { QING_char_promote_standing = { pop = 3  prom = 3 } }
	}

	# A workmanlike clearing of the backlog under a lesser secretary — modest influence
	option = {
		name = "qing_secretariat.1.b"
		custom_tooltip = "qing_secretariat.1.b.tt"
		trigger = { scope:secretary = { charisma < 9 } }
		add_political_influence = 6
	}
}

# ======================================================================
# qing_secretariat.2 — THE RESCRIPT BACKLOG (積壓): the chancery has fallen behind
# and edicts pile up unanswered — the machinery of state stalls. Drive the
# secretaries hard to clear it (influence recovered, but the Grand Secretary is
# worn and resentful) or let it slide (the influence pool bleeds, unrest ticks up).
# ======================================================================
qing_secretariat.2 = {
	type = country_event
	title = "qing_secretariat.2.t"
	desc = "qing_secretariat.2.desc"
	picture = chinese_throne_room
	left_portrait = root.current_ruler
	right_portrait = scope:secretary
	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_office_grand_secretariat_holder
		var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
		political_influence < 40
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_secretariat.2 (the rescript backlog)" }
		var:qing_office_grand_secretariat_holder = { save_scope_as = secretary }
	}

	# Drive the chancery hard — clear the backlog; influence recovered, the secretary worn
	option = {
		name = "qing_secretariat.2.a"
		custom_tooltip = "qing_secretariat.2.a.tt"
		add_political_influence = 12
		scope:secretary = { add_loyalty = -6 }
	}

	# Let it slide — the pool bleeds and the localities feel the drift
	option = {
		name = "qing_secretariat.2.b"
		custom_tooltip = "qing_secretariat.2.b.tt"
		add_political_influence = -8
		add_stability = -1
	}
}

# ======================================================================
# qing_secretariat.3 — THE DRAFTING SCANDAL (擬旨舞弊): a rescript has been doctored
# in the drafting — a faction slipped its own language into an imperial edict. The
# throne can make a hard example of the chancery (root out the culprit: corruption
# eases, but the Grand Secretary is tainted by the scandal on his watch) or hush it
# up (the edict stands, but venality festers and the influence pool suffers).
# ======================================================================
qing_secretariat.3 = {
	type = country_event
	title = "qing_secretariat.3.t"
	desc = "qing_secretariat.3.desc"
	picture = senate_debate
	left_portrait = root.current_ruler
	right_portrait = scope:secretary
	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_office_grand_secretariat_holder
		var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
		OR = {
			var:qing_corruption_level >= 45
			var:qing_office_grand_secretariat_holder = { OR = { corruption >= 25  has_trait = corrupt } }
		}
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_secretariat.3 (the drafting scandal)" }
		var:qing_office_grand_secretariat_holder = { save_scope_as = secretary }
	}

	# Make an example — root out the doctored language; corruption eases, the secretary tainted
	option = {
		name = "qing_secretariat.3.a"
		custom_tooltip = "qing_secretariat.3.a.tt"
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -6 }
		add_stability = 1
		scope:secretary = { add_prominence = -5 }
	}

	# Hush it up — the edict stands; venality festers, influence bleeds
	option = {
		name = "qing_secretariat.3.b"
		custom_tooltip = "qing_secretariat.3.b.tt"
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 5 }
		add_political_influence = -6
	}
}

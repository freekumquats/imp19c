########################################################################
# qing_frontier_migration_events.txt — the DEMOGRAPHIC & FRONTIER-FLAVOUR wave:
# nomad in-migrations, the New World crop revolution, and two anachronistic
# opportunity chains (Hong Kong as a Chinese entrepot; the naming of Vietnam).
#
#   ROOT / this = the Qing overlord (CHI, player)
#
# All events are QING-ONLY + player-only (gated by the flavour roll, which already
# requires tag = CHI + is_ai = no). Every state-change is logged to sys = QING and,
# where it touches the health of the empire, nudges the decline/reform counters so
# these flavour beats feed the same simulation as the rest of the suite.
#
# Namespace = qing_migration
#   .10  Torghut return 土爾扈特東歸 (1771) — the great Kalmyk homecoming
#   .11  a lesser nomad band seeks Qing protection (Torghut-inspired, generic)
#   .20  New World crops arrive 番薯/玉米/落花生/番茄 — the double-edged boom
#   .21  the boom RECKONING — hidden router: sends the boom to blessing or crisis
#        per dynastic health (#79 traceability fix: was mislabelled a resolution)
#   .22  the population boom RESOLVES — BLESSING (strong dynasty, turbocharged growth)
#   .23  the population boom RESOLVES — CRISIS  (weak dynasty, Malthusian collapse)
#   .30  Hong Kong 香港 — develop the fishing island into a Chinese entrepot
#   .31  Hong Kong flourishes (follow-up if the player invested)
#   .40  The naming of Vietnam 越南 (1804, Jiaqing) — bestow the name on Gia Long
#   .41  Vietnamese self-strengthening spillover (follow-up influence beat)
#
# Sources: Ho Ping-ti, Studies on the Population of China 1368-1953 (Harvard,
#   1959) — the sweet-potato/maize demographic thesis; Perdue, China Marches West
#   (2005) — the Torghut return & Qing Inner Asia; Peter Purdue & Millward,
#   Beyond the Pass; Carroll, A Concise History of Hong Kong (2007); Woodside,
#   Vietnam and the Chinese Model (1988); 大南實錄 / 清實錄 on the 1804 investiture.
########################################################################

########################################################################
# .10  THE TORGHUT RETURN  (土爾扈特東歸, 1771)
# History: the Torghut Mongols (a Kalmyk/Oirat people) had migrated to the lower
# Volga in the 1630s. In the winter of 1770-71, oppressed by Russian levies and
# Orthodox pressure, Ubashi Khan (渥巴錫) led ~170,000 of them on an epic, ruinous
# trek back east; perhaps a third survived to reach Qing Xinjiang, where the
# Qianlong Emperor settled them and commemorated the return on a stele at Chengde
# (土爾扈特全部歸順記). Here it can fire anachronistically whenever the Qing looks
# like a worthy refuge.
########################################################################
qing_migration.10 = {
	type = country_event
	title = "qing_migration.10.t"
	desc = "qing_migration.10.desc"
	picture = looting

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.10 (Torghut return) [ROOT.GetTag]" }
	}

	# --- Settle them on the frontier as banner pastoralists ----------------
	option = {
		name = "qing_migration.10.a"
		custom_tooltip = "qing_migration.10.a.tt"
		# a great host of hardy horsemen: manpower, light cavalry, and new subjects
		add_manpower = 15
		add_country_modifier = { name = qing_migr_torghut_return  duration = -1 }
		# settle them: MOVE real Oirat-Kalmyk-Mongol tribesmen out of the Russian
		# steppe onto our frontier (create_state_pop fallback if the source is dry),
		# then make that settling ground a durable draw for further migration.
		QING_MIGR_nomad_ingather = { count = 2 }
		current_ruler = { add_prestige = 25 }
		# a magnanimous ingathering steadies the multiethnic realm
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = -8 }
	}

	# --- Welcome them but strain the treasury to feed them -----------------
	option = {
		name = "qing_migration.10.b"
		custom_tooltip = "qing_migration.10.b.tt"
		trigger = { treasury >= 40 }
		add_treasury = -40
		add_manpower = 10
		add_country_modifier = { name = qing_migr_torghut_return  duration = -1 }
		# a smaller settled host, likewise drawn from the real steppe source.
		QING_MIGR_nomad_ingather = { count = 1 }
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = -10 }
		add_stability = 1
	}

	# --- Turn them away — the frontier cannot bear the burden --------------
	option = {
		name = "qing_migration.10.c"
		custom_tooltip = "qing_migration.10.c.tt"
		current_ruler = { add_prestige = -10 }
		# a refuge refused: word spreads that the Qing is no protector of peoples
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = 6 }
	}
}

########################################################################
# .20  THE NEW WORLD CROPS ARRIVE  (番薯 / 玉米 / 落花生 / 番茄)
# History (Ho Ping-ti): sweet potato (番薯/甘薯), maize (玉米), peanut (落花生)
# and other American cultigens, arriving via the Columbian Exchange and the
# Manila galleon in the 16th-17th c., let Chinese farmers colonise dry uplands and
# sandy soils that rice and wheat could not use. The result was one of history's
# great population booms — China roughly tripled, ~150m (1700) to ~430m (1850).
# But that same pressure crowded marginal highlands, stripped forests, and bred
# the landless unrest that fed the White Lotus and, later, the Taiping. This
# event PLANTS the boom; .21 / .22 RESOLVE it as blessing or curse per the health
# of the dynasty. Can fire anachronistically — earlier OR later than history.
########################################################################
qing_migration.20 = {
	type = country_event
	title = "qing_migration.20.t"
	desc = "qing_migration.20.desc"
	picture = farming

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		NOT = { has_variable = qing_newworld_crops }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.20 (New World crops) [ROOT.GetTag]" }
		set_variable = { name = qing_newworld_crops  value = 1 }
	}

	# --- Promote the new crops across the empire ---------------------------
	option = {
		name = "qing_migration.20.a"
		custom_tooltip = "qing_migration.20.a.tt"
		# the boom begins: an empire-wide growth modifier, and a resolution roll
		# some years hence that reads the state of the dynasty.
		add_country_modifier = { name = qing_migr_crop_boom  duration = -1 }
		# turn some marginal land into new farm pops now
		if = {
			limit = { any_owned_province = { count >= 1 } }
			ordered_owned_province = {
				order_by = total_population
				max = 2
				create_state_pop = freemen
			}
		}
		# the swelling population starts to seek land: mark the crowded core so
		# its surplus out-migrates to the frontier via the migration pulse
		# (se_QING_COLON.txt) — the demographic engine of the frontier settlement.
		QING_COLON_heartland_push = { count = 3 }
		# #78 (2026-07-05): the boom also RAISES the carrying ceiling on every crop
		# province (structural capacity, not just growth) so the extra people have
		# somewhere to live — the piece the crop coupling was missing.
		QING_COLON_apply_nwcrop_capacity = yes
		# the reckoning comes later (anachronistic: 5-12 years out)
		trigger_event = { id = qing_migration.21  days = { 1825 4380 } }
	}

	# --- Let them spread on their own, unmanaged ---------------------------
	option = {
		name = "qing_migration.20.b"
		custom_tooltip = "qing_migration.20.b.tt"
		add_country_modifier = { name = qing_migr_crop_boom  duration = -1 }
		# unmanaged growth tilts the reckoning toward crisis (see .21 gate)
		set_variable = { name = qing_crop_boom_unmanaged  value = 1 }
		# unmanaged, the surplus still spills toward the frontier
		QING_COLON_heartland_push = { count = 3 }
		# #78 (2026-07-05): capacity ceiling rises with the crops even when unmanaged
		QING_COLON_apply_nwcrop_capacity = yes
		trigger_event = { id = qing_migration.21  days = { 1825 4380 } }
	}
}

########################################################################
# .21  THE RECKONING OF THE BOOM  (router — routes to blessing or crisis)
# Reads the health of the dynasty at the moment the swollen population comes due.
# A STRONG, well-run Qing turns the numbers into wealth and armies (.22 path a);
# a WEAK, corrupt, sect-riven one reaps famine and rebellion (.22 path b). This
# is the double-edged sword the design calls for.
########################################################################
qing_migration.21 = {
	type = country_event
	hidden = yes

	is_triggered_only = yes

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.21 (boom reckoning router) [ROOT.GetTag]" }
		if = {
			# STRONG dynasty: stable, not too corrupt, low sect pressure, and the
			# boom was actively managed. Turbocharged growth.
			limit = {
				stability >= 40
				var:qing_corruption_level < 50
				var:qing_sect_pressure < 50
				NOT = { has_variable = qing_crop_boom_unmanaged }
			}
			trigger_event = { id = qing_migration.22  days = { 3 12 } }
		}
		else = {
			# WEAK dynasty (or unmanaged boom): Malthusian crisis.
			trigger_event = { id = qing_migration.23  days = { 3 12 } }
		}
	}
}

########################################################################
# .22  THE BOOM AS BLESSING  (strong dynasty — turbocharged growth)
########################################################################
qing_migration.22 = {
	type = country_event
	title = "qing_migration.22.t"
	desc = "qing_migration.22.desc"
	picture = farming

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.22 (boom BLESSING) [ROOT.GetTag]" }
		# swap the neutral boom modifier for the golden one
		remove_country_modifier = qing_migr_crop_boom
		add_country_modifier = { name = qing_migr_crop_boom_golden  duration = -1 }
	}

	# --- A teeming, prosperous empire --------------------------------------
	option = {
		name = "qing_migration.22.a"
		custom_tooltip = "qing_migration.22.a.tt"
		add_manpower = 20
		current_ruler = { add_prestige = 20 }
		add_stability = 2
		# prosperity eases the pressures that would otherwise mount
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -6 }
		# a prosperous core holds its people: ease the out-migration push back
		# to the organic baseline (se_QING_COLON.txt).
		QING_COLON_clear_heartland_push = yes
		# seed a few more thriving communities
		if = {
			limit = { any_owned_province = { count >= 1 } }
			ordered_owned_province = {
				order_by = total_population
				max = 3
				create_state_pop = freemen
			}
		}
	}
}

########################################################################
# .23  THE BOOM AS CURSE  (weak dynasty — Malthusian crisis)
# Overcrowded highlands, deforestation, landless migrants, and the millenarian
# sects (White Lotus 白蓮教, later the Taiping) that recruit among them.
########################################################################
qing_migration.23 = {
	type = country_event
	title = "qing_migration.23.t"
	desc = "qing_migration.23.desc"
	picture = revolt

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.23 (boom CRISIS) [ROOT.GetTag]" }
		remove_country_modifier = qing_migr_crop_boom
		add_country_modifier = { name = qing_migr_overpopulation  duration = -1 }
	}

	# --- Relief works and resettlement (costly, mitigates) -----------------
	option = {
		name = "qing_migration.23.a"
		custom_tooltip = "qing_migration.23.a.tt"
		trigger = { treasury >= 80 }
		add_treasury = -80
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -8 }
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = -4 }
		# relief works and resettlement relieve the crowding: clear the push.
		QING_COLON_clear_heartland_push = yes
		add_stability = 1
	}

	# --- Let the land absorb them as it may (the pressures mount) ----------
	option = {
		name = "qing_migration.23.b"
		custom_tooltip = "qing_migration.23.b.tt"
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = 10 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 5 }
		# unrelieved, the crowding drives an even harder exodus to the frontier.
		QING_COLON_heartland_push = { count = 4 }
		add_stability = -2
	}
}

########################################################################
# .30  HONG KONG — A CHINESE ENTREPOT  (香港, anachronistic)
# History: Hong Kong island (香港, 香港仔 "Fragrant Harbour") was a cluster of
# fishing and incense-trading villages ceded to Britain in 1842, which built it
# into one of the world's great ports. The alternate-history lever: a
# self-strengthening Qing develops the deep-water harbour ITSELF — a Chinese-run
# treaty-port and free-trade entrepot — rather than losing it. Gated on owning
# Xianggang (province 8245).
########################################################################
qing_migration.30 = {
	type = country_event
	title = "qing_migration.30.t"
	desc = "qing_migration.30.desc"
	picture = trade_port

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		owns = 8245
		NOT = { has_variable = qing_hongkong_developed }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.30 (Hong Kong) [ROOT.GetTag]" }
	}

	# --- Build the harbour into a modern entrepot --------------------------
	option = {
		name = "qing_migration.30.a"
		custom_tooltip = "qing_migration.30.a.tt"
		trigger = { treasury >= 100 }
		add_treasury = -100
		set_variable = { name = qing_hongkong_developed  value = 1 }
		p:8245 = {
			set_city_status = city
			add_building_level = port_building
			add_building_level = URB_commerce_district
			create_state_pop = citizen
			create_state_pop = freemen
			add_province_modifier = { name = qing_migr_hongkong_boom  duration = -1 }
		}
		add_country_modifier = { name = qing_migr_hongkong_trade  duration = -1 }
		# a booming free port some years on
		trigger_event = { id = qing_migration.31  days = { 1095 2555 } }
	}

	# --- A modest customs station will do ----------------------------------
	option = {
		name = "qing_migration.30.b"
		custom_tooltip = "qing_migration.30.b.tt"
		p:8245 = {
			add_building_level = port_building
		}
		add_treasury = 20
	}

	# --- Leave the fishing villages be -------------------------------------
	option = {
		name = "qing_migration.30.c"
		add_stability = 1
	}
}

########################################################################
# .40  THE NAMING OF VIETNAM  (越南, 1804 — anachronistic)
# History: after Nguyen Anh unified the country and founded the Nguyen dynasty as
# Emperor Gia Long (嘉隆帝), he sought investiture from the Qing. He proposed the
# name "南越 Nam Việt"; the Jiaqing Emperor (嘉慶帝), wary of the old Nam Viet's
# claim to Guangdong/Guangxi, inverted it to "越南 Việt Nam" and invested Gia Long
# in 1804. Here the Qing player performs that investiture — and may follow it with
# influence and reform spillover. Gated on VIE existing.
########################################################################
qing_migration.40 = {
	type = country_event
	title = "qing_migration.40.t"
	desc = "qing_migration.40.desc"
	picture = chinese_throne_room

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		exists = c:VIE
		NOT = { has_variable = qing_named_vietnam }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_migration.40 (naming Vietnam) [ROOT.GetTag]" }
		set_variable = { name = qing_named_vietnam  value = 1 }
	}

	# --- Bestow the name 越南 and invest Gia Long --------------------------
	option = {
		name = "qing_migration.40.a"
		custom_tooltip = "qing_migration.40.a.tt"
		current_ruler = { add_prestige = 20 }
		# the tributary bond affirmed: prestige of the Son of Heaven as arbiter
		add_country_modifier = { name = qing_migr_tributary_suzerain  duration = -1 }
		if = {
			limit = { exists = c:VIE }
			c:VIE = { add_opinion = { modifier = qing_investiture_opinion  target = ROOT } }
		}
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = -3 }
		# open the influence follow-up
		trigger_event = { id = qing_migration.41  days = { 730 1825 } }
	}

	# --- Insist on the grander "南越" they asked for (provocative) ---------
	option = {
		name = "qing_migration.40.b"
		custom_tooltip = "qing_migration.40.b.tt"
		current_ruler = { add_prestige = 10 }
		add_tyranny = 2
		# an overbearing suzerain: the tributary bristles
		if = {
			limit = { exists = c:VIE }
			c:VIE = { add_opinion = { modifier = qing_overbearing_opinion  target = ROOT } }
		}
	}
}

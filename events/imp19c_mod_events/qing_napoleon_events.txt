########################################################################
# qing_napoleon_events.txt — THE EMPEROR EMERITUS (太上皇): NAPOLEON AT THE QING
# COURT (task #65). The counterfactual reform chain — after Waterloo the exiled
# Napoleon is received in Beijing, given the revived 太上皇 (Retired/Grand Emperor)
# office Qianlong once held, and from that seat drives a slate of radical
# Enlightenment reforms while steering Qing foreign policy toward his own revenge
# on Britain and Russia. Machinery: se_QING_NAPOLEON.txt (all QING_napoleon_*
# verbs) + qing_mechanics_modifiers.txt (the reform/office/backlash modifiers) +
# 00_napoleon.txt (the Grande Armée military tradition, gated on the flag the
# arrival sets). Design + scope: memory imp19c-napoleon-in-china.
#
# CHI-only, player-only. All logged sys = QING. KICKED OFF by RECEIVING the Amherst
# embassy (qing_embassy.2, 1816 — the sole British embassy that occurs in play, and
# the historically-correct caller at St Helena): the envoy lets fall the news of
# Waterloo, the court expresses interest, the envoy carries the invitation to St
# Helena, Britain humours the Qing, and Napoleon comes east for a "brief" visit that
# becomes permanent. The overture (qing_napoleon.5) is gated to the early window
# (<1821.5.5); once committed, the downstream beats (.6, .1) run to completion
# regardless of date (guarded once-only by qing_napoleon_arrived). Rebuffing or
# ignoring the Amherst embassy forecloses the chain.
#
#   ROOT / this = the Qing overlord (CHI, player); root.current_ruler = the emperor
#   (the Jiaqing Emperor at the 1815 start); scope:napoleon = the conjured Napoleon;
#   char:227 = the heir Mianning (the future Daoguang Emperor).
#
# Namespace = qing_napoleon
#   .5  Un mot de Waterloo — the OVERTURE at the British reception (the kickoff)
#   .6  Le rocher        — St Helena; the envoy relays the invitation, Britain humours
#   .1  L'arrivée        — Napoleon comes to stay; offered the throne's right hand (from .6)
#   .2  Le grand œuvre   — the reform hub: enact the Napoleonic reform slate + tilt
#   .3  La réaction      — the conservative Manchu backlash against the foreigner
#   .4  道光主義          — the Daoguang Doctrine capstone, proclaimed by the heir
########################################################################

namespace = qing_napoleon

########################################################################
# .1  L'ARRIVÉE — HE COMES TO STAY, AND IS OFFERED THE THRONE'S RIGHT HAND
# Fed by .6: Napoleon arrives in China for what was meant to be a brief sojourn,
# openly and with Britain's hesitant leave. Taken with the court — and it with him —
# he resolves to stay at the Qing's invitation, and London grudgingly consents. Now
# the court must decide what he shall BE: revive the 太上皇 office for him, or thank
# him and let him sail back to his rock. Carries the sleeping-giant apocryphon.
########################################################################
qing_napoleon.1 = {
	type = country_event
	title = "qing_napoleon.1.t"
	desc = "qing_napoleon.1.desc"
	picture = chinese_throne_room

	left_portrait = root.current_ruler

	is_triggered_only = yes

	# review-fix (#80 MEDIUM): reached only from .6 (never a roll). NO date guard — the
	# early-window gate lives on the overture (.5); once committed the chain must land.
	# Once-only held by NOT qing_napoleon_arrived.
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_napoleon_arrived }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_napoleon.1 (Napoleon offered the Qing court) [ROOT.GetTag]" }
	}

	# --- Receive him, and revive the 太上皇 office for him ------------------
	option = {
		name = "qing_napoleon.1.a"
		custom_tooltip = "qing_napoleon.1.a.tt"
		set_variable = qing_napoleon_arrived
		QING_napoleon_spawn = yes
		scope:napoleon = { QING_napoleon_take_office = yes }
		# he begins to work at once — the reform council convenes within the year.
		trigger_event = { id = qing_napoleon.2  days = { 120 300 } }
		# and the conservative reaction gathers.
		trigger_event = { id = qing_napoleon.3  days = { 200 500 } }
	}

	# --- Thank him, and let him sail back to St Helena when the visit ends --
	option = {
		name = "qing_napoleon.1.b"
		custom_tooltip = "qing_napoleon.1.b.tt"
		# the visit stays a visit; the chance passes for good.
		set_variable = qing_napoleon_arrived
		add_stability = 1
		LOG_line = { sys = QING  msg = "napoleon: the court let Napoleon return to St Helena for [ROOT.GetTag]" }
	}
}

########################################################################
# .2  LE GRAND ŒUVRE — THE REFORM COUNCIL
# The recurring hub. From the Emperor Emeritus's seat the Napoleonic programme is
# laid before the throne: the Code, central banking, mass schooling, the levée —
# and the foreign-policy tilt that keeps the old soldier engaged. Each reform verb
# self-guards once-only; the event re-offers itself until the slate is complete,
# then hands off to the capstone. Requires Napoleon alive and in office.
########################################################################
qing_napoleon.2 = {
	type = country_event
	title = "qing_napoleon.2.t"
	desc = "qing_napoleon.2.desc"
	picture = council_country

	left_portrait = root.current_ruler
	right_portrait = scope:napoleon

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_napoleon_char
		var:qing_napoleon_char = { is_alive = yes }
	}

	immediate = {
		# re-establish the right-portrait scope for a re-fired instance.
		var:qing_napoleon_char = { save_scope_as = napoleon }
		LOG_line = { sys = QING  msg = "EVENT qing_napoleon.2 (reform council; reforms=[ROOT.MakeScope.GetVariable('qing_napoleon_reforms_done').GetValue|0]) [ROOT.GetTag]" }
	}

	# --- 拿破崙法典 The Napoleonic Code ------------------------------------
	option = {
		name = "qing_napoleon.2.a"
		custom_tooltip = "qing_napoleon.2.a.tt"
		trigger = { NOT = { has_country_modifier = qing_reform_napoleonic_code } }
		QING_napoleon_reform_code = yes
		trigger_event = { id = qing_napoleon.2  days = { 180 480 } }
	}

	# --- 中央銀行 Central Banking ------------------------------------------
	option = {
		name = "qing_napoleon.2.b"
		custom_tooltip = "qing_napoleon.2.b.tt"
		trigger = { NOT = { has_country_modifier = qing_reform_central_bank } }
		QING_napoleon_reform_bank = yes
		trigger_event = { id = qing_napoleon.2  days = { 180 480 } }
	}

	# --- 國民教育 Primary Education ----------------------------------------
	option = {
		name = "qing_napoleon.2.c"
		custom_tooltip = "qing_napoleon.2.c.tt"
		trigger = { NOT = { has_country_modifier = qing_reform_primary_education } }
		QING_napoleon_reform_education = yes
		trigger_event = { id = qing_napoleon.2  days = { 180 480 } }
	}

	# --- 全民皆兵 Levée en masse -------------------------------------------
	option = {
		name = "qing_napoleon.2.d"
		custom_tooltip = "qing_napoleon.2.d.tt"
		trigger = { NOT = { has_country_modifier = qing_reform_levee_en_masse } }
		QING_napoleon_reform_levee = yes
		trigger_event = { id = qing_napoleon.2  days = { 180 480 } }
	}

	# --- 以夷制夷 Follow his line against Britain and Russia ----------------
	# The price of his engagement: the anti-GBR/RUS pro-FRA tilt. Keeps his loyalty
	# high (scored by QING_napoleon_loyalty_pulse). Always available while he lives.
	option = {
		name = "qing_napoleon.2.e"
		custom_tooltip = "qing_napoleon.2.e.tt"
		QING_napoleon_revenge_tilt = { severity = 8 }
		trigger_event = { id = qing_napoleon.2  days = { 180 480 } }
	}

	# --- 道光主義 Consolidate the reforms into a lasting doctrine -----------
	# The capstone hand-off, unlocked once the slate is substantially complete.
	option = {
		name = "qing_napoleon.2.f"
		custom_tooltip = "qing_napoleon.2.f.tt"
		trigger = {
			var:qing_napoleon_reforms_done >= 3
			NOT = { has_country_modifier = qing_napoleon_daoguang_doctrine }
		}
		trigger_event = { id = qing_napoleon.4  days = { 30 90 } }
	}

	# --- Let the council rest for now --------------------------------------
	option = {
		name = "qing_napoleon.2.g"
		# no re-trigger — the player reconvenes it from the next council prompt.
	}
}

########################################################################
# .3  LA RÉACTION — THE CONSERVATIVE MANCHU BACKLASH
# A foreign usurper-emperor forcing Enlightenment reform enrages the Manchu old
# guard (the qing_manchu_staunch band the backlash already bled at take-office).
# The throne must choose how to face them. Fires once, a while after the arrival.
########################################################################
qing_napoleon.3 = {
	type = country_event
	title = "qing_napoleon.3.t"
	desc = "qing_napoleon.3.desc"
	picture = dagger_behind_back

	left_portrait = root.current_ruler
	right_portrait = scope:napoleon

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_napoleon_char
		var:qing_napoleon_char = { is_alive = yes }
		NOT = { has_variable = qing_napoleon_reaction_faced }
	}

	immediate = {
		set_variable = qing_napoleon_reaction_faced
		var:qing_napoleon_char = { save_scope_as = napoleon }
		LOG_line = { sys = QING  msg = "EVENT qing_napoleon.3 (conservative backlash) [ROOT.GetTag]" }
	}

	# --- Stand with the Emperor Emeritus; face down the reaction -----------
	# Back reform: relieve reform pressure and steady him, at a stability cost.
	option = {
		name = "qing_napoleon.3.a"
		custom_tooltip = "qing_napoleon.3.a.tt"
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = -10 }
		var:qing_napoleon_char = { add_loyalty = loyalty_qing_congenial }
		add_stability = -1
	}

	# --- Rein him in to placate the old guard ------------------------------
	# Soothe the conservatives (ease ethnic tension, gain stability) — but he sours
	# at being checked (loyalty falls; the loyalty pulse will compound it).
	option = {
		name = "qing_napoleon.3.b"
		custom_tooltip = "qing_napoleon.3.b.tt"
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = -6 }
		var:qing_napoleon_char = { add_loyalty = loyalty_qing_estranged }
		add_stability = 1
	}
}

########################################################################
# .4  道光主義 — THE DAOGUANG DOCTRINE (THE CAPSTONE)
# The heir Napoleon tutored — Mianning, the future Daoguang Emperor (char:227) —
# proclaims the reforms and the anti-GBR/RUS pro-FRA line a lasting doctrine of
# state. Grants the permanent apex modifier and seals the chain.
########################################################################
qing_napoleon.4 = {
	type = country_event
	title = "qing_napoleon.4.t"
	desc = "qing_napoleon.4.desc"
	picture = chinese_throne_room

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		var:qing_napoleon_reforms_done >= 3
		NOT = { has_country_modifier = qing_napoleon_daoguang_doctrine }
	}

	immediate = {
		# put the heir on the stage if he is still alive to be shown.
		if = {
			limit = { exists = char:227  char:227 = { is_alive = yes } }
			char:227 = { save_scope_as = daoguang_heir }
		}
		LOG_line = { sys = QING  msg = "EVENT qing_napoleon.4 (Daoguang Doctrine) [ROOT.GetTag]" }
	}

	# --- Proclaim the Daoguang Doctrine ------------------------------------
	option = {
		name = "qing_napoleon.4.a"
		custom_tooltip = "qing_napoleon.4.a.tt"
		QING_napoleon_proclaim_doctrine = yes
	}
}

########################################################################
# .5  UN MOT DE WATERLOO — THE OVERTURE (the kickoff, from the British reception)
# When the court RECEIVES the Amherst embassy (qing_embassy.2, 1816 — the departing
# envoy, making conversation of the wider world), he lets fall the news of Waterloo
# and the Corsican's exile to a rock in the southern ocean. The Qing, intrigued by
# so famous a fallen conqueror, may express a wish to host him. Historically Lord
# Amherst DID call at St Helena on his homeward voyage (1817) and met Napoleon; here
# that call carries an invitation. CHI-only; early window only; once.
########################################################################
qing_napoleon.5 = {
	type = country_event
	title = "qing_napoleon.5.t"
	desc = "qing_napoleon.5.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		exists = c:GBR
		NOT = { has_variable = qing_napoleon_arrived }
		NOT = { has_variable = qing_napoleon_overture_opened }
		current_date < 1821.5.5
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_napoleon.5 (Waterloo overture at the British reception) [ROOT.GetTag]" }
	}

	# --- Express interest in hosting the famous exile ----------------------
	option = {
		name = "qing_napoleon.5.a"
		custom_tooltip = "qing_napoleon.5.a.tt"
		set_variable = qing_napoleon_overture_opened
		# the envoy agrees to carry the court's word when he calls at St Helena.
		trigger_event = { id = qing_napoleon.6  days = { 60 150 } }
		LOG_line = { sys = QING  msg = "napoleon: the court asks the British envoy to carry an invitation to St Helena for [ROOT.GetTag]" }
	}

	# --- A curiosity, nothing more -----------------------------------------
	option = {
		name = "qing_napoleon.5.b"
		custom_tooltip = "qing_napoleon.5.b.tt"
		# the overture is not made; the singular chance passes.
		set_variable = qing_napoleon_overture_opened
		LOG_line = { sys = QING  msg = "napoleon: the court let the overture pass for [ROOT.GetTag]" }
	}
}

########################################################################
# .6  LE ROCHER — ST HELENA, AND BRITAIN HUMOURED
# The British envoy, homeward bound, calls at St Helena and lays the Qing invitation
# before Napoleon — and word before his own government. London, weighing it, decides
# to HUMOUR the Qing: to permit the exile a visit east, in the hope of winning at
# Peking the influence the embassy itself had failed to gain. Napoleon, for his part,
# consents to come — for a brief sojourn only. Hands off to .1 (the arrival proper).
# CHI-only; early window only; once.
########################################################################
qing_napoleon.6 = {
	type = country_event
	title = "qing_napoleon.6.t"
	desc = "qing_napoleon.6.desc"
	picture = chinese_throne_room

	left_portrait = root.current_ruler

	is_triggered_only = yes

	# review-fix (#80 MEDIUM): NO date guard here. The overture (.5) is the sole
	# early-window gate; once the player has committed to it, the delayed hand-offs
	# (.5->.6 60-150d, .6->.1 90-200d) must run to completion or a promised arrival
	# silently dead-ends when the ~1yr of delays crosses 1821.5.5. Once-only is held
	# by NOT qing_napoleon_arrived.
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_napoleon_arrived }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_napoleon.6 (St Helena; Britain humours the Qing) [ROOT.GetTag]" }
	}

	# --- Prepare a reception befitting an emperor --------------------------
	option = {
		name = "qing_napoleon.6.a"
		custom_tooltip = "qing_napoleon.6.a.tt"
		trigger_event = { id = qing_napoleon.1  days = { 90 200 } }
		LOG_line = { sys = QING  msg = "napoleon: Britain consents to the visit; the court prepares to receive Napoleon for [ROOT.GetTag]" }
	}
}

########################################################################
# qing_keju_events.txt — THE IMPERIAL EXAMINATION (科舉), task #9.
#
# The keju is the beating heart of the Qing civil order: the ladder by which
# the empire recruited its officials, conferred its degrees, and bound the
# gentry to the throne. This roster makes the examination a living cycle the
# player HOSTS — the triennial provincial (鄉試) and metropolitan (會試)
# examinations culminating in the palace examination (殿試) under the emperor's
# own eye — and ties its outcomes to the men who sit on the Grand Council and
# hold the great offices (se_QING_COUNCIL.txt).
#
# WHAT THE PLAYER FEELS:
#   • HOSTING: a triennial exam prompt. Fund it well and it renews the
#     bureaucracy with merit (qing_keju_fresh_intake) and confers real degrees
#     — the shengyuan / juren / gongshi / jinshi status traits (00_imp19c.txt) —
#     on his courtiers, raising their office eligibility (office_eligibility.*).
#   • RANK: the top graduate is the 狀元 (zhuangyuan, optimus); the event names
#     a laureate, awards the jinshi degree, and binds him to the throne
#     (loyalty_qing_exam_laureate — the 天子門生, "the Son of Heaven's disciple").
#   • CORRUPTION AFFECTING SCORES: a high qing_corruption_level makes bought
#     degrees (捐納) and rigged rankings likely — the examiner-bribery scandals
#     that periodically rocked the Qing. The player chooses whether to prosecute.
#   • OLD RIVALRIES: examination cohorts (同年) and the examiner–examinee bond
#     (座師/門生) were the sinews of Qing factionalism; a passed-over scholar can
#     become a rival, or worse — the archetype being Hong Xiuquan (洪秀全), who
#     failed the Canton exams four times and raised the Taiping heaven.
#
# CHI-only, player-only. Fired from QING_frontier_flavour_roll. All logged
# sys = QING. Effects verified: add_trait (status degrees), add_loyalty (named
# objects in 00_imp19c_loyalty.txt), add_rival, add_country_modifier, the
# QING_DECLINE_nudge counters, and stock country/character effects.
#
#   ROOT / this = the Qing overlord (CHI, player)
#
# Namespace = qing_keju
#   .1  the triennial examination convenes (the host prompt)
#   .2  the palace examination & the naming of the 狀元 (award the laureate)
#   .3  an examination scandal — corruption has reached the graders (bought ranks)
#   .4  a brilliant scholar passed over — old rivalry, or the road to rebellion
#   .5  a grace examination (恩科) proclaimed at a jubilee (optional largesse)
#   .6  a scholar of the pool raised to office (掄才 — drawing on the talent)  [#321]
#   .7  the abolition of 捐納 pressed — purge purchased office, or keep the silver  [#321 deferred]
#
# Sources: Benjamin Elman, A Cultural History of Civil Examinations in Late
#   Imperial China (2000) & Civil Examinations and Meritocracy (2013); Ichisada
#   Miyazaki, China's Examination Hell (tr. 1976); Chung-li Chang, The Chinese
#   Gentry (1955); Jonathan Spence, God's Chinese Son (1996, on Hong Xiuquan);
#   Elisabeth Kaske, "The Price of an Office" (2011, on 捐納).
########################################################################

namespace = qing_keju

########################################################################
# .1  THE TRIENNIAL EXAMINATION CONVENES  (鄉試 / 會試)
# The examination cycle comes round. The throne decides how it will be run:
# funded richly and by the book, run on a shoestring, or — the fiscal
# temptation — thrown open to purchased degrees for ready silver.
########################################################################
qing_keju.1 = {
	type = country_event
	title = "qing_keju.1.t"
	desc = "qing_keju.1.desc"
	picture = great_library

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_keju.1 (triennial exam convenes, corruption=[ROOT.MakeScope.GetVariable('qing_corruption_level').GetValue|0]) [ROOT.GetTag]" }
	}

	# --- Host it in full state, by the book (the meritocratic ideal) -----
	option = {
		name = "qing_keju.1.a"
		custom_tooltip = "qing_keju.1.a.tt"
		trigger = { treasury >= 90 }
		add_treasury = -90
		add_country_modifier = { name = qing_keju_fresh_intake  duration = 3650 }
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -4 }
		# funding the prefectural & provincial halls (院試/鄉試) properly clears
		# the lower ladder so able men rise — the abstracted 生員/舉人 tiers.
		QING_DECLINE_nudge = { var = qing_exam_ladder  amount = 8 }
		add_stability = 1
		# the cycle culminates in the palace examination — name the laureate.
		trigger_event = { id = qing_keju.2  days = { 60 150 } }
	}

	# --- Hold a modest examination (adequate, unglamorous) ---------------
	option = {
		name = "qing_keju.1.b"
		custom_tooltip = "qing_keju.1.b.tt"
		trigger = { treasury >= 30 }
		add_treasury = -30
		add_political_influence = 10
		trigger_event = { id = qing_keju.2  days = { 60 150 } }
	}

	# --- Sell degrees to fill the war chest (捐納) — the rot sets in ------
	option = {
		name = "qing_keju.1.c"
		custom_tooltip = "qing_keju.1.c.tt"
		QING_sell_offices = yes
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 6 }
		# bought shengyuan/juren places clog the lower ladder from the base up
		QING_DECLINE_nudge = { var = qing_exam_ladder  amount = -10 }
		# a purchased cohort still yields a nominal laureate cycle
		trigger_event = { id = qing_keju.2  days = { 60 150 } }
	}
}

########################################################################
# .2  THE PALACE EXAMINATION & THE 狀元  (殿試)
# The metropolitan graduates are ranked before the throne. The first-place
# graduate is the zhuangyuan (狀元). The event picks the ablest eligible
# courtier, confers the jinshi degree, and binds him to the emperor as his
# examiner-sovereign (天子門生). If corruption is high, the ranking may be
# suspect — a hook into .3.
########################################################################
qing_keju.2 = {
	type = country_event
	title = "qing_keju.2.t"
	desc = "qing_keju.2.desc"
	picture = chinese_throne_room

	left_portrait = root.current_ruler
	right_portrait = scope:laureate

	is_triggered_only = yes

	immediate = {
		# the laureate: the ablest courtier not already a jinshi (the new degree
		# is his making). Fall back to any able adult courtier.
		if = {
			limit = {
				any_character = {
					employer = ROOT  is_ruler = no  is_adult = yes  is_alive = yes
					NOT = { has_trait = jinshi }
				}
			}
			# [#98-fix task #98] order_by is IGNORED inside random_character, so the
			# laureate was picked uniformly at random, not by ability. ordered_character
			# + max = 1 honours order_by and selects the ablest courtier (the proven
			# ablest-picker idiom, se_QING_COUNCIL.txt:128).
			ordered_character = {
				limit = {
					employer = ROOT  is_ruler = no  is_adult = yes  is_alive = yes
					NOT = { has_trait = jinshi }
				}
				order_by = combined_stats_council_svalue
				check_range_bounds = no
				max = 1
				save_scope_as = laureate
			}
		}
		else = {
			# [#98-fix task #98] same order_by-in-random_character bug on the fallback path.
			ordered_character = {
				limit = { employer = ROOT  is_ruler = no  is_adult = yes  is_alive = yes }
				order_by = combined_stats_council_svalue
				check_range_bounds = no
				max = 1
				save_scope_as = laureate
			}
		}
		LOG_line = { sys = QING  msg = "EVENT qing_keju.2 (palace exam, laureate=[ROOT.MakeScope.GetVariable('qing_corruption_level').GetValue|0] corruption) [ROOT.GetTag]" }
	}

	# --- Confer the jinshi degree on the 狀元 (bind him to the throne) ----
	option = {
		name = "qing_keju.2.a"
		custom_tooltip = "qing_keju.2.a.tt"
		scope:laureate = {
			add_trait = jinshi
			add_loyalty = loyalty_qing_exam_laureate
			add_popularity = 15
			add_character_experience = 10
			# passing under the emperor's own eye seasons a man's charm and learning
			# (the 天子門生 bond) — cultivate the factors, then bind him to the throne.
			QING_char_cultivate = { stat = charisma  amount = 1 }
			QING_char_cultivate = { stat = finesse  amount = 1 }
			# the degree confers a clientele of students & protégés (門生故吏) — a
			# real power base, but one bound to the throne that examined him.
			add_loyal_veterans = 6
			QING_char_bind = yes
		}
		# [#321] beyond the named laureate, the cycle mints a COHORT of graduate
		# characters into the hiring pool (scaled by the ladder's health) — the
		# concrete talent pipeline office-appointment draws on first.
		QING_exam_graduate_cohort = yes
		add_stability = 1
		# a suspect ranking in a corrupt court may spark a scandal
		if = {
			limit = { var:qing_corruption_level >= 45 }
			trigger_event = { id = qing_keju.3  days = { 30 120 } }
		}
	}

	# --- Confer it, and seat the new laureate on the Grand Council -------
	option = {
		name = "qing_keju.2.b"
		custom_tooltip = "qing_keju.2.b.tt"
		trigger = {
			var:qing_council_filled_count < 11
			political_influence >= 15
		}
		add_political_influence = -15
		scope:laureate = {
			add_trait = jinshi
			add_loyalty = loyalty_qing_exam_laureate
			add_popularity = 15
			# the top laureate is placed in the first vacant great office — his
			# Grand Council seat (council-is-offices model)
			QING_office_appoint_first_vacant = yes
		}
		# [#321] the cohort still graduates into the pool (the laureate is placed
		# directly; the rest wait in the pool for the next vacancy).
		QING_exam_graduate_cohort = yes
		QING_council_recompute = yes
		add_stability = 1
	}
}

########################################################################
# .3  AN EXAMINATION SCANDAL  (bought ranks / leaked papers)
# Fires when corruption has reached the examination halls — the graders have
# sold rankings or leaked the questions. The great scandals (e.g. the 1858
# Shuntian fraud, for which a Grand Secretary was executed) were prosecuted
# ferociously. The throne must choose severity against fiscal/factional cost.
########################################################################
qing_keju.3 = {
	type = country_event
	title = "qing_keju.3.t"
	desc = "qing_keju.3.desc"
	picture = senate_debate

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_keju.3 (exam scandal) [ROOT.GetTag]" }
	}

	# --- Prosecute to the utmost (execute the graders — the 1858 precedent) -
	option = {
		name = "qing_keju.3.a"
		custom_tooltip = "qing_keju.3.a.tt"
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -10 }
		add_tyranny = 3
		add_stability = 1
		add_country_modifier = { name = qing_censorate_vigilant  duration = 1825 }
		# a purged examiner among the courtiers, disgraced
		if = {
			limit = {
				any_character = { employer = ROOT  is_ruler = no  is_adult = yes  is_alive = yes  has_trait = corrupt }
			}
			random_character = {
				limit = { employer = ROOT  is_ruler = no  is_adult = yes  is_alive = yes  has_trait = corrupt }
				add_loyalty = loyalty_qing_disgraced
				add_popularity = -30
			}
		}
	}

	# --- Quietly annul the tainted results and re-sit the exam -----------
	option = {
		name = "qing_keju.3.b"
		custom_tooltip = "qing_keju.3.b.tt"
		trigger = { treasury >= 40 }
		add_treasury = -40
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -4 }
	}

	# --- Hush it up (the degrees stand; the rot compounds) ---------------
	option = {
		name = "qing_keju.3.c"
		custom_tooltip = "qing_keju.3.c.tt"
		add_country_modifier = { name = qing_keju_scandal  duration = 3650 }
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 5 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 5 }
	}
}

########################################################################
# .4  A BRILLIANT SCHOLAR PASSED OVER  (old rivalries / the road to rebellion)
# A gifted candidate has failed the examinations — whether by ill luck, by a
# rigged grade, or by the sheer brutal odds (the metropolitan pass rate was ~1%).
# History's warning is Hong Xiuquan, four times failed at Canton, who read a
# Christian tract in his despair and raised the Taiping Heavenly Kingdom.
########################################################################
qing_keju.4 = {
	type = country_event
	title = "qing_keju.4.t"
	desc = "qing_keju.4.desc"
	picture = great_library

	left_portrait = root.current_ruler
	right_portrait = scope:failed_scholar

	is_triggered_only = yes

	immediate = {
		# conjure the disappointed scholar (a Han commoner of real ability)
		create_character = {
			age = 26
			culture = han
			religion = root.religion
			add_martial = 6
			add_finesse = 9
			add_charisma = 8
			add_zeal = 7
			save_scope_as = failed_scholar
		}
		LOG_line = { sys = QING  msg = "EVENT qing_keju.4 (scholar passed over) [ROOT.GetTag]" }
	}

	# --- A special appointment — recognise the talent outside the ladder --
	option = {
		name = "qing_keju.4.a"
		custom_tooltip = "qing_keju.4.a.tt"
		trigger = { political_influence >= 15 }
		add_political_influence = -15
		scope:failed_scholar = {
			add_trait = juren
			add_loyalty = loyalty_qing_elevated
			add_popularity = 10
			set_home_country = ROOT
			# recognised and taken into service against the odds — a debt of
			# gratitude. Score him fresh in his new station and bind him.
			QING_char_bind = yes
		}
		add_stability = 1
	}

	# --- Grant a re-examination (a fair second chance) -------------------
	option = {
		name = "qing_keju.4.b"
		custom_tooltip = "qing_keju.4.b.tt"
		trigger = { treasury >= 30 }
		add_treasury = -30
		scope:failed_scholar = {
			add_trait = shengyuan
			set_home_country = ROOT
		}
		# a fair re-sit signals the lower ladder still works — the gentry notice.
		QING_DECLINE_nudge = { var = qing_exam_ladder  amount = 4 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = -2 }
	}

	# --- Turn him away — and reap the bitterness (the Hong Xiuquan road) --
	option = {
		name = "qing_keju.4.c"
		custom_tooltip = "qing_keju.4.c.tt"
		scope:failed_scholar = {
			add_loyalty = loyalty_qing_exam_embittered
			add_rival = root.current_ruler
			add_popularity = -10
			# shut out of imperial office, he raises a power base of his OWN — the
			# disaffected-gentry militia that grew into the Taiping (a hostile,
			# independent power base, the mirror of the loyal laureate's).
			add_loyal_veterans = 8
		}
		# a passed-over scholar-class swells the ranks of the disaffected
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = 6 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 4 }
	}
}

########################################################################
# .6  A SCHOLAR OF THE POOL RAISED TO OFFICE  (掄才 — drawing on the talent)
# [#321] Fires when a great office stands vacant AND the hiring pool holds a
# graduate waiting for a post. This is where the examination pipeline PAYS OFF:
# the throne draws the ablest pool scholar into the empty seat, rather than a
# stranger conjured from nowhere. If the throne declines, the seat stays open
# and the scholar waits (or, in time, retires — the pool tick).
########################################################################
qing_keju.6 = {
	type = country_event
	title = "qing_keju.6.t"
	desc = "qing_keju.6.desc"
	picture = chinese_throne_room

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		# a real vacancy exists AND the pool has at least one waiting scholar
		has_variable = qing_council_vacancy
		var:qing_council_vacancy > 0
		has_variable = qing_scholar_pool_count
		var:qing_scholar_pool_count > 0
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_keju.6 (pool scholar raised to office, pool=[ROOT.MakeScope.GetVariable('qing_scholar_pool_count').GetValue|0]) [ROOT.GetTag]" }
	}

	# --- Raise the ablest waiting scholar into the first vacant seat -----
	option = {
		name = "qing_keju.6.a"
		custom_tooltip = "qing_keju.6.a.tt"
		# fill the FIRST vacant office from the pool. QING_office_appoint_first_vacant
		# resolves which seat is empty; we route it through the pool-first picker by
		# pre-selecting the ablest scholar and appointing him to that seat.
		QING_exam_fill_first_vacant_from_pool = yes
		add_stability = 1
	}

	# --- Leave the seat empty; let the scholar wait his turn -------------
	option = {
		name = "qing_keju.6.b"
		custom_tooltip = "qing_keju.6.b.tt"
		add_political_influence = 5
	}
}

########################################################################
# .7  THE ABOLITION OF 捐納 PRESSED  (purchased office vs the merit ladder)
# [#321 deferred] The sale of offices (捐納), a fiscal expedient the throne could reach
# for in a war-chest crunch (qing_keju.1 option c / QING_sell_offices), corroded the
# examination order it competed with: every place bought for silver was a place denied
# to a degree-holder, and the reform-minded gentry pressed hard for its abolition. This
# fires when the purchased-ranks regime is ENTRENCHED (the modifier live) AND reform
# pressure has built — the throne must choose between the ready silver and the integrity
# of the ladder. Abolishing it is the concrete lever that REMOVES qing_office_purchased_ranks
# (otherwise only time expires it) and lifts the pass-rate; keeping it deepens the rot.
########################################################################
qing_keju.7 = {
	type = country_event
	title = "qing_keju.7.t"
	desc = "qing_keju.7.desc"
	picture = senate_debate

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_country_modifier = qing_office_purchased_ranks
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_keju.7 (abolition of juanna pressed, corruption=[ROOT.MakeScope.GetVariable('qing_corruption_level').GetValue|0]) [ROOT.GetTag]" }
	}

	# --- Abolish the sale of offices — restore the ladder (the reformer's choice) ----
	option = {
		name = "qing_keju.7.a"
		custom_tooltip = "qing_keju.7.a.tt"
		# purge the purchased-ranks regime outright (the concrete removal the sale otherwise
		# leaves to decay), and clean the intake: corruption falls, reform pressure eases.
		remove_country_modifier = qing_office_purchased_ranks
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -8 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = -6 }
		# a purged ladder passes more merit — recompute the pass-rate on the new footing.
		QING_exam_compute_pass_rate = yes
		add_stability = 1
		# [review-fix] add_legitimacy is a COUNTRY-scope effect (every other occurrence in
		# this codebase is bare — cf. qing_emeritus_events.txt:119, which wraps add_prestige in
		# current_ruler but applies add_legitimacy bare). The current_ruler wrapper here would
		# silently no-op + log a scope error; the tooltip promises +5 Legitimacy, so apply it bare.
		add_legitimacy = 5
	}

	# --- Reform the practice without ending it (a half-measure) ----------------------
	option = {
		name = "qing_keju.7.b"
		custom_tooltip = "qing_keju.7.b.tt"
		trigger = { treasury >= 40 }
		# buy back the worst abuses without ending the revenue stream: partial relief.
		add_treasury = -40
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -4 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = -2 }
	}

	# --- Keep the silver flowing — the war chest needs it (the rot compounds) --------
	option = {
		name = "qing_keju.7.c"
		custom_tooltip = "qing_keju.7.c.tt"
		# refresh the purchased-ranks modifier (the practice entrenches further) and take
		# the fiscal boon, at the cost of a ladder that clogs from the base and rising
		# reform pressure — the classic late-Qing fiscal-vs-merit trap.
		add_country_modifier = { name = qing_office_purchased_ranks  duration = 3650 }
		CURRENCY_grant_country_wealth = { thousands = 60 }
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 5 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 6 }
		QING_DECLINE_nudge = { var = qing_exam_ladder  amount = -8 }
	}
}

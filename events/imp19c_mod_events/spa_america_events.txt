########################################################################
# [boot #17] SPANISH AMERICA — the Napoleonic-era disruption of Spain's New World empire,
# 1800-1811, and a Qing opportunity to fish in troubled waters.
#
# The existing spa_bourbon arc scripts SPAIN'S OWN 1763->1815 trajectory (Jesuit expulsion,
# San Ildefonso, Dos de Mayo, Ferdinand's return); the existing qing_americas.2 already gives
# the Qing a "seize the Californian shore while Madrid is prostrate" beat. This arc fills the
# GAP the user asked for: the COLONIAL-SIDE disruption in Spanish America — the retrocession
# of Louisiana, the collapse of viceregal authority after the Bayonne abdications (the juntas),
# the loss of West Florida, and the outbreak of the Mexican independence war — as SPA-side
# posture beats, PLUS one further Qing-intervention opening (spa_america.5).
#
# Seeded from common/on_action/spa_america_on_actions.txt (day-offsets from the 1763.2.16
# START_DATE). AI-autonomous: self-gated on `tag = SPA` (never is_ai = no), each beat guarded
# by a per-beat has_variable done-flag, ai_chance-weighted options. The engine handles the
# actual wars/secessions; these beats nudge posture (stability / prestige / aggressive_expansion
# / political_influence) toward the historical trajectory that the MEX / mex_instability arc
# then presumes. The Qing beat (spa_america.5) is CHI-only, gated on a Pacific presence, and
# fires the proven qing_americas colonial hook. All logged sys = SPA (Qing beat sys = QING).
#
# Dates (offsets from 1763.2.16):
#   1800.10.01  San Ildefonso — Louisiana retroceded to France        -> 13741
#   1803.12.20  Louisiana Purchase — the USA buys it from France       -> 14916
#   1808.05.06  Bayonne abdications — viceregal juntas rise            -> 16515
#   1810.09.16  Grito de Dolores — the Mexican war of independence      -> 17378
#   1810.09.26  Republic of West Florida / US annexation (Oct 27)       -> 17388
#   1808.06.10  (Qing) Spain prostrate — a Pacific opening (CHI)        -> 16550
#
# Sources: .research/1763_north_america_map_groundtruth.md §3.
########################################################################

# --- .1  1800.10.01  The Retrocession of Louisiana (San Ildefonso) --------------------
# Godoy trades distant, costly Louisiana back to Napoleon's France for an Italian kingdom
# for the queen's brother — a paper triumph that strips Spain of the Mississippi buffer.
spa_america.1 = {
	type = country_event
	title = "spa_america.1.t"
	desc = "spa_america.1.desc"
	picture = throneroom

	left_portrait = root.current_ruler
	is_triggered_only = yes

	trigger = {
		tag = SPA
		NOT = { has_variable = spa_america_beat_retrocession }
	}

	immediate = {
		set_variable = spa_america_beat_retrocession
		add_political_influence = 5
		add_prestige = -4
		LOG_line = { sys = SPA  msg = "EVENT spa_america.1 (Retrocession of Louisiana, San Ildefonso) for [ROOT.GetTag]" }
	}

	# Trust the French alliance — the Italian crown is worth a wilderness.
	option = {
		name = "spa_america.1.a"
		add_prestige = 4
		ai_chance = { factor = 10 }
	}
	# Rue the bargain — the Mississippi was a shield, now sold.
	option = {
		name = "spa_america.1.b"
		add_stability = -1
		ai_chance = { factor = 3 }
	}
}

# --- .4  1810.09  The Cry of Dolores & the Loss of West Florida -----------------------
# The empire fractures: Father Hidalgo raises the peasantry of the Bajío under the banner of
# Guadalupe (16 Sept), and on the Gulf the Anglo settlers of West Florida seize Baton Rouge and
# declare a republic (26 Sept), which the United States annexes within weeks (27 Oct).
spa_america.4 = {
	type = country_event
	title = "spa_america.4.t"
	desc = "spa_america.4.desc"
	picture = revolt

	left_portrait = root.current_ruler
	is_triggered_only = yes

	trigger = {
		tag = SPA
		NOT = { has_variable = spa_america_beat_dolores }
	}

	immediate = {
		set_variable = spa_america_beat_dolores
		add_stability = -4
		add_prestige = -6
		# [#31 MEX-merge 2026-07-22] THE 1810 RE-LANDING OF MEXICO. At the 1763 start there was ONE
		# Viceroyalty of New Spain (the NSP tag); the separate "Mexico" (MEX) tag was made landless-
		# inert (its 32 core provinces folded into NSP — see setup/main/00_default.txt). The Grito de
		# Dolores (16 Sep 1810) opens the Mexican war of independence, so THIS is where MEX is carved
		# back out: grant its historic provinces from NSP to MEX, then bind MEX as a Spanish colony so
		# the mex_instability arc (Iturbide 1822 etc., gated on tag = MEX) fires on a landed MEX.
		# Guarded on MEX still being landless so it fires exactly once and is save-safe.
		if = {
			limit = {
				exists = c:MEX
				c:MEX = { NOT = { any_owned_province = { count >= 1 } } }
			}
			hidden_effect = {
				# The 32 Viceroyalty-of-New-Spain provinces that were the historic MEX core.
				p:7416 = { set_owned_by = c:MEX }
				p:7283 = { set_owned_by = c:MEX }
				p:1024 = { set_owned_by = c:MEX }
				p:1809 = { set_owned_by = c:MEX }
				p:2706 = { set_owned_by = c:MEX }
				p:5731 = { set_owned_by = c:MEX }
				p:7974 = { set_owned_by = c:MEX }
				p:558  = { set_owned_by = c:MEX }
				p:4292 = { set_owned_by = c:MEX }
				p:8247 = { set_owned_by = c:MEX }
				p:9785 = { set_owned_by = c:MEX }
				p:13263 = { set_owned_by = c:MEX }
				p:414  = { set_owned_by = c:MEX }
				p:2691 = { set_owned_by = c:MEX }
				p:3545 = { set_owned_by = c:MEX }
				p:4378 = { set_owned_by = c:MEX }
				p:6812 = { set_owned_by = c:MEX }
				p:8572 = { set_owned_by = c:MEX }
				p:912  = { set_owned_by = c:MEX }
				p:1787 = { set_owned_by = c:MEX }
				p:4175 = { set_owned_by = c:MEX }
				p:6360 = { set_owned_by = c:MEX }
				p:7138 = { set_owned_by = c:MEX }
				p:7439 = { set_owned_by = c:MEX }
				p:7878 = { set_owned_by = c:MEX }
				p:8119 = { set_owned_by = c:MEX }
				p:8948 = { set_owned_by = c:MEX }
				p:9801 = { set_owned_by = c:MEX }
				p:6511 = { set_owned_by = c:MEX }
				p:1187 = { set_owned_by = c:MEX }
				p:2066 = { set_owned_by = c:MEX }
				p:5759 = { set_owned_by = c:MEX }
				# Bind the freshly-landed MEX as a Spanish colony (mirrors the pre-merge SPA->MEX
				# client_colony). FUNC_make_subject binds an existing landed country; the grant above
				# gives it land first. mex_instability then walks it to independence (~1821).
				FUNC_make_subject = { overlord = ROOT  target = c:MEX  type = client_colony }
			}
			LOG_line = { sys = SPA  msg = "EVENT spa_america.4: re-landed + colonised MEX from NSP (1810 Grito de Dolores MEX-merge) for [ROOT.GetTag]" }
		}
		LOG_line = { sys = SPA  msg = "EVENT spa_america.4 (Grito de Dolores + West Florida) for [ROOT.GetTag]" }
	}

	# Loose the royalist army on the insurgents — Hidalgo will hang within the year (history).
	option = {
		name = "spa_america.4.a"
		add_aggressive_expansion = 6
		ai_chance = { factor = 9 }
	}
	# The empire is slipping — husband what strength remains for Spain itself.
	option = {
		name = "spa_america.4.b"
		add_stability = 1
		ai_chance = { factor = 3 }
	}
}

# ======================================================================
# .5  1808.06.10  (QING) SPAIN PROSTRATE — A PACIFIC OPENING (西班牙傾覆・太平洋之機)
# With Madrid fallen to Bonaparte and her American empire convulsed by juntas, the far Pacific
# reaches of New Spain — the Manila galleon's terminus, the Californian missions, the
# Nootka-coast claims — lie all but ungoverned. A Qing throne with a Pacific presence may press
# its own advantage while Spain cannot lift a hand. CHI-only; gated on a Pacific foothold; fires
# the proven qing_americas colonial hook (se_QING_AMERICAS.txt). Distinct from qing_americas.2
# (the Californian shore) — this is the wider "Spain is finished in the Pacific" opening.
# ======================================================================
spa_america.5 = {
	type = country_event
	title = "spa_america.5.t"
	desc = "spa_america.5.desc"
	picture = harbour

	left_portrait = root.current_ruler
	is_triggered_only = yes

	trigger = {
		tag = CHI
		NOT = { has_variable = spa_america_beat_qing_pacific }
		OR = {
			has_country_modifier = qing_col_silver_road
			has_country_modifier = qing_col_fur_coast
			has_country_modifier = qing_col_pacific_trade
			has_country_modifier = qing_col_golden_shore
			any_owned_province = {
				OR = {
					is_in_region = California
					is_in_region = Cascadia
					is_in_region = Alaska
					is_in_region = Pacific_Mexico
				}
			}
		}
	}

	immediate = {
		set_variable = spa_america_beat_qing_pacific
		LOG_line = { sys = QING  msg = "EVENT spa_america.5 (Spain prostrate, a Pacific opening) for [ROOT.GetTag]" }
	}

	# PRESS THE PACIFIC COAST — plant a foothold on the shore Spain can no longer hold.
	option = {
		name = "spa_america.5.a"
		custom_tooltip = "spa_america.5.a.tt"
		add_treasury = -120
		QING_americas_press_pacific_coast = yes
		add_prestige = 10
		current_ruler = { add_popularity = 8 }
		ai_chance = { factor = 5 }
	}
	# STAND ASIDE — the throne will not chase a wilderness across the world's widest ocean.
	option = {
		name = "spa_america.5.b"
		ai_chance = { factor = 10 }
	}
}

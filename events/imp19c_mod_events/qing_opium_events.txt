########################################################################
# qing_opium_events.txt — [#366] THE SILVER / OPIUM CRISIS (白銀外流 / 鴉片貿易).
# The player-facing surface of the opium monetary model (se_QING_OPIUM.txt): the memorials
# and decisions that dramatise the silver drain, the Lin Zexu crackdown, the treaty-forced
# legalization, and the road to domestic production. CHI-only.
#
# .1  THE SILVER DRAIN MEMORIAL (銀荒之憂) — a censor's warning as the drain first bites:
#     PROHIBIT harder (appoint a commissioner) vs TOLERATE-FOR-REVENUE.
# .2  THE HUMEN DECISION (虎門銷煙) — Lin Zexu has seized the foreign opium: DESTROY it
#     (legitimacy + the road to war with Britain) vs RELEASE it (the drain rolls on).
# .3  THE TREATY LEGALIZATION (條約弛禁) — forced post-Opium-War: accept legal, taxed opium
#     (a revenue offset, a worse epidemic) — gated on the unequal-treaty order being in force.
# .4  THE EPIDEMIC CRISIS (煙毒蔓延) — the addiction has hollowed the realm: crack down at a
#     social cost vs let the poppy pay (domestic production).
#
# Sources: se_QING_OPIUM.txt header; Hsü, Rise of Modern China; Spence, Search for Modern
# China; Greenberg, British Trade and the Opening of China.
########################################################################

namespace = qing_opium

# ======================================================================
# .1  THE SILVER DRAIN MEMORIAL (銀荒之憂). As opium imports climb and silver bleeds west,
# a censor memorialises the throne: the copper-silver exchange is turning against the
# common people, the land tax bites harder in real terms, and the treasury's reserves
# thin. Queued once by QING_opium_pulse (import index >= 40, still prohibiting, no
# commissioner). Re-checks its own state.
# ======================================================================
qing_opium.1 = {
	type = country_event
	title = "qing_opium.1.t"
	desc = "qing_opium.1.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		var:qing_opium_import_index >= 40
		var:qing_opium_posture = 0
	}

	immediate = {
		# [#366-R review-fix] set the once-only offered flag HERE (only when the event actually
		# fires), not in QING_opium_pulse — so if this event's re-check trigger fails during the
		# 5-20 day delay (e.g. the player switches posture), the flag is never stranded set and
		# the memorial can be re-offered when the drain bites again.
		set_variable = { name = qing_opium_warning_seen  value = 1 }
		LOG_line = { sys = QING  msg = "EVENT qing_opium.1 (silver drain memorial) fired for" }
	}

	# --- PROHIBIT HARDER (嚴禁) — heed the memorial and appoint an Imperial Commissioner to
	# stamp out the traffic. Suppresses the import ramp while prohibition holds; sets the
	# stage for the Humen decision. A resolute throne gains legitimacy.
	option = {
		name = "qing_opium.1.prohibit"
		custom_tooltip = "qing_opium.1.prohibit.tt"
		QING_opium_appoint_commissioner = yes
		LOG_line = { sys = QING  msg = "opium: memorial heeded, commissioner appointed for" }
	}

	# --- TOLERATE FOR REVENUE (弛禁徵稅) — wink at the trade and tax it instead. Eases the
	# fisc a touch now (the customs take), but the drain and the addiction run unchecked, and
	# the graft it breeds feeds corruption.
	option = {
		name = "qing_opium.1.tolerate"
		custom_tooltip = "qing_opium.1.tolerate.tt"
		QING_opium_set_posture = { posture = 1 }
		add_treasury = 20
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 4 }
		LOG_line = { sys = QING  msg = "opium: trade tolerated for revenue for" }
	}
}

# ======================================================================
# .2  THE HUMEN DECISION (虎門銷煙). Commissioner Lin Zexu has blockaded the foreign
# factories at Canton and forced the surrender of some 20,000 chests of opium. The throne
# must decide their fate: destroy them in the pits at Humen — a thunderclap of imperial
# resolve that Britain will answer with gunboats — or release them and let the traffic, and
# the silver drain, roll on. Queued by QING_opium_pulse (index >= 60, commissioner in place).
# ======================================================================
qing_opium.2 = {
	type = country_event
	title = "qing_opium.2.t"
	desc = "qing_opium.2.desc"
	picture = looting

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_lin_zexu_appointed
		var:qing_opium_posture = 0
		NOT = { has_variable = qing_opium_crackdown_resolved }
	}

	immediate = {
		set_variable = { name = qing_opium_crackdown_resolved  value = 1 }
		LOG_line = { sys = QING  msg = "EVENT qing_opium.2 (Humen decision) fired for" }
	}

	# --- DESTROY IT (銷煙) — the Humen destruction. A sharp cut to the traffic + a surge of
	# legitimacy (the throne looks resolute against the barbarian poison), at the price of
	# Britain's fury — the road to the First Opium War.
	option = {
		name = "qing_opium.2.destroy"
		custom_tooltip = "qing_opium.2.destroy.tt"
		QING_opium_crackdown_destroy = yes
		LOG_line = { sys = QING  msg = "opium: the Humen destruction ordered for" }
	}

	# --- RELEASE IT (妥協) — return the chests and let the traffic continue. No war provoked,
	# but the drain rolls on and the court loses face before the whole realm.
	option = {
		name = "qing_opium.2.release"
		custom_tooltip = "qing_opium.2.release.tt"
		QING_opium_crackdown_release = yes
		LOG_line = { sys = QING  msg = "opium: the seized stock released for" }
	}
}

# ======================================================================
# .3  THE TREATY LEGALIZATION (條約弛禁). The wars are lost, the treaty order imposed, and
# now the victors demand what they came for: the legal, tariffed import of opium (the 1858
# Tianjin settlement). The throne can no longer refuse — only choose how to make the best of
# it. Queued by the decline flavour roll when the treaty system is in force but opium is not
# yet legalized. Re-checks its own state.
# ======================================================================
qing_opium.3 = {
	type = country_event
	title = "qing_opium.3.t"
	desc = "qing_opium.3.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_treaty_system_imposed
		NOT = { var:qing_opium_posture = 2 }
	}

	immediate = {
		# [#366-R review-fix] set the once-only offered flag HERE (only when the event fires).
		set_variable = { name = qing_opium_legalize_offered  value = 1 }
		LOG_line = { sys = QING  msg = "EVENT qing_opium.3 (treaty legalization) fired for" }
	}

	# --- ACCEPT THE TARIFF (抽釐徵稅) — legalize and tax the trade. The tariff revenue offsets
	# the fisc (the drain is now at least partly recouped as customs), but the epidemic, no
	# longer even nominally checked, spreads unchecked.
	option = {
		name = "qing_opium.3.accept"
		custom_tooltip = "qing_opium.3.accept.tt"
		QING_opium_treaty_legalize = yes
		LOG_line = { sys = QING  msg = "opium: legalized and tariffed by treaty for" }
	}
}

# ======================================================================
# .4  THE EPIDEMIC CRISIS (煙毒蔓延). A generation on, the poppy has hollowed the realm:
# addicts in every county, labour lost to the pipe, the secret societies swollen with the
# ruined. The throne faces the epidemic itself, not the trade balance. Queued by
# QING_opium_pulse when the addicted share is severe.
# ======================================================================
qing_opium.4 = {
	type = country_event
	title = "qing_opium.4.t"
	desc = "qing_opium.4.desc"
	picture = farming

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		var:qing_opium_addicted_share >= 55
	}

	immediate = {
		# [#366-R review-fix] set the once-only offered flag HERE (only when the event fires).
		set_variable = { name = qing_opium_epidemic_offered  value = 1 }
		LOG_line = { sys = QING  msg = "EVENT qing_opium.4 (epidemic crisis) fired for" }
	}

	# --- MORAL SUASION + SUPPRESSION (勸戒并禁) — throw the state at the epidemic: interdiction,
	# rehabilitation edicts, the gentry mobilised. A real cost in coin and unrest now (the
	# ruined resist), but the addicted share is cut back.
	option = {
		name = "qing_opium.4.suppress"
		custom_tooltip = "qing_opium.4.suppress.tt"
		trigger = { treasury >= 40 }
		add_treasury = -40
		QING_opium_nudge = { var = qing_opium_addicted_share  amount = -15 }
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = 4 }
		add_legitimacy = 5
		LOG_line = { sys = QING  msg = "opium: epidemic suppression campaign mounted for" }
	}

	# --- LET THE POPPY PAY (以土抵洋) — unlock domestic Yunnan/Sichuan cultivation. Home-grown
	# opium substitutes for imports (the silver drain eases) but the epidemic deepens: the
	# drug is now cheaper and everywhere.
	option = {
		name = "qing_opium.4.domestic"
		custom_tooltip = "qing_opium.4.domestic.tt"
		QING_opium_unlock_domestic = yes
		QING_opium_nudge = { var = qing_currency_stress  amount = -5 }
		LOG_line = { sys = QING  msg = "opium: domestic cultivation unlocked to stem the silver drain for" }
	}
}

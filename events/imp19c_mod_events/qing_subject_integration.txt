########################################################################
# qing_subject_integration.txt — reaction events for the Qing subject
# integration process (the promote/demote/integrate/absorb feature).
#
# Fired from se_SUBJECT_QING.txt (SUBJ_QING_roll_reaction) either right after
# a deliberate integration push, or on the ambient pulse for any subject with
# an active, unsuspended integration. In every event:
#     ROOT / this  = the overlord (player, Qing / CHI)
#     scope:target = the subject governorship being integrated
#
# The chain is deliberately Qing-only and player-only; AI never sees it.
# Each event is anchored to real Qing frontier history and cites it in the
# localisation. Options let the player push through, delay, suspend, or
# accelerate the process, and are wired into existing mechanics: treasury /
# currency, industrialisation, trade, pop happiness/unrest, stability,
# character loyalty, and the subject-relationship modifiers.
#
# Namespace = qing_integ
#   .10  NEGATIVE — popular unrest (Three Feudatories / White Lotus echo)
#   .11  NEGATIVE — corruption scandal (Heshen archetype)
#   .12  NEGATIVE — ethnic strife (Miao / Panthay / Dungan risings)
#   .20  POSITIVE — bountiful frontier harvest (tuntian colonies)
#   .21  POSITIVE — investiture festival / imperial favour
########################################################################

namespace = qing_integ

########################################################################
# .10  POPULAR UNREST
# Anchor: Revolt of the Three Feudatories (三藩之亂, 1673-81) — Kangxi's move
# to revoke the autonomous southern fiefs triggered an eight-year revolt; and
# the White Lotus Rebellion (白蓮教起義, 1794-1804), where a Daoguang decree
# admitted "it was extortion by local officials that goaded the people into
# rebellion" (官吏苛索逼民為亂). Tightening the screws provokes a rising.
########################################################################
qing_integ.10 = {
	type = country_event
	title = "qing_integ.10.t"
	desc = "qing_integ.10.desc"
	picture = revolt

	left_portrait = root.current_ruler
	right_portrait = scope:target.current_ruler

	goto_location = scope:unrest_province

	is_triggered_only = yes

	# [playtest loc-fix] Self-abort if the subject vanished during the 3-10 day
	# delay (absorbed to landless / deleted) — otherwise scope:target orphans and
	# the desc renders ERROR:[scope:target.GetName]. Mirrors the qing_integ.30 guard.
	trigger = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
	}

	immediate = {
		LOG_line = { sys = SUBJ  msg = "EVENT qing_integ.10 (unrest) overlord=[ROOT.GetTag] subject=[scope:target.GetTag]" }
		# Anchor the event on the subject's most populous province, if any.
		if = {
			limit = { scope:target = { any_owned_province = { count >= 1 } } }
			scope:target = {
				ordered_owned_province = {
					order_by = total_population
					max = 1
					save_scope_as = unrest_province
				}
			}
		}
	}

	# --- Push through with a show of force: crush it, but at a cost -------
	option = {
		name = "qing_integ.10.a"
		custom_tooltip = "qing_integ.10.a.tt"
		add_stability = -5
		add_war_exhaustion = 2
		# The subject network resents the heavy hand (Three Feudatories echo).
		add_country_modifier = {
			name = qing_integ_frontier_resentment
			duration = 1825
		}
		if = {
			limit = { exists = scope:unrest_province }
			scope:unrest_province = {
				add_province_modifier = {
					name = local_unrest_harsh
					duration = 730
				}
			}
		}
		# But a crushed revolt does not stop the process.
	}

	# --- Suspend the integration and let tempers cool --------------------
	option = {
		name = "qing_integ.10.b"
		custom_tooltip = "qing_integ.10.b.tt"
		SUBJ_QING_suspend_integration = yes
		SUBJ_QING_reduce_integration = { steps = 1 }
		add_stability = 2
	}

	# --- Give alms and remit taxes to pacify the frontier (rou yuan 柔遠) -
	option = {
		name = "qing_integ.10.c"
		custom_tooltip = "qing_integ.10.c.tt"
		trigger = { treasury >= 100 }
		add_treasury = -100
		if = {
			limit = { exists = scope:unrest_province }
			scope:unrest_province = {
				add_province_modifier = {
					name = positive_food_modifier
					duration = 730
				}
			}
		}
		# Applied to the overlord: subject_loyalty tightens the bond, and integrate_speed
		# now feeds the ENGINE's subject-integration timer (post-#subj-integ-rework: once the
		# 改土歸流 capstone hands a governorship to the engine's native integration, integrate_speed
		# genuinely accelerates the gradual absorption — it was inert on the old flat-+1 meter).
		add_country_modifier = {
			name = qing_integ_imperial_favor
			duration = 1095
		}
	}
}

########################################################################
# .11  CORRUPTION SCANDAL
# Anchor: Heshen (和珅, d. 1799), "the most corrupt official in Chinese
# history" — his confiscated estate (~1.1bn taels) reputedly equalled twelve
# years of imperial revenue. His graft diverted funds meant for canals and
# dams, worsening floods and famine (Hummel 1943). A grandee has been skimming
# the integration funds.
########################################################################
qing_integ.11 = {
	type = country_event
	title = "qing_integ.11.t"
	desc = "qing_integ.11.desc"
	picture = dagger_behind_back

	left_portrait = root.current_ruler
	right_portrait = scope:corrupt_official

	is_triggered_only = yes

	# [playtest loc-fix] Self-abort if the subject vanished during the delay — the
	# LOG line and (via .a) the desc reference scope:target. corrupt_official is
	# re-derived with a fallback below, but target cannot be. Mirrors qing_integ.30.
	trigger = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
	}

	immediate = {
		LOG_line = { sys = SUBJ  msg = "EVENT qing_integ.11 (scandal) overlord=[ROOT.GetTag] subject=[scope:target.GetTag]" }
		# Find a disloyal official at court to pin the scandal on; fall back
		# to any adult courtier if none are outright disloyal.
		if = {
			limit = {
				any_character = {
					employer = ROOT
					is_ruler = no
					is_adult = yes
					prisoner = no
					loyalty < 50
				}
			}
			random_character = {
				limit = {
					employer = ROOT
					is_ruler = no
					is_adult = yes
					prisoner = no
					loyalty < 50
				}
				weight = {
					modifier = { factor = 2  prominence >= 50 }
					modifier = { factor = 2  has_any_office = yes }
				}
				save_scope_as = corrupt_official
			}
		}
		else_if = {
			limit = {
				any_character = { employer = ROOT  is_ruler = no  is_adult = yes  prisoner = no }
			}
			random_character = {
				limit = { employer = ROOT  is_ruler = no  is_adult = yes  prisoner = no }
				save_scope_as = corrupt_official
			}
		}
		else = {
			# Last-resort fallback so scope:corrupt_official is ALWAYS set — the
			# desc text and right_portrait reference it, and an unset scope would
			# render a broken portrait and an unresolved name. The ruler stands in.
			current_ruler = { save_scope_as = corrupt_official }
			LOG_line = { sys = SUBJ  msg = "qing_integ.11 no courtier available; ruler stands in as corrupt_official" }
		}
	}

	# --- Impeach and confiscate (the Jiaqing solution) -------------------
	option = {
		name = "qing_integ.11.a"
		custom_tooltip = "qing_integ.11.a.tt"
		trigger = { exists = scope:corrupt_official }
		# Seize the grandee's fortune — a Heshen-scale windfall to the treasury.
		add_treasury = 150
		add_stability = 3
		add_tyranny = 5
		scope:corrupt_official = {
			add_corruption = 20
			add_loyalty = inspired_disloyalty_other_l
		}
	}

	# --- Quietly settle it, losing money and momentum --------------------
	option = {
		name = "qing_integ.11.b"
		custom_tooltip = "qing_integ.11.b.tt"
		add_treasury = -80
		SUBJ_QING_reduce_integration = { steps = 1 }
		add_country_modifier = {
			name = qing_integ_corruption_scandal
			duration = 1095
		}
	}

	# --- Institute yanglian yin (養廉銀), "nourish honesty" stipends ------
	option = {
		name = "qing_integ.11.c"
		custom_tooltip = "qing_integ.11.c.tt"
		trigger = { treasury >= 120 }
		add_treasury = -120
		current_ruler = { add_popularity = 5 }
		if = {
			limit = { exists = scope:corrupt_official }
			scope:corrupt_official = {
				add_character_modifier = {
					name = qing_integ_bribed_official
					duration = 1825
				}
			}
		}
	}
}

########################################################################
# .20  BOUNTIFUL FRONTIER HARVEST
# Anchor: the tuntian (屯田) military-agricultural colonies that anchored Qing
# control in Xinjiang, Inner Mongolia and Heilongjiang (Kim, Borderland
# Capitalism, 2016). A bumper harvest feeds the garrisons and settlers who hold
# the frontier — and warms the subject to absorption.
########################################################################
qing_integ.20 = {
	type = country_event
	title = "qing_integ.20.t"
	desc = "qing_integ.20.desc"
	picture = farming

	left_portrait = root.current_ruler

	is_triggered_only = yes

	# [playtest loc-fix] Self-abort if the subject vanished during the delay —
	# scope:target drives the desc. Mirrors qing_integ.30.
	trigger = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
	}

	immediate = {
		LOG_line = { sys = SUBJ  msg = "EVENT qing_integ.20 (harvest) overlord=[ROOT.GetTag] subject=[scope:target.GetTag]" }
	}

	# --- Store the grain surplus in the Ever-Normal Granaries (常平倉) ----
	# [playtest fix task #22] The surplus is LITERAL GRAIN, not silver — it feeds the
	# ever-normal granary reserve (the concrete qing_granary_food pool), NOT the treasury.
	# The old CURRENCY_grant_country_wealth path granted 0 gold at a 1763 start (the
	# currency backing was not yet resolvable) AND, left open, re-threw every frame — the
	# ~25k-line wealth_to_grant flood. This option shows only when a granary EXISTS to
	# receive the grain (qing_granary_count >= 1, so the pool's cap is already > 0 and the
	# deposit survives the pulse clamp); otherwise the build option below is offered instead.
	option = {
		name = "qing_integ.20.a"
		trigger = {
			has_variable = qing_granary_count
			var:qing_granary_count >= 1
		}
		custom_tooltip = "qing_integ.20.a.tt"
		# deposit the harvest into the ever-normal granary pool (food, not money). The pool
		# is derived/clamped to [0, count*200] each quarter by QING_DECLINE_granary_pool, so a
		# deposit into an already-counted pool persists; init to 0 if the pulse has not yet run.
		if = { limit = { NOT = { has_variable = qing_granary_food } }  set_variable = { name = qing_granary_food  value = 0 } }
		change_variable = { name = qing_granary_food  add = 150 }
		add_country_modifier = {
			name = qing_integ_bountiful_frontier
			duration = 730
		}
		# A prosperous year advances the process a further step for free.
		SUBJ_QING_advance_integration = { steps = 1 }
	}

	# --- No granary yet: build a new Ever-Normal Granary to hold the surplus ----
	# [playtest fix task #22] Shown only when the realm holds NO ever-normal granary
	# (qing_granary_count unset or 0). We do NOT deposit grain here: the granary count is
	# retallied by the Works sweep, and the DECLINE pulse (which runs first and clamps the
	# pool to [0, count*200]) would zero any grain banked before the count catches up. The
	# reward is the granary infrastructure itself — it then fills organically from surplus
	# states via the pool sweep. QING_revenue_stock_granary raises one 糧倉 in the Yellow-
	# River heartland (北京/天津/保定/開封/濟南).
	option = {
		name = "qing_integ.20.a2"
		trigger = {
			OR = {
				NOT = { has_variable = qing_granary_count }
				var:qing_granary_count < 1
			}
		}
		custom_tooltip = "qing_integ.20.a2.tt"
		QING_revenue_stock_granary = yes
		add_country_modifier = {
			name = qing_integ_bountiful_frontier
			duration = 730
		}
		# A prosperous year advances the process a further step for free.
		SUBJ_QING_advance_integration = { steps = 1 }
	}

	# --- Invest the surplus in a state farm / industrial estate ----------
	option = {
		name = "qing_integ.20.b"
		custom_tooltip = "qing_integ.20.b.tt"
		trigger = { scope:target = { any_owned_province = { count >= 1 } } }
		scope:target = {
			ordered_owned_province = {
				order_by = total_population
				max = 1
				add_building_level = IND_industrial_estate
			}
		}
		# Applied to the overlord: subject_loyalty tightens the bond, and integrate_speed
		# now feeds the ENGINE's subject-integration timer (post-#subj-integ-rework: once the
		# 改土歸流 capstone hands a governorship to the engine's native integration, integrate_speed
		# genuinely accelerates the gradual absorption — it was inert on the old flat-+1 meter).
		add_country_modifier = {
			name = qing_integ_imperial_favor
			duration = 730
		}
	}
}

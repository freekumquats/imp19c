########################################################################
# qing_roster_events.txt — THE ANACHRONISTIC HISTORICAL-CHARACTER ROSTER (task #13).
#
# Real late-Qing figures arrive on the stage keyed to MECHANIC CONTEXT rather
# than to their historical birth date (see se_QING_ROSTER.txt for the design and
# the naming reality). Each event conjures the man with historically-flavoured
# stats + traits, pins his real name as a nickname (add_nickname), refers to him
# by full name in the text, finalizes him into the court (QING_roster_finalize),
# and offers the throne a choice of how to employ or dispose of him — cross-wired
# to the Grand Council (se_QING_COUNCIL), the offices, the Self-Strengthening
# track (se_QING_SELFSTR), the decline/reform meters (se_QING_DECLINE), and the
# affinity model (se_QING_AFFINITY).
#
# Dispatched by QING_roster_roll (fired from QING_frontier_flavour_roll); each
# figure guarded by a once-only ROOT flag set in the event's immediate.
#
# CHI-only, player-only. All logged sys = QING. Stat scale: the roster memory's
# 0-9 axis maps directly onto Imperator's add_martial/finesse/charisma/zeal.
# Trait keys verified against common/traits/ (00_personality/military/imp19c).
#
#   ROOT / this = the Qing overlord (CHI, player); root.current_ruler = emperor.
#
# Namespace = qing_roster
#   .1  和珅 Heshen           — the corrupt favourite (decline anchor)
#   .2  林則徐 Lin Zexu        — the incorruptible commissioner
#   .3  恭親王奕訢 Prince Gong — the reform & diplomacy patron
#   .4  李鴻章 Li Hongzhang    — the Self-Strengthening hub
#   .5  曾國藩 Zeng Guofan     — the statesman-general
#   .6  左宗棠 Zuo Zongtang    — the frontier reconqueror
#   .7  張之洞 Zhang Zhidong   — the ti-yong industrialiser
#   .8  赫德 Robert Hart       — the foreign inspector-general of customs
#   .9  戈登 Charles Gordon    — the Ever-Victorious Army commander
#   .10 康有為 Kang Youwei      — the constitutional reformer
#   .11 孫中山 Sun Yat-sen     — the republican revolutionary
#   .12 袁世凱 Yuan Shikai      — the ambitious militarist (can betray)
#   .13 洪秀全 Hong Xiuquan     — the Taiping Heavenly King (rebel foil)
########################################################################

namespace = qing_roster

########################################################################
# .1  和珅 HESHEN — THE OVERMIGHTY FAVOURITE
# The corrupt Manchu guardsman Qianlong raised over the whole apparatus of state.
# Arrives when the court is already running to graft; the throne may raise him
# (efficient, but the corruption deepens — the Heshen bargain) or shun him.
########################################################################
qing_roster.1 = {
	type = country_event
	title = "qing_roster.1.t"
	desc = "qing_roster.1.desc"
	picture = council_country

	left_portrait = root.current_ruler
	right_portrait = scope:figure

	is_triggered_only = yes

	immediate = {
		set_variable = qing_spawned_heshen
		create_character = {
			age = 35
			culture = manchu
			religion = root.religion
			add_finesse = 9
			add_charisma = 7
			add_martial = 2
			add_zeal = 3
			add_trait = corrupt
			add_trait = silver_tongued
			add_trait = ambitious
			save_scope_as = figure
		}
		scope:figure = { QING_roster_finalize = { nick = NICKNAME_QING_HESHEN } }
		LOG_line = { sys = QING  msg = "EVENT qing_roster.1 (Heshen 和珅 arrives) [ROOT.GetTag]" }
	}

	# --- Raise him over the apparatus of state (the Qianlong bargain) ----
	option = {
		name = "qing_roster.1.a"
		custom_tooltip = "qing_roster.1.a.tt"
		trigger = { var:qing_council_filled_count < 11 }
		scope:figure = {
			# raise him straight to the Grand Chancellorship (領班軍機大臣) — the
			# summit office is itself the head of the Grand Council (council-is-offices).
			QING_office_appoint = { office = chancellor }
			add_loyalty = loyalty_qing_indulged
		}
		QING_council_recompute = yes
		# an able chief runs the government smoothly — but the graft compounds
		QING_DECLINE_nudge = { var = qing_council_effectiveness  amount = 8 }
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 10 }
		set_variable = { name = qing_council_overmighty_flag  value = 1 }
	}

	# --- Keep him at a wary distance (a minor post only) -----------------
	option = {
		name = "qing_roster.1.b"
		custom_tooltip = "qing_roster.1.b.tt"
		scope:figure = { add_popularity = 5 }
		add_political_influence = 5
	}
}

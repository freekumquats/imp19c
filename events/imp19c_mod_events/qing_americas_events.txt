# qing_americas_events.txt — [scripted-revolutions 2nd pass] THE NORTH AMERICAN OPPORTUNITY.
# The Napoleonic Wars and the War of 1812 (usa_1812.*, fra_revolution.*) fling the North
# American frontier open; a Qing that has planted a Pacific foothold (the Great Pacific
# Enterprise colonisation tree) may exploit the distraction. Player-facing CHOICE events
# (the Qing is the player, D0 — every one OFFERS, never forces). CHI-gated, one-shot,
# done-flag-guarded. Machinery: se_QING_AMERICAS.txt.
#
#   ROOT / this = c:CHI.
#
# GATE — every event requires a genuine Chinese presence in the Americas (see each
# trigger): a Pacific-coast foothold modifier from the colonisation tree
# (qing_col_silver_road / _gulf_gate / _fur_coast / _pacific_trade / _golden_shore), or an
# actually-held North American province. Without a foothold none of these fire — the Qing
# cannot be offered the New World from an empty Pacific.
#
# CLAIM & SPHERE, NOT CONQUEST — the .1 Louisiana offer grants the preemption CLAIM to the
# watershed + draws the sovereign Plains nations into the Qing sphere as tributaries; it
# does NOT annex Native-held land (the interior was held by the Osage/Comanche/Lakota/…
# into the 1820s-30s — Hämäläinen; DuVal). The Pacific beats (.2 Fort Ross window, .3
# Astoria, .4 Russian America) lay coastal claims + grant footholds, mirroring the real
# Russian (Fort Ross 1812) and British (Astoria 1813) moves into the same vacuum.
#
# Seeded from qing_americas_on_actions.txt at the §4 / 2nd-pass offsets (1763.2.16 anchor):
#   .1  1803.04.30 The Louisiana Offer         -> 14682   (needs NA presence)
#   .2  1808.06.01 Spain Prostrate / California -> 16537   (Peninsular War vacuum)
#   .3  1813.10.23 The Fall of Astoria          -> 18511   (War-of-1812 opening)
#   .4  1812.09.11 The Tsar's Colony Stripped   -> 18104   (Russia pinned west)
#
# Sources (research agents, EN academic): Hämäläinen, The Comanche Empire (2008, Yale) &
#   Lakota America (2019, Yale); DuVal, The Native Ground (2006); Gibson, Otter Skins,
#   Boston Ships, and China Goods (maritime fur trade to Canton); Tikhmenev, A History of
#   the Russian-American Company (1978); standard Louisiana Purchase / Fort Ross / Astoria
#   scholarship. See scripted_revolutions.md §7-§9.
########################################################################

# ======================================================================
# .1  1803.04.30  THE LOUISIANA OFFER (路易斯安那之議). Napoleon, his Saint-Domingue army
# dead of yellow fever and war with Britain looming, cannot hold a territory across the
# Atlantic and needs silver. Because the Qing has planted a foothold in the Americas, the
# same offer that went to Washington comes to Peking: the whole trans-Mississippi claim.
# But — as the court's own memorialists note — France sells only a PAPER TITLE: the land is
# held by the horse-lords of the plains (the Osage, the Comanche, the Lakota), who bow to
# no European. The choice: buy the claim and draw the plains nations into the tributary
# sphere; buy it as bald title alone; or decline (the American republic takes it — history).
# ======================================================================
qing_americas.1 = {
	type = country_event
	title = "qing_americas.1.t"
	desc = "qing_americas.1.desc"
	picture = chinese_throne_room
	left_portrait = root.current_ruler
	is_triggered_only = yes
	trigger = {
		tag = CHI
		NOT = { has_variable = qing_americas_beat_louisiana }
		# a genuine Chinese presence in the Americas.
		OR = {
			has_country_modifier = qing_col_silver_road
			has_country_modifier = qing_col_gulf_gate
			has_country_modifier = qing_col_fur_coast
			has_country_modifier = qing_col_pacific_trade
			has_country_modifier = qing_col_golden_shore
			any_owned_province = {
				OR = {
					is_in_region = California
					is_in_region = Cascadia
					is_in_region = Alaska
					is_in_region = Pacific_Mexico
					is_in_region = Great_Plains
				}
			}
		}
	}
	immediate = {
		set_variable = qing_americas_beat_louisiana
		LOG_line = { sys = QING  msg = "EVENT qing_americas.1 (The Louisiana Offer, 1803) for [ROOT.GetTag]" }
	}
	# BUY THE CLAIM & COURT THE PLAINS NATIONS — silver for the preemption right, and the
	# horse-lords drawn under the wing as tributaries; the trans-Mississippi west becomes a
	# contested Qing sphere (NOT annexed land).
	option = {
		name = "qing_americas.1.a"
		custom_tooltip = "qing_americas.1.a.tt"
		add_treasury = -200
		QING_americas_claim_louisiana = yes
		QING_americas_embrace_plains = yes
		add_prestige = 15
		current_ruler = { add_popularity = 15 }
		ai_chance = { factor = 5 }
	}
	# BUY THE PAPER TITLE ALONE — take the claim, but do not entangle the throne with the
	# plains nations (a bald preemption right the Qing may or may not ever make good).
	option = {
		name = "qing_americas.1.b"
		custom_tooltip = "qing_americas.1.b.tt"
		add_treasury = -120
		QING_americas_claim_louisiana = yes
		add_prestige = 6
		ai_chance = { factor = 6 }
	}
	# DECLINE — the throne will not pour silver into a wilderness held by mounted barbarians
	# across the world's widest ocean. The young American republic buys it instead (history).
	option = {
		name = "qing_americas.1.c"
		ai_chance = { factor = 10 }
	}
}

########################################################################
# qing_tribute_events.txt — THE TRIBUTE EMBASSY (朝貢), [#322 / overnight-build-#2].
#
# The positive rhythm of the tributary order at its zenith: the periodic embassies
# the vassal states sent to Peking bearing tribute — the Korean 燕行使, the Vietnamese,
# Ryukyu, Nepalese and Taiwan-aborigine missions — each a real ENVOY the throne
# receives. Driven by se_QING_TRIBUTE.txt's shared quarterly scheduler off the ACTUAL
# tributary subjects (KOR / VIE / NEP / RYU / TNN). CHI-only, player-only.
#
#   ROOT / this = the Qing overlord (CHI, player)
#   scope:trib_sender = the tributary COUNTRY that sent the mission
#   scope:trib_envoy  = the envoy CHARACTER minted for it
#
# Namespace = qing_tribute
#   .1  a tribute embassy arrives at Peking (audience-with-gifts vs perfunctory-dismiss)
#
# Sources: Fairbank (ed.), The Chinese World Order (1968); Hamashita, China, East
#   Asia and the Global Economy (2008); Larsen, Tradition, Treaties, and Trade (2008).
########################################################################

namespace = qing_tribute

########################################################################
# .1  A TRIBUTE EMBASSY ARRIVES  (朝貢使節入覲)
# The mission reaches the capital. The throne may receive it in full ceremony — a
# lavish audience that affirms the whole tributary order and returns the customary
# imperial largesse — or dismiss it perfunctorily, taking the tribute but sparing the
# expense and the ritual, at a cost to the order's prestige.
########################################################################
qing_tribute.1 = {
	type = country_event
	title = "qing_tribute.1.t"
	desc = "qing_tribute.1.desc"
	picture = council_country

	left_portrait = root.current_ruler
	# [BT-7b logfix] portrait fields resolve at FIRE time, BEFORE immediate saves the scope, so
	# scope:trib_envoy was empty here -> ERROR:[scope:trib_envoy.GetName]. Point it at the PERSISTED
	# var instead (proven: right_portrait = var:qing_office_justice_holder). The desc/options still
	# read scope:trib_sender/trib_envoy, which immediate re-saves before the body renders.
	right_portrait = var:qing_trib1_envoy

	is_triggered_only = yes

	# [BT-7 logfix] self-guard on the PERSISTED vars, not raw saved scopes: a scope
	# saved at the scheduler site can go stale across the 5-20d delay, so the arrival
	# rendered ERROR:[scope:trib_sender.GetName]/[scope:trib_envoy.GetName]. Guard that
	# the subject is still ours and the envoy still exists; else cancel cleanly.
	trigger = {
		tag = CHI
		has_variable = qing_trib1_sender
		has_variable = qing_trib1_envoy
		var:qing_trib1_sender = { is_subject_of = ROOT }
		exists = var:qing_trib1_envoy
	}

	immediate = {
		# [BT-7 logfix] re-save the persisted vars into the named scopes the desc/portraits
		# and options read (scope:trib_sender / scope:trib_envoy) BEFORE render.
		var:qing_trib1_sender = { save_scope_as = trib_sender }
		var:qing_trib1_envoy  = { save_scope_as = trib_envoy }
		LOG_line = { sys = QING  msg = "EVENT qing_tribute.1 (a tribute embassy arrives at Peking) [ROOT.GetTag]" }
	}

	# --- Receive it in full ceremony (the lavish audience, largesse returned) ----
	option = {
		name = "qing_tribute.1.a"
		custom_tooltip = "qing_tribute.1.a.tt"
		trigger = { treasury >= 20 }
		# the return-gift ceremony costs something, but the tribute-trade profit + the
		# affirmation of the order more than repay it.
		add_treasury = -20
		QING_tribute_receive = { sender = scope:trib_sender }
		# a full audience firms the order (receive is prestige-neutral now, so this is
		# the whole prestige gain — #322 review-fix).
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 7 }
		current_ruler = { add_popularity = 8 }
		add_stability = 1
		# [#322 review-fix] the envoy's mission is done — dismiss him home so he does not
		# accumulate in the Qing court over the game.
		QING_tribute_dismiss_envoy = yes
	}

	# --- Receive the tribute, but dismiss the mission perfunctorily ---------------
	option = {
		name = "qing_tribute.1.b"
		custom_tooltip = "qing_tribute.1.b.tt"
		# still take the tribute-trade profit, but the curt reception strains the order
		# (receive is prestige-neutral now, so this -3 is a real decline — #322 review-fix).
		QING_tribute_receive = { sender = scope:trib_sender }
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -3 }
		add_political_influence = 5
		QING_tribute_dismiss_envoy = yes
	}
}

########################################################################
# qing_tribute_events.txt — THE TRIBUTE EMBASSY (朝貢), [#322 / overnight-build-#2].
#
# The positive rhythm of the tributary order at its zenith: the periodic embassies
# the vassal states sent to Peking bearing tribute — the Korean 燕行使, the Vietnamese,
# Ryukyu, Nepalese and Taiwan-aborigine missions — each a real ENVOY the throne
# receives. Driven by se_QING_TRIBUTE.txt's shared quarterly scheduler off the ACTUAL
# tributary subjects (KOR / VIE / NEP / RYU / TNN). CHI-only, player-only.
#
#   ROOT / this = the Qing overlord (CHI, player)
#   scope:trib_sender = the tributary COUNTRY that sent the mission
#   scope:trib_envoy  = the envoy CHARACTER minted for it
#
# Namespace = qing_tribute
#   .1  a tribute embassy arrives at Peking (audience-with-gifts vs perfunctory-dismiss)
#   .2  a tributary's new ruler awaits investiture (冊封)  [#322 deferred]
#   .3  a restless tributary tests the bond (breakaway crisis, low prestige)  [#322 deferred]
#   .4  a great power meddles in a Qing tributary (以夷制夷)  [#322 deferred]
#
# Sources: Fairbank (ed.), The Chinese World Order (1968); Hamashita, China, East
#   Asia and the Global Economy (2008); Larsen, Tradition, Treaties, and Trade (2008).
########################################################################

namespace = qing_tribute

########################################################################
# .1  A TRIBUTE EMBASSY ARRIVES  (朝貢使節入覲)
# The mission reaches the capital. The throne may receive it in full ceremony — a
# lavish audience that affirms the whole tributary order and returns the customary
# imperial largesse — or dismiss it perfunctorily, taking the tribute but sparing the
# expense and the ritual, at a cost to the order's prestige.
########################################################################
qing_tribute.1 = {
	type = country_event
	title = "qing_tribute.1.t"
	desc = "qing_tribute.1.desc"
	picture = council_country

	left_portrait = root.current_ruler
	right_portrait = scope:trib_envoy

	is_triggered_only = yes

	# self-guard: only fire if the mission's scopes survived (a subject freed between
	# dispatch and arrival, or the envoy died, cancels it cleanly).
	trigger = {
		tag = CHI
		exists = scope:trib_sender
		exists = scope:trib_envoy
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_tribute.1 (a tribute embassy arrives at Peking) [ROOT.GetTag]" }
	}

	# --- Receive it in full ceremony (the lavish audience, largesse returned) ----
	option = {
		name = "qing_tribute.1.a"
		custom_tooltip = "qing_tribute.1.a.tt"
		trigger = { treasury >= 20 }
		# the return-gift ceremony costs something, but the tribute-trade profit + the
		# affirmation of the order more than repay it.
		add_treasury = -20
		QING_tribute_receive = { sender = scope:trib_sender }
		# a full audience firms the order (receive is prestige-neutral now, so this is
		# the whole prestige gain — #322 review-fix).
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 7 }
		current_ruler = { add_prestige = 8 }
		add_stability = 1
		# [#322 review-fix] the envoy's mission is done — dismiss him home so he does not
		# accumulate in the Qing court over the game.
		QING_tribute_dismiss_envoy = yes
	}

	# --- Receive the tribute, but dismiss the mission perfunctorily ---------------
	option = {
		name = "qing_tribute.1.b"
		custom_tooltip = "qing_tribute.1.b.tt"
		# still take the tribute-trade profit, but the curt reception strains the order
		# (receive is prestige-neutral now, so this -3 is a real decline — #322 review-fix).
		QING_tribute_receive = { sender = scope:trib_sender }
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -3 }
		add_political_influence = 5
		QING_tribute_dismiss_envoy = yes
	}
}

########################################################################
# .2  A TRIBUTARY'S NEW RULER AWAITS INVESTITURE  (冊封)
# [#322 deferred] Fired from QING_tribute_invest_new_ruler off the tributary-side
# on_ruler_change: a Qing tributary's throne has passed to a new sovereign, who awaits
# the imperial patent + seal that renewed the tributary bond each reign. The throne may
# invest him promptly (affirms the order), sell the recognition dear (a price for the
# patent), or withhold it (a snub that strains the tie — and invites a rival's meddling).
########################################################################
qing_tribute.2 = {
	type = country_event
	title = "qing_tribute.2.t"
	desc = "qing_tribute.2.desc"
	picture = council_country

	left_portrait = root.current_ruler
	right_portrait = scope:trib_invest_vassal.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		exists = scope:trib_invest_vassal
		scope:trib_invest_vassal = { is_subject_of = ROOT }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_tribute.2 (a tributary's new ruler awaits investiture) [ROOT.GetTag]" }
	}

	# --- Invest him promptly — confer the patent and seal (affirm the order) ---------
	option = {
		name = "qing_tribute.2.a"
		custom_tooltip = "qing_tribute.2.a.tt"
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 6 }
		current_ruler = { add_prestige = 5 }
		add_stability = 1
		# the confirmed vassal is bound the tighter to Peking (subject-facing loyalty).
		scope:trib_invest_vassal = { add_country_modifier = { name = qing_tributary_invested  duration = 3650 } }
	}

	# --- Confer it, but extract a price for the recognition --------------------------
	option = {
		name = "qing_tribute.2.b"
		custom_tooltip = "qing_tribute.2.b.tt"
		# a lesser affirmation, but a fee for the patent (the recognition sold dear).
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 2 }
		if = {
			limit = { scope:trib_invest_vassal = { exists = current_ruler } }
			QING_tribute_receive = { sender = scope:trib_invest_vassal }
		}
	}

	# --- Withhold recognition — let him prove his submission first (a snub) ----------
	option = {
		name = "qing_tribute.2.c"
		custom_tooltip = "qing_tribute.2.c.tt"
		# the unrecognised sovereign is slighted; the tie strains and reform pressure ticks.
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -5 }
		add_political_influence = 10
	}
}

########################################################################
# .3  A RESTLESS TRIBUTARY TESTS THE BOND  (breakaway crisis)
# [#322 deferred] Fired from QING_tribute_breakaway_check when suzerain prestige has
# fallen low: a tributary, sensing the centre weaken, hangs back from its obligations
# and tests whether the bond still holds. The throne must overawe it (a costly show of
# strength), buy its loyalty back, or let it drift toward independence.
########################################################################
qing_tribute.3 = {
	type = country_event
	title = "qing_tribute.3.t"
	desc = "qing_tribute.3.desc"
	picture = senate_debate

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		exists = scope:trib_restless
		scope:trib_restless = { is_subject_of = ROOT }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_tribute.3 (a restless tributary tests the bond) [ROOT.GetTag]" }
	}

	# --- Overawe it — a show of strength on the marches (reassert suzerainty) ---------
	option = {
		name = "qing_tribute.3.a"
		custom_tooltip = "qing_tribute.3.a.tt"
		trigger = { treasury >= 60 }
		add_treasury = -60
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 12 }
		add_stability = 1
		scope:trib_restless = { add_country_modifier = { name = qing_tributary_invested  duration = 3650 } }
	}

	# --- Buy its loyalty back — largesse and remitted tribute -------------------------
	option = {
		name = "qing_tribute.3.b"
		custom_tooltip = "qing_tribute.3.b.tt"
		trigger = { treasury >= 30 }
		add_treasury = -30
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 5 }
		scope:trib_restless = { add_country_modifier = { name = qing_tributary_invested  duration = 3650 } }
	}

	# --- Let it drift — the empire cannot answer every insolence (the order frays) ----
	option = {
		name = "qing_tribute.3.c"
		custom_tooltip = "qing_tribute.3.c.tt"
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -8 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 4 }
	}
}

########################################################################
# .4  A GREAT POWER MEDDLES IN A QING TRIBUTARY  (以夷制夷)
# [#322 deferred] Fired from QING_tribute_gp_interference_check when a great power's
# China tension has risen: the meddling power (named in qing_tribute_meddler) presses
# into a Qing tributary's affairs — the historical arc of Western/Japanese encroachment
# on the Sinocentric order (France in Vietnam, Britain on the Burmese/Nepalese marches).
# The throne answers with a firm warning (raises tension with that power but holds the
# tributary), a diplomatic bargain (eases tension, cedes influence), or acquiescence.
########################################################################
qing_tribute.4 = {
	type = country_event
	title = "qing_tribute.4.t"
	desc = "qing_tribute.4.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		exists = scope:trib_meddled
		scope:trib_meddled = { is_subject_of = ROOT }
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_tribute.4 (a great power meddles in a Qing tributary) [ROOT.GetTag]" }
	}

	# --- Warn the meddling power off — reassert the tributary is ours -----------------
	option = {
		name = "qing_tribute.4.a"
		custom_tooltip = "qing_tribute.4.a.tt"
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 6 }
		# the warning raises tension with the named meddler (以夷制夷 pushback).
		if = { limit = { var:qing_tribute_meddler = flag:britain } QING_DECLINE_nudge = { var = qing_gp_tension_britain  amount = 8 } }
		else_if = { limit = { var:qing_tribute_meddler = flag:france } QING_DECLINE_nudge = { var = qing_gp_tension_france  amount = 8 } }
		else = { QING_DECLINE_nudge = { var = qing_gp_tension_russia  amount = 8 } }
		scope:trib_meddled = { add_country_modifier = { name = qing_tributary_invested  duration = 3650 } }
	}

	# --- Strike a bargain — cede a measure of influence to ease the tension -----------
	option = {
		name = "qing_tribute.4.b"
		custom_tooltip = "qing_tribute.4.b.tt"
		# a face-saving accommodation: tension with the meddler eases, but the order slips.
		if = { limit = { var:qing_tribute_meddler = flag:britain } QING_DECLINE_nudge = { var = qing_gp_tension_britain  amount = -6 } }
		else_if = { limit = { var:qing_tribute_meddler = flag:france } QING_DECLINE_nudge = { var = qing_gp_tension_france  amount = -6 } }
		else = { QING_DECLINE_nudge = { var = qing_gp_tension_russia  amount = -6 } }
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -4 }
		add_political_influence = 10
	}

	# --- Acquiesce — the empire cannot contest the barbarians here (the order frays) --
	option = {
		name = "qing_tribute.4.c"
		custom_tooltip = "qing_tribute.4.c.tt"
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -10 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 6 }
	}
}

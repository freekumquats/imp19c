########################################################################
# qing_frontier_events.txt — [#346] the WAR MINISTRY ↔ LIFAN YUAN turf war.
#
# A Qing (Beijing-commanded) garrison stationed on an Inner-Asian dependency's soil is
# a direct military occupation that steps on the Lifan Yuan's indirect-rule remit (理藩院
# governs through its resident Amban 駐防大臣, not by the 兵部's bayonets). When both the
# Grand Minister of War (兵部) and the Grand Director of the Lifan Yuan (理藩院) sit, and a
# resident Amban shares the ground with the garrison, the two courts clash over who
# governs the frontier. Queued by QING_fgar_apply_occupation (se_QING_FRONTIER.txt).
#
# Fires in CHI scope with:
#   scope:qing_frontier_war_min    — the Grand Minister of War (backs the garrison)
#   scope:qing_frontier_lifan_dir  — the Grand Director of the Lifan Yuan (backs the Amban)
#   scope:qing_frontier_amban      — the resident Amban on the contested ground
#   scope:qing_frontier_subject    — the occupied dependency
#
# Namespace: qing_frontier. country_event, is_triggered_only = yes. se_LOG sys = QING.
########################################################################

namespace = qing_frontier

# ======================================================================
# qing_frontier.1 — the turf war. The throne chooses whose hand governs the frontier:
# the army (兵部, direct rule — integration presses on, the Lifan Yuan is affronted) or
# the resident (理藩院, indirect rule — the dependency is soothed, the War Ministry
# chafes), or a division of authority (both mollified a little, neither satisfied).
# ======================================================================
qing_frontier.1 = {
	type = country_event
	title = "qing_frontier.1.t"
	desc = "qing_frontier.1.desc"
	picture = chinese_throne_room   # a court/deliberation scene
	left_portrait = scope:qing_frontier_war_min
	right_portrait = scope:qing_frontier_lifan_dir
	is_triggered_only = yes

	trigger = {
		tag = CHI
		exists = scope:qing_frontier_war_min
		exists = scope:qing_frontier_lifan_dir
		exists = scope:qing_frontier_subject
	}

	immediate = {
		LOG_line = { sys = QING  msg = "EVENT qing_frontier.1 (War Ministry vs Lifan Yuan turf war) firing" }
	}

	# OPTION A: govern by the ARMY (兵部) — direct rule. Integration presses on; the War
	# Minister is gratified, the Lifan Director affronted (his indirect-rule remit shrinks).
	option = {
		name = "qing_frontier.1.a"
		scope:qing_frontier_subject = { save_scope_as = target }
		SUBJ_QING_advance_integration = { steps = 1 }
		scope:qing_frontier_war_min = { add_loyalty = 5 }
		scope:qing_frontier_lifan_dir = { add_loyalty = -6 }
		LOG_line = { sys = QING  msg = "turf war resolved: direct military rule (兵部), Lifan Yuan affronted" }
	}

	# OPTION B: govern by the RESIDENT (理藩院) — indirect rule. The dependency is soothed
	# (occupation modifier lifted, opinion recovers); the Lifan Director is gratified, the
	# War Minister chafes at the curb on his garrison.
	option = {
		name = "qing_frontier.1.b"
		scope:qing_frontier_subject = {
			if = {
				limit = { has_country_modifier = qing_frontier_occupied }
				remove_country_modifier = qing_frontier_occupied
			}
			add_opinion = { modifier = qing_amban_cooperation_opinion  target = ROOT }
		}
		scope:qing_frontier_lifan_dir = { add_loyalty = 5 }
		scope:qing_frontier_war_min = { add_loyalty = -6 }
		LOG_line = { sys = QING  msg = "turf war resolved: indirect rule by resident (理藩院), garrison reined in" }
	}

	# OPTION C: divide authority — a joint civil-military condominium. Neither court is
	# satisfied (both lose a little standing) but the frontier is not inflamed; no change
	# to integration or the occupation modifier.
	option = {
		name = "qing_frontier.1.c"
		scope:qing_frontier_war_min = { add_loyalty = -2 }
		scope:qing_frontier_lifan_dir = { add_loyalty = -2 }
		LOG_line = { sys = QING  msg = "turf war resolved: divided civil-military authority, neither court satisfied" }
	}
}

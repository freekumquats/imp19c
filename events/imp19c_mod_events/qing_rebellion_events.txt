########################################################################
# qing_rebellion_events.txt — the ANACHRONISTIC REBELLION ROSTER (task #14).
#
# The great risings of the long Qing crisis, offered to the player keyed to
# MECHANIC CONTEXT rather than to their historical dates (the same anachronistic
# principle as the character roster, qing_roster_events.txt). A realm rotten with
# corruption and sect pressure may face the White Lotus in 1818 or the Taiping in
# 1830; a realm straining its frontier minorities may face the Panthay or Dungan
# risings whenever the ethnic tension boils over.
#
# The research digest (Kuhn, Naquin, Spence, Atwill, Hodong Kim, Perry, Esherick,
# Cambridge History of China v.10-11) sorts the risings into FOUR thematic
# flavours, each modelled by its own rebellion modifier (se_QING_REFORM.txt /
# qing_mechanics_modifiers.txt):
#   • sectarian / millenarian      White Lotus 白蓮教, Eight Trigrams 天理教
#   • theocratic (Taiping)         太平天國 — an existential rival kingdom
#   • ethnic / communal frontier   Panthay 回變, Dungan 陝甘回變, Miao 苗變
#   • restorationist secret-society Red Turban 紅巾/天地會, Nian 捻軍
#   • (late outliers)              Boxers 義和團 (pro-dynastic), Xinhai 辛亥 (republican)
#
# The Taiping rising is the seed of the Christian-theocracy end-state: its "come
# to terms" option feeds qing_taiping_overture / qing_taiping_progress (task #14,
# qing_reform.33). The others are decline pressure the player must weather.
#
# CHI-only, player-only (fired from QING_frontier_flavour_roll, itself gated
# tag = CHI + is_ai = no). All logged sys = QING. Each rising self-guards once via
# a spawned flag so it is a rare, memorable crisis, not churn.
#
#   ROOT / this = the Qing overlord (CHI, player)
#
# Namespace = qing_rebellion
#   .1  白蓮教   White Lotus (sectarian)        gate: corruption + sect
#   .2  天理教   Eight Trigrams (sectarian; palace strike)  gate: sect + corruption
#   .3  太平天國 Taiping (theocratic)            gate: sect high (+ Hong spawned)
#   .4  捻軍     Nian (restorationist/banditry)  gate: sect + banner decay
#   .5  回變     Panthay — Yunnan Hui (ethnic)   gate: ethnic tension
#   .6  陝甘回變 Dungan + Yaqub Beg (ethnic)     gate: ethnic tension high
#   .7  紅巾     Red Turban — Triads (restorationist)  gate: corruption + sect
#   .8  苗變     Guizhou Miao (ethnic)           gate: ethnic tension
#   .9  義和團   Boxers (pro-dynastic anti-foreign)  gate: currency/foreign stress
#   .10 辛亥     Xinhai / Wuchang (republican)   gate: reform pressure very high
########################################################################

namespace = qing_rebellion

########################################################################
# .1  白蓮教起義 — THE WHITE LOTUS (sectarian / millenarian)
# 1794-1804 in the Hubei-Sichuan-Shaanxi highlands. Wang Cong'er 王聰兒; Eternal
# Venerable Mother 無生老母, coming Maitreya 彌勒. The rising that ended the High
# Qing and exposed the rot of the banners. (Kuhn, Rebellion and Its Enemies.)
########################################################################
qing_rebellion.1 = {
	type = country_event
	title = "qing_rebellion.1.t"
	desc = "qing_rebellion.1.desc"
	picture = revolt

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		set_variable = qing_rebel_white_lotus
		LOG_line = { sys = QING  msg = "EVENT qing_rebellion.1 (White Lotus 白蓮教) [ROOT.GetTag]" }
		QING_rebellion_begin = { flavour = sectarian  sect = 8 }
	}

	# --- 堅壁清野 fortify hamlets and clear the countryside (the historical win)
	option = {
		name = "qing_rebellion.1.a"
		custom_tooltip = "qing_rebellion.1.a.tt"
		trigger = { treasury >= 80 }
		add_treasury = -80
		QING_rebellion_end = { flavour = sectarian }
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -12 }
		add_stability = 1
	}

	# --- Raise gentry militia (鄉勇) — cheaper, but the banners are shamed ---
	option = {
		name = "qing_rebellion.1.b"
		custom_tooltip = "qing_rebellion.1.b.tt"
		add_manpower = 3
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -6 }
		QING_DECLINE_nudge = { var = qing_banner_decay  amount = 6 }
	}

	# --- Rely on the hereditary armies (they are not what they were) --------
	option = {
		name = "qing_rebellion.1.c"
		custom_tooltip = "qing_rebellion.1.c.tt"
		add_war_exhaustion = 2
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 4 }
	}
}

# .3  太平天國 — THE TAIPING (Christian-inflected revolutionary theocracy)
# 1850-1864. Hong Xiuquan 洪秀全, God-Worshipping Society 拜上帝會; capital fixed at
# Nanjing as the Heavenly Capital 天京. Broken not by the banners but by the Han
# gentry armies of Zeng Guofan 曾國藩 and Li Hongzhang 李鴻章. Deadliest civil war
# in history. This rising is the fork to the Christian-theocracy end-state.
# (Spence, God's Chinese Son; Cambridge History v.10.)
########################################################################
qing_rebellion.3 = {
	type = country_event
	title = "qing_rebellion.3.t"
	desc = "qing_rebellion.3.desc"
	picture = revolt

	left_portrait = root.current_ruler

	is_triggered_only = yes

	immediate = {
		set_variable = qing_rebel_taiping_risen
		LOG_line = { sys = QING  msg = "EVENT qing_rebellion.3 (Taiping 太平天國) [ROOT.GetTag]" }
		QING_rebellion_begin = { flavour = taiping  sect = 10 }
		QING_taiping_init = yes
	}

	# --- Loose the Han gentry armies (曾國藩/李鴻章 — the historical course) ---
	option = {
		name = "qing_rebellion.3.a"
		custom_tooltip = "qing_rebellion.3.a.tt"
		trigger = { treasury >= 120 }
		add_treasury = -120
		QING_rebellion_end = { flavour = taiping }
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -15 }
		# the provincial armies that save the dynasty also empower the Han elite
		QING_DECLINE_nudge = { var = qing_banner_decay  amount = 8 }
		add_stability = 1
	}

	# --- Take Western help against the God Worshippers (常勝軍) --------------
	option = {
		name = "qing_rebellion.3.b"
		custom_tooltip = "qing_rebellion.3.b.tt"
		QING_rebellion_end = { flavour = taiping }
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -10 }
		QING_selfstr_advance = { amount = 6 }
	}

	# --- Come to terms with the Heavenly Kingdom (the Christian road) -------
	# Seeds / deepens the theocracy path — the unthinkable accommodation.
	option = {
		name = "qing_rebellion.3.c"
		custom_tooltip = "qing_rebellion.3.c.tt"
		set_variable = qing_taiping_overture
		QING_taiping_advance = { amount = 25 }
		QING_DECLINE_nudge = { var = qing_sect_pressure  amount = -6 }
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 8 }
	}
}

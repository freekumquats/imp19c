# qing_xinjiang_events.txt — THE XINJIANG CONSOLIDATION ARC (新疆善後), #367.
# The player-facing surface of the consolidation engine (se_QING_XINJIANG.txt): the
# khoja revolt-scare that a neglected far west invites. CHI-only.
#
# This is the EARLY-consolidation counterpart to the late Ili-crisis set-piece
# (qing_ili_events.txt): where that chain resolves a full-blown rising, this scare is the
# WARNING the consolidation levers exist to prevent. The White-Mountain khojas (backed
# from Kokand, KOK — the neighbouring Turkic power, per the separatism-backer rule) stir
# when the grip is low AND the beg corps is empty or disloyal. Queued by QING_xj_pulse.
#
# .1  THE KHOJA STIRS (張格爾蠢動) — suppress now (spend, grip up) or let it fester
#     (grip down, ethnic tension up, a beg turns, Kokand emboldened).
#
# Sources: se_QING_XINJIANG.txt header; Kim 2004 (Holy War in China); Newby 2005.
########################################################################

namespace = qing_xinjiang

# ======================================================================
# .1  THE KHOJA STIRS (張格爾蠢動). A White-Mountain pretender, sheltered in Kokand, tests
# the New Dominion's undermanned oases. The throne may pour silver and troops into a swift
# suppression (a real cost now, the grip firmed) or trust the frontier to hold itself —
# and risk the rising taking root: the grip slips, the borderlands seethe, a wavering beg
# goes over, and Kokand reads Peking's inaction as weakness. Queued by the pulse; re-checks
# its own state (the pending flag is cleared here so a later scare can queue).
# ======================================================================
qing_xinjiang.1 = {
	type = country_event
	title = "qing_xinjiang.1.t"
	desc = "qing_xinjiang.1.desc"
	picture = revolt

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		has_variable = qing_xinjiang_control
		var:qing_xinjiang_control <= 30
	}

	immediate = {
		# clear the pending flag so the pulse can queue a fresh scare later.
		if = { limit = { has_variable = qing_xj_khoja_pending }  remove_variable = qing_xj_khoja_pending }
		LOG_line = { sys = QING  msg = "EVENT qing_xinjiang.1 (khoja stirs) fired for" }
	}

	# --- SUPPRESS NOW (進剿) — pour silver and the Ili banner garrison into a swift strike.
	# A real fiscal cost, but the pretender is scattered and the grip firms. Gated on the
	# treasury (the historical suppressions were expensive).
	option = {
		name = "qing_xinjiang.1.suppress"
		custom_tooltip = "qing_xinjiang.1.suppress.tt"
		trigger = { treasury >= 50 }
		add_treasury = -50
		QING_DECLINE_nudge = { var = qing_xinjiang_control  amount = 12 }
		# a firm hand steadies the frontier's loyalty — recompute the score at once.
		QING_xj_recompute_consolidation = yes
		LOG_line = { sys = QING  msg = "xinjiang: the khoja scare is suppressed for" }
	}

	# --- LET IT FESTER (姑息) — trust the far oases to hold themselves. The rising takes
	# root: the grip slips, the borderlands seethe (ethnic tension), a wavering beg goes
	# over to the khoja, and Kokand — reading inaction as weakness — sours on Peking.
	option = {
		name = "qing_xinjiang.1.fester"
		custom_tooltip = "qing_xinjiang.1.fester.tt"
		QING_DECLINE_nudge = { var = qing_xinjiang_control  amount = -10 }
		QING_DECLINE_nudge = { var = qing_ethnic_tension  amount = 6 }
		# a beg turns (marked venal — a discipline target the panel can later cashier).
		QING_xj_turn_a_beg = yes
		# Kokand (KOK), the khoja's shelter and the neighbouring Turkic power, is emboldened
		# — its opinion of Peking sours (the separatism backer is the NEIGHBOUR, never a
		# far-flung power, per the standing rule). Guarded on KOK existing.
		if = {
			limit = { exists = c:KOK }
			c:KOK = { add_opinion = { modifier = qing_xj_kokand_emboldened  target = ROOT } }
		}
		QING_xj_recompute_consolidation = yes
		LOG_line = { sys = QING  msg = "xinjiang: the khoja scare is left to fester for" }
	}
}

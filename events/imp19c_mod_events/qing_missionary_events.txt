########################################################################
# qing_missionary_events.txt — MISSIONARY INCIDENTS (教案).
# The player-facing surface of the missionary engine (se_QING_MISSIONARY.txt): the
# anti-Christian incidents that force the throne to choose between the foreign
# consuls and the outraged gentry — the grievance loop that culminates in the Boxer
# rising (qing_rebellion.9, escalated to by the pulse). CHI-only, fired from the
# flavour roll while the anti-Christian sentiment simmers.
#
# .1  a missionary incident (choose an aggrieved power at fire-time; suppress or indulge)
#
# Sources: se_QING_MISSIONARY.txt header (Cohen 1963; Esherick 1987).
########################################################################

namespace = qing_missionary

# ======================================================================
# .1  A MISSIONARY INCIDENT (教案) — a dispute over converts, mission property, or a
# child-killing rumour flares into riot. Fired by the flavour roll once anti-Christian
# sentiment is simmering. The aggrieved power is chosen at fire-time: France (the
# Catholic protector) if it exists, else Britain, else Russia — set into a variable by
# the immediate block so the options can pass it to the suppress/indulge effects.
# ======================================================================
qing_missionary.1 = {
	type = country_event
	title = "qing_missionary.1.t"
	desc = "qing_missionary.1.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		OR = { exists = c:FRA  exists = c:GBR  exists = c:RUS }
	}

	immediate = {
		QING_missionary_init = yes
		# choose the aggrieved power: France (Catholic protector) first, then Britain,
		# then Russia — whichever is on the map.
		if = { limit = { exists = c:FRA }  set_variable = { name = qing_missionary_power  value = flag:france } }
		else_if = { limit = { exists = c:GBR }  set_variable = { name = qing_missionary_power  value = flag:britain } }
		else = { set_variable = { name = qing_missionary_power  value = flag:russia } }
		LOG_line = { sys = QING  msg = "EVENT qing_missionary.1 (教案 incident) [ROOT.GetTag]" }
	}

	# --- Suppress the incident — satisfy the consuls (indemnity + punishment) ------
	option = {
		name = "qing_missionary.1.suppress"
		custom_tooltip = "qing_missionary.1.suppress.tt"
		if = { limit = { var:qing_missionary_power = flag:france }  QING_missionary_suppress = { power = france } }
		else_if = { limit = { var:qing_missionary_power = flag:britain }  QING_missionary_suppress = { power = britain } }
		else = { QING_missionary_suppress = { power = russia } }
	}

	# --- Indulge the fury — let the magistrates look away --------------------------
	option = {
		name = "qing_missionary.1.indulge"
		custom_tooltip = "qing_missionary.1.indulge.tt"
		if = { limit = { var:qing_missionary_power = flag:france }  QING_missionary_indulge = { power = france } }
		else_if = { limit = { var:qing_missionary_power = flag:britain }  QING_missionary_indulge = { power = britain } }
		else = { QING_missionary_indulge = { power = russia } }
	}
}

# ======================================================================
# [#309] .2  A LOCAL HETERODOXY SCARE (地方教案, PRE-TREATY). A domestic incident with NO
# foreign power involved: gentry in a province where Christian communities have grown press
# the magistrates to act against the "evil sect" (邪教). This is the quiet pre-treaty face of
# the missions — the throne could tolerate or suppress at will, with zero international
# consequence (contrast the treaty-era qing_missionary.1, where a consul stands behind the
# mission). Fires only pre-treaty, from the flavour roll, once social friction has built.
# ======================================================================
qing_missionary.2 = {
	type = country_event
	title = "qing_missionary.2.t"
	desc = "qing_missionary.2.desc"
	picture = council_country

	left_portrait = root.current_ruler

	is_triggered_only = yes

	trigger = {
		tag = CHI
		NOT = { has_variable = qing_treaty_system_imposed }
		has_variable = qing_mission_social_friction
		var:qing_mission_social_friction >= 25
	}

	immediate = {
		QING_missionary_init = yes
		LOG_line = { sys = QING  msg = "EVENT qing_missionary.2 (pre-treaty local heterodoxy scare) [ROOT.GetTag]" }
	}

	# --- Tolerate — leave the quiet communities be (the pragmatic High-Qing norm) --------
	option = {
		name = "qing_missionary.2.tolerate"
		custom_tooltip = "qing_missionary.2.tolerate.tt"
		QING_missionary_tolerate_local = yes
	}

	# --- Suppress — enforce the proscription (expel the priests, scatter the flock) ------
	option = {
		name = "qing_missionary.2.suppress"
		custom_tooltip = "qing_missionary.2.suppress.tt"
		QING_missionary_suppress_local = yes
	}
}

# [#62 2026-07-23] NON-INTEGRATED culture decisions (Design 4 — Culture → Nationalism / citizenship).
# The "grant citizenship" surface: decisions available on an OUT-GROUP (non-integrated) culture present
# in the realm, moving it toward the national in-group. Uses ONLY proven verbs:
#   • integrate_country_culture = <culture>  — the real engine citizenship grant (already used in
#     se_QING_DECLINE.txt:640; TI form_miao.txt); flips is_integrated, admitting the culture to the citizen tier.
#   • add_country_culture_modifier — the graduated-rights layer (the TI/Invictus culture-decision pattern).
# NOTE: change_pop_type_right is a GUI-only engine action (oracle-verified: NO scriptable effect exists in
# vanilla / TI / Invictus), so citizenship is expressed through integrate_country_culture + culture modifiers,
# NOT that effect. scope:target_culture = the country_culture the decision is opened on (culture_view.gui provides
# it); scope:target_culture.culture = the underlying culture (the .culture accessor is used the same way in
# imp19c_general_culture_decisions.txt).

culture_decisions = {

	# ---- GRANT CITIZENSHIP (歸化) — naturalise an out-group culture into the national community ----
	qing_grant_citizenship = {
		potential = {
			trigger_if = {
				limit = { exists = scope:target_culture }
				# only meaningful for a culture that is NOT our primary and is NOT yet integrated.
				NOT = { primary_culture = scope:target_culture.culture }
				scope:target_culture = { is_integrated = no }
			}
		}
		allow = {
			can_pay_price = major_cultural_decision_price
			# not already naturalised by this decision.
			scope:target_culture = { NOT = { has_country_culture_modifier = qing_naturalised_citizenship } }
		}
		effect = {
			pay_price = major_cultural_decision_price
			# the real engine citizenship grant — admits the culture to the integrated (in-group) tier.
			integrate_country_culture = scope:target_culture.culture
			# the standing naturalised marker + dividend.
			scope:target_culture = {
				add_country_culture_modifier = {
					name = "qing_naturalised_citizenship"
					duration = -1
				}
			}
			# admitting an out-group people is a concession that unsettles the old order briefly.
			add_stability = -3
		}
		ai_will_do = {
			factor = 0
		}
	}

	# ---- EXTEND LOCAL RIGHTS (優容) — a lesser step short of full citizenship: ease an out-group ----
	# culture's lot with the vanilla rights_increased modifier (a happiness lift), without integrating it.
	qing_extend_local_rights = {
		potential = {
			trigger_if = {
				limit = { exists = scope:target_culture }
				NOT = { primary_culture = scope:target_culture.culture }
				scope:target_culture = { is_integrated = no }
			}
		}
		allow = {
			can_pay_price = minor_cultural_decision_price
			scope:target_culture = { NOT = { has_country_culture_modifier = rights_increased } }
		}
		effect = {
			pay_price = minor_cultural_decision_price
			scope:target_culture = {
				add_country_culture_modifier = {
					name = "rights_increased"
					duration = -1
				}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
}

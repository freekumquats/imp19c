# [#62 2026-07-23] PRIMARY-CULTURE decisions (Design 4 — Culture → Nationalism).
# The nationalism concept made a player action: PROCLAIM THE NATION — a primary-culture decision that,
# once the realm has cohered (a recognised nationalism + a high civic-identity meter), formally awakens
# the modern national identity and grants the standing national-awakening country modifier.
# Gate idiom (primary_culture = scope:target_culture.culture) matches TI/Invictus primary_culture_decisions.
# The recognised-nationalism test uses the has_recognised_nationalism_country_trigger from
# 00_nationalism_groups.txt; qing_civic_identity is the existing National Integration meter (se_QING_DECLINE).

culture_decisions = {

	# ---- PROCLAIM THE NATION (立國族) — awaken the modern national identity ----
	qing_proclaim_nation = {
		potential = {
			trigger_if = {
				limit = { exists = scope:target_culture }
				# only on the realm's OWN primary culture (the national core), and only for the Qing player.
				primary_culture = scope:target_culture.culture
				tag = CHI
			}
		}
		allow = {
			# the realm must actually read as a coherent nation and have cohered civically.
			custom_tooltip = {
				text = "qing_proclaim_nation_recognised_tt"
				has_recognised_nationalism_country_trigger = yes
			}
			custom_tooltip = {
				text = "qing_proclaim_nation_civic_tt"
				has_variable = qing_civic_identity
				# a literal RHS is legal (only a VAR on the RHS needs a named script_value); 50 = the
				# "emerging nation" band threshold in QING_DECLINE_apply_civic_band.
				var:qing_civic_identity >= 50
			}
			NOT = { has_country_modifier = qing_national_awakening }
			can_pay_price = major_cultural_decision_price
		}
		effect = {
			pay_price = major_cultural_decision_price
			add_country_modifier = {
				name = "qing_national_awakening"
				duration = -1
			}
			add_legitimacy = 10
		}
		ai_will_do = {
			factor = 0
		}
	}
}

# [#62 2026-07-23] INTEGRATED culture decisions (Design 4 — Culture → Nationalism / citizenship).
# The "revoke citizenship" surface: a decision on an INTEGRATED, non-primary culture that strips the
# naturalised-citizenship standing and imposes the vanilla integration_status_removed penalty.
#
# ENGINE LIMIT (oracle-verified): there is NO unintegrate_country_culture effect and change_pop_type_right
# is GUI-only, so a script cannot forcibly flip is_integrated back to FALSE. This decision therefore models
# revocation as a RIGHTS revocation: it removes the qing_naturalised_citizenship dividend and applies the
# vanilla integration_status_removed / rights_revoked culture modifiers (the happiness/output penalty the
# on_culture_reduced_right reaction already uses). The culture may remain engine-"integrated" until the
# integration system itself changes it, but its standing and dividend are stripped — the in-group privileges
# are withdrawn, which is the player-facing meaning of revocation. Documented so it isn't mistaken for a bug.
# scope:target_culture = the country_culture the decision is opened on.

culture_decisions = {

	# ---- REVOKE CITIZENSHIP (削籍) — strip an integrated out-origin culture of its naturalised standing ----
	qing_revoke_citizenship = {
		potential = {
			trigger_if = {
				limit = { exists = scope:target_culture }
				# never applicable to our own primary culture (always in-group by definition).
				NOT = { primary_culture = scope:target_culture.culture }
				# only where the naturalised-citizenship standing was actually granted.
				scope:target_culture = { has_country_culture_modifier = qing_naturalised_citizenship }
			}
		}
		allow = {
			can_pay_price = minor_cultural_decision_price
		}
		effect = {
			pay_price = minor_cultural_decision_price
			scope:target_culture = {
				remove_country_culture_modifier = qing_naturalised_citizenship
				# the vanilla revocation penalty (the same modifier the on_culture_reduced_right reaction applies).
				add_country_culture_modifier = {
					name = "integration_status_removed"
					duration = 3650
				}
			}
			# withdrawing citizenship from a people already inside the national community is destabilising.
			add_stability = -8
		}
		ai_will_do = {
			factor = 0
		}
	}
}

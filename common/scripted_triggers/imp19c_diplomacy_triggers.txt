	# Scope: Province (diplomatic play)
	# Instigator is considering going to war
	var:war_assessment <= DIPLOMACY_play_war_willing
	var:war_assessment >= DIPLOMACY_play_war_considering
}

	# Scope: Province (diplomatic play)
	# Instigator is not willing to go to war

}

# ======================================================================
# QING_gp_can_partition — is the Qing strong enough to "split the difference"
# with a great power (an amicable partition) rather than merely provoke it?
# Measured by GDP and MILITARY STRENGTH, using the mod's own power model:
#   • DIPLOMACY_power_from_economy  — GDP / trade / currency weight
#   • DIPLOMACY_power_from_military — army manpower scaled by military tech
#
# HOME-FIELD ADVANTAGE: the Qing does NOT need to be a global peer of Britain,
# France or Russia to bargain with them in Asia. It fights on its own ground,
# at the end of their supply lines, while each European power can spare only a
# FRACTION of its strength for the Far East — the bulk of their armies stay in
# Europe watching one another (the very rivalry the 以夷制夷 layer exploits).
# So the bar is only a fraction of true great-power heft (cf. EE_scripted_guis.txt:
# a DIPLOMACY_power < 200 is a "minor"; a full great power clears ~200+ composite).
# A solid regional heavyweight — well short of parity — suffices. Scope: Country.
# ======================================================================
QING_gp_can_partition = {
	# a respectable economy — regional-heavyweight GDP, not global-peer GDP
	DIPLOMACY_power_from_economy >= 80
	# a credible army on home soil — enough to make partition cheaper than war
	DIPLOMACY_power_from_military >= 30
}

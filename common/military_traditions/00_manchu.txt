########################################################################
# 00_manchu.txt — THE TEN GREAT CAMPAIGNS (十全武功), the Manchu military
# tradition of the Qing. Qianlong styled himself 十全老人 — "the Old Man of
# the Ten Complete Victories" — after the ten frontier campaigns his reign
# proclaimed. This tree commemorates them AND grants their martial lessons as
# real bonuses (the user's brief: commemorative, but it applies buffs).
#
# Gated on the jurchen culture group (CHI = manchu, see 00_jurchen.txt) via the
# same began_with_tradition_group idiom the arabic/indian/japanese trees use.
#
# Three theatres + a capstone:
#   A 平定西域  Pacification of the Western Regions — the two Dzungar campaigns
#              (1755, 1757) and the Altishahr/Muslim-tribes campaign (1759).
#              Steppe cavalry, supply over distance, desert war.
#   B 大小金川  The Jinchuan Wars and the Island — the two Jinchuan suppressions
#              (1747-49, 1771-76) and Taiwan / Lin Shuangwen (1787-88).
#              Mountain siege, assault, pacification of revolt.
#   C 外藩征討  The Tributary Frontiers — Burma (1765-69), Vietnam (1788-89) and
#              the two Gurkha/Nepal campaigns (1788-89, 1790-92). Expeditionary
#              war in hostile terrain: attrition, reinforcement, endurance.
#   Capstone 十全老人  requires the three theatres' final campaigns.
#
# Icons are PLACEHOLDERS — every icon points at a .dds that already exists in
# gfx/interface/icons/military_traditions/ so no node renders blank (the user's
# brief). Swap for bespoke art later. Only modifier keys already proven valid in
# the mod's own populated trees (arabic/indian/japanese) are used.
#
# Sources: Qianlong's 御製十全記 (Record of the Ten Complete Victories); Perdue,
# China Marches West; the campaign roster in [[imp19c-qing-history-and-mechanics]].
########################################################################

manchu_shiquan = {
	color = hsv { 0.08 0.9 0.9 }
	image = arabic_philosophy_start_bonus   # placeholder start image (existing .dds)

	allow = {
		trigger_if = {
			limit = {
				country_culture_group = jurchen
			}
			custom_tooltip = {
				text = "began_with_tradition_group"
			}
			country_culture_group = jurchen
		}
	}

	## START ####################################################################
	# 十全老人 — the reign that proclaimed the ten victories. The opening bonus is
	# the manpower reserve of the Eight Banners and Green Standard that made the
	# campaigns possible.
	shiquan_start = {
		icon = indian_philosophy_start_bonus   # placeholder (existing .dds)
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 5 }
			}
		}
		modifier = {
			global_manpower_modifier = 0.05
		}
	}

	## PATH A — 平定西域  PACIFICATION OF THE WESTERN REGIONS ####################
	# The two Dzungar campaigns and the Altishahr campaign that annexed the whole
	# Tarim and Ili to the empire as the New Dominion (Xinjiang).

	# First Dzungar Campaign (1755) — the lightning march on Ili that broke the
	# Dzungar Khanate. Reward: steppe cavalry reach and supply over distance.
	shiquan_dzungar_1 = {
		icon = arabic_peninsula_path_1   # placeholder
		requires = { shiquan_start }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 5 }
			}
		}
		modifier = {
			light_cavalry_offensive = 0.1
			global_supply_limit_modifier = 0.1
		}
	}

	# Second Dzungar Campaign (1757) — the crushing of Amursana's revolt and the
	# final pacification of Dzungaria. Reward: cavalry discipline and endurance in
	# hostile country.
	shiquan_dzungar_2 = {
		icon = arabic_peninsula_path_5   # placeholder
		requires = { shiquan_dzungar_1 }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 6 }
			}
		}
		modifier = {
			light_cavalry_discipline = 0.1
			hostile_attrition = 1
		}
	}

	# Pacification of the Muslim Tribes / Altishahr (1759) — the reduction of the
	# Tarim oasis cities and the Khoja brothers. Reward: desert combat and the
	# manpower of the incorporated west.
	shiquan_altishahr = {
		icon = arabic_levantine_path_1   # placeholder
		requires = { shiquan_dzungar_2 }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 6 }
			}
		}
		modifier = {
			archers_desert_combat_bonus = 0.1
			global_manpower_modifier = 0.05
			# Path-A capstone reward. This is a FLAT tree (arabic/indian/japanese
			# pattern, no path-container level), so a separate `bonus = {}` block
			# would have no path to attach to and be silently dropped — the reward is
			# folded straight into this final node's modifier instead.
			land_morale_modifier = 0.05
		}
	}

	## PATH B — 大小金川  THE JINCHUAN WARS AND THE ISLAND #######################
	# The two brutal suppressions of the Gyalrong hill fortresses of Jinchuan and
	# the reconquest of rebel Taiwan — the empire's hardest siege warfare.

	# First Jinchuan Campaign (1747-49) — the first assault on the stone watch-
	# towers (碉) of the Jinchuan valleys. Reward: siege doctrine.
	shiquan_jinchuan_1 = {
		icon = arabic_levantine_path_4   # placeholder
		requires = { shiquan_start }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 5 }
			}
		}
		modifier = {
			siege_ability = 0.1
		}
	}

	# Second Jinchuan Campaign (1771-76) — the long, ruinous reduction of the
	# fortress-towers with massed cannon and sappers. Reward: assault and engineers.
	shiquan_jinchuan_2 = {
		icon = arabic_african_path_5   # placeholder
		requires = { shiquan_jinchuan_1 }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 6 }
			}
		}
		modifier = {
			assault_ability = 0.1
			siege_engineers = 0.1
		}
	}

	# Pacification of Taiwan — the Lin Shuangwen rising (1787-88), the largest of
	# the era's revolts, put down by an amphibious expedition. Reward: order at
	# home and blockade of the rebel coast.
	shiquan_taiwan = {
		icon = indian_northern_path_3   # placeholder
		requires = { shiquan_jinchuan_2 }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 6 }
			}
		}
		modifier = {
			global_unrest = -1
			blockade_efficiency = 0.1
			# Path-B capstone reward, folded in (see shiquan_altishahr on why a flat
			# tree cannot use a separate `bonus = {}` block).
			siege_ability = 0.05
		}
	}

	## PATH C — 外藩征討  THE TRIBUTARY FRONTIERS ################################
	# The expeditionary campaigns beyond the frontier — Burma, Vietnam and the two
	# marches over the Himalaya into Nepal. Wars won as much against terrain and
	# disease as against the enemy.

	# The Burma Campaign (1765-69) — four costly expeditions into the jungle
	# against the Konbaung dynasty, ending in a face-saving tributary peace.
	# Reward: resistance to the attrition that wrecked those armies.
	shiquan_burma = {
		icon = indian_southern_path_1   # placeholder
		requires = { shiquan_start }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 5 }
			}
		}
		modifier = {
			land_unit_attrition = -0.1
			hostile_attrition = 1
		}
	}

	# The Vietnam Campaign (1788-89) — the intervention to restore the Lê against
	# the Tây Sơn, driven back from Thăng Long. Reward: the reinforcement and
	# recovery a far expedition depends on.
	shiquan_vietnam = {
		icon = indian_southern_path_7   # placeholder
		requires = { shiquan_burma }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 6 }
			}
		}
		modifier = {
			cohort_reinforcement_speed = 0.1
		}
	}

	# The Two Gurkha Campaigns (1788-89, 1790-92) — the marches over the Himalaya
	# that drove the Gurkhas from Tibet to the gates of Kathmandu, the farthest the
	# banners ever reached. Reward: supply and morale at the end of the world.
	shiquan_gurkha = {
		icon = indian_western_path_2   # placeholder
		requires = { shiquan_vietnam }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 6 }
			}
		}
		modifier = {
			global_supply_limit_modifier = 0.1
			land_morale_modifier = 0.05
			# Path-C capstone reward, folded in (see shiquan_altishahr on why a flat
			# tree cannot use a separate `bonus = {}` block).
			war_exhaustion = -0.02
		}
	}

	## CAPSTONE — 十全老人  THE OLD MAN OF THE TEN COMPLETE VICTORIES ############
	# Requires the final campaign of each of the three theatres — the ten victories
	# complete. The summit commemoration: the martial prestige of the high Qing.
	shiquan_laoren = {
		icon = arabic_philosophy_start_bonus   # placeholder
		requires = { shiquan_altishahr shiquan_taiwan shiquan_gurkha }
		ai_will_do = {
			modifier = {
				trigger = { country_culture_group = jurchen }
				add = { value = 8 }
			}
		}
		modifier = {
			land_morale_modifier = 0.1
			global_manpower_modifier = 0.05
			ruler_popularity_gain = 0.1
		}
	}
}

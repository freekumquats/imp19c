# §11 Layer 4: reconquest of claimed / de jure land. A cheap take_province war
# available only when the attacker actually holds a claim on the defender's land
# (is_core_of read-back of add_claim, seeded by Layers 1-3). This is the war the
# irredentist claim exists to justify — taking back land the attacker considers
# rightfully its own. Discounted conquest cost on that land; not a subject war.
reconquest_wargoal = {
	type = take_province

	allow = {
		is_subject = no
		# only offer it when there is genuinely claimed land to reclaim: the
		# attacker holds a core/claim on at least one province the defender owns.
		# Both scope forms here are verified against Invictus' maurya_revival_wargoal,
		# whose allow block uses `scope:defender = { any_owned_province = {...} }` and
		# whose top-level tag/is_subject tests read the attacker — so in a wargoal
		# allow block scope:defender is the target country and ROOT is the attacker.
		# add_claim is read back via is_core_of (repo-wide convention: see
		# se_CLAIM_HOSTILITY / se_MIGRATION / se_DEJURE). NOTE the qualifying core
		# is the ATTACKER's own — either a Layer 1-3 add_claim granted directly to
		# it (the separatism liberation path grants the rebel claims on other
		# empires' de jure land) OR a vanilla engine auto-core on land it recently
		# held; it is NOT the kin-state claims seeded elsewhere. On the separatism
		# spawn path the war is forced through the scripted declare (which bypasses
		# this allow), so this gate matters mainly for AI/diplomacy selection.
		scope:defender = {
			any_owned_province = {
				is_core_of = root
			}
		}
	}

	attacker = {
		conquer_cost = -0.4
	}
	defender = {
	}
	ticking_war_score = 1
}

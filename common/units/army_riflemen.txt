riflemen = {
    army = yes
	assault = yes
	support = yes

	# [#290] Elite rifle-armed skirmisher/sharpshooter infantry. Historically (incl. the
	# 1763 start) rifles were rare, slow-loading arms issued ONLY to skirmishers, jaeger and
	# sharpshooters — line infantry carried smoothbore muskets and did not adopt rifles en
	# masse until rifling was industrialised (Minie ball / percussion, ~1850s). So riflemen
	# are a MINORITY elite arm, NOT a line-infantry replacement (see ai_max_percentage below).
	#
	# Units are gated by an `allow = {}` trigger inside the def (the engine has no
	# enable_unit_type verb — see army_artillery.txt:9). Two gates, both intentional:
	#   invention = tech_rifles      — cannot field riflemen before rifling tech exists
	#   trade_good_surplus = rifles  — must actually have rifles in surplus to raise them
	# This is the CONCRETE consumer of the rifles trade good: rifle demand is driven by the
	# count of these units (DEMAND_rifles -> num_of_unit_type = riflemen), so the good is
	# consumed by the specific troops that carry it rather than abstractly by all cohorts.
	allow = {
		invention = tech_rifles
		trade_good_surplus = { target = rifles  value > 0 }
	}

	maneuver = 4
	movement_speed = 2.6
	build_time = 60

	# Combat profile: strong on the offensive (aimed fire), a light-infantry skirmisher that
	# outshoots line infantry but is fragile in a stand-up melee and few in number.
	regular_infantry = 1.25
	conscripts = 1.4
	light_infantry = 1.2
	heavy_infantry = 0.9
	heavy_cavalry = 0.8
	light_cavalry = 1.1

	attrition_weight = 0.7
	attrition_loss = 0.015

	food_consumption = 0.22
	food_storage = 2.8

	morale_damage_taken = 1.1
	build_cost = {
		gold = 16
		manpower = 1
	}

	# Elite minority arm: cap the AI's rifle fraction so it supplements, never replaces,
	# musket/regular line infantry (matches the historical skirmisher-screen role).
	ai_max_percentage = 15
}

# §2/§3). Five concrete on-map PRODUCTION buildings for the 1815 Qing specialty-goods
# economy. Each is a real building that MULTIPLIES its province's raw-good output
# (base_resources, proven on IND_resource_gathering_operation:33) and adds the strata
# whose labour the craft actually used. Schema follows arsenal_building /
# IND_resource_gathering_operation / qing_granary_building. Behaviour lives here as
# flat local modifiers; the buildings are placed by the 1815 seed (oa_economy_setup)
# and by integration hooks (Board of Works, Self-Strengthening) — NOT by the player
# build menu at start (the allow gate only restricts that menu; add_building_level
# bypasses it). Concrete-over-abstract per standing rule.
#
# BOM / UTF-8, matching its common/buildings/ siblings.
########################################################################

	# employing skilled urban artisans. Boosts silk output and the skilled middle/upper
	# strata that staffed the looms; a mark of refined administration.

	local_upper_strata_desired_pop_ratio = 0.01
	local_monthly_civilization = 0.006
	base_resources = 2

	cost = 90
	time = 240

	# [#193-fix task #193] Build-menu gate (add_building_level bypasses this, so the
	# 1815 seed / Board-of-Works / Self-Strengthening spawns are unaffected): restrict

qing_opium_poppy_farm_building = {
	# 罌粟田 — the Yunnan/Sichuan poppy farm. Opium poppy was known in China from the
	# Tang and grown for medicine long before the 19th-century boom; this is the
	# domestic cultivation that (historically, from the mid-1800s) substituted for the
	# Bengal import and reversed the silver drain (以土抵洋). Labour-heavy hill-country
	# estate: boosts the lower/indentured strata that worked the poppy fields. The
	# concrete map counterpart of the opium balance-of-trade model — its output feeds
	# GOODS_national_production_opium, which se_QING_OPIUM.txt subtracts from national
	# consumption to size the (import) silver drain. Schema mirrors the five sibling
	# Qing production works above; the ROW row_plantation_building is barred to Chinese
	# cultures, so this is the Qing's own poppy estate.
	local_lower_strata_output = 0.5
	local_indentured_output = 0.4
	local_lower_strata_desired_pop_ratio = 0.05
	local_monthly_civilization = 0.002
	# [#423] base_resources = 3 (punchier than the base-2 tea/salt works) so each poppy farm
	# meaningfully lifts the province's opium yield. Like every building, this is MULTI-BUILDABLE
	# — the player can raise several poppy farms in one province (up to its building slots), and
	# base_resources stacks per instance, so a poppy-country province can be turned over to serious
	# cultivation. The more the empire grows, the more GOODS_national_production_opium offsets the
	# import silver drain (#423 balance-of-trade model).
	base_resources = 3

	cost = 75
	time = 200

	# Build-menu gate (add_building_level bypasses this): a Chinese-culture-group state
	# on a province already growing opium. Buildable from the start (poppy was long
	# cultivated in China) — the throne turns the province to poppy via the opium panel's
	# domestic-production lever (QING_opium_unlock_domestic) or the Burma conquest, then
	# raises as many poppy farms here as the province's slots allow.
	potential = {
		owner = { country_culture_group = chinese_group }
		trade_goods = opium
	}

	allow = {
		owner = { invention = tech_construction }
		sufficient_job_slots = yes
	}

	modification_display = {
	}
}

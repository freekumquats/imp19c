# [#7 2026-07-18] FOREIGN buildings — the CONCRETE on-map face of the Christian-mission mechanic
# (se_QING_MISSIONARY_STATIONS.txt). Per the concrete-over-abstract rule, the missionary system no
# longer tracks presence by a qing_mission_station PROVINCE MODIFIER: it spawns/destroys a REAL
# building, and the mission's mechanical effect lives ON THE BUILDING (buildings carry local
# modifiers natively) rather than on a separate province modifier. TWO tiers, matching the two
# historical modes the mechanic distinguishes:
#
#   • qing_mission_underground_building (地下傳教據點) — the PRE-TREATY tolerated/underground mission
#     (1601-1860): chapel-in-a-courtyard, a catechist, quiet charity. Small, discreet, low friction.
#   • qing_mission_public_building (公開傳教據點) — the POST-TREATY open mission (1858/60+): the
#     treaty-sanctioned chapel + school + dispensary compound flying the foreign flag. Larger reach
#     and civilisation/research, but a lightning-rod for grievance (the 教案 / Boxer feeling) — so it
#     carries the anti-Christian HAPPINESS penalty the old qing_mission_station modifier used to.
#
# The effects folded here FROM the retired qing_mission_station modifier (was local_monthly_civilization
# 0.02 + local_population_happiness -0.04): the open/public tier carries them (the friction was a
# treaty-era, high-visibility phenomenon); the underground tier is discreet (a small civ/research
# lift, no happiness penalty — a tolerated community drew little ire). add_building_level does NOT
# enforce the `allow` tech gate, so the mechanic seeds either tier directly; `allow` governs only the
# player's own build-menu click. UTF-8 BOM present. Keys proven building-scope (mirror
# qing_granary_building / the governance buildings; local_population_happiness on a building is
# vanilla, e.g. the base-game happiness buildings).

# ---- PRE-TREATY: the underground/tolerated mission (地下傳教據點) ----------------------
qing_mission_underground_building = {
	# a discreet presence: a little Western learning + charity goodwill, minimal footprint,
	# and NO happiness penalty (a tolerated underground community drew little public ire).
	local_monthly_civilization = 0.01
	local_research_points_modifier = 0.03

	cost = 40
	time = 180

	potential = { has_city_status = yes }
	allow = {
		has_city_status = yes
		sufficient_job_slots = yes
	}

	modification_display = {
		0 = local_monthly_civilization
		1 = local_research_points_modifier
	}
}

# ---- POST-TREATY: the open, treaty-sanctioned mission (公開傳教據點) --------------------
qing_mission_public_building = {
	# the full compound — chapel, mission-school, dispensary: a real conduit of Western science,
	# medicine and letters, and of trade contact. Carries the anti-Christian HAPPINESS penalty
	# (the 教案 grievance) the retired qing_mission_station modifier used to apply.
	local_monthly_civilization = 0.03
	local_research_points_modifier = 0.08
	local_state_trade_routes = 0.15
	local_population_happiness = -0.04

	cost = 60
	time = 180

	potential = { has_city_status = yes }
	allow = {
		has_city_status = yes
		sufficient_job_slots = yes
	}

	modification_display = {
		0 = local_monthly_civilization
		1 = local_research_points_modifier
		2 = local_population_happiness
	}
}

# ---- FOREIGN TRADING CONCESSION (夷館 / a European feitoria) --------------------------
# [#51 2026-07-22] The on-map face of a European trading concession held on Qing soil under
# Qing sovereignty — the canonical case is Portuguese MACAU (澳門, settled 1557, held as a leased
# enclave paying ground-rent 地租 to the Qing until Portugal gained sovereignty only in the 1887
# Sino-Portuguese Treaty). Macau is therefore modelled as a CHI-owned city carrying this concession
# building (the Portuguese factory/feitoria), NOT as Portuguese territory. Generic (a treaty-port
# concession building), seeded on Macau at setup; also player-buildable in a Qing treaty-port city.
# A conduit of foreign commerce + a little Western contact, with mild local friction (a walled
# foreign quarter under imperial suffrance). Same building-scope keys as the mission buildings; same
# has_city_status gate (Macau is bumped settlement->city for #51, matching its real walled-port status).
qing_foreign_concession_building = {
	local_state_trade_routes = 0.25
	local_monthly_civilization = 0.02
	local_research_points_modifier = 0.04
	local_population_happiness = -0.02

	cost = 60
	time = 180

	potential = { has_city_status = yes }
	# [#51 fix 2026-07-22] NO sufficient_job_slots gate. The archetype is MACAU — a small (~5,000-person,
	# ~1 pop-unit at the Guangdong scale) Portuguese enclave that genuinely cannot meet a mass job-slot
	# floor. The boot test showed exactly this: "too few people in Macau for another building". A foreign
	# trading concession is a compact walled compound (factories + a resident merchant community), NOT a
	# mass-employment industry, so it should stand in a tiny port. Dropping the job-slot requirement is
	# the historically-correct fix — the alternative (inflating Macau's pops ~30x) was rejected as
	# ahistorical. has_city_status is kept (a concession implies a chartered port, not a bare settlement).
	allow = {
		has_city_status = yes
	}

	modification_display = {
		0 = local_state_trade_routes
		1 = local_monthly_civilization
		2 = local_research_points_modifier
		3 = local_population_happiness
	}
}

# prestige, Boxer sentiment) is hung off `has_building` by an event/pulse in
# se_QING_FOREIGNBUILD.txt / qing_foreignbuild_events.txt, NEVER baked into the building.
# All keys below verified building-scope against existing buildings (qing_works /
# qing_governance / 00_military / 00_infrastructure). add_building_level seeds them

# ---- SUBJECT INDUSTRIALIZATION AID (洋務局) -------------------------------------------
# The Self-Strengthening works-bureau the overlord plants in a SUBJECT's own city to help it
# modernize (中體西用, 1861-95: Jiangnan Arsenal / Fuzhou Shipyard model). The subject KEEPS the
# province; the building lifts the subject's civilisation/research/promotion locally, while the
# Qing earns a standing works-patron country modifier (qing_fbuild_works_patron) for as long as
# it stands (applied by QING_fbuild_subject_works, not by the building). No happiness penalty —
# it is aid, welcomed. has_city_status kept (an arsenal implies a real urban base).
qing_foreign_works_building = {
	local_monthly_civilization = 0.04
	local_research_points_modifier = 0.10
	local_pop_promotion_speed_modifier = 0.05

	cost = 80
	time = 180

	# [#52 review-fix 2026-07-23] NOT player-buildable from the province menu — the 洋務局 is planted
	# ONLY by the authorization effect QING_fbuild_subject_works (in a SUBJECT's city, spending
	# treasury+PI and taking the works-patron country modifier). An open build-menu allow would let a
	# player self-build it and, via the upkeep sweep's has_building read, hold the patron modifier for
	# free — so the gate is closed, matching the treaty-port pattern. add_building_level bypasses allow.
	potential = { always = yes }   # [BT-D1-3] list the type in EVERY province (authorization-only; allow gates buildability)
	allow = {
		# [#52 BT-D1-3] Planted only by its authorization effect, never player-built. potential (below) lists
		# the type; this allow keeps it un-buildable by hand. custom_tooltip only gives the greyed entry a
		# readable reason — it is TRANSPARENT to the false verdict (same as a bare always=no), NOT a
		# visibility lever. Listing is governed by potential, which is widened to always=yes below.
		custom_tooltip = {
			text = "qing_fbuild_authorization_only_tt"
			always = no
		}
	}

	modification_display = {
		0 = local_monthly_civilization
		1 = local_research_points_modifier
		2 = local_pop_promotion_speed_modifier
	}
}

	local_population_growth = 0.05
	local_migration_attraction = 0.10
	local_monthly_civilization = 0.02

	}

########################################################################
# row_production_buildings.txt — [bookmark-1763 #291] REST-OF-WORLD specialty
# production buildings.
#
# The Qing ship five SPECIFIC, historically-named specialty works (silk filature,
# porcelain kiln, tea workshop, cotton works, salt yard — qing_production_buildings.txt),
# gated on country_culture_group = chinese_group. The rest of the world had no
# equivalent on-map production buildings. Per the granular-China / broad-ROW ruling,
# this adds TWO GENERIC, abstract production buildings that any non-Chinese state can
# raise on the appropriate good — deliberately lower-granularity than the Qing set
# (no per-good named variants, no special features): a manufactory for finished/craft
# goods and a plantation for cash crops. They MULTIPLY the province's raw-good output
# and add the strata whose labour the trade used, exactly like the Qing works.
#
# Schema mirrors qing_production_buildings.txt (which mirrors arsenal_building /
# IND_resource_gathering_operation). add_building_level bypasses the `allow` gate, so
# these can also be seeded; the `potential` gate keeps them off Chinese states (which
# use their own specialty works) and off unrelated goods. Concrete-over-abstract.
#
# BOM / UTF-8, matching common/buildings/ siblings.
########################################################################

# --- ROW MANUFACTORY: finished / craft goods (cloth, linen, wool, cotton, silk,
#     porcelain, dye). The generic European/Indian mill-and-workshop — Lancashire and
#     Flemish textile towns, Saxon/Bohemian works, Bengali muslin, Gobelins-style
#     luxury craft. Boosts the skilled middle/upper strata that staffed manufactories. ---
row_manufactory_building = {
	local_middle_strata_output = 0.4
	local_upper_strata_output = 0.3
	local_lower_strata_output = 0.3
	local_middle_strata_desired_pop_ratio = 0.03
	local_upper_strata_desired_pop_ratio = 0.01
	local_monthly_civilization = 0.005
	base_resources = 2

	cost = 90
	time = 240

	# Build-menu gate (add_building_level bypasses it): any NON-Chinese state (the Qing
	# use their own named specialty works) producing a finished/craft good. Broad on
	# purpose — one building for the whole family of manufactured goods. Goods verified
	# live + province-assigned (common/trade_goods/00_imp19c.txt): cotton/linen are
	# flagged "# to be removed" and porcelain is Jingdezhen-only (Chinese, excluded by the
	# culture gate), so none of those are listed here. textile_fibres is the raw-textile
	# good the Qing cotton works itself processes (se_QING_BUILDINGS.txt:48).
	potential = {
		owner = { NOT = { country_culture_group = chinese_group } }
		OR = {
			trade_goods = cloth
			trade_goods = wool
			trade_goods = silk
			trade_goods = dye
			trade_goods = textile_fibres
			trade_goods = industrial_fibres
		}
	}

	allow = {
		owner = { invention = tech_manufactories }
		sufficient_job_slots = yes
	}

	modification_display = {
	}
}

# --- ROW PLANTATION: cash crops (sugar, tobacco, coffee, chocolate, cotton, spices,
#     opium). The generic New-World / colonial-tropics estate — Caribbean sugar,
#     Chesapeake tobacco, Brazilian coffee, East-Indies spice. Labour-heavy: boosts the
#     lower strata + the indentured/servile labour that worked the estates. ---
row_plantation_building = {
	local_lower_strata_output = 0.5
	local_indentured_output = 0.4
	local_middle_strata_output = 0.2
	local_lower_strata_desired_pop_ratio = 0.05
	local_monthly_civilization = 0.003
	base_resources = 2

	cost = 80
	time = 220

	# Build-menu gate (bypassed by add_building_level): any NON-Chinese state producing
	# a plantation cash crop. Goods verified live + province-assigned; chocolate (0
	# provinces) and cotton ("# to be removed") are omitted.
	potential = {
		owner = { NOT = { country_culture_group = chinese_group } }
		OR = {
			trade_goods = sugar
			trade_goods = tobacco
			trade_goods = coffee
			trade_goods = spices
			trade_goods = opium
		}
	}

	allow = {
		owner = { invention = tech_construction }
		sufficient_job_slots = yes
	}

	modification_display = {
	}
}

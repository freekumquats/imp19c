########################################################################
# qing_granary_buildings.txt — the ever-normal granary (常平倉) building [#91 item K / #435].
#
# A concrete on-map counterpart to the abstract qing_granary_stock counter
# (se_QING_DECLINE.txt:25,66): where the counter is the summary/AI layer, this is
# a real building the decline engine raises in the Yellow River (黃河) provinces
# in good years / when the stock is high, and tears down (remove_building_level)
# in famine as the reserves are eaten. Additive: the counter is KEPT, the building
# hangs off it.
#
# Schema follows the mod's own arsenal_building (00_military_buildings.txt:20) and
# the food-capacity + monthly-food shape of Terra-Indomita's population_building
# (#Granary). Event-spawned via add_building_level, so no macro_builder/config
# entry is needed; the potential/allow gate is for the player build menu only.
#
# BOM / UTF-8, matching its common/buildings/ siblings.
########################################################################

qing_granary_building = {
	# [#435] the EVER-NORMAL GRANARY (常平倉) — not a plain grain store but the distinctive Qing
	# PRICE-STABILISATION institution: it bought grain dear in glut and sold cheap in dearth, so
	# its teeth are (a) food storage + accumulation (the buffer against famine), AND (b) market
	# stabilisation — a settled grain market that steadies the countryside. Modelled with the
	# storage keys (food_capacity/food_modifier, ~ TI population_building #Granary), PLUS the
	# stabilisation character the plain granary lacked: a trade-route (market reach), a loyalty
	# lift (a fed, price-steadied populace is a quiet one — the famine-riot prevention the 常平倉
	# was built for), and happiness. All keys proven building-scope (local_state_trade_routes /
	# local_monthly_state_loyalty on the customs-house + dike buildings).
	local_food_capacity = 200
	local_monthly_food_modifier = 0.05
	local_state_trade_routes = 0.2
	local_monthly_state_loyalty = 0.02
	local_population_happiness = 0.03
	local_monthly_civilization = 0.01

	cost = 60
	time = 180

	# Build-menu gate only (add_building_level does not enforce allow): a granary
	# needs an urban settlement to administer it and free job slots to staff it.
	potential = {
		has_city_status = yes
	}
	allow = {
		has_city_status = yes
		sufficient_job_slots = yes
	}

	modification_display = {
		0 = local_food_capacity
		1 = local_monthly_food_modifier
		2 = local_state_trade_routes
	}
}

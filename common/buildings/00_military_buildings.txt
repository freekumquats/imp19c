arsenal_building = {
	# [logistics P2] the arsenal is no longer a bare demand-sink: it now (a) shows a
	# real defensive output on-map and (b) contributes LOCAL munitions PRODUCTION via
	# the scripted building-count term in GOODS_governorship_early_munitions_produced
	# (GOODS_svalues.txt). The munitions accrual is scripted, not a province-modifier,
	# because munitions is a scripted trade good with no engine province-output key —
	# the arsenal is already COUNTED this way on the demand side (DEMAND_svalues.txt:1448),
	# so counting it symmetrically on the supply side is the concrete, matching hook.
	local_defensive = 0.05

	cost = 50
	time = 180

	allow = {
		owner = { invention = tech_firearms }
		civilization_value >= 10
		sufficient_job_slots = yes
	}
	
	modification_display = {
		0 = local_defensive
	}
}

military_depot_building = {
	# [logistics P2] the depot is the forward-supply half of the pair: it stockpiles
	# and forwards munitions, contributing a LOCAL munitions production term (see
	# GOODS_governorship_early_munitions_produced) and steadying supply near the front.
	# Cheaper and less gated than the arsenal (it manufactures nothing — it warehouses),
	# so it needs no firearms invention, only a city to administer it. Schema follows the
	# mod's own qing_granary_building (event-spawnable via add_building_level; the
	# potential/allow gate is the build-menu only). num_of_military_depot_building is the
	# engine counter the production term reads, exactly as num_of_arsenal_building is used.
	local_defensive = 0.02

	cost = 40
	time = 150

	potential = {
		has_city_status = yes
	}
	allow = {
		has_city_status = yes
		sufficient_job_slots = yes
	}

	modification_display = {
		0 = local_defensive
	}
}

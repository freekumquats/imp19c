INF_canal = {
	local_population_capacity_modifier = 0.01
	local_monthly_civilization = 0.02
	# [#441 logistics-wiring] a canal is a transport artery — troops and supply move
	# on it — so it grants army_movement_speed like the railway (0.1,
	# 00_infrastructure_buildings.txt:68) and the Qing canal buildings
	# (qing_grand_canal_building 0.25, qing_canal_depot_building 0.10). Previously the
	# generic canal was the only water/rail transport work with NO movement teeth.
	army_movement_speed = 0.1
	local_lower_strata_desired_pop_ratio = 0.05
	local_middle_strata_desired_pop_ratio = 0.05

	# [#441] A canal is the cheapest way to MOVE goods (water carriage far out-hauls early
	# rail per ton-mile — hence canal_bonus > railway_bonus in MOVEMENT_svalues), but the
	# dearest and slowest to BUILD: it must be dug by hand over a long haul of corvee labour.
	# The build system has no manpower-cost field (only cost/time), so that digging burden is
	# expressed as a higher cost and a much longer build time than the railway (cost 50 /
	# time 180). Cost 90 + time 540 = 3x the railway's build time.
	cost = 90
	time = 540

	allow = {
		owner = {invention = tech_artificial_canals}
		sufficient_job_slots = yes
	}
	
	modification_display = {
		0 = local_food_capacity
	}
}

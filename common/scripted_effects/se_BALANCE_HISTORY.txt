########################################################################
# se_BALANCE_HISTORY.txt — quarterly net-balance history recorder (#162).
#
# Records the national net quarterly balance (INCOME_national_total_quarterly,
# computed in INCOME_update_treasury_country, se_INCOME.txt:54-58) into an
# 8-quarter ROLLING window on the country, so the balance-history panel in
# economy_view.gui (the "balance_history" flowcontainer) renders a histogram of
# the last two years of surplus / deficit.
#
# STORAGE (per country):
#   balance_hist_q0 .. balance_hist_q7   — raw net balance, q0 = most recent.
#   balance_hist_max_abs                 — max |value| over the 8 slots, floored
#                                          at 1 (the histogram's full-scale ref).
#   balance_hist_pos_q0..q7 / _neg_q0..q7 — pre-normalised 0.0..1.0 bar heights.
#                                          A surplus fills the green (positive)
#                                          bar; a deficit fills the red (negative)
#                                          bar. Pre-computed HERE so the GUI does
#                                          no arithmetic — progressbars take a
#                                          0..1 value directly.
#
# Called once per quarter, per country, from INCOME_update_treasury_country
# (right after INCOME_national_total_quarterly is set). se_LOG-traced.
#
# Prefix = BALHIST_
########################################################################

# ---- Seed the window to zero (idempotent; first-quarter / new-country safe) --
BALHIST_init = {
	# Scope: Country
	if = {
		limit = { NOT = { has_variable = balance_hist_seeded } }
		set_variable = { name = balance_hist_q0  value = 0 }
		set_variable = { name = balance_hist_q1  value = 0 }
		set_variable = { name = balance_hist_q2  value = 0 }
		set_variable = { name = balance_hist_q3  value = 0 }
		set_variable = { name = balance_hist_q4  value = 0 }
		set_variable = { name = balance_hist_q5  value = 0 }
		set_variable = { name = balance_hist_q6  value = 0 }
		set_variable = { name = balance_hist_q7  value = 0 }
		set_variable = { name = balance_hist_max_abs  value = 1 }
		set_variable = { name = balance_hist_seeded  value = 1 }
	}
}

# ---- Record one quarter: shift the window and push the newest balance --------
BALHIST_record_quarter = {
	# Scope: Country. Expects INCOME_national_total_quarterly to be current.
	LOG_enter = { sys = INCOME  fn = "BALHIST_record_quarter" }
	BALHIST_init = yes

	# Shift q6->q7, q5->q6, ... q0->q1 (oldest drops off q7). Do it high-to-low
	# so each slot is read before it is overwritten.
	set_variable = { name = balance_hist_q7  value = var:balance_hist_q6 }
	set_variable = { name = balance_hist_q6  value = var:balance_hist_q5 }
	set_variable = { name = balance_hist_q5  value = var:balance_hist_q4 }
	set_variable = { name = balance_hist_q4  value = var:balance_hist_q3 }
	set_variable = { name = balance_hist_q3  value = var:balance_hist_q2 }
	set_variable = { name = balance_hist_q2  value = var:balance_hist_q1 }
	set_variable = { name = balance_hist_q1  value = var:balance_hist_q0 }
	# newest quarter into q0
	set_variable = { name = balance_hist_q0  value = var:INCOME_national_total_quarterly }

	BALHIST_recompute_scale = yes
	LOG_line = { sys = INCOME  msg = "BALHIST q0=[This.MakeScope.GetVariable('balance_hist_q0').GetValue|0] scale=[This.MakeScope.GetVariable('balance_hist_max_abs').GetValue|0] for [This.GetTag]" }
	LOG_exit = { sys = INCOME  fn = "BALHIST_record_quarter"  result = OK }
}

# ---- Find the full-scale reference, then pre-normalise every slot ------------
# max_abs = max over the window of |balance_hist_qN|, floored at 1 so the
# per-slot normalise can never divide by zero (a flat-zero history renders as
# empty bars, not NaN).
BALHIST_recompute_scale = {
	# Scope: Country
	set_variable = { name = balance_hist_max_abs  value = 1 }
	BALHIST_fold_abs = { slot = balance_hist_q0 }
	BALHIST_fold_abs = { slot = balance_hist_q1 }
	BALHIST_fold_abs = { slot = balance_hist_q2 }
	BALHIST_fold_abs = { slot = balance_hist_q3 }
	BALHIST_fold_abs = { slot = balance_hist_q4 }
	BALHIST_fold_abs = { slot = balance_hist_q5 }
	BALHIST_fold_abs = { slot = balance_hist_q6 }
	BALHIST_fold_abs = { slot = balance_hist_q7 }

	# Pre-normalise every slot to 0.0..1.0 split into a positive and negative bar.
	BALHIST_normalise_slot = { slot = 0 }
	BALHIST_normalise_slot = { slot = 1 }
	BALHIST_normalise_slot = { slot = 2 }
	BALHIST_normalise_slot = { slot = 3 }
	BALHIST_normalise_slot = { slot = 4 }
	BALHIST_normalise_slot = { slot = 5 }
	BALHIST_normalise_slot = { slot = 6 }
	BALHIST_normalise_slot = { slot = 7 }
}

# ---- Fold one slot's |value| into the running max ----------------------------
# $slot$ = a balance_hist_qN variable name. Updates balance_hist_max_abs to the
# larger of itself and |slot|.  |x| = max(x, -x).
BALHIST_fold_abs = {
	# Scope: Country
	# |x| = max(x, -x). In PDX svalue math `min = V` is the FLOOR (raise up to V),
	# so min of (-slot) raises the running value to the larger of slot and -slot.
	set_variable = {
		name = balhist_scratch_abs
		value = {
			value = var:$slot$
			min = {
				value = var:$slot$
				multiply = -1
			}
		}
	}
	if = {
		limit = { var:balhist_scratch_abs > var:balance_hist_max_abs }
		set_variable = { name = balance_hist_max_abs  value = var:balhist_scratch_abs }
	}
}

# ---- Normalise slot N to 0.0..1.0 positive/negative bar heights --------------
# $slot$ = 0..7. Writes balance_hist_pos_q$slot$ and balance_hist_neg_q$slot$.
# positive = value/max_abs clamped [0,1] (0 for a deficit quarter);
# negative = (-value)/max_abs clamped [0,1] (0 for a surplus quarter).
BALHIST_normalise_slot = {
	# Scope: Country
	# clamp to [0,1]: min = 0 is the FLOOR, max = 1 is the CEILING.
	set_variable = {
		name = balance_hist_pos_q$slot$
		value = {
			value = var:balance_hist_q$slot$
			divide = var:balance_hist_max_abs
			min = 0
			max = 1
		}
	}
	set_variable = {
		name = balance_hist_neg_q$slot$
		value = {
			value = var:balance_hist_q$slot$
			multiply = -1
			divide = var:balance_hist_max_abs
			min = 0
			max = 1
		}
	}
}

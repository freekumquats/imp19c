########################################################################
# se_QING_EARLYINDUS.txt — [bookmark-1763] THE EARLY-INDUSTRIALIZATION DIVERGENCE (洋務先聲).
#
# The great counterfactual of the 1763 start: what if the Qianlong Emperor, receiving
# the Macartney embassy in 1793, had NOT written "the Celestial Empire possesses all
# things and needs nothing", but instead embraced the machines the English brought —
# launching China's industrial turn ~67 years BEFORE the historical Self-Strengthening
# Movement (洋務運動, 1861-1895)?
#
# This module is that alt-history track. It is launched from the Macartney embassy's
# EMBRACE option (qing_embassy.1, "learn the barbarians' arts"), and it deliberately
# REUSES the Self-Strengthening engine (se_QING_SELFSTR.txt) rather than duplicating it:
#   • the SAME 0..100 qing_selfstr_progress counter — so an early start simply means the
#     movement is already far along by the time the historical 1860s tree unlocks; the
#     SS bands (nascent/advancing/golden_age), the great-work WONDER, and the "refraction"
#     hollow-test all key off this shared counter and apply for free.
#   • QING_selfstr_advance — which already HALVES progress gains when bureau capacity /
#     council effectiveness are low. In 1793 they are: a pre-modern bureaucracy cannot
#     carry an industrial program at full speed. That damping IS the historical friction
#     (no central coordination) — reused verbatim, no new logic.
#   • QING_selfstr_build — drops the founded institution as a REAL on-map building.
#
# What differs from the 1860s tree: the INSTITUTIONS are era-appropriate 1790s foundations
# (an imperial cannon foundry, a translation/science bureau, a steam-experiment yard, a
# state machine-works), and every founding carries a HEAVY conservative backlash — a Manchu
# court in Qianlong's twilight recoils at the Son of Heaven bowing to barbarian craft. The
# reward for pushing through it is a China that meets the 19th century already industrial.
#
# Self-guards on the qing_early_industrialization era flag, so it is inert unless the
# divergence was actually chosen (i.e. it never fires at the 1815 develop start, where
# Macartney does not fire at all). All steps logged (sys = QING). NO BOM (se_*.txt convention).
#   ROOT / this = the Qing overlord (CHI).
# Prefix = QING_earlyindus_
########################################################################

# ======================================================================
# LAUNCH — the divergence itself. Called from the Macartney EMBRACE option.
# Sets the era flag, seeds the shared SS progress counter, founds the first
# institution (the imperial cannon foundry — the English gift of cannon and
# muskets was the opening wedge), grants the harbinger modifier, and pays the
# conservative price of so radical a break with the tribute worldview.
# ======================================================================
QING_earlyindus_launch = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_launch" }
	if = {
		limit = { NOT = { has_variable = qing_early_industrialization } }
		set_variable = { name = qing_early_industrialization  value = 1 }
		# seed the SHARED Self-Strengthening counter (the early program IS the movement,
		# begun early) and give it a running head start for the imperial commitment.
		QING_selfstr_init = yes
		add_country_modifier = { name = qing_earlyindus_harbinger  duration = -1 }
		# the English cannon and muskets are the first foundation — a real arsenal.
		QING_earlyindus_found_foundry = yes
		# the radical break: the court's conservative wing recoils (the Manchu old guard
		# who hold that 國語騎射 — the Manchu Way — is sufficient). A one-time launch shock
		# on top of the per-institution backlash below.
		QING_earlyindus_backlash = { severity = 8 }
		add_legitimacy = -8
		LOG_line = { sys = QING  msg = "EARLYINDUS: the Macartney divergence — early industrialization launched (洋務先聲) for" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_earlyindus_launch"  reason = "early industrialization already launched for" }
	}
	LOG_exit = { sys = QING  fn = "QING_earlyindus_launch"  result = OK }
}

# ======================================================================
# CONSERVATIVE BACKLASH — the shared cost of pushing modernization a lifetime early.
# In the 1860s the movement had the trauma of the Opium Wars + Taiping to justify it;
# in the 1790s there is no such shock, so every foundation is a naked provocation of the
# tribute-worldview conservatives. Expressed as a reform-pressure spike (feeds the reform
# track), a stability dip, and — while the program is young — a lasting court-tension band.
# $severity$ = the size of the provocation (launch 8; per-institution 3-5).
# ======================================================================
QING_earlyindus_backlash = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_backlash" }
	QING_DECLINE_nudge = { var = qing_reform_pressure  amount = $severity$ }
	add_stability = -3
	# the conservative-court tension band (a lasting drag until the program proves itself).
	# Re-applied, not stacked (a single instance stands while the divergence is contested).
	if = {
		limit = { NOT = { has_country_modifier = qing_earlyindus_conservative_reaction } }
		add_country_modifier = { name = qing_earlyindus_conservative_reaction  duration = -1 }
	}
	LOG_line = { sys = QING  msg = "EARLYINDUS: conservative backlash (severity $severity$) — reform pressure + stability hit for" }
	LOG_exit = { sys = QING  fn = "QING_earlyindus_backlash"  result = OK }
}

# ======================================================================
# THE FOUR EARLY INSTITUTIONS — each reuses QING_selfstr_advance (shared counter,
# auto-damped by weak machinery) + QING_selfstr_build (real building), applies a
# lasting institution modifier, and pays a conservative backlash. Each once-guarded on
# its own modifier so a repeated call is a safe no-op.
# ======================================================================

# 火器局 (Imperial Cannon Foundry) — the English gift of cannon/muskets made a
# domestic arm. The opening foundation (founded at launch). Real arsenal + the early
# firearms tech the foundry embodied. Advance is modest — a single works, run by a
# bureaucracy that does not yet understand it.
QING_earlyindus_found_foundry = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_found_foundry" }
	if = {
		limit = { NOT = { has_country_modifier = qing_earlyindus_foundry } }
		add_country_modifier = { name = qing_earlyindus_foundry  duration = -1 }
		QING_selfstr_build = { building = arsenal_building }
		# the metallurgy + small-arms the foundry taught (a subset of the later Jiangnan
		# unlocks — the 1790s reach is shorter than the 1860s arsenal's).
		unlock_invention = tech_metalworking
		unlock_invention = tech_bombard_cannons
		unlock_invention = tech_wheeled_cannons
		LOG_line = { sys = QING  msg = "EARLYINDUS: 火器局 Imperial Cannon Foundry founded for" }
		QING_selfstr_advance = { amount = 8 }
	}
	LOG_exit = { sys = QING  fn = "QING_earlyindus_found_foundry"  result = OK }
}

# 譯館 (Imperial Translation & Science Bureau) — the true seed. Macartney's planetarium,
# telescopes and clocks were the wonder of the embassy; a bureau to translate and teach
# the science behind them is the foundation of everything else. A real school + the early
# modern-science tech. The Jesuit court astronomers (still at Peking in 1793) staff it.
QING_earlyindus_found_translation = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_found_translation" }
	if = {
		limit = { NOT = { has_country_modifier = qing_earlyindus_translation } }
		add_country_modifier = { name = qing_earlyindus_translation  duration = -1 }
		QING_selfstr_build = { building = EDU_school }
		unlock_invention = tech_scientific_method
		unlock_invention = tech_technical_drawings
		unlock_invention = tech_astronomy
		LOG_line = { sys = QING  msg = "EARLYINDUS: 譯館 Translation & Science Bureau founded for" }
		QING_selfstr_advance = { amount = 10 }
		QING_earlyindus_backlash = { severity = 3 }
	}
	LOG_exit = { sys = QING  fn = "QING_earlyindus_found_translation"  result = OK }
}

# 機器船廠 (Imperial Steam-Experiment Yard) — a coastal yard to copy the paddle-steamer
# the West was just beginning to build. A real port + the early steam/ship tech. The most
# provocative foundation (a Manchu court funding a navy the tribute worldview says it does
# not need) — the heaviest per-institution backlash.
QING_earlyindus_found_yard = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_found_yard" }
	if = {
		limit = { NOT = { has_country_modifier = qing_earlyindus_yard } }
		add_country_modifier = { name = qing_earlyindus_yard  duration = -1 }
		if = {
			limit = { any_owned_province = { is_coastal = yes } }
			ordered_owned_province = {
				limit = { is_coastal = yes }
				order_by = total_population
				max = 1
				add_building_level = port_building
			}
			LOG_line = { sys = QING  msg = "EARLYINDUS: 機器船廠 Steam-Experiment Yard founded at top coastal province for" }
		}
		else = {
			LOG_fail = { sys = QING  fn = "QING_earlyindus_found_yard"  reason = "no coastal province owned; yard modifier granted but no port built for" }
		}
		unlock_invention = tech_shipyards
		unlock_invention = tech_copper_plating
		QING_selfstr_advance = { amount = 10 }
		QING_earlyindus_backlash = { severity = 5 }
	}
	LOG_exit = { sys = QING  fn = "QING_earlyindus_found_yard"  result = OK }
}

# 機器製造局 (Imperial Machine-Works) — a state manufactory to make the machines that
# make machines (the industrial-revolution insight). A real resource/industry operation +
# the steam-engine + construction tech. Raises development capital (a working state works
# turned a profit), the payoff for having pushed through the backlash. 官辦 (purely state-run,
# unlike the later 官督商辦 merchant model — the 1790s had no merchant-capital partners).
QING_earlyindus_found_manufactory = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_found_manufactory" }
	if = {
		limit = { NOT = { has_country_modifier = qing_earlyindus_manufactory } }
		add_country_modifier = { name = qing_earlyindus_manufactory  duration = -1 }
		QING_selfstr_build = { building = IND_resource_gathering_operation }
		unlock_invention = tech_construction
		unlock_invention = tech_rotative_beam_engine
		unlock_invention = tech_blast_furnace
		LOG_line = { sys = QING  msg = "EARLYINDUS: 機器製造局 Imperial Machine-Works founded for" }
		# a working state works turned a profit — development capital via the currency layer.
		CURRENCY_grant_country_wealth = { thousands = 50 }
		QING_selfstr_advance = { amount = 12 }
		QING_earlyindus_backlash = { severity = 4 }
	}
	LOG_exit = { sys = QING  fn = "QING_earlyindus_found_manufactory"  result = OK }
}

# ======================================================================
# CONSOLIDATE — the program has proved itself; the conservative reaction fades.
# Once all four foundations stand AND the movement has real momentum (shared SS
# progress >= 40), the court accepts the new order: lift the conservative-reaction
# band. Called from the program hub. Once-guarded on the flag.
# ======================================================================
QING_earlyindus_consolidate = {
	LOG_enter = { sys = QING  fn = "QING_earlyindus_consolidate" }
	if = {
		limit = {
			NOT = { has_variable = qing_earlyindus_consolidated }
			has_country_modifier = qing_earlyindus_foundry
			has_country_modifier = qing_earlyindus_translation
			has_country_modifier = qing_earlyindus_yard
			has_country_modifier = qing_earlyindus_manufactory
			# [#304 review-fix] gate 40 -> 20. The four foundings advance the shared counter by
			# 8+10+10+12 = 40 at FULL rate, but QING_selfstr_advance HALVES each gain when bureau
			# capacity / council effectiveness < 50 (the intended 1790s friction), so the realistic
			# total is 20-40. A >=40 gate had zero headroom — one halved step stranded the player
			# below consolidation with the malus stuck. 20 is reachable even fully-halved
			# (4+5+5+6), while the four-modifier requirement above still proves the whole program
			# was actually built. Mirror-updated at qing_embassy.10's consolidate option trigger.
			var:qing_selfstr_progress >= 20
		}
		set_variable = { name = qing_earlyindus_consolidated  value = 1 }
		remove_country_modifier = qing_earlyindus_conservative_reaction
		QING_DECLINE_nudge = { var = qing_reform_pressure  amount = -10 }
		add_stability = 3
		add_legitimacy = 10
		LOG_line = { sys = QING  msg = "EARLYINDUS: the divergence consolidated — the court accepts the new order for" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_earlyindus_consolidate"  reason = "consolidation preconditions not met (need all four foundations + progress>=40) for" }
	}
	LOG_exit = { sys = QING  fn = "QING_earlyindus_consolidate"  result = OK }
}

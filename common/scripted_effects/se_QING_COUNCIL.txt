# se_QING_COUNCIL.txt — the Grand Council (軍機處) hub and the character-held
# great offices (領班軍機大臣 / 六部尚書 / 都察院 / 理藩院 / 總理衙門 / 內務府).
#
# REDESIGN (council-is-offices model): the Grand Council is NOT a separate pool
# of "grand councillors" seated into abstract chairs — historically the 軍機處
# was staffed BY serving board presidents, grand secretaries and the like,
# concurrently with their offices. So HERE the Grand Council IS EXACTLY the set
# of filled great offices. You join the council by being APPOINTED TO AN OFFICE;
# there is no seat action and no councillor population.
#
# The appointable offices (13):
#   chancellor  領班軍機大臣  Grand Chancellor      (summit; all four skills)
#   personnel   吏部          Grand Minister of Personnel   (finesse)
#   revenue     戶部          Grand Minister of Revenue     (finesse)
#   rites       禮部          Grand Minister of Rites       (zeal)
#   war         兵部          Grand Minister of War         (martial)
#   justice     刑部          Grand Minister of Justice     (zeal)
#   works       工部          Grand Minister of Works       (finesse)
#   censor      都察院        Grand Inspector of the Censorate (finesse)
#   lifanyuan   理藩院        Grand Director of the Lifan Yuan (charisma)
#   chamberlain 內務府    Grand Secretary of the Imperial Household (charisma)
#   zongli      總理衙門      Grand Director of the Zongli Yamen (charisma)
# Plus the event-only 太上皇 Emperor Emeritus (emeritus) — a read-only overlay
# card shown when Napoleon (#65) sits above the banners; not autofilled, not an
# ordinary appointable seat, not counted toward vacancy.
#
# Effectiveness (0..100, in se_QING_GOVERNANCE) is derived from the office-
# holders' OFFICE-RELEVANT skills (a war minister is judged on martial, a rites
# minister on zeal, the Grand Chancellor on all four), plus a Manchu/Han balance
# bonus and a per-vacant-office penalty. It feeds the abstracted bureaucracy in
# se_QING_GOVERNANCE.txt (QING_GOV_pulse calls QING_council_recompute +
# QING_council_apply_band) and thereby the whole decline/reform machine.
#
# State (all on CHI unless noted):
#   qing_office_<key>_holder    the character holding each office
#   qing_council_effectiveness  0..100 derived meter (init in se_QING_GOVERNANCE)
#   qing_council_members        DERIVED list of office-holders, rebuilt each
#                               recompute (informational; drives the GUI roster)
#   qing_council_manchu_count / qing_council_han_count   culture split of holders
#   qing_council_filled_count   number of filled appointable offices
# Character-scope marks:
#   qing_office_held            flag: which office a character holds
#
# Prefix = QING_council_ / QING_office_
########################################################################

# ======================================================================
# AUTOFILL — staff every VACANT appointable office with the ablest unassigned
# courtier. Used at game start and by the "convene the council" verb. Appoints
# to each empty office in turn (chancellor first, then the six boards and the
# frontier/foreign yamen); a courtier already holding an office is skipped.
# ======================================================================
QING_council_autofill = {
	# [#103] se_LOG enter/exit wiring (council-is-offices rewrite)
	LOG_enter = { sys = QING  fn = "QING_council_autofill" }
	QING_council_init = yes
	# [BT-appointment-redesign 2026-07-22] $degree$ per office: the two MARTIAL seats — War (兵部) and
	# the Commandant of the Imperial Guard (領侍衛內大臣) — get the martial degree 武進士 (wu_jinshi);
	# the eleven civil offices get the civil 進士 (jinshi). So an autofilled War minister reads as a
	# martial-degree holder, not a mis-assigned civil scholar (the BT-N congruence, done at CREATION now).
	QING_council_autofill_office = { office = chancellor        degree = jinshi }
	QING_council_autofill_office = { office = personnel         degree = jinshi }
	QING_council_autofill_office = { office = revenue           degree = jinshi }
	QING_council_autofill_office = { office = rites             degree = jinshi }
	QING_council_autofill_office = { office = war               degree = wu_jinshi }
	QING_council_autofill_office = { office = justice           degree = jinshi }
	QING_council_autofill_office = { office = works             degree = jinshi }
	QING_council_autofill_office = { office = censor            degree = jinshi }
	QING_council_autofill_office = { office = lifanyuan         degree = jinshi }
	QING_council_autofill_office = { office = chamberlain       degree = jinshi }
	QING_council_autofill_office = { office = zongli            degree = jinshi }
	QING_council_autofill_office = { office = grand_secretariat degree = jinshi } # [#273]
	QING_council_autofill_office = { office = guard_commandant  degree = wu_jinshi }  # [#273]
	QING_council_recompute = yes
	LOG_line = { sys = QING  msg = "council autofilled (offices staffed) for" }
	LOG_exit = { sys = QING  fn = "QING_council_autofill"  result = OK }
}

# ======================================================================
# [#352 F3/F8] STALE-SEAT PRUNE — one office. A seat var (qing_office_$office$_holder)
# can outlive its holder when a minister dies without the on_character_death handler
# resolving him, or simply LEAVES CHI employ (a departure fires no death event). The
# var then stays SET but points at a dead / foreign character: the GUI renders a filled
# card off the raw .IsSet, while recompute's is_alive+employer guard counts the seat
# vacant — the "13 shown / 12 counted / 1 Unassigned" the boot-test surfaced. Clear the
# var whenever its holder is no longer a living CHI employee. Country scope.
QING_council_prune_seat = {
	if = {
		limit = {
			has_variable = qing_office_$office$_holder
			NOT = { var:qing_office_$office$_holder = { is_alive = yes  employer = ROOT } }
		}
		remove_variable = qing_office_$office$_holder
	}
	# [BT-50 / #396] also RELIEVE a seat whose holder has taken a CONFLICTING role — become an
	# army/navy commander or a provincial governor since he was seated. This is the reported
	# symptom's cure at the START STATE: the 1763 OOB attach (qing_force_setup.1, day 30) makes
	# commanders AFTER the game-start autofill already seated the ablest men, so those councillors
	# are also commanders until this prune (run each recompute + right after the OOB attach) relieves
	# them.
	# [BT-recursion-fix 2026-07-23] MUST use the NO-BACKFILL vacate twin here, NOT QING_office_vacate_dispatch.
	# The appointment redesign made QING_office_vacate_dispatch auto-refill a vacated seat via
	# QING_council_autofill_office — and QING_council_autofill_office calls THIS effect (QING_council_prune_seat)
	# at its top. So `prune_seat -> QING_office_vacate_dispatch -> QING_council_autofill_office -> prune_seat`
	# is an infinite cycle: at a 1763 start the OOB seats commander-ministers, so this branch fires, and the
	# loop recursed past the engine's 16-deep scripted-block limit → startup HANG (error.log:
	# "Too deeply nested scripted block", chain vacate_dispatch->autofill_office->prune_seat repeating).
	# The no-backfill twin only tears the seat down (no autofill_office call), breaking the cycle. Refill is
	# not lost: when prune runs inside QING_council_autofill_office the very next block re-fills the now-empty
	# seat with a fresh created official; when prune runs from a recompute/prune_all sweep the seat is left
	# for re-staffing (consistent with the redesign's "only death/departure auto-creates" rule).
	if = {
		limit = {
			has_variable = qing_office_$office$_holder
			var:qing_office_$office$_holder = {
				is_alive = yes
				employer = ROOT
				OR = { is_general = yes  is_admiral = yes  is_governor = yes }
			}
		}
		var:qing_office_$office$_holder = { QING_office_vacate_dispatch_nobackfill = yes }
	}
}

# Prune ALL thirteen appointable office seats in one call. Country scope. Cheap: O(13)
# var checks. Called at the top of QING_council_recompute so a seat whose holder died /
# departed this tick is cleared BEFORE the GUI and the count read it.
QING_council_prune_all_seats = {
	QING_council_prune_seat = { office = chancellor }
	QING_council_prune_seat = { office = personnel }
	QING_council_prune_seat = { office = revenue }
	QING_council_prune_seat = { office = rites }
	QING_council_prune_seat = { office = war }
	QING_council_prune_seat = { office = justice }
	QING_council_prune_seat = { office = works }
	QING_council_prune_seat = { office = censor }
	QING_council_prune_seat = { office = lifanyuan }
	QING_council_prune_seat = { office = chamberlain }
	QING_council_prune_seat = { office = zongli }
	QING_council_prune_seat = { office = grand_secretariat }
	QING_council_prune_seat = { office = guard_commandant }
}

# Fill ONE office if vacant with the ablest unassigned courtier. Country scope.
# $office$ = the office key. Reuses combined_stats_council_svalue as the ranking
# (the same overall-worth measure the old pool used); the office's own governing
# skill is what then MATTERS for effectiveness (QING_council_recompute), but a
# well-rounded man is the sensible default pick when the throne fills a post.
QING_council_autofill_office = {
	# [BT-appointment-redesign review-fix 2026-07-22] capture the COUNTRY from `this` (CHI in EVERY call
	# path: the day-32 country_event, and inside QING_office_vacate_dispatch's `employer = {}` block). The
	# created official's set_home_country below MUST use this saved country, NOT ROOT — on the runtime
	# backfill paths (on_character_death → QING_office_vacate_dispatch; QING_justice_strip_for_trial) ROOT
	# is the DYING/ACCUSED CHARACTER, not CHI, so `set_home_country = ROOT` would hand a character to a
	# country arg → the new official gets no employer → the following QING_office_appoint's `employer = {}`
	# no-ops and the seat silently fails to refill (the headline death-backfill would be dead). `this` is
	# CHI on all paths, so saving it here is correct and path-agnostic.
	save_scope_as = qing_autofill_country
	# [#352 F3/F8] prune a stale-set seat (holder dead / left CHI employ) before filling,
	# so a var pointing at an invalid character does not read as "already filled" and block
	# the autofill (the "1 Unassigned that Convene never cleared" boot-test bug).
	QING_council_prune_seat = { office = $office$ }
	if = {
		limit = { NOT = { has_variable = qing_office_$office$_holder } }
		# [BT-appointment-redesign 2026-07-22] CREATE a purpose-built official for the vacant seat and
		# assign him — do NOT draw an existing courtier. Assigning existing people is what caused the
		# double-booking mess (autofill running more than once, or grabbing minted/court characters,
		# seated men who were or later became diplomats/censors/commanders). A freshly-created official
		# exists ONLY for this office and is added to NO pool, so nothing else can ever pick him and he
		# can never double-book. Autofill runs EXACTLY ONCE at game start (deferred to day 32, AFTER the
		# historical figures are assigned); a later vacancy is refilled one seat at a time by the vacate
		# handler the same way. $degree$ = the exam degree the office requires (wu_jinshi for the martial
		# War/Guard seats, jinshi for the civil offices) — add_trait inside create_character is #90-safe
		# and off the construction crash path (this runs in a runtime event). religion=confucianism (a
		# scope-chain religion field like root.religion is an invalid create_character field — floods).
		create_character = {
			age = 40
			culture = han
			religion = confucianism
			add_finesse = 6
			add_charisma = 5
			add_martial = 3
			add_zeal = 4
			add_trait = $degree$
			save_scope_as = qing_autofill_official
		}
		scope:qing_autofill_official = {
			set_home_country = scope:qing_autofill_country
			QING_office_appoint = { office = $office$ }
		}
	}
}

# ======================================================================
# RECOMPUTE EFFECTIVENESS
# Effectiveness is a 0..100 blend of:
#   - the average OFFICE-RELEVANT skill of the filled offices (main driver):
#     each domain office contributes its ONE governing skill; the Grand
#     Chancellor (a generalist at the summit) contributes the average of all
#     four. "All skills don't matter for all offices" — a brilliant general is
#     wasted at the Board of Rites.
#   - a dyarchic-balance bonus when Manchu and Han are both represented
#   - a vacancy penalty when appointable offices sit empty
# Also detects the "overmighty grandee" state (a Heshen-style Grand Chancellor)
# for the band applier. All O(offices) counter maths; no pop or province sweeps.
# ======================================================================
# ======================================================================
# [1:1 fix 2026-07-21] QING_validate_one_position — VALIDATION SWEEP for the 1:1 character:position
# rule. That rule is SUPPOSED to be enforced by the appointment/enrol gates, but has been broken by
# accident (boot-test: a Lifan Yuan director ALSO marked Palace Eunuch; a Works minister ALSO an
# Inspecting Censor). This LOGs any CHI character carrying MORE THAN ONE position marker, so a
# violation is CAUGHT + visible in the log rather than silently masked by the card's first-match
# display. Read-only (LOG only — does NOT auto-strip; a strip could hide the real assignment bug).
# Scope: COUNTRY (CHI), called from QING_GOV_pulse. O(court); the count compare uses a LITERAL RHS
# (> 1), the only legal relational-RHS form (see the RHS-comparison rule). Scratch var removed per char.
# ======================================================================
QING_validate_one_position = {
	every_character = {
		limit = { employer = ROOT  is_alive = yes }
		set_variable = { name = qing_pos_marker_ct  value = 0 }
		if = { limit = { has_variable = qing_office_held }           change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_zongli_diplomat }       change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_is_censor_inspector }   change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_is_imperial_guardsman } change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_is_southernstudy }      change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_is_upperstudy }         change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_amban_marker }          change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_is_palace_eunuch }      change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = { limit = { has_variable = qing_is_harem_consort }      change_variable = { name = qing_pos_marker_ct  add = 1 } }
		if = {
			limit = { var:qing_pos_marker_ct > 1 }
			LOG_line = { sys = QING  msg = "1:1 VIOLATION — character holds MORE THAN ONE court position (count above); check appointment/enrol gates for" }
		}
		remove_variable = qing_pos_marker_ct
	}
}

QING_council_recompute = {
	# [#103] se_LOG enter/exit wiring (council-is-offices rewrite)
	LOG_enter = { sys = QING  fn = "QING_council_recompute" }
	QING_council_init = yes

	# [#352 F3/F8] prune any seat whose holder died / left CHI employ this tick, so the
	# GUI (raw .IsSet) and the count (is_alive+employer guard) agree — no phantom filled card.
	QING_council_prune_all_seats = yes

	# --- reset accumulators + rebuild the DERIVED office-holder roster
	set_variable = { name = qing_council_skill_total  value = 0 }
	set_variable = { name = qing_council_filled_count  value = 0 }
	set_variable = { name = qing_council_figurehead_count  value = 0 }
	set_variable = { name = qing_council_manchu_count  value = 0 }
	set_variable = { name = qing_council_han_count  value = 0 }
	# [#308 D] a THIRD ethnic bucket. The old split was Manchu (dynasty culture) vs Han
	# (EVERYTHING else) — which wrongly lumped Mongol, Tibetan and Turkic bannermen / frontier
	# officials in with the Han. The court was in truth Manchu / Han / OTHER (the Lifan Yuan
	# 理藩院 seated Mongol and Tibetan grandees). This counter holds the non-Han, non-Manchu
	# members; the balance bonus below now rewards a genuinely broad court, not a two-bloc one.
	set_variable = { name = qing_council_other_count  value = 0 }
	clear_variable_list = qing_council_members

	# [#157 fix] MANCHU REFERENCE for the 滿漢 split. The score functions compare a
	# holder's culture against the dynasty's own — but `culture = ROOT.primary_culture`
	# is invalid on a character scope (primary_culture is a country TRIGGER, not a
	# navigable culture link), so it PostValidated false and every holder fell to the
	# Han branch (manchu_count stuck at 0 => the +15 滿漢並用 bonus and the
	# qing_council_dyarchic_balance modifier could never fire, and the GUI split read
	# 0/N). The ruler IS the Manchu reference; save him so the proven
	# `culture = scope:X.culture` idiom (00_ambitions.txt:1606) can be used instead.
	if = {
		limit = { exists = current_ruler }
		current_ruler = { save_scope_as = qing_council_manchu_ref }
	}

	# --- score every filled office on its governing skill (rebuilds the roster,
	# the culture split and the filled count as it goes)
	QING_council_score_chancellor = yes
	QING_council_score_office = { office = personnel  skill = finesse }
	QING_council_score_office = { office = revenue    skill = finesse }
	QING_council_score_office = { office = rites      skill = zeal }
	QING_council_score_office = { office = war        skill = martial }
	QING_council_score_office = { office = justice    skill = zeal }
	QING_council_score_office = { office = works      skill = finesse }
	QING_council_score_office = { office = censor      skill = finesse }
	QING_council_score_office = { office = lifanyuan  skill = charisma }
	QING_council_score_office = { office = chamberlain      skill = charisma }
	QING_council_score_office = { office = zongli     skill = charisma }
	# [#273 develop] two new great offices. Grand Secretary of the Central Secretariat
	# (內閣大學士, charisma) and Grand Commandant of the Imperial Guard (領侍衛內大臣, martial).
	QING_council_score_office = { office = grand_secretariat  skill = charisma }
	QING_council_score_office = { office = guard_commandant   skill = martial }

	# --- [#158b] fold the DYNASTIC FIGUREHEADS into the effectiveness base (user
	# request): the Emperor 皇帝, Empress Dowager 皇太后, Crown Prince 皇太子 and
	# Grand Regent 攝政 all shape whether the council can govern, each judged on ALL
	# FOUR skills averaged (like the summit Grand Chancellor). The Emperor carries
	# extra weight (the throne sets the tone above every office); when a minority /
	# incapacity has installed a Grand Regent, the regent — who actually wields the
	# throne's authority — carries that same extra weight. The Dowager and Crown
	# Prince weigh one seat each. Each guarded for presence exactly as the SEATS /
	# DYNASTY code guards them, so an absent figurehead is a clean no-op.
	if = {
		limit = { exists = current_ruler }
		QING_council_score_figurehead = { who = current_ruler  weight = 3  skill = all }
	}
	if = {
		limit = { QING_dynasty_has_dowager = yes }
		QING_council_score_figurehead = { who = current_ruler.mother  weight = 1  skill = all }
	}
	if = {
		limit = { QING_dynasty_has_crownprince = yes }
		QING_council_score_figurehead = { who = var:qing_office_crownprince_holder  weight = 1  skill = all }
	}
	if = {
		limit = {
			has_variable = qing_office_regent_holder
			var:qing_office_regent_holder = { is_alive = yes  employer = ROOT }
		}
		QING_council_score_figurehead = { who = var:qing_office_regent_holder  weight = 3  skill = all }
	}
	# [#272 develop / BT-48] EMPRESS 皇后 (current_ruler.spouse) — the consort sits on the
	# throne row and shapes the court, judged on CHARISMA at weight 1 (a consort's influence is
	# her court sway, not a martial/finesse average). Guarded on the spouse existing + alive as the
	# SEATS code guards her; an unmarried or widowed emperor is a clean no-op.
	if = {
		limit = {
			exists = current_ruler
			exists = current_ruler.spouse
			current_ruler.spouse = { is_alive = yes }
		}
		QING_council_score_figurehead = { who = current_ruler.spouse  weight = 1  skill = charisma }  # [BT-48] Empress judged on charisma (a consort's court sway), not the 4-skill average
	}

	# --- [#270/#271 develop] KEEP EACH SEATED MEMBER'S IMPERIAL-FAVOUR SCORE LIVE.
	# se_QING_AFFINITY.txt's qing_char_affinity (0..100, 50 = neutral) is otherwise only
	# computed inside events on demand, so a minister who has never been through an event
	# has no score for the Grand Council title-row favour square (#270) to read. Re-score
	# every member of the freshly-rebuilt roster here, on the same pulse that recomputes
	# effectiveness, so the square is always current. QING_char_affinity is a PURE READ
	# (no loyalty/relation side effects — that is QING_char_bind, which we deliberately do
	# NOT call here), so this only refreshes the stored variable. Guarded on ruler
	# presence since affinity is scored against root.current_ruler.
	if = {
		limit = { exists = current_ruler }
		every_in_list = {
			variable = qing_council_members
			QING_char_affinity = yes
		}
		# [#270 develop] The dynastic figureheads shown on the throne row (Crown Prince,
		# Regent) are NOT in qing_council_members — they are scored for effectiveness via
		# QING_council_score_figurehead but never appended to the roster list. Their favour
		# squares (#270) still read qing_char_affinity, so re-score them here too, guarded on
		# each being present + alive + still employed. The Emperor is deliberately excluded:
		# affinity is scored against current_ruler, so the reigning Emperor's favour with
		# himself is undefined — his throne card carries no favour square.
		if = {
			limit = {
				has_variable = qing_office_crownprince_holder
				var:qing_office_crownprince_holder = { is_alive = yes  employer = ROOT }
			}
			var:qing_office_crownprince_holder = { QING_char_affinity = yes }
		}
		if = {
			limit = {
				has_variable = qing_office_regent_holder
				var:qing_office_regent_holder = { is_alive = yes  employer = ROOT }
			}
			var:qing_office_regent_holder = { QING_char_affinity = yes }
		}
		# [#272 develop] EMPRESS 皇后 is a FIGUREHEAD (not in qing_council_members) — like the
		# Crown Prince / Regent she needs an explicit affinity re-score for her throne-row favour
		# square. Scored against current_ruler; guarded on the spouse existing + alive.
		if = {
			limit = {
				exists = current_ruler.spouse
				current_ruler.spouse = { is_alive = yes }
			}
			current_ruler.spouse = { QING_char_affinity = yes }
		}
	}

	# --- [#158] recompute the conservative-vs-reformist faction meters over the
	# freshly-rebuilt roster + the three dynastic figureheads (se_QING_FACTION.txt).
	# Must run AFTER the score passes above (which rebuild qing_council_members) and
	# BEFORE the eff_target math below, which reads the deadlock penalty it derives.
	QING_faction_recompute = yes

	# --- [#308 E] SINICIZATION DRIFT of the council. Over the generations the Manchu
	# conquest elite steadily assimilated to Han ways (漢化) — language, dress, letters —
	# a drift the throne alternately fought (國語騎射) and could not stop. Model it on the
	# court itself: once a year, every seated member (and the throne row) edges toward the
	# Han pole, EXCEPT the conservatives — traditionalists and staunch zealots dig in, and
	# some even harden the other way. Self-throttled ~yearly so the drift is generational,
	# not quarterly. Reuses QING_char_shift_identity (se_QING_MECHANICS.txt).
	if = {
		limit = { NOT = { has_variable = qing_council_sinicize_cooldown } }
		set_variable = { name = qing_council_sinicize_cooldown  days = 365 }
		if = {
			limit = { has_variable_list = qing_council_members }
			every_in_list = {
				variable = qing_council_members
				limit = { is_alive = yes  employer = ROOT }
				QING_council_sinicize_drift_one = yes
			}
		}
		# the throne row drifts too (the Qianlong-to-Puyi arc of an ever-more-sinophone court).
		if = { limit = { exists = current_ruler }  current_ruler = { QING_council_sinicize_drift_one = yes } }
		if = {
			limit = {
				exists = current_ruler.spouse
				current_ruler.spouse = { is_alive = yes }
			}
			current_ruler.spouse = { QING_council_sinicize_drift_one = yes }
		}
		# [#308/user] on the same yearly cadence, the council's FACTION alignments drift too:
		# officials occasionally shift reformist/conservative (usually bandwagoning to the
		# ascendant wing, sometimes bucking it) — the mechanic that makes the office-name
		# colours in the GC panel change over a reign. (se_QING_FACTION.txt)
		QING_faction_drift_alignments = yes
	}

	# --- base = weighted-average governing skill of the filled offices AND the
	# dynastic figureheads, scaled to ~0..100. A council of solid specialists
	# (governing skill ~9) lands near the effective band; brilliant men clear it,
	# mediocrities fall to dysfunction. [#158b] the divisor is the office count PLUS
	# the figurehead seat-weights (Emperor 3 + Regent 3 + Dowager 1 + Prince 1),
	# since QING_council_score_figurehead added their weighted skill to the total —
	# so the throne's own ability moves the average, heavily for the Emperor/Regent.
	set_variable = { name = qing_council_eff_denom  value = var:qing_council_filled_count }
	change_variable = { name = qing_council_eff_denom  add = var:qing_council_figurehead_count }
	set_variable = { name = qing_council_eff_target  value = 0 }
	if = {
		limit = { var:qing_council_eff_denom > 0 }
		set_variable = { name = qing_council_eff_target  value = var:qing_council_skill_total }
		change_variable = { name = qing_council_eff_target  divide = var:qing_council_eff_denom }
		change_variable = { name = qing_council_eff_target  multiply = 7 }
	}

	# --- [#308 D] balance bonus: the 滿漢並用 dyarchy (both Manchu AND Han seated) is the
	# core +15; a genuinely BROAD court that ALSO seats a non-Han/non-Manchu grandee (Mongol/
	# Tibetan/Turkic — the multi-ethnic empire governing itself, 因俗而治) earns a further +5.
	if = {
		limit = {
			var:qing_council_manchu_count >= 1
			var:qing_council_han_count >= 1
		}
		change_variable = { name = qing_council_eff_target  add = 15 }
		if = {
			limit = { var:qing_council_other_count >= 1 }
			change_variable = { name = qing_council_eff_target  add = 5 }
		}
	}

	# --- vacancy penalty: 13 appointable offices (12 domain boards + chancellor,
	# [#273] up from 11); -3 per empty one.
	set_variable = { name = qing_council_vacancy  value = 13 }
	change_variable = { name = qing_council_vacancy  subtract = var:qing_council_filled_count }
	if = { limit = { var:qing_council_vacancy < 0 }  set_variable = { name = qing_council_vacancy  value = 0 } }
	set_variable = { name = qing_council_vacancy_pen  value = var:qing_council_vacancy }
	change_variable = { name = qing_council_vacancy_pen  multiply = 3 }
	change_variable = { name = qing_council_eff_target  subtract = var:qing_council_vacancy_pen }

	# --- [#157] dynastic-harmony coupling: the family dynamics of the imperial
	# house (emperor / crown prince / empress dowager) bleed into whether the
	# offices can actually govern. Read qing_dynastic_harmony (0..100, midpoint
	# 50, seeded + moved by se_QING_DYNASTY.txt + the qing_dynasty events) and
	# fold its deviation from the midpoint into the target: a united house (>50)
	# lifts council effectiveness, a court split between the heir's faction and
	# the dowager's clique drags it down. Scaled by /5 so the full 0..100 swing
	# is a +/-10 band — meaningful but never dominant over skill/vacancy.
	if = {
		limit = { has_variable = qing_dynastic_harmony }
		set_variable = { name = qing_council_harmony_mod  value = var:qing_dynastic_harmony }
		change_variable = { name = qing_council_harmony_mod  subtract = 50 }
		change_variable = { name = qing_council_harmony_mod  divide = 5 }
		change_variable = { name = qing_council_eff_target  add = var:qing_council_harmony_mod }
	}

	# --- [#158] faction-deadlock penalty: a Grand Council split hard between a
	# reformist bloc and a conservative bloc governs worse — the gridlock of the
	# 1898-style court where every edict is fought over. qing_council_polar_pen is
	# 0 for a UNITED council (whether solidly reformist OR solidly conservative) and
	# rises (capped at 20) only as the two blocs approach parity. Sits beside the
	# dynastic-harmony coupling: the two "court politics" levers together. Computed
	# in QING_faction_recompute (called above, after the roster rebuild).
	if = {
		limit = { has_variable = qing_council_polar_pen }
		change_variable = { name = qing_council_eff_target  subtract = var:qing_council_polar_pen }
	}

	# --- [#331 user] OFFICER-CORPS coupling: the quality of the standing army's commanders
	# (the historical 1763 garrison generals — Fuheng, Zhaohui, Mingrui et al.) bleeds into
	# how well the War office (兵部) and, for the capital, the Guard Commandant (領侍衛內大臣)
	# can govern the military. Aggregate the MARTIAL of every seated legion commander employed
	# by us: count the corps and sum martial, then fold the AVERAGE-martial deviation from a
	# competent baseline (7) into the target — a corps of brilliant generals lifts effectiveness,
	# a present-but-mediocre officer corps (avg martial < 7) drags it; a corps with NO living
	# commanders at all is a clean no-op (guarded on count > 0, which also avoids /0). Scaled to a modest +/-6 band (avg 0..~15 → dev
	# -7..+8, /1.2) so it never dominates skill/vacancy. Cheap: one every_unit walk, guarded on
	# the War office being filled (no minister = no coupling — the corps has no political voice).
	# Iterator = every_unit + has_commander + commander = { ... } (the codebase-proven country-scope
	# army walk: QING_army_apply_decay_debuffs uses every_unit; military_colonies.txt reads a unit's
	# commander via `commander = { ... }`). We DON'T filter is_navy=no here — a fleet flag officer
	# (the Fujian/Guangdong water-force commanders) is as much a part of the officer corps as a
	# banner general, and the historical naval commanders are seeded too.
	set_variable = { name = qing_officer_corps_count  value = 0 }
	set_variable = { name = qing_officer_corps_martial  value = 0 }
	every_unit = {
		limit = { has_commander = yes  commander = { is_alive = yes  employer = ROOT } }
		commander = {
			# proven idiom (this file, chancellor score ~L582): <char> = { ROOT = { ... add = prev.martial } }
			# — inside the ROOT block, prev = the commander character scope.
			ROOT = {
				change_variable = { name = qing_officer_corps_count  add = 1 }
				change_variable = { name = qing_officer_corps_martial  add = prev.martial }
			}
		}
	}
	if = {
		# only couple when there IS an officer corps AND a War minister to answer for it.
		limit = {
			var:qing_officer_corps_count > 0
			has_variable = qing_office_war_holder
		}
		set_variable = { name = qing_officer_corps_mod  value = var:qing_officer_corps_martial }
		change_variable = { name = qing_officer_corps_mod  divide = var:qing_officer_corps_count }  # average martial
		change_variable = { name = qing_officer_corps_mod  subtract = 7 }    # deviation from the competent baseline
		change_variable = { name = qing_officer_corps_mod  divide = 1.2 }    # scale to ~+/-6 band
		change_variable = { name = qing_council_eff_target  add = var:qing_officer_corps_mod }
	}

	# --- [#347+ / D8] MINISTRY-PERFORMANCE FOLD. Each ministry bureaucracy panel (War #349,
	# Lifan Yuan #350, Works #351, Rites #353, Zongli #354, Revenue #355, Personnel #365,
	# Justice #363, Censorate #362, Central Secretariat #361, Imperial Household #359, Imperial
	# Guard #364) drives one country var qing_min_perf_<office> (0..100, 50 = adequate). A
	# well-run ministry raises its Minister's council standing; a poorly-run one drags it. This
	# single helper averages the deviation-from-50 across every office that BOTH reports a perf
	# var AND has a live filled holder, so the term stays bounded no matter how many panels ship
	# — panels can be added incrementally without destabilising the band. A clean no-op until at
	# least one ministry reports (n=0), so pre-panel saves are unchanged.
	QING_council_fold_ministry_perf = yes

	# --- clamp target 0..100
	if = { limit = { var:qing_council_eff_target < 0 }   set_variable = { name = qing_council_eff_target  value = 0 } }
	if = { limit = { var:qing_council_eff_target > 100 } set_variable = { name = qing_council_eff_target  value = 100 } }

	# --- smooth drift toward target (+/-3 per pulse)
	if = {
		limit = { var:qing_council_effectiveness < qing_council_eff_target_cmpsvalue }
		QING_DECLINE_nudge = { var = qing_council_effectiveness  amount = 3 }
	}
	else_if = {
		limit = { var:qing_council_effectiveness > qing_council_eff_target_cmpsvalue }
		QING_DECLINE_nudge = { var = qing_council_effectiveness  amount = -3 }
	}

	# --- [#273 develop] GUARD COMMANDANT -> EMPEROR POWER BASE (idempotent reconcile).
	# The 領侍衛內大臣 puts the palace guard corps behind the throne, so a seated
	# commandant lifts the EMPEROR's own power base (defensive clout — loyalist civil-war
	# weight + coup deterrent; the ruler can't rebel against himself, so no downside).
	# Modelled with add_loyal_veterans, which is ADDITIVE and would stack if placed in the
	# pulse — so we reconcile it here ONCE PER STATE TRANSITION via a stored recipient-ruler
	# ref (qing_guard_emperor_pb_ruler). STRIP first (office emptied OR the recipient ruler
	# died/changed), then GRANT (office filled + no grant outstanding) so a same-pulse
	# ruler-change re-homes the +4 to the new sovereign in one pass. Recompute runs at the
	# tail of every appoint/vacate, so this behaves as an appoint/vacate grant + self-heals.
	if = {
		limit = {
			has_variable = qing_guard_emperor_pb_ruler
			OR = {
				NOT = { has_variable = qing_office_guard_commandant_holder }
				NOT = { var:qing_office_guard_commandant_holder = { is_alive = yes  employer = ROOT } }
				NOT = { exists = current_ruler }
				NOT = { current_ruler = { this = ROOT.var:qing_guard_emperor_pb_ruler } }
			}
		}
		# only mirror the -4 off the ORIGINAL recipient if he is still alive (a dead man
		# needs no strip); then clear the grant marker so a fresh grant can follow.
		if = {
			limit = { var:qing_guard_emperor_pb_ruler = { is_alive = yes } }
			var:qing_guard_emperor_pb_ruler = { add_loyal_veterans = -4 }
		}
		remove_variable = qing_guard_emperor_pb_ruler
	}
	if = {
		limit = {
			has_variable = qing_office_guard_commandant_holder
			var:qing_office_guard_commandant_holder = { is_alive = yes  employer = ROOT }
			exists = current_ruler
			NOT = { has_variable = qing_guard_emperor_pb_ruler }
		}
		current_ruler = { add_loyal_veterans = 4 }
		set_variable = { name = qing_guard_emperor_pb_ruler  value = current_ruler }
	}

	# --- overmighty-grandee detection: a corrupt realm + a Grand Chancellor who
	# has amassed dangerous personal clout (the Heshen pattern). Clout is read
	# from the engine's own character POWER BASE (holdings + loyal veterans) as
	# well as court prominence — a chancellor with an independent power base is
	# the real threat, not merely a popular one. Re-pointed from the old chief
	# slot onto the Grand Chancellor office-holder.
	set_variable = { name = qing_council_overmighty_flag  value = 0 }
	if = {
		limit = {
			has_variable = qing_office_chancellor_holder
			# [PERF-AUDIT 2026-07-18 C5] guard the corruption read with has_variable, matching every
			# sibling read in se_QING_MINISTRY. Seeded at boot by QING_DECLINE_init so effectively always
			# set, but an unset-var read on a comparison RHS is the illegal-read class; make it explicit.
			has_variable = qing_corruption_level
			var:qing_corruption_level >= 45
			var:qing_office_chancellor_holder = {
				OR = {
					prominence >= 60
					power_base >= 15
				}
			}
		}
		set_variable = { name = qing_council_overmighty_flag  value = 1 }
	}
	LOG_exit = { sys = QING  fn = "QING_council_recompute"  result = OK }
}

# ======================================================================
# [#308 D] CLASSIFY A HOLDER's ETHNICITY into one of THREE buckets. Call INSIDE the
# holder's character scope (this = the office-holder); ROOT = CHI. Manchu = shares the
# dynasty's own culture (the conquest elite); Han = the chinese_group (the literati-
# official majority); Other = everyone else (Mongol/Tibetan/Turkic bannermen + frontier
# grandees, esp. the Lifan Yuan). Uses the proven scope-link culture compare + the
# has_culture_group idiom (both verified in-mod). Bumps the matching country counter.
# ======================================================================
QING_council_classify_ethnicity = {
	if = {
		limit = {
			exists = scope:qing_council_manchu_ref
			culture = scope:qing_council_manchu_ref.culture
		}
		ROOT = { change_variable = { name = qing_council_manchu_count  add = 1 } }
	}
	else_if = {
		limit = { has_culture_group = chinese_group }
		ROOT = { change_variable = { name = qing_council_han_count  add = 1 } }
	}
	else = {
		ROOT = { change_variable = { name = qing_council_other_count  add = 1 } }
	}
}

# ======================================================================
# [#308 E] SINICIZATION DRIFT — nudge ONE character's Manchu identity. Call INSIDE the
# character scope (this = the member / throne-row figure). Most of the court drifts toward
# the Han pole (漢化); the conservatives resist or even harden. Guarded so it only acts on
# characters who actually carry a Manchu identity to lose (Han/other members sit near the
# sinicized floor already and QING_char_shift_identity clamps at 0, so calling them is a
# cheap no-op). Reuses QING_char_shift_identity (se_QING_MECHANICS.txt; +=Manchu, -=Han).
# ======================================================================
QING_council_sinicize_drift_one = {
	QING_char_init_identity = yes
	# CONSERVATIVES dig in: a traditionalist, a fervent zealot, or an already-staunch
	# Manchu (identity >= 66) resists the tide — and the hardest cases harden further.
	if = {
		limit = {
			OR = {
				has_trait = traditionalist
				zeal >= 12
				var:qing_manchu_identity >= 80
			}
		}
		# the staunchest reassert the Manchu Way (a small hardening); the merely conservative
		# simply hold the line (no drift this year).
		if = {
			limit = { OR = { has_trait = traditionalist  var:qing_manchu_identity >= 80 } }
			QING_char_shift_identity = { amount = 2 }
		}
	}
	else = {
		# the rest of the court drifts Han-ward. A reformer drifts faster (he embraces it).
		if = {
			limit = { has_trait = reformer }
			QING_char_shift_identity = { amount = -5 }
		}
		else = {
			QING_char_shift_identity = { amount = -3 }
		}
	}
}

# Score one DOMAIN office on its single governing skill. Country scope.
# $office$ = office key, $skill$ = martial|finesse|charisma|zeal. Adds the
# holder's governing skill to the running total, counts him toward the filled
# count and the Manchu/Han split, and appends him to the derived roster list.
QING_council_score_office = {
	if = {
		limit = {
			has_variable = qing_office_$office$_holder
			var:qing_office_$office$_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_$office$_holder = {
			# add the governing skill onto the country total (prev = country here,
			# the proven prev.<skill> value-read idiom from the old list recompute).
			ROOT = {
				change_variable = { name = qing_council_skill_total  add = prev.$skill$ }
				change_variable = { name = qing_council_filled_count  add = 1 }
				add_to_variable_list = { name = qing_council_members  target = prev }
			}
			# culture balance (滿漢): dynasty-culture holders count Manchu, others Han.
			# [#157 fix] compare against the saved ruler's culture via the proven
			# `culture = scope:X.culture` link idiom (ROOT.primary_culture is invalid here).
			# [#308 D] three-bucket 滿漢蒙回藏 classification (was Manchu / else-Han).
			QING_council_classify_ethnicity = yes
		}
	}
}

# Score the Grand Chancellor (領班軍機大臣) — a summit generalist judged on ALL
# FOUR skills, so his contribution is the average of the four (kept comparable
# to a single-skill domain office: one seat, one averaged contribution).
QING_council_score_chancellor = {
	if = {
		limit = {
			has_variable = qing_office_chancellor_holder
			var:qing_office_chancellor_holder = { is_alive = yes  employer = ROOT }
		}
		set_variable = { name = qing_council_tmp  value = 0 }
		var:qing_office_chancellor_holder = {
			ROOT = {
				change_variable = { name = qing_council_tmp  add = prev.martial }
				change_variable = { name = qing_council_tmp  add = prev.finesse }
				change_variable = { name = qing_council_tmp  add = prev.charisma }
				change_variable = { name = qing_council_tmp  add = prev.zeal }
				add_to_variable_list = { name = qing_council_members  target = prev }
			}
			# [#157 fix] see QING_council_score_office: scope-link culture compare.
			# [#308 D] three-bucket 滿漢蒙回藏 classification (was Manchu / else-Han).
			QING_council_classify_ethnicity = yes
		}
		change_variable = { name = qing_council_tmp  divide = 4 }
		change_variable = { name = qing_council_skill_total  add = var:qing_council_tmp }
		change_variable = { name = qing_council_filled_count  add = 1 }
	}
}

# ======================================================================
# [#158b] SCORE A DYNASTIC FIGUREHEAD onto the council's effectiveness base.
# The Emperor (皇帝), Empress Dowager (皇太后), Crown Prince (皇太子) and Grand
# Regent (攝政) all sit above the Grand Council and shape whether it can govern —
# so each PRESENT figurehead contributes their ability to the effectiveness
# TARGET, judged (like the summit Grand Chancellor) on ALL FOUR skills averaged.
# Country scope; $who$ = a character scope expression (current_ruler /
# current_ruler.mother / var:qing_office_*_holder), $weight$ = seat-equivalents
# (a positive integer). Adds his 4-skill average, MULTIPLIED by $weight$, to
# qing_council_skill_total and counts $weight$ toward the separate
# qing_council_figurehead_count — so a heavier figurehead pulls the base average
# harder in BOTH numerator and denominator (a wise Emperor lifts the council more,
# a foolish one drags it more). The Emperor (皇帝) and, when the post is filled,
# the Grand Regent (攝政) carry extra weight — the throne (or whoever wields it in
# a minority) sets the tone above all the great offices; the Dowager and Crown
# Prince weigh one seat each. Deliberately does NOT touch qing_council_filled_count
# (which drives the office-vacancy penalty) nor the Manchu/Han office split —
# figureheads are not appointable offices. Mirrors the prev.<skill> value-read
# idiom of QING_council_score_chancellor exactly.
# ======================================================================
QING_council_score_figurehead = {
	set_variable = { name = qing_council_tmp  value = 0 }
	# [logfix] the branch selector was `"$skill$" = charisma` (quoted string macro vs bareword) which
	# the engine misread as an unknown trigger (charisma$/all$) at load. Seed the passed skill as a flag
	# scratch var, then compare var: = flag: (proven idiom, QING_techtransfer_dispatch).
	set_variable = { name = qing_fig_skill  value = flag:$skill$ }
	# [BT-48] OPTIONAL $skill$ = a single governing attribute (e.g. charisma for the Empress —
	# a consort's sway over the court is her charisma, not a martial/finesse/zeal average). When
	# passed, score that one skill; otherwise fall back to the all-four-skill average used by the
	# Emperor / Crown Prince / Regent / Dowager (they are judged on every competence).
	if = {
		limit = { exists = $who$  var:qing_fig_skill = flag:charisma }
		$who$ = { ROOT = { change_variable = { name = qing_council_tmp  add = prev.charisma } } }
	}
	else_if = {
		limit = { exists = $who$  var:qing_fig_skill = flag:finesse }
		$who$ = { ROOT = { change_variable = { name = qing_council_tmp  add = prev.finesse } } }
	}
	else_if = {
		limit = { exists = $who$  var:qing_fig_skill = flag:martial }
		$who$ = { ROOT = { change_variable = { name = qing_council_tmp  add = prev.martial } } }
	}
	else_if = {
		limit = { exists = $who$  var:qing_fig_skill = flag:zeal }
		$who$ = { ROOT = { change_variable = { name = qing_council_tmp  add = prev.zeal } } }
	}
	else_if = {
		# default: the all-four-skill average. [BOOT-CRASHFIX 2026-07-18] MUST guard `exists = $who$`
		# exactly like the four skill-specific branches above — this else was UNGUARDED, so scoping
		# `$who$ = { ... }` on a $who$ that resolves to a non-existent character (a figurehead var
		# holding a dead/cleared handle, or current_ruler.mother on a motherless sovereign that slips
		# a caller guard) is a null-scope deref = EXCEPTION_ACCESS_VIOLATION at QING_council_recompute
		# (fired at on_game_initialized). The four typed branches never crashed because each carries
		# its own `exists = $who$`; only the all-skill default (the path Emperor/Dowager/Crown Prince/
		# Regent take) was exposed. Convert else -> guarded else_if so an absent figurehead is a clean no-op.
		limit = { exists = $who$ }
		$who$ = {
			ROOT = {
				change_variable = { name = qing_council_tmp  add = prev.martial }
				change_variable = { name = qing_council_tmp  add = prev.finesse }
				change_variable = { name = qing_council_tmp  add = prev.charisma }
				change_variable = { name = qing_council_tmp  add = prev.zeal }
			}
		}
		change_variable = { name = qing_council_tmp  divide = 4 }
	}
	# apply the seat-weight (both to the skill total and the count, so the average
	# is a true weighted mean over offices + figureheads).
	change_variable = { name = qing_council_tmp  multiply = $weight$ }
	change_variable = { name = qing_council_skill_total  add = var:qing_council_tmp }
	change_variable = { name = qing_council_figurehead_count  add = $weight$ }
}

# ======================================================================
# BAND APPLICATION — effectiveness bands + the two "character" states.
# ======================================================================
QING_council_apply_band = {
	# [#103] se_LOG enter/exit wiring (council-is-offices rewrite)
	LOG_enter = { sys = QING  fn = "QING_council_apply_band" }
	remove_country_modifier = qing_council_effective
	remove_country_modifier = qing_council_dysfunctional
	remove_country_modifier = qing_council_overmighty
	remove_country_modifier = qing_council_dyarchic_balance

	if = {
		limit = { var:qing_council_effectiveness >= 66 }
		add_country_modifier = { name = qing_council_effective  duration = -1 }
		LOG_line = { sys = QING  msg = "band council = effective for (effectiveness=)" }
	}
	else_if = {
		limit = { var:qing_council_effectiveness <= 33 }
		add_country_modifier = { name = qing_council_dysfunctional  duration = -1 }
		LOG_line = { sys = QING  msg = "band council = dysfunctional for (effectiveness=)" }
	}

	# balance + overmighty are independent overlays on the effectiveness band
	if = {
		limit = {
			var:qing_council_manchu_count >= 1
			var:qing_council_han_count >= 1
		}
		add_country_modifier = { name = qing_council_dyarchic_balance  duration = -1 }
	}
	if = {
		limit = { var:qing_council_overmighty_flag = 1 }
		add_country_modifier = { name = qing_council_overmighty  duration = -1 }
		# [#91 item F] CONCRETE FACE of the overmighty grandee. The counter/flag above
		# stays the AI/summary layer; here we hang a REAL object off it. If the
		# overmighty Grand Chancellor has ALSO turned disloyal he stops being a mere
		# loyalty number and raises a personal army — a visible rebel-army seed whose
		# cohorts answer to HIM, not the throne (the loyal-cohorts civil-war seed).
		# The flip is guarded inside the effect so a merely-popular loyal chancellor
		# never revolts. Re-pointed from the old chief slot onto the Grand Chancellor.
		if = { limit = { has_variable = qing_office_chancellor_holder }  var:qing_office_chancellor_holder = { QING_council_grandee_flip = yes } }
	}
	LOG_exit = { sys = QING  fn = "QING_council_apply_band"  result = OK }
}

# ======================================================================
# [#91 item F] THE OVERMIGHTY GRANDEE TURNS — CONCRETE REBEL-ARMY SEED
# Called inside the Grand Chancellor's CHARACTER scope from QING_council_apply_band
# (this = the 領班 chancellor; ROOT = CHI). The overmighty flag is the AI/summary
# layer; this hangs the REAL object off it. If the overmighty chancellor has ALSO
# turned disloyal (loyalty < 40, the AFFINITY "estranged" band), his personal power
# base — the loyal veterans the office conferred — becomes a REAL personal army:
# a legion raised under him whose cohorts are set_personal_loyalty to HIM, not the
# throne (the loyal-cohorts civil-war seed — the Heshen/warlord danger given a
# military face). Idiom proven: raise_legion+create_unit+add_subunit and the
# add_to_legion/set_as_commander commander attach (se_QING_SELFSTR.txt:320-342,
# se_QING_NAPOLEON.txt:357-375); every_sub_unit + set_personal_loyalty on the raised
# cohorts (common/unit_abilities/military_colonies.txt:57-69; TI warfare_events.txt:269-271).
# Fires ONCE per chancellor (qing_grandee_rebelled marker). KEEPS all council counters:
# effectiveness / overmighty flag / roster are untouched here — we only convert
# part of his abstract loyal-veteran pool into the concrete legion (mirror -N).
# ======================================================================
QING_council_grandee_flip = {
	LOG_enter = { sys = QING  fn = "QING_council_grandee_flip" }
	if = {
		limit = {
			is_alive = yes
			loyalty < 40
			NOT = { has_variable = qing_grandee_rebelled }
		}
		# fire-once marker on the man himself.
		set_variable = { name = qing_grandee_rebelled  value = 1 }
		save_scope_as = grandee_rebel_commander
		# convert part of his abstract power base into the concrete legion: the loyal
		# veterans the office granted march off as his personal army (mirror of the
		# +4 office grant). The 0..100 council counters are KEPT.
		add_loyal_veterans = -5
		# raise the real force under him. Prefer the capital; guarded so a landless Qing
		# still gets the marker (and the log) without a scope error.
		if = {
			limit = { exists = ROOT.capital_scope }
			ROOT.capital_scope = { state.governorship = { QING_council_raise_grandee_legion = yes } }
			LOG_line = { sys = QING  msg = "grandee flip: overmighty chancellor turned — personal army raised at capital (cohorts loyal to HIM) for" }
		}
		else = {
			LOG_fail = { sys = QING  fn = "QING_council_grandee_flip"  reason = "no capital_scope; grandee marked rebel but no legion raised for" }
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_council_grandee_flip"  reason = "chancellor still loyal (loyalty>=40) or already flipped; no rebel army for" }
	}
	LOG_exit = { sys = QING  fn = "QING_council_grandee_flip"  result = OK }
}

# [#91 item F] Raise the grandee's personal legion. Called in governorship scope;
# scope:grandee_rebel_commander = the flipped chancellor. Mirrors QING_selfstr_raise_evarmy_legion:
# base sub_unit + add_subunit, commander attached via add_to_legion=PREV +
# set_as_commander. The one addition is the loyal-cohorts bind: every raised sub_unit
# is set_personal_loyalty to the grandee, so the cohorts answer to HIM — the concrete
# civil-war seed rather than a loyalty number.
QING_council_raise_grandee_legion = {
	raise_legion = {
		create_unit = {
			# [#91-fix task #91] quoted LITERAL display name — the proven in-mod idiom
			# (QING_selfstr_raise_fleet name="Beiyang Fleet", se_QING_SELFSTR.txt:180;
			# Invictus name="Basilike Taxis"). The former "qing_grandee_rebel_army" relied
			# on an unquoted-key auto-resolve that this mod never demonstrates, so it would
			# have shown the raw string; its loc line has been retired.
			name = "A Grandee's Private Army"
			location = ROOT.capital_scope
			sub_unit = regular_infantry
			save_scope_as = qing_grandee_legion
			add_subunit = regular_infantry
			add_subunit = regular_infantry
			add_subunit = regular_infantry
			# [#91-fix task #91] was 'archers' — an UNDEFINED unit type (army_archers.txt is
			# a 3-byte BOM-only stub, no 'archers = {}' anywhere in common/units); it threw a
			# runtime error and dropped the 5th cohort. regular_infantry is the mod's only
			# line-infantry sub_unit, matching the other four cohorts of this field host.
			add_subunit = regular_infantry
		}
		if = {
			limit = { exists = scope:grandee_rebel_commander }
			scope:grandee_rebel_commander = { add_to_legion = PREV }
			random_legion_unit = { set_as_commander = scope:grandee_rebel_commander }
		}
	}
	# bind the raised cohorts to the grandee personally (loyal-cohorts civil-war seed).
	if = {
		limit = { exists = scope:qing_grandee_legion  exists = scope:grandee_rebel_commander }
		scope:qing_grandee_legion = {
			every_sub_unit = { set_personal_loyalty = scope:grandee_rebel_commander }
		}
	}
}

# [grand-council-tab] CANDIDATE CACHE for the Grand Council view's appoint
# pickers. The government-view GUI has no engine courtier datamodel, and the
# office system stores holders as character vars, so the tab needs a
# GUI-iterable list of ELIGIBLE courtiers to populate its pickers. We cache it
# on the country as qing_council_candidates and let the GUI walk it via
# MakeScope.GetList('qing_council_candidates').
#
# COST: one every_character walk over the court, bounded to the top
# QING_COUNCIL_CANDIDATE_CAP courtiers by combined_stats_council_svalue. Court is
# dozens-to-low-hundreds of characters — trivial beside the economy sim's
# province/pop sweeps. It is refreshed LAZILY: only when the player opens the tab
# (QING_council_refresh_candidates called from the tab-open scripted_gui) and
# once on the quarterly QING_GOV_pulse. NEVER on a monthly/per-character hot path.
#
# Eligibility: adult, alive, employed by ROOT, and NOT already holding a great
# office (office-holders are shown in their own roster slots, not offered as
# fresh candidates).
# ======================================================================
QING_council_refresh_candidates = {
	# Scope: country (CHI).
	LOG_enter = { sys = QING  fn = "QING_council_refresh_candidates" }
	QING_council_init = yes
	clear_variable_list = qing_council_candidates
	# top-N by council skill; check_range_bounds=no so the ordered set is not
	# silently pruned to a default window (matches QING_council_autofill_office).
	ordered_character = {
		limit = {
			employer = ROOT
			is_adult = yes
			is_alive = yes
			# [#8 fix 2026-07-18] EXCLUDE harem consorts (後宮妃嬪). A consort is created employed by CHI,
			# adult and alive (se_QING_HAREM.txt QING_harem_mint_consort), so without this she matched the
			# generic bench filter and flowed into the shared Appoint picker; appointing her stamped
			# qing_zongli_diplomat and she surfaced in the Zongli diplomat roster — the reported
			# 'diplomat list replaced by harem women' bug. The inner court is not a pool of officials.
			NOT = { has_variable = qing_is_harem_consort }
			# [#6 fix 2026-07-17] EXCLUDE the conflicting NON-council roles the appoint verb's is_valid
			# (QING_office_eligible_candidate) rejects — a serving army/navy commander, a provincial
			# governor, the reigning ruler, the primary heir, or a vanilla great-office holder. Without
			# this the picker LISTED such men (e.g. a general like Sun Yuting/Zhang Xu) but the row-click
			# silently failed is_valid — the reported 'listed but un-appointable' bug. Cross-COUNCIL-office
			# holders stay (one-office-per-man reshuffle is intended); only the conflicting roles are cut.
			is_ruler = no
			is_general = no
			is_admiral = no
			is_governor = no
			NOT = { AND = { exists = ROOT.primary_heir  this = ROOT.primary_heir } }
			NOT = { has_office = office_foreign_minister }
			NOT = { has_office = office_royal_tutor }
			NOT = { has_office = office_marshal }
			NOT = { has_office = office_master_of_the_guard }
			NOT = { has_office = office_high_priest_monarchy }
			NOT = { has_office = office_philosopher }
			NOT = { has_office = office_steward }
			NOT = { has_office = office_physician }
			# [R2-7 fix] MUST match the inclusive eligibility of QING_council_refresh_candidates_by
			# (the B7 fix): do NOT exclude ALL office-holders — autofill staffs every office at start,
			# so `NOT has_variable qing_office_held` left the list EMPTY (the boot-test empty picker).
			# This plain refresh runs from the pulse + autofill, so if the window happens to read a
			# pulse-built list it must be populated too. Only the two dynastic FIGUREHEAD seats that
			# ride on qing_office_held — Grand Regent (攝政) + Emperor Emeritus (太上皇) — stay excluded
			# (they hold by regency/abdication, not appointment). Cross-office holders are valid
			# reshuffle candidates (QING_office_appoint one-office-per-man relieves the prior seat).
			NOT = {
				AND = {
					has_variable = qing_office_held
					OR = { var:qing_office_held = flag:regent  var:qing_office_held = flag:emeritus }
				}
			}
			# [BT-diplomat=minister fix 2026-07-22] When filling a SUB-POST (駐外使臣/御史/侍衛),
			# exclude every sitting great-office holder — a great office + sub-post are mutually
			# exclusive, and the sub-post stamp path does NOT relieve the great office, so a sitting
			# minister appointed as a diplomat was left double-booked. Fires only for the three
			# sub-post target flags; council reshuffle candidates are unaffected. (Mirror of the
			# guard added to QING_council_refresh_candidates_by.)
			NOT = {
				AND = {
					ROOT = { has_variable = qing_gc_picker_office_var }
					OR = {
						ROOT = { var:qing_gc_picker_office_var = flag:zongli_diplomat }
						ROOT = { var:qing_gc_picker_office_var = flag:censor_inspector }
						ROOT = { var:qing_gc_picker_office_var = flag:imperial_guardsman }
					}
					has_variable = qing_office_held
				}
			}
		}
		order_by = combined_stats_council_svalue
		check_range_bounds = no
		max = 12   # QING_COUNCIL_CANDIDATE_CAP — picker shows the ablest dozen
		# the list is COUNTRY-held: add via ROOT with the courtier as prev (mirror of
		# the appoint verb's `employer = { ... target = prev }`).
		ROOT = { add_to_variable_list = { name = qing_council_candidates  target = prev } }
	}
	LOG_line = { sys = QING  msg = "council candidates refreshed for" }
	LOG_exit = { sys = QING  fn = "QING_council_refresh_candidates"  result = OK }
}

			# governor, the reigning ruler, the primary heir, or a vanilla great-office holder. Without

# ======================================================================
# OFFICER CHARACTER-BUFFS applied by the quarterly pulse (after recompute, so the
# derived roster qing_council_members is fresh):
#   • PERSONNEL 吏部 — the Grand Minister of Personnel grooms the officialdom.
#     He lends FINESSE to his FELLOW office-holders, scaled by his OWN finesse
#     (>=12 -> +2, >=6 -> +1, else none). Applied to every OTHER filled office.
#   • GRAND SECRETARY 內務府 — the closest man to the throne sharpens the
#     sovereign's own judgement, buffing ALL FOUR of the RULER's skills, scaled
#     by the Grand Secretary's OWN charisma (>=12 -> +2, >=6 -> +1).
# Both are re-derived from scratch each pulse: strip from all current holders /
# the ruler, then re-apply. (Vacate + death also strip, so a dropped holder never
# keeps a stale buff.)
# ======================================================================
QING_council_apply_officer_buffs = {
	LOG_enter = { sys = QING  fn = "QING_council_apply_officer_buffs" }

	# --- PERSONNEL finesse buff -------------------------------------------------
	# strip from every current office-holder (clean slate)
	if = {
		limit = { has_variable_list = qing_council_members }
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes }
			remove_character_modifier = qing_personnel_cultivated_minor
			remove_character_modifier = qing_personnel_cultivated_major
		}
	}
	# re-apply based on the Personnel holder's own finesse, to every OTHER holder
	if = {
		limit = {
			has_variable = qing_office_personnel_holder
			var:qing_office_personnel_holder = { is_alive = yes  employer = ROOT  finesse >= 12 }
		}
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  NOT = { this = ROOT.var:qing_office_personnel_holder } }
			add_character_modifier = { name = qing_personnel_cultivated_major  duration = -1 }
		}
	}
	else_if = {
		limit = {
			has_variable = qing_office_personnel_holder
			var:qing_office_personnel_holder = { is_alive = yes  employer = ROOT  finesse >= 6 }
		}
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  NOT = { this = ROOT.var:qing_office_personnel_holder } }
			add_character_modifier = { name = qing_personnel_cultivated_minor  duration = -1 }
		}
	}

	# --- GRAND SECRETARY counsel buff on the RULER ------------------------------
	if = {
		limit = { exists = current_ruler }
		current_ruler = {
			remove_character_modifier = qing_grand_secretary_counsel_minor
			remove_character_modifier = qing_grand_secretary_counsel_major
		}
	}
	if = {
		limit = {
			exists = current_ruler
			has_variable = qing_office_chamberlain_holder
			var:qing_office_chamberlain_holder = { is_alive = yes  employer = ROOT  charisma >= 12 }
		}
		current_ruler = { add_character_modifier = { name = qing_grand_secretary_counsel_major  duration = -1 } }
	}
	else_if = {
		limit = {
			exists = current_ruler
			has_variable = qing_office_chamberlain_holder
			var:qing_office_chamberlain_holder = { is_alive = yes  employer = ROOT  charisma >= 6 }
		}
		current_ruler = { add_character_modifier = { name = qing_grand_secretary_counsel_minor  duration = -1 } }
	}

	# --- [#273 develop] CENSORATE oversight buff -------------------------------
	# The Grand Inspector 都察院 steadies his FELLOW council members' personal
	# loyalty (monthly_character_loyalty) — the character-scope replacement for the
	# state-loyalty grant his office modifier used to carry. Mirrors the Personnel
	# finesse buff EXACTLY: strip both tiers from every current member, then re-apply
	# one tier to every OTHER member, scaled by the censor's own finesse (>=12 major,
	# >=6 minor). Runs on the same fresh qing_council_members roster.
	if = {
		limit = { has_variable_list = qing_council_members }
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes }
			remove_character_modifier = qing_censorate_oversight_minor
			remove_character_modifier = qing_censorate_oversight_major
		}
	}
	if = {
		limit = {
			has_variable = qing_office_censor_holder
			var:qing_office_censor_holder = { is_alive = yes  employer = ROOT  finesse >= 12 }
		}
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  NOT = { this = ROOT.var:qing_office_censor_holder } }
			add_character_modifier = { name = qing_censorate_oversight_major  duration = -1 }
		}
	}
	else_if = {
		limit = {
			has_variable = qing_office_censor_holder
			var:qing_office_censor_holder = { is_alive = yes  employer = ROOT  finesse >= 6 }
		}
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  NOT = { this = ROOT.var:qing_office_censor_holder } }
			add_character_modifier = { name = qing_censorate_oversight_minor  duration = -1 }
		}
	}

	# --- [#273 develop] IMPERIAL GUARD sovereign-vitality buff on the RULER -----
	# The Grand Commandant 領侍衛內大臣 keeps the emperor's person safe and well — a
	# small pulse-managed health boost on current_ruler, scaled by the commandant's
	# own martial (>=12 major, >=6 minor). Strip-then-reapply like the counsel buff so
	# a dropped/weakened commandant never leaves a stale grant. (His OTHER contribution
	# — the emperor's POWER BASE — is a one-time add_loyal_veterans grant on appoint /
	# strip on vacate, NOT here: add_loyal_veterans is additive and would stack every pulse.)
	if = {
		limit = { exists = current_ruler }
		current_ruler = {
			remove_character_modifier = qing_guard_vitality_minor
			remove_character_modifier = qing_guard_vitality_major
		}
	}
	if = {
		limit = {
			exists = current_ruler
			has_variable = qing_office_guard_commandant_holder
			var:qing_office_guard_commandant_holder = { is_alive = yes  employer = ROOT  martial >= 12 }
		}
		current_ruler = { add_character_modifier = { name = qing_guard_vitality_major  duration = -1 } }
	}
	else_if = {
		limit = {
			exists = current_ruler
			has_variable = qing_office_guard_commandant_holder
			var:qing_office_guard_commandant_holder = { is_alive = yes  employer = ROOT  martial >= 6 }
		}
		current_ruler = { add_character_modifier = { name = qing_guard_vitality_minor  duration = -1 } }
	}
	LOG_exit = { sys = QING  fn = "QING_council_apply_officer_buffs"  result = OK }
}

# ======================================================================
# GREAT OFFICES  (領班軍機大臣 / 六部尚書 / 都察院 / 理藩院 / 總理衙門 / 內務府)
# Each office is a single character seat. Appointing a capable holder grants a
# country modifier from qing_governance_modifiers.txt; a vacancy strains that
# domain. $office$ = chancellor|personnel|revenue|rites|war|justice|works|censor|lifanyuan|chamberlain|zongli
# Call QING_office_appoint inside the appointee's character scope; the country
# is reached via `employer` so it is call-site-agnostic.
# QING_office_vacate is call-site-agnostic too: call it in EITHER the country
# scope or the holder's character scope — it resolves the country via a helper.
# ======================================================================
QING_office_appoint = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = QING  fn = "QING_office_appoint" }
	# $office$ = the office key.  this = the appointee.
	# [#175 fix] ONE-OFFICE-PER-MAN. If the appointee ALREADY holds a different great
	# office, relieve him of it FIRST — otherwise the old office's
	# qing_office_<old>_holder country var keeps pointing at him while his
	# qing_office_held flag is overwritten to the new office, so he silently "holds"
	# both (only the newest is vacatable). QING_office_vacate_dispatch reads his
	# current qing_office_held flag and cleanly tears down whichever office that is
	# (clears the country holder var, strips his officeholder modifier + the +4 power
	# base, adds vacancy strain). Guarded to a DIFFERENT office so re-appointing to
	# the same seat (a no-op refresh) doesn't needlessly churn. The primary guard is
	# the GUI is_valid gate (a man holding any office is not offered elsewhere), so
	# this is the belt-and-suspenders for event / autofill call paths.
	if = {
		limit = {
			has_variable = qing_office_held
			NOT = { var:qing_office_held = flag:$office$ }
		}
		# [BT-appointment-redesign 2026-07-22] use the NO-BACKFILL dispatcher: appointing a sitting
		# minister to another seat is a MANUAL reshuffle, and the user's rule is "clicking Appoint
		# creates NO characters." The backfilling QING_office_vacate_dispatch would spawn a fresh
		# official for the seat this man just left; the no-backfill twin tears it down and leaves it
		# empty for the player to fill by hand (only death/departure auto-creates a replacement).
		QING_office_vacate_dispatch_nobackfill = yes
	}
	employer = {
		# vacate the previous holder of this office if any
		if = {
			limit = { has_variable = qing_office_$office$_holder }
			var:qing_office_$office$_holder = {
				remove_character_modifier = qing_officeholder
				# [#103] strip the officer character-buffs the pulse may have hung on
				# the displaced man, so he leaves office clean (no stale finesse/counsel).
				remove_character_modifier = qing_personnel_cultivated_minor
				remove_character_modifier = qing_personnel_cultivated_major
				# [#273 develop] also strip the Censorate oversight loyalty buff, so a
				# displaced holder keeps no stale personal-loyalty grant.
				remove_character_modifier = qing_censorate_oversight_minor
				remove_character_modifier = qing_censorate_oversight_major
				remove_variable = qing_office_held
				# the displaced holder loses the power base the office gave him
				# (mirror of the +4 the new appointee gains below).
				add_loyal_veterans = -4
			}
		}
		remove_country_modifier = qing_office_vacancy_strain
		set_variable = { name = qing_office_$office$_holder  value = prev }
		add_country_modifier = { name = qing_office_$office$_active  duration = -1 }
	}
	add_character_modifier = { name = qing_officeholder  duration = -1 }
	set_variable = { name = qing_office_held  value = flag:$office$ }
	# [#364-R2 review-fix] a man seated in a great office leaves any inner-court CORPS he served
	# in — the corps rosters (QING_ministry_recompute_perf_*) all exclude office-holders
	# (NOT has_variable=qing_office_held), so a lingering corps mark would let an ex-officer
	# silently rejoin the corps on dismissal WITHOUT passing its enrol gate, bypassing the
	# Imperial Guard's hard cap of 6 (enrol-count excludes office-holders, so an officed guardsman
	# is a hidden 7th). Strip all three corps marks here so the enrol caps stay exact. Self-guarded
	# no-ops for the ~all appointees who were never in a corps.
	if = { limit = { has_variable = qing_is_imperial_guardsman } remove_variable = qing_is_imperial_guardsman }
	if = { limit = { has_variable = qing_is_censor_inspector }   remove_variable = qing_is_censor_inspector }
	if = { limit = { has_variable = qing_zongli_diplomat }       remove_variable = qing_zongli_diplomat }
	# [#336/#337 review-fix] the inner-court ACADEMIES (南書房 Southern Study #336, 上書房 Upper
	# Study #337) are the same shape: their rosters exclude office-holders + enrol under a hard cap
	# (8 / 6), so an elevated 首席/總師傅 who kept his study mark would silently rejoin the corps on
	# a later dismissal, bypassing the cap (the 擢用 Elevate lever is exactly that promote-out path).
	if = { limit = { has_variable = qing_is_southernstudy }      remove_variable = qing_is_southernstudy }
	if = { limit = { has_variable = qing_is_upperstudy }         remove_variable = qing_is_upperstudy }
	# [1:1 fix 2026-07-21] ALSO strip the palace-eunuch (內廷太監) + amban (駐紮大臣) markers — the two
	# sub-position markers the strip block above MISSED. Boot-test: the Grand Director of the Lifan Yuan
	# rendered ALSO as "Palace Eunuch"; a man promoted INTO a great office kept his old sub-post marker,
	# so the character-card court-position title (which reads these raw markers) showed two positions (a
	# 1:1 violation). Strip both here so the rule holds. Self-guarded no-ops for non-eunuch/non-amban appointees.
	if = { limit = { has_variable = qing_is_palace_eunuch }      remove_variable = qing_is_palace_eunuch }
	if = { limit = { has_variable = qing_amban_marker }          remove_variable = qing_amban_marker }
	# [#5 fix 2026-07-17; BOOT-CRASHFIX 2026-07-18] MARTIAL-OFFICE EXAM CONGRUENCE — the wu_jinshi
	# grant that USED to live here was REMOVED. QING_office_appoint is reached at GAMESTATE
	# CONSTRUCTION (on_game_initialized -> QING_council_autofill -> QING_council_autofill_office{war,
	# guard_commandant} -> here; the war/guard seats are not pre-seated, so autofill fills them at
	# boot). add_trait mutates the character's trait/opposites machinery (the appointee usually holds
	# the civil jinshi, an OPPOSITE of wu_jinshi), and a trait mutation on a character during
	# construction access-violates with NO logged line — the exact class of the two prior boot
	# crashfixes (b385452a4 castrated eunuch, ade0b498f). At the last CLEAN boot QING_office_appoint
	# did NO add_trait; the surrounding add_character_modifier / add_loyal_veterans are simple attaches
	# that don't touch trait recomputation and boot fine. The martial-degree congruence is now granted
	# at RUNTIME instead — qing_force_setup.1 (day 30, is_triggered_only) stamps wu_jinshi on the
	# serving war/guard office-holders alongside the field commanders it already handles.
	# heading a great office of state builds the holder a power base of his own —
	# the board or yamen brings a staff, clients, and clout (power_base). Grant it
	# here; QING_office_vacate strips it symmetrically when he is relieved.
	add_loyal_veterans = 4
	LOG_line = { sys = QING  msg = "office $office$ -> (+4 loyal veterans -> power base) for" }
	# [#321 review-fix] if this appointee was an examination-pool scholar, drop him from
	# the hiring pool NOW — the single pool-exit-on-seating path, so EVERY appointment
	# route (autofill, manual, laureate, keju.6) keeps the pool + count honest and never
	# leaves a seated man listed/counted. Self-guards on qing_is_pool_scholar (a no-op
	# for the ~all appointees who were never pool scholars). (se_QING_EXAM.txt)
	QING_exam_pool_drop_member = yes
	# [#175 fix] rebuild the derived roster AND the eligible-candidate cache right away
	# so the GUI updates immediately: the just-appointed man drops out of the Eligible
	# Courtiers list (its filter is NOT has_variable=qing_office_held, now set on him)
	# and appears in his office slot, without waiting for the next tab-open / pulse.
	# [logfix BT-13/#373] MUST run the recompute with ROOT = CHI, not ROOT = the appointee.
	# This effect is called from the character-scope GUI verbs (QING_governance_actions.txt),
	# so ROOT = the clicked courtier; the recompute machinery is written for ROOT = CHI and
	# its employer=ROOT guards otherwise compared country-vs-character (2072 err/boot) AND
	# mis-pruned live seats (the mass-vacate). Fire it through the hidden CHI country_event
	# qing_office.40 (re-roots to CHI), exactly as the pulse / event callers root it.
	employer = { trigger_event = { id = qing_office.40 } }
	# [BT-N fix 2026-07-20] MARTIAL-OFFICE EXAM CONGRUENCE on EVERY seating, not just day 30.
	# The War Minister (兵部) and Commandant of the Guard (領侍衛內大臣) should read as 武科 laureates,
	# but the wu_jinshi grant only lived in qing_force_setup.1 (a ONE-SHOT day-30 event). These two
	# seats ROTATE (autofill re-seats them as holders die/reshuffle), so every holder appointed after
	# day 30 was a civil-degree courtier who never got the martial degree — the reported BT-N bug
	# (boot log 2026-07-20 proved the day-30 grant itself works: 30/30 wu_jinshi STUCK). Grant it here
	# on EVERY appointment instead, via a RUNTIME trampoline (qing_office.41), NOT inline: an add_trait
	# in QING_office_appoint runs at gamestate CONSTRUCTION (autofill) and access-violates on the
	# trait/opposites machinery — the exact crash the 1284-1294 note documents. trigger_event defers
	# past construction (same safety qing_office.40 relies on). No office-gate needed: the event reads
	# the war/guard holder vars directly + self-guards on "already has a wuju degree", so it is a
	# cheap no-op when the appointment was to some other office (or the holder is already a laureate).
	employer = { trigger_event = { id = qing_office.41 } }
	LOG_exit = { sys = QING  fn = "QING_office_appoint"  result = OK }
}

# loyalty; the tie there is the marriage_subject_tie, not a lobby). Fires qing_office.30 against the
# first such member found, saving the councillor + her homeland as scopes. Self-throttled by the
# pulse's own 90-day cadence + a low chance in the on_action; re-checks presence so an empty court is
# a clean no-op. LOG-instrumented per the standing rule.
# ======================================================================
QING_council_foreign_spouse_check = {
	LOG_enter = { sys = QING  fn = "QING_council_foreign_spouse_check" }
	if = {
		limit = {
			has_variable_list = qing_council_members
			# at least one seated member is a foreign spouse whose homeland still stands
			any_in_list = {
				variable = qing_council_members
				is_alive = yes
				has_variable = marriage_origin_country
				var:marriage_origin_country = {
					exists = yes
					NOT = { this = ROOT }
					NOT = { is_subject_of = ROOT }   # a subject is bound by the marriage_subject_tie, not a lobby
				}
			}
		}
		# pick ONE qualifying councillor (bounded single fire per pulse). random_in_list is the
		# proven variable-list single-pick idiom (ordered_in_list in this codebase is only used with
		# a named `list =` + order_by; the members roster is a variable_list iterated via `variable =`).
		random_in_list = {
			variable = qing_council_members
			limit = {
				is_alive = yes
				has_variable = marriage_origin_country
				var:marriage_origin_country = {
					exists = yes
					NOT = { this = ROOT }
					NOT = { is_subject_of = ROOT }
				}
			}
			save_scope_as = qing_lobby_councillor
			var:marriage_origin_country = { save_scope_as = qing_lobby_homeland }
		}
		if = {
			limit = { exists = scope:qing_lobby_councillor  exists = scope:qing_lobby_homeland }
			ROOT = { set_variable = { name = qing_gc_event_slot_used  value = 1 } }  # [BT-10] claim the quarterly GC-event slot on the fire path only (this check self-guards, so a no-op pulse leaves the slot free for another GC event)
			trigger_event = { id = qing_office.30 }
			LOG_line = { sys = QING  msg = "foreign-spouse council lobby fired (a councillor petitions for her homeland)" }
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_council_foreign_spouse_check"  reason = "no foreign-spouse councillor with an extant independent homeland" }
	}
	LOG_exit = { sys = QING  fn = "QING_council_foreign_spouse_check"  result = OK }
}

# ======================================================================
# [#331 user] QING_council_officer_report_check — quarterly roll: the Grand Minister of War
# (兵部尚書) reports to the throne on a serving garrison commander. If the chosen commander leads
# the CAPITAL garrison (a legion at Beijing p:8363), the report instead comes from the Grand
# Commandant of the Imperial Guard (領侍衛內大臣, guard_commandant) who answers for the metropolitan
# banners. Scope: COUNTRY (CHI). Self-guards: needs a War minister seated (or Guard Commandant for
# the capital case) AND at least one living legion commander in our employ. Fires qing_office.31.
# ======================================================================
QING_council_officer_report_check = {
	LOG_enter = { sys = QING  fn = "QING_council_officer_report_check" }
	if = {
		limit = {
			has_variable = qing_office_war_holder
			any_character = { is_general = yes  is_alive = yes  employer = ROOT }
		}
		# [logfix] pick the reporting officer as a CHARACTER, not via random_unit -> commander.
		# A scope saved off a unit's `commander` LINK (the prior `random_unit = { commander =
		# { save_scope_as } }`) resolves in script triggers/effects (exists/portrait/add_loyalty
		# all worked) but is NOT promotable by pdx_data_factory for a loc/LOG `.GetName` read:
		# `[scope:officer_report_commander.GetName]` threw "Failed to find type ... Could not find
		# promote" ~79k/session (in the desc, both tooltips, and the immediate LOG), while the
		# minister — saved from a COUNTRY VARIABLE — resolved fine. A character saved via
		# random_character IS promotable (proven: qing_war.1's able_commander, 0 loc errors;
		# is_general is a valid random_character filter, internal_politics_monarchy.txt). A general
		# is by definition a live unit commander, so this picks the same population.
		random_character = {
			limit = { is_general = yes  is_alive = yes  employer = ROOT }
			save_scope_as = officer_report_commander
			# is this the CAPITAL garrison? (his unit stationed on the Qing capital province) —
			# a character's `.location` is his unit's province (proven, recruit_general.txt:85).
			if = {
				limit = { exists = ROOT.capital_scope  exists = location  location = { this = ROOT.capital_scope } }
				ROOT = { set_variable = { name = officer_report_is_capital  value = 1 } }
			}
		}
		if = {
			limit = { exists = scope:officer_report_commander }
			# choose the reporting minister: Guard Commandant for the capital garrison (if seated),
			# else the War minister.
			if = {
				limit = {
					has_variable = officer_report_is_capital
					has_variable = qing_office_guard_commandant_holder
					var:qing_office_guard_commandant_holder = { is_alive = yes  employer = ROOT }
				}
				var:qing_office_guard_commandant_holder = { save_scope_as = officer_report_minister }
			}
			else_if = {
				# match the file-wide holder idiom: only a LIVING, employed war minister reports.
				limit = { var:qing_office_war_holder = { is_alive = yes  employer = ROOT } }
				var:qing_office_war_holder = { save_scope_as = officer_report_minister }
			}
			if = {
				limit = { exists = scope:officer_report_minister }
				ROOT = { set_variable = { name = qing_gc_event_slot_used  value = 1 } }  # [BT-10] claim the quarterly GC-event slot on the fire path only (this check self-guards, so a no-op pulse leaves the slot free for another GC event)
				trigger_event = { id = qing_office.31 }
				LOG_line = { sys = QING  msg = "War-minister officer report fired on [scope:officer_report_commander.GetName]" }
			}
		}
		if = { limit = { has_variable = officer_report_is_capital }  remove_variable = officer_report_is_capital }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_council_officer_report_check"  reason = "no War minister or no living legion commander" }
	}
	LOG_exit = { sys = QING  fn = "QING_council_officer_report_check"  result = OK }
}

# ======================================================================
# [#331 user] QING_council_on_commander_death — lifecycle: when a serving legion COMMANDER dies,
# the Grand Minister of War brings a replacement recommendation before the throne (qing_office.32).
# Scope: the DYING CHARACTER (called from on_character_death). Guards: the dead char was a legion
# commander in CHI's employ (had qing_officer_marker, stamped when attached — see below) AND a War
# minister is seated to make the recommendation. Employer = CHI.
# ======================================================================
QING_council_on_commander_death = {
	if = {
		limit = {
			has_variable = qing_officer_marker
			# [review-fix] only a general who STILL held a command at death fires the "a commander
			# falls, a garrison stands without its general" event — a marked officer long since
			# relieved / reassigned to a civil post is not a falling commander. is_general is the
			# engine's "leads a unit" test (00_hardcoded.txt).
			is_general = yes
			exists = employer
			employer = {
				tag = CHI
				has_variable = qing_office_war_holder
				var:qing_office_war_holder = { is_alive = yes }
			}
			# [review-fix] guard the minister-is-not-the-corpse overlap: if the dying character IS
			# himself the seated war minister, there is no living minister to bring a recommendation.
			NOT = { this = employer.var:qing_office_war_holder }
		}
		employer = {
			var:qing_office_war_holder = { save_scope_as = officer_report_minister }
			trigger_event = { id = qing_office.32 }
			LOG_line = { sys = QING  msg = "a serving garrison commander died; War-minister replacement recommendation fired" }
		}
	}
}

# ======================================================================
# [#347+ / D8] QING_council_fold_ministry_perf — the MINISTRY-PERFORMANCE FOLD.
# Scope: COUNTRY (ROOT = CHI), called from QING_council_recompute after the officer-
# corps fold. Each ministry bureaucracy panel drives one 0..100 var qing_min_perf_<office>
# (50 = adequate). This folds the NET average deviation-from-50 across all reporting
# offices into qing_council_eff_target, so a well-run set of ministries lifts the
# council and a poorly-run set drags it — and each Minister answers on the Grand
# Council for how his own ministry is run (the D3 coupling).
#
# BOUNDED BY AVERAGING: we sum (perf-50) only over offices that BOTH have the perf var
# set AND a live, employed office-holder, count them, and add avg/5 (so the whole fold
# stays within ~[-10,+10] however many ministries report). n=0 -> clean no-op (pre-panel
# saves unchanged). The applied term is stored in qing_min_perf_council_mod so the GC
# tooltip + each panel can surface "this ministry's contribution to the council".
# All O(offices) counter maths; no pop/province sweeps.
# ======================================================================
QING_council_fold_ministry_perf = {
	set_variable = { name = qing_min_perf_sum    value = 0 }
	set_variable = { name = qing_min_perf_count  value = 0 }
	# accumulate each reporting office's deviation-from-50 (guarded on a live holder).
	QING_council_perf_accumulate = { office = war }
	QING_council_perf_accumulate = { office = personnel }
	QING_council_perf_accumulate = { office = revenue }
	QING_council_perf_accumulate = { office = rites }
	QING_council_perf_accumulate = { office = justice }
	QING_council_perf_accumulate = { office = works }
	QING_council_perf_accumulate = { office = censor }
	QING_council_perf_accumulate = { office = lifanyuan }
	QING_council_perf_accumulate = { office = chamberlain }
	QING_council_perf_accumulate = { office = zongli }
	QING_council_perf_accumulate = { office = grand_secretariat }
	QING_council_perf_accumulate = { office = guard_commandant }
	# fold the bounded average into the effectiveness target.
	if = {
		limit = { var:qing_min_perf_count > 0 }
		set_variable = { name = qing_min_perf_council_mod  value = var:qing_min_perf_sum }
		change_variable = { name = qing_min_perf_council_mod  divide = var:qing_min_perf_count }  # average deviation
		change_variable = { name = qing_min_perf_council_mod  divide = 5 }                        # scale to ~+/-10 band
		change_variable = { name = qing_council_eff_target  add = var:qing_min_perf_council_mod }
		LOG_line = { sys = QING  msg = "ministry-performance fold applied to council effectiveness target" }
	}
	else = {
		# no ministry has reported a performance var yet: the fold contributes nothing.
		set_variable = { name = qing_min_perf_council_mod  value = 0 }
	}
}

# ----------------------------------------------------------------------
# QING_council_perf_accumulate — add ONE office's (perf-50) to the running sum + count,
# but ONLY when that office both reports a perf var AND has a live, employed holder (an
# unstaffed or unreported ministry contributes nothing). Scope: COUNTRY. $office$ = key.
# ----------------------------------------------------------------------
QING_council_perf_accumulate = {
	if = {
		limit = {
			has_variable = qing_min_perf_$office$
			has_variable = qing_office_$office$_holder
			var:qing_office_$office$_holder = { is_alive = yes  employer = ROOT }
		}
		change_variable = { name = qing_min_perf_sum   add = var:qing_min_perf_$office$ }
		change_variable = { name = qing_min_perf_sum   subtract = 50 }
		change_variable = { name = qing_min_perf_count add = 1 }
		# [council-contribution truthful per-office display] Store THIS office's OWN scaled
		# deviation-from-adequate: qing_min_perf_council_mod_<office> = (perf-50)/5, a ±10 signed
		# term the office's own ministry panel surfaces as "this ministry's contribution to the
		# council" — the realm-wide qing_min_perf_council_mod is the AVERAGE of these, so a single
		# office's share is exactly this figure before the /count averaging. (Dividing by count too
		# would round to 0 under integer maths, so the per-office read is the pre-average pull.)
		set_variable = { name = qing_min_perf_council_mod_$office$  value = var:qing_min_perf_$office$ }
		change_variable = { name = qing_min_perf_council_mod_$office$  subtract = 50 }
		change_variable = { name = qing_min_perf_council_mod_$office$  divide = 5 }
	}
	else = {
		# unreported / unstaffed office contributes nothing — keep its display term at 0.
		set_variable = { name = qing_min_perf_council_mod_$office$  value = 0 }
	}
}

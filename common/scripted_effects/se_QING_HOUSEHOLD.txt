# se_QING_HOUSEHOLD.txt — the Grand Secretary's charge: the Imperial Household
# Department (內務府 / 總管內務府大臣), the privy purse (內帑), the eunuchs
# (太監), and the imperial workshops (造辦處).
#
# The Grand Secretary (內務府) heads the throne's PERSONAL establishment — the
# privy purse (內帑, the emperor's private treasury, distinct from the state
# 戶部 treasury), the palace eunuchs (太監, a perennial danger — the Qing kept
# them tightly checked after past dynasties fell to eunuch power), and the
# imperial workshops (造辦處). The closest man to the sovereign; his charge is
# the emperor's own household and standing.
#
# A capable, loyal, high-charisma Grand Secretary keeps the privy purse full
# (a private reserve the throne can tap), the eunuchs in check (no eunuch
# faction), and the workshops productive (prestige/luxury); a venal or vacant
# office means eunuch overreach, a drained privy purse, and a diminished sovereign.
#
# STATE (CHI country scope):
#   [#424] THE PRIVY PURSE IS NOW current_ruler.wealth — the emperor's REAL vanilla character
#   wealth (內帑 = the sovereign's private silver, exactly what `wealth` models), NOT a 0..100
#   country meter. Every purse flow is add_gold on the emperor; every read is his wealth. Scale:
#   the old meter baseline 50 maps to a comfortable ruler wealth ~200 (old×4); the 內帑 passes to
#   the successor on a change of sovereign (QING_household_carryover_purse, on_ruler_change).
#   qing_eunuch_power      0..100 faction strength (NOT stored — derived from
#                          the presence of a flagged eunuch character; 0 when
#                          no eunuch faction exists)
# Character-scope mark on the faction-leading eunuch (if one exists):
#   qing_eunuch_faction_leader   flag (the dangerous eunuch)
#
# PREFIX: QING_household_
########################################################################

# ======================================================================
# INIT — seed the privy purse. Called once at game start from on_game_initialized.
# ======================================================================
QING_household_init = {
	# [#424] the privy purse is now the emperor's own vanilla wealth (current_ruler.wealth),
	# which the engine already seeds, so there is nothing to initialise. Kept as a no-op stub
	# so the many callers (QING_household_pulse etc.) need no change. The old 0..100
	# qing_privy_purse country meter is RETIRED; the 內帑 is the sovereign's real private silver.
	LOG_enter = { sys = QING  fn = "QING_household_init" }
	# [boot #12] seed the opening palace-eunuch corps (太監) once at game start.
	QING_household_seed_eunuchs = yes
	LOG_exit = { sys = QING  fn = "QING_household_init"  result = OK }
}

# ======================================================================
# [boot #12] PALACE-EUNUCH CORPS (太監). The user saw "The Eunuchs" as a field in the
# Household panel but only a threat-meter stood behind it — no actual eunuch characters.
# The Qing kept a large palace establishment of eunuchs (the 敬事房 administered them);
# they are the concrete objects the eunuch mechanic should manage (concrete-over-abstract
# house rule), and the faction-danger mechanic promotes one of THESE rather than conjuring a
# fresh stranger. Seed a small opening corps + keep a self-healing roster, mirroring the
# proven harem consort roster (se_QING_HAREM.txt QING_harem_mint_consort / _recompute_roster).
#
# STATE:  qing_household_eunuchs  country variable_list of the living palace eunuchs
#         qing_eunuch_count       their count (GUI + perf reads)
# Char:   qing_is_palace_eunuch   marker flag on each eunuch character
# ======================================================================
QING_household_seed_eunuchs = {
	if = {
		limit = { NOT = { has_variable = qing_eunuchs_initialized } }
		set_variable = { name = qing_eunuchs_initialized  value = 1 }
		set_variable = { name = qing_eunuch_count  value = 0 }
		clear_variable_list = qing_household_eunuchs
		# an opening establishment of four palace eunuchs (only if the throne is filled).
		if = {
			limit = { exists = current_ruler }
			QING_household_mint_eunuch = yes
			QING_household_mint_eunuch = yes
			QING_household_mint_eunuch = yes
			QING_household_mint_eunuch = yes
		}
		LOG_line = { sys = QING  msg = "household: seeded the opening palace-eunuch corps for" }
	}
}

# QING_household_mint_eunuch — create ONE palace eunuch and add him to the roster.
# Scope: COUNTRY (CHI). Palace eunuchs were overwhelmingly Han commoners bound to the
# throne's household. #90 rule: NO modifiers granted inside create_character; markers set
# in a follow-up scope. Mirrors QING_harem_mint_consort.
QING_household_mint_eunuch = {
	create_character = {
		age = 30
		culture = han            # palace eunuchs were overwhelmingly Han commoners
		religion = confucianism
		# a eunuch's gifts are administrative/social, not martial.
		add_finesse = 4
		add_charisma = 3
		add_zeal = 1
		# [B12] a palace eunuch IS a eunuch — give him the castrated trait (the only eunuch trait
		# the game ships, common/traits/00_health.txt:179). MUST be set INSIDE create_character (the
		# proven idiom — se_QING_AMBAN/se_QING_EXAM/se_MEXICO all add_trait here): granting a trait
		# to the character in a SEPARATE follow-up scope is the #90 crash pattern (access-violation
		# at gamestate construction — this effect seeds 4 eunuchs at boot via QING_household_init).
		add_trait = castrated
		save_scope_as = qing_new_eunuch
	}
	scope:qing_new_eunuch = {
		set_home_country = ROOT
		set_variable = { name = qing_is_palace_eunuch  value = 1 }
	}
	add_to_variable_list = { name = qing_household_eunuchs  target = scope:qing_new_eunuch }
	change_variable = { name = qing_eunuch_count  add = 1 }
	LOG_line = { sys = QING  msg = "household: a eunuch enters the palace establishment for" }
}

# QING_household_recompute_eunuchs — rebuild the eunuch roster fresh from the live
# characters carrying the marker (alive + still in the household). Self-healing (prunes the
# dead/departed), keeps the count honest for the GUI. Bounded O(characters); run on the
# household pulse + on panel open. Scope: COUNTRY (CHI). Mirrors QING_harem_recompute_roster.
QING_household_recompute_eunuchs = {
	clear_variable_list = qing_household_eunuchs
	set_variable = { name = qing_eunuch_count  value = 0 }
	every_character = {
		limit = {
			has_variable = qing_is_palace_eunuch
			is_alive = yes
			employer = ROOT
		}
		save_scope_as = qing_roster_eunuch
		ROOT = {
			add_to_variable_list = { name = qing_household_eunuchs  target = scope:qing_roster_eunuch }
			change_variable = { name = qing_eunuch_count  add = 1 }
		}
	}
}

# ======================================================================
# [#424] QING_household_carryover_purse — the 內帑 is an institution of the throne: on a
# change of sovereign it passes to the successor, not into the void with the dead emperor's
# personal wealth. Called from on_ruler_change (scope:former_ruler is set there). Reads the
# former ruler's wealth into a country var, then grants it to the new current_ruler — the
# proven inheritance idiom (TI on_character_death_inv: set_variable value=<X.wealth>, then
# add_gold = { value = <var> }). Scope: COUNTRY (CHI).
# ======================================================================
QING_household_carryover_purse = {
	LOG_enter = { sys = QING  fn = "QING_household_carryover_purse" }
	if = {
		limit = { exists = scope:former_ruler  exists = current_ruler }
		set_variable = { name = qing_purse_carryover_tmp  value = scope:former_ruler.wealth }
		# only carry a POSITIVE balance (a bankrupt emperor bequeaths no privy purse; the new
		# sovereign is not saddled with the dead one's personal debts).
		if = {
			limit = { var:qing_purse_carryover_tmp > 0 }
			current_ruler = { add_gold = { value = ROOT.var:qing_purse_carryover_tmp } }
			LOG_line = { sys = QING  msg = "household: the privy purse passes to the new sovereign for" }
		}
		remove_variable = qing_purse_carryover_tmp
	}
	LOG_exit = { sys = QING  fn = "QING_household_carryover_purse"  result = OK }
}

# ======================================================================
# PULSE — quarterly drift of the privy purse and eunuch danger, gated on the
# Grand Secretary holder's charisma + loyalty. Called from the quarterly
# governance pulse (after accountability recompute, so the metric is fresh).
# ======================================================================
QING_household_pulse = {
	LOG_enter = { sys = QING  fn = "QING_household_pulse" }
	QING_household_init = yes
	# [boot #12] refresh the palace-eunuch roster (prunes the dead/departed, keeps the count honest).
	QING_household_recompute_eunuchs = yes

	# --- PRIVY PURSE DRIFT: capable loyal high-charisma Grand Secretary fills it
	# (+2/quarter), vacant or venal office drains it (-3/quarter).
	if = {
		limit = {
			has_variable = qing_office_chamberlain_holder
			var:qing_office_chamberlain_holder = {
				is_alive = yes
				employer = ROOT
				charisma >= 8
				loyalty >= 60
			}
		}
		if = { limit = { exists = current_ruler }  current_ruler = { add_gold = 8 } }
		LOG_line = { sys = QING  msg = "household pulse: privy purse +8 wealth (capable Grand Secretary) for" }
	}
	else_if = {
		limit = {
			OR = {
				NOT = { has_variable = qing_office_chamberlain_holder }
				var:qing_office_chamberlain_holder = {
					OR = {
						NOT = { is_alive = yes }
						NOT = { employer = ROOT }
						loyalty < 40
						has_trait = corrupt
					}
				}
			}
		}
		if = { limit = { exists = current_ruler }  current_ruler = { add_gold = -12 } }
		LOG_line = { sys = QING  msg = "household pulse: privy purse -12 wealth (vacant/venal office) for" }
	}

	# --- EUNUCH DANGER CHECK: if the Grand Secretary is absent/weak/disloyal AND
	# corruption is high, a eunuch faction MAY arise (one-time spawn when no faction
	# leader exists yet). CONCRETE: create a real eunuch character, flag him, and
	# grant him a power base (the eunuch danger is a man with clout, not a counter).
	if = {
		limit = {
			var:qing_corruption_level >= 50
			OR = {
				NOT = { has_variable = qing_office_chamberlain_holder }
				var:qing_office_chamberlain_holder = {
					OR = {
						NOT = { is_alive = yes }
						NOT = { employer = ROOT }
						charisma < 6
						loyalty < 50
					}
				}
			}
			# no eunuch faction already exists
			NOT = {
				any_character = {
					employer = ROOT
					has_variable = qing_eunuch_faction_leader
				}
			}
		}
		random = {
			chance = 20
			LOG_line = { sys = QING  msg = "household pulse: eunuch faction arising (corrupt + weak Grand Secretary) for" }
			# [boot #12] CONCRETE: the faction leader rises from the EXISTING palace-eunuch corps
			# (the danger is one of the seeded 太監 growing overmighty, not a stranger materialising).
			# Guard against an empty corps (all dead/departed) by minting one first if needed.
			if = {
				limit = {
					NOT = {
						any_character = { employer = ROOT  has_variable = qing_is_palace_eunuch  is_alive = yes }
					}
				}
				QING_household_mint_eunuch = yes
			}
			# promote the most capable serving eunuch to faction leader + give him a power base.
			ordered_character = {
				limit = {
					employer = ROOT
					has_variable = qing_is_palace_eunuch
					is_alive = yes
					NOT = { has_variable = qing_eunuch_faction_leader }
				}
				order_by = finesse
				max = 1
				set_variable = { name = qing_eunuch_faction_leader  value = 1 }
				add_trait = ambitious
				add_loyal_veterans = 3
				add_gold = 50
				add_corruption = 20
				LOG_line = { sys = QING  msg = "household pulse: a palace eunuch leads a faction (power base +3) for" }
			}
		}
	}

	LOG_exit = { sys = QING  fn = "QING_household_pulse"  result = OK }
}

# ======================================================================
# REPLENISH PRIVY PURSE — top up the emperor's private silver. $amount$ (int, 10-30)
# and $cost$ (treasury gold or prestige). Called from events. Country scope.
# ======================================================================
QING_household_replenish_purse = {
	LOG_enter = { sys = QING  fn = "QING_household_replenish_purse" }
	# [#424] grant the replenishment to the emperor's own purse (his wealth). $amount$ is the
	# old-meter delta; callers pass ×4 wealth values now (see qing_household_events).
	if = { limit = { exists = current_ruler }  current_ruler = { add_gold = $amount$ } }
	add_treasury = $cost$
	LOG_line = { sys = QING  msg = "household: privy purse += $amount$ (cost $cost$ treasury) for" }
	LOG_exit = { sys = QING  fn = "QING_household_replenish_purse"  result = OK }
}

# ======================================================================
# TAP PRIVY PURSE — the throne draws on the private reserve to bail out the state
# treasury (a crisis measure). $relief$ (treasury gold gained), $prestige_cost$
# (the personal-prestige price of raiding the privy purse). Drains the purse by
# half. Country scope.
# ======================================================================
QING_household_tap_purse = {
	LOG_enter = { sys = QING  fn = "QING_household_tap_purse" }
	# [#424] the throne raids its OWN privy purse (the emperor's wealth) to bail out the state.
	# Drain HALF the sovereign's wealth into a country var, then subtract it from him. Proven
	# inheritance/transfer idiom (TI on_character_death_inv: read wealth into a var, add_gold={value}).
	if = {
		limit = { exists = current_ruler }
		set_variable = { name = qing_purse_drain_tmp  value = { value = current_ruler.wealth  multiply = 0.5 } }
		if = { limit = { var:qing_purse_drain_tmp < 0 }  set_variable = { name = qing_purse_drain_tmp  value = 0 } }
		current_ruler = { add_gold = { value = 0  subtract = ROOT.var:qing_purse_drain_tmp } }
	}
	add_treasury = $relief$
	if = {
		limit = { exists = current_ruler }
		current_ruler = {
			add_popularity = $prestige_cost$
			LOG_line = { sys = QING  msg = "household: privy purse tapped (ruler popularity $prestige_cost$) for" }
		}
	}
	LOG_exit = { sys = QING  fn = "QING_household_tap_purse"  result = OK }
}

# ======================================================================
# EUNUCH FACTION CHECK — the throne deals with the dangerous eunuch. Character
# scope (this = the eunuch). $outcome$ = check|indulge|instrument.
#   check    — the Qing way: strip his power base hard (remove loyal veterans,
#              confiscate wealth), clear the faction, nudge corruption down.
#   indulge  — buy him off: grant wealth, he stays powerful (danger deferred).
#   instrument — co-opt him as the throne's private agent: he becomes a rival
#              of the Grand Secretary (friction), gains loyal veterans, but
#              loyalty to the throne rises (the dirty-work lever).
# ======================================================================
QING_household_eunuch_check = {
	LOG_enter = { sys = QING  fn = "QING_household_eunuch_check" }
	if = {
		limit = { has_variable = qing_eunuch_faction_leader }
		# strip the flag + power base
		remove_variable = qing_eunuch_faction_leader
		add_loyal_veterans = -3
		add_gold = -50
		add_corruption = -20
		employer = {
			QING_DECLINE_nudge = { var = qing_corruption_level  amount = -3 }
		}
		LOG_line = { sys = QING  msg = "household: eunuch checked (power stripped) for" }
	}
	LOG_exit = { sys = QING  fn = "QING_household_eunuch_check"  result = OK }
}

QING_household_eunuch_indulge = {
	LOG_enter = { sys = QING  fn = "QING_household_eunuch_indulge" }
	add_gold = 80
	add_corruption = 10
	employer = {
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 2 }
	}
	LOG_line = { sys = QING  msg = "household: eunuch indulged (wealth +80, danger deferred) for" }
	LOG_exit = { sys = QING  fn = "QING_household_eunuch_indulge"  result = OK }
}

QING_household_eunuch_instrument = {
	LOG_enter = { sys = QING  fn = "QING_household_eunuch_instrument" }
	add_loyal_veterans = 2
	add_loyalty = loyalty_qing_delta_p20
	# [#126 review-fix] set him as rival of the Grand Secretary (if one exists) —
	# friction between the private instrument and the official household head. The
	# prior code used `add_rival = prev` / `is_rival = prev` from inside
	# employer{}->var:...holder{}; since if/limit do not push scope, `prev` there
	# resolved to the COUNTRY (one level up), not the eunuch — so the rivalry was
	# created against the country and the guards tested the wrong scope. Save the
	# eunuch explicitly and reference him by scope so the intended eunuch<->Grand
	# Secretary rivalry is actually formed.
	save_scope_as = qing_eunuch_instrument
	if = {
		limit = {
			employer = {
				has_variable = qing_office_chamberlain_holder
				var:qing_office_chamberlain_holder = {
					is_alive = yes
					employer = ROOT
					NOT = { is_rival = scope:qing_eunuch_instrument }
				}
			}
		}
		employer = {
			var:qing_office_chamberlain_holder = {
				add_rival = scope:qing_eunuch_instrument
			}
		}
	}
	LOG_line = { sys = QING  msg = "household: eunuch co-opted as instrument (loyal veterans +2, loyalty +20) for" }
	LOG_exit = { sys = QING  fn = "QING_household_eunuch_instrument"  result = OK }
}

# ======================================================================
# WORKSHOP COMMISSION — the Grand Secretary's charisma gates a prestige project
# (the imperial workshops craft a luxury / great work). $prestige_gain$, $cost$.
# Country scope.
# ======================================================================
QING_household_workshop_commission = {
	LOG_enter = { sys = QING  fn = "QING_household_workshop_commission" }
	add_treasury = $cost$
	add_legitimacy = $prestige_gain$
	add_country_modifier = { name = qing_household_workshop_flourish  duration = 1825 }
	LOG_line = { sys = QING  msg = "household: workshop commission (legitimacy +$prestige_gain$, cost $cost$) for" }
	LOG_exit = { sys = QING  fn = "QING_household_workshop_commission"  result = OK }
}

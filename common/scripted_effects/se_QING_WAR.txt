# se_QING_WAR.txt — the Grand Minister of War (兵部) and military commanders.
#
# Historical anchor: the 兵部 administered the Green Standard Army (綠營),
# military examinations (武舉), the courier relays, and officer appointments.
# Army and navy commanders were its charges. The 19th-century danger: regional
# armies (湘軍/淮軍) whose commanders answered to their patrons, not the throne.
#
# COUPLING to the Grand Minister of War (兵部):
#   - Commanders (characters with military power_base, the loyal-cohorts model)
#     are scored against the War office-holder via QING_pair_friction (the same
#     agree/disagree mechanism governors/residents use).
#   - A capable, congenial War minister keeps his commanders loyal and effective
#     (loyalty up, martial cultivated).
#   - A vacant War office or a clashing pair → the commander drifts toward
#     personal loyalty (the warlord seed); loyalty risk, possible review event.
#   - The effect GUARDS every deref (is_alive / employer / has_variable checks)
#     and does NOT double-count the War accountability metric (war exhaustion /
#     cohort count already measured in se_QING_ACCOUNTABILITY.txt).
#
# State (all on CHI):
#   qing_office_war_holder    the character holding the War office (from se_QING_COUNCIL.txt)
#
# Events (qing_war namespace):
#   qing_war.1   officer review / promotion board under capable War minister
#   qing_war.2   commander too close to his troops (warlord danger)
#   qing_war.3   naval/army commission decision gated on War minister
#   qing_war.4   military-examination (武舉) intake of fresh officers
#
# Prefix = QING_war_
########################################################################

# ======================================================================
# REVIEW COMMANDERS — quarterly pulse (country scope, CHI).
# Iterate the realm's military commanders (characters with military power_base,
# the loyal-cohorts model). Score each against the War office-holder via
# QING_pair_friction and against the throne via QING_char_affinity. A strong,
# loyal War minister + congenial commander = well-managed (loyalty/martial up).
# A vacant War office or a clashing pair = drift toward personal loyalty (the
# warlord seed) — loyalty risk, chance to queue an officer-review event.
# ======================================================================
QING_war_review_commanders = {
	LOG_enter = { sys = QING  fn = "QING_war_review_commanders" }

	# --- guard: War office must be filled and the holder alive
	if = {
		limit = {
			has_variable = qing_office_war_holder
			var:qing_office_war_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_war_holder = { save_scope_as = war_minister }

		# --- iterate commanders: characters with martial focus AND power_base (loyal cohorts model)
		# This is a conservative proxy: commanders who have been granted loyal veterans
		# (via office, regional army sanction, or mission rewards) have power_base >= 1.
		# We filter for martial >= 8 to focus on actual military leaders, not just
		# courtiers with holdings.
		every_character = {
			limit = {
				employer = ROOT
				is_alive = yes
				is_adult = yes
				is_ruler = no
				martial >= 8
				power_base >= 1
			}
			save_scope_as = commander

			# --- score the commander against the War minister (pair friction)
			ROOT = {
				QING_pair_friction = { a = scope:war_minister  b = scope:commander }
			}

			# --- also score the commander against the throne (personal affinity)
			QING_char_affinity = yes

			# --- HIGH FRICTION or LOW AFFINITY: the commander drifts from central control
			if = {
				limit = {
					OR = {
						ROOT = { var:qing_pair_friction >= 50 }
						var:qing_char_affinity <= 35
					}
				}
				# loyalty risk
				add_loyalty = loyalty_qing_estranged
				# chance to queue a warlord-danger event (low prob each pulse to avoid spam)
				ROOT = {
					random = {
						chance = 15
						trigger_event = { id = qing_war.2  days = 30 }
						LOG_line = { sys = QING  msg = "commander [scope:commander.GetName] drifting (friction/low affinity); queued warlord review event for [ROOT.GetTag]" }
					}
				}
			}
			# --- LOW FRICTION and HIGH AFFINITY: well-managed commander
			else_if = {
				limit = {
					ROOT = { var:qing_pair_friction < 50 }
					var:qing_char_affinity >= 60
				}
				# loyalty and martial boost (capable War minister cultivates his officers)
				add_loyalty = loyalty_qing_congenial
				if = {
					limit = { scope:war_minister = { martial >= 10 } }
					# a brilliant War minister sharpens his commanders' martial skill
					QING_char_cultivate = { stat = martial  amount = 1 }
					LOG_line = { sys = QING  msg = "commander [scope:commander.GetName] well-managed under War minister [scope:war_minister.GetName] (low friction, high affinity) for [ROOT.GetTag]" }
				}
			}
		}

		# --- trigger officer review event if the War minister is capable (quarterly chance)
		if = {
			limit = {
				scope:war_minister = { martial >= 10  loyalty >= 50 }
			}
			random = {
				chance = 20
				trigger_event = { id = qing_war.1  days = 45 }
				LOG_line = { sys = QING  msg = "queued officer review board event (capable War minister) for [ROOT.GetTag]" }
			}
		}

		# --- trigger military-examination event if the War minister is effective
		if = {
			limit = {
				scope:war_minister = { martial >= 12  finesse >= 8 }
			}
			random = {
				chance = 12
				trigger_event = { id = qing_war.4  days = 60 }
				LOG_line = { sys = QING  msg = "queued military-examination event (effective War minister) for [ROOT.GetTag]" }
			}
		}
	}
	# --- VACANT War office: commanders drift (no oversight)
	else = {
		every_character = {
			limit = {
				employer = ROOT
				is_alive = yes
				is_adult = yes
				is_ruler = no
				martial >= 8
				power_base >= 1
			}
			save_scope_as = commander
			# no War minister → officers become wayward (loyalty drift)
			add_loyalty = loyalty_qing_estranged
			# higher chance of warlord-danger event when the office is vacant
			ROOT = {
				random = {
					chance = 25
					trigger_event = { id = qing_war.2  days = 30 }
					LOG_line = { sys = QING  msg = "commander [scope:commander.GetName] drifting (War office vacant); queued warlord review event for [ROOT.GetTag]" }
				}
			}
		}
		LOG_line = { sys = QING  msg = "War office vacant; commanders drifting without oversight for [ROOT.GetTag]" }
	}

	LOG_exit = { sys = QING  fn = "QING_war_review_commanders"  result = OK }
}

CONSUME_all_stockpiles = {
	# Scope: Governorship
	# Function: trigger CONSUME_from_stockpile for every named tradegood stockpile in a governorship
	every_tradegood_complex = {
		APPLY = CONSUME_from_stockpile
	}
}

CONSUME_from_stockpile = {
	# Scope: Governorship
	# Function: Get the percentage shortage of a tradegood in this governorship
	change_variable = {
		name = $tradegood$_stockpile
		subtract = DEMAND_$tradegood$
	}
	CONSUME_update_shortage = {
		tradegood = $tradegood$
	}
}

CONSUME_update_shortage = {
	if = {
		limit = {
			has_variable = $tradegood$_stockpile
		}
		if = {
			limit = {
				var:$tradegood$_stockpile < 0
			}
			set_variable = {
				name = shortage_$tradegood$
				value = var:$tradegood$_stockpile
			}
			change_variable = {
				name = shortage_$tradegood$
				multiply = -1 # Make positive
			}
			change_variable = {
				name = shortage_$tradegood$
				divide = {
					value = DEMAND_$tradegood$
					min = 0.001
				}
			}
			# [logistics-phys] Snapshot the PURE physical-deficit ratio here, BEFORE the
			# deflation and elasticity terms below inflate shortage_$tradegood$. The military-
			# logistics layer (se_LOGISTICS.txt) must react only to a genuine equipment
			# shortfall, not to monetary deflation or a wealth-decline elasticity swing, so it
			# reads this snapshot rather than the composite shortage_ var. Set only for goods
			# actually in physical deficit; removed in lockstep in the surplus branch below.
			set_variable = {
				name = shortage_phys_$tradegood$
				value = var:shortage_$tradegood$
			}
			set_variable = {
				name = $tradegood$_stockpile
				value = 0
			}
		}
		else_if = {
			limit = {
				has_variable = shortage_$tradegood$
				OR = {
					var:$tradegood$_stockpile > 0
					DEMAND_$tradegood$ = 0
				}
			}
			remove_variable = shortage_$tradegood$
			# [logistics-phys] keep the physical-only snapshot in lockstep with the composite
			remove_variable = shortage_phys_$tradegood$
		}
		if = {
			limit = {
				owner.CURRENCY_amt_circulated_deflation > 0
				has_variable = shortage_$tradegood$
			}
			change_variable = {
				name = shortage_$tradegood$
				add = owner.CURRENCY_amt_circulated_deflation
			}
			if = {
				limit = {
					var:shortage_$tradegood$ > 1
				}
				set_variable = {
					name = shortage_$tradegood$
					value = 1
				}
			}
		}
		else_if = {
			limit = {
				owner.CURRENCY_amt_circulated_deflation > 0
				NOT = { has_variable = shortage_$tradegood$ }
			}
			set_variable = {
				name = shortage_$tradegood$
				value = owner.CURRENCY_amt_circulated_deflation
			}
		}
		# Add shortage from price elasticity impact
		# Elasticity impact is typically 1 - X where X is a compounded decrease in governorship wealth
		# So 1 - DEMAND_elasticity_impact gives a % shortage value to add
		if = {
			limit = {
				has_variable = shortage_$tradegood$
			}
			change_variable = {
				name = shortage_$tradegood$
				add = {
					value = 1
					subtract = DEMAND_elasticity_impact
				}
			}
		}
		if = {
			limit = {
				has_variable = shortage_$tradegood$
				trigger_if = {
					limit = {
						has_variable = shortage_$tradegood$
					}
					var:shortage_$tradegood$ > 1
				}
			}
			set_variable = {
				name = shortage_$tradegood$
				value = 1
			}
		}
	}

}

########################################################################
# se_USA_ROSTER.txt — THE ANTEBELLUM HISTORICAL CAST (task #93, concrete layer).
#
# The named men of the sectional crisis step onto the American stage as REAL
# on-map characters, keyed to the beat / band that is their moment: Clay founds
# the balance, Calhoun preaches nullification, Douglas repeals the line, Brown
# bleeds Kansas, Lincoln names the house divided, Davis / Lee / Grant fight the
# war. This is the CONCRETE-OVER-ABSTRACT layer (standing rule): the sectional
# counters are the summary, but every crisis hangs a real, named agitator or
# captain off the map — the direct twin of se_QING_ROSTER.txt.
#
# ENGINE REALITY (identical to the Qing roster): the mod's create_character
# takes no name argument and there is no set_first_name effect, so a conjured
# character gets a random, culture-appropriate GetName. We carry the historical
# identity two ways the engine DOES support:
#   (1) add_nickname = "<loc key>" pins the real name as his epithet, so he is
#       findable in the court list as e.g. Nathaniel "Abraham Lincoln".
#   (2) the beat that spawns him already names him in its title/desc (Clay,
#       Calhoun, Douglas, Lincoln, Brown, Sumner, Taney...), so the player
#       always sees who has arrived.
#
# Every spawn is guarded by a once-only ROOT flag (usa_spawned_*), so a figure
# who is summoned early by a band milestone is NOT summoned again by his dated
# beat (and vice versa) — the two drivers share one idempotent code path.
#
# create_character sets culture / religion / age explicitly (the oracle gotcha).
# Trait keys verified against common/traits/ (orator, abolitionist, silver_tongued,
# stubborn, proud, righteous, zealous, brave, aggressive, tactician, disciplined,
# conqueror, honest, just, ambitious). Stats on Imperator's four axes (0-9).
#
# AI-OWNED: called from the AI-autonomous US arc (se_USA_SECTION / usa_section
# events); no is_ai gate anywhere. All paths log sys = USA. Prefix = USA_spawn_.
#   ROOT / this = c:USA throughout; the fresh character is saved as scope:usa_figure.
#
# Sources: Potter, The Impending Crisis (1976); McPherson, Battle Cry of Freedom
#   (1988); Freehling, The Road to Disunion (1990/2007); Remini, Henry Clay (1991);
#   Bartlett, John C. Calhoun (1993); Oates, To Purge This Land with Blood (1970).
########################################################################

# ======================================================================
# FINALIZE — shared tail for every spawn. Runs in the new character's scope.
# Roots him in the American court and pins his historical name. $nick$ = the
# NICKNAME_USA_* loc key.
# ======================================================================
USA_roster_finalize = {
	set_home_country = ROOT
	add_nickname = "$nick$"
	save_scope_as = usa_figure
	LOG_line = { sys = USA  msg = "roster: [this.GetName] ($nick$) entered the [ROOT.GetTag] stage" }
}

# ---- HENRY CLAY — "the Great Compromiser" (Missouri, 1850) ------------
# The Kentucky Whig who bought the Union three reprieves. Master orator, national
# broker. Arrives to found the balance at the Missouri Compromise.
USA_spawn_clay = {
	if = {
		limit = { NOT = { has_variable = usa_spawned_clay } }
		set_variable = usa_spawned_clay
		create_character = {
			age = 43  culture = yankee  religion = evangelical
			add_charisma = 9  add_finesse = 9  add_martial = 1  add_zeal = 3
			add_trait = silver_tongued  add_trait = orator  add_trait = ambitious
			USA_roster_finalize = { nick = "NICKNAME_USA_CLAY" }
		}
	}
	else = { LOG_fail = { sys = USA  fn = "USA_spawn_clay"  reason = "Henry Clay already on the stage for [ROOT.GetTag]" } }
}

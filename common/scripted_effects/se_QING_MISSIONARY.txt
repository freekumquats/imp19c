########################################################################
# se_QING_MISSIONARY.txt — MISSIONARY INCIDENTS & THE ROAD TO THE BOXERS (教案).
#
# The treaties that opened China also opened it to the CHRISTIAN MISSIONS: the
# Tianjin treaties (1858/60) won foreigners the right to propagate their faith and
# hold property in the interior, and the French claimed a protectorate over all
# Catholic missions. The result was a running series of ANTI-CHRISTIAN INCIDENTS
# (教案, jiào'àn) — disputes over converts who claimed exemption from local custom,
# over mission property on temple land, over rumours of child-killing — that flared
# into riot and massacre: the Tianjin Massacre (天津教案, 1870), and hundreds of
# lesser cases the local gentry quietly encouraged. Each incident forced the throne
# to CHOOSE between the foreign consuls (who demanded indemnities and executions) and
# the outraged gentry and populace (扶清滅洋). The unresolved accumulation of this
# grievance is what finally boiled over into the Boxer catastrophe of 1900.
#
# THE BARGAIN THIS MODELS: an ANTI-CHRISTIAN SENTIMENT meter (仇教情緒) rises with
# the treaty burden, sect pressure, and each mishandled incident. At an incident the
# player SUPPRESSES it (satisfy the consuls with indemnity + punishment — foreign
# goodwill, but sentiment climbs as the gentry seethe at the humiliation) or INDULGES
# the popular fury (spare the humiliation, relieve sentiment a touch — but affront the
# powers and feed reform pressure). When sentiment boils over, the BOXER MOMENT
# arrives: this engine escalates directly into the existing Boxer rising
# (qing_rebellion.9), the culmination of the whole grievance.
#
# Design mirrors the suite: a 0..100 COUNTER on CHI, event-driven, read O(1) by the
# governance pulse (QING_missionary_pulse), all logged.
#   qing_antichristian_sentiment  0..100  仇教情緒 — anti-missionary/anti-foreign feeling
#
#   ROOT / this = the Qing overlord (CHI, human)
# Prefix = QING_missionary_
#
# Sources: Paul Cohen, "China and Christianity: The Missionary Movement and the
#   Growth of Chinese Antiforeignism, 1860-1870" (1963); Joseph Esherick, "The
#   Origins of the Boxer Uprising" (1987); Spence, The Search for Modern China; John
#   King Fairbank (ed.), The Missionary Enterprise in China and America (1974).
########################################################################

# ======================================================================
# INIT — seed the sentiment meter once. Called by QING_missionary_pulse.
# ======================================================================
QING_missionary_init = {
	# audit-fix (#79): peer inits (QING_selfstr_init, QING_taiping_init) carry the
	# LOG_enter/LOG_exit pair; wire it here so the sentiment-meter seed is traceable.
	LOG_enter = { sys = QING  fn = "QING_missionary_init" }
	if = { limit = { NOT = { has_variable = qing_antichristian_sentiment } } set_variable = { name = qing_antichristian_sentiment  value = 0 } }
	LOG_exit = { sys = QING  fn = "QING_missionary_init"  result = OK }
}

# ======================================================================
# SUPPRESS AN INCIDENT — satisfy the foreign consuls: indemnity, punishment of the
# rioters, protection of the mission. Wins foreign goodwill (eases great-power
# tension) but the visible kowtow to the barbarians inflames the gentry — sentiment
# CLIMBS. $power$ = the aggrieved power (britain|france|russia). Country scope.
# ======================================================================
QING_missionary_suppress = {
	LOG_enter = { sys = QING  fn = "QING_missionary_suppress" }
	QING_missionary_init = yes
	# the humiliation of yielding to the consuls stokes the anti-Christian fury.
	QING_DECLINE_nudge = { var = qing_antichristian_sentiment  amount = 10 }
	# ...but the powers are placated (eases tension where tracked).
	QING_missionary_ease_power = { power = $power$ }
	add_tyranny = 1
	LOG_line = { sys = QING  msg = "missionary: incident SUPPRESSED (consuls satisfied, gentry seethe) vs $power$ for [ROOT.GetTag]" }
	LOG_exit = { sys = QING  fn = "QING_missionary_suppress"  result = OK }
}

# ======================================================================
# INDULGE THE FURY — let the local magistrates look away, spare the humiliation of
# indemnity. Relieves sentiment a touch (the populace vents) but affronts the power
# (tension spikes) and feeds reform pressure (the disorder). $power$ = the power.
# Country scope.
# ======================================================================
QING_missionary_indulge = {
	LOG_enter = { sys = QING  fn = "QING_missionary_indulge" }
	QING_missionary_init = yes
	QING_DECLINE_nudge = { var = qing_antichristian_sentiment  amount = -6 }
	QING_missionary_pressure_power = { power = $power$ }
	QING_DECLINE_nudge = { var = qing_reform_pressure  amount = 4 }
	LOG_line = { sys = QING  msg = "missionary: incident INDULGED (populace vents, power affronted) vs $power$ for [ROOT.GetTag]" }
	LOG_exit = { sys = QING  fn = "QING_missionary_indulge"  result = OK }
}

# Ease / pressure the aggrieved power's great-power tension. Britain/France/Russia
# have counters; a no-op-with-log for any unmapped power.
QING_missionary_ease_power = {
	# audit-fix (#79): enter/exit so the power-dispatch step traces even on the
	# mapped-power branches (previously only the unmapped else logged).
	LOG_enter = { sys = QING  fn = "QING_missionary_ease_power ($power$)" }
	if = { limit = { $power$ = britain }  QING_DECLINE_nudge = { var = qing_gp_tension_britain  amount = -6 } }
	else_if = { limit = { $power$ = france }  QING_DECLINE_nudge = { var = qing_gp_tension_france  amount = -6 } }
	else_if = { limit = { $power$ = russia }  QING_DECLINE_nudge = { var = qing_gp_tension_russia  amount = -6 } }
	else = { LOG_line = { sys = QING  msg = "missionary: no tension counter for $power$ (ease noted)" } }
	LOG_exit = { sys = QING  fn = "QING_missionary_ease_power"  result = OK }
}
QING_missionary_pressure_power = {
	# audit-fix (#79): enter/exit so the power-dispatch step traces even on the
	# mapped-power branches (previously only the unmapped else logged).
	LOG_enter = { sys = QING  fn = "QING_missionary_pressure_power ($power$)" }
	if = { limit = { $power$ = britain }  QING_DECLINE_nudge = { var = qing_gp_tension_britain  amount = 12 } }
	else_if = { limit = { $power$ = france }  QING_DECLINE_nudge = { var = qing_gp_tension_france  amount = 12 } }
	else_if = { limit = { $power$ = russia }  QING_DECLINE_nudge = { var = qing_gp_tension_russia  amount = 12 } }
	else = { LOG_line = { sys = QING  msg = "missionary: no tension counter for $power$ (pressure noted)" } }
	LOG_exit = { sys = QING  fn = "QING_missionary_pressure_power"  result = OK }
}

# ======================================================================
# THE PULSE — drift sentiment toward a target driven by the treaty burden and sect
# pressure (the accumulating grievance), apply the sentiment band, and — when the
# meter boils over — escalate to the Boxer moment (the existing qing_rebellion.9).
# Called from QING_GOV_pulse. All O(1).
# ======================================================================
QING_missionary_pulse = {
	LOG_enter = { sys = QING  fn = "QING_missionary_pulse" }
	QING_missionary_init = yes

	# --- sentiment drifts toward a blend of the treaty burden (the humiliation the
	# missions embody) and sect pressure (the popular unrest they inflame). Both are
	# existing suite meters; if the treaty system is not imposed the burden reads 0.
	set_variable = { name = qing_antichr_target  value = 0 }
	if = { limit = { has_variable = qing_treaty_burden } change_variable = { name = qing_antichr_target  add = var:qing_treaty_burden } }
	change_variable = { name = qing_antichr_target  add = var:qing_sect_pressure }
	change_variable = { name = qing_antichr_target  divide = 2 }
	if = { limit = { var:qing_antichr_target > 100 } set_variable = { name = qing_antichr_target  value = 100 } }
	if = {
		limit = { var:qing_antichristian_sentiment < var:qing_antichr_target }
		QING_DECLINE_nudge = { var = qing_antichristian_sentiment  amount = 1 }
	}
	else_if = {
		limit = { var:qing_antichristian_sentiment > var:qing_antichr_target }
		QING_DECLINE_nudge = { var = qing_antichristian_sentiment  amount = -1 }
	}

	QING_missionary_apply_band = yes

	# [#91 item P] give the fury a FACE: at fever-pitch sentiment, conjure a real
	# anti-missionary agitator (once). The call is gated on the same guard the effect
	# carries so the pulse does not spam the LOG_fail branch every tick.
	if = {
		limit = {
			var:qing_antichristian_sentiment >= 66
			NOT = { has_variable = qing_spawned_antichr_agitator }
		}
		QING_missionary_spawn_agitator = yes
	}

	# --- the boil-over: once sentiment is at fever pitch AND the Boxers have not yet
	# risen, the accumulated grievance erupts into the Boxer moment. We route into the
	# EXISTING Boxer rising (qing_rebellion.9), whose own guard (qing_rebel_boxer_rose)
	# ensures it fires once. Guarded here on the same flag so we do not re-trigger.
	if = {
		limit = {
			var:qing_antichristian_sentiment >= 85
			NOT = { has_variable = qing_rebel_boxer_rose }
			NOT = { has_variable = qing_active_rebellion }
		}
		LOG_line = { sys = QING  msg = "missionary: sentiment boils over -> Boxer moment for [ROOT.GetTag]" }
		trigger_event = { id = qing_rebellion.9  days = { 5 20 } }
	}

	LOG_exit = { sys = QING  fn = "QING_missionary_pulse"  result = OK }
}

# ======================================================================
# BAND APPLICATION — one active modifier matching the sentiment meter. High
# sentiment is a running danger to the missions and the foreign standing; it also
# lends a grim martial temper to the populace (a two-edged thing).
# ======================================================================
QING_missionary_apply_band = {
	# audit-fix (#79): enter/exit so the band-application step traces on ALL paths,
	# including the "cleared to baseline" case (sentiment < 40, no band added).
	LOG_enter = { sys = QING  fn = "QING_missionary_apply_band" }
	remove_country_modifier = qing_antichristian_simmering
	remove_country_modifier = qing_antichristian_fever
	if = {
		limit = {
			var:qing_antichristian_sentiment >= 66
		}
		add_country_modifier = { name = qing_antichristian_fever  duration = -1 }
		LOG_line = { sys = QING  msg = "band antichristian = fever for [ROOT.GetTag] (sentiment=[ROOT.MakeScope.GetVariable('qing_antichristian_sentiment').GetValue|0])" }
	}
	else_if = {
		limit = {
			var:qing_antichristian_sentiment >= 40
		}
		add_country_modifier = { name = qing_antichristian_simmering  duration = -1 }
		LOG_line = { sys = QING  msg = "band antichristian = simmering for [ROOT.GetTag] (sentiment=[ROOT.MakeScope.GetVariable('qing_antichristian_sentiment').GetValue|0])" }
	}
	LOG_exit = { sys = QING  fn = "QING_missionary_apply_band"  result = OK }
}

# ======================================================================
# [#91 item P] THE AGITATOR TAKES THE STAGE (concrete-over-abstract).
# ORACLE VERDICT: use the CHARACTER path — a pop cannot carry a hostile trait nor be
# targeted/killed, so the anti-Christian sentiment counter (kept intact as the summary
# layer) HANGS A REAL MAN off itself: a named 仇教 firebrand who preaches against the
# missions and the converts. Built with the proven roster idiom (create_character in
# country scope + QING_roster_finalize, exactly as the #90 regional magnate at
# se_QING_MECHANICS.txt:201-218). zealous/cruel/stubborn are all real trait keys
# (common/traits/). Guarded once by qing_spawned_antichr_agitator so he cannot be
# conjured twice. Country scope; ROOT = the Qing.
# ======================================================================
QING_missionary_spawn_agitator = {
	LOG_enter = { sys = QING  fn = "QING_missionary_spawn_agitator" }
	QING_missionary_init = yes
	if = {
		limit = { NOT = { has_variable = qing_spawned_antichr_agitator } }
		# lock the once-only guard BEFORE the spawn so a re-entrant pulse cannot double-conjure.
		set_variable = qing_spawned_antichr_agitator
		# a Han preacher-agitator of the anti-foreign cause; zeal to inflame, cruelty in
		# the pogrom, stubbornness that will not be talked down.
		create_character = {
			age = 41
			culture = han
			religion = ROOT.religion
			add_martial = 4
			add_finesse = 5
			add_zeal = 8
			add_charisma = 7
			add_trait = zealous
			add_trait = cruel
			add_trait = stubborn
			save_scope_as = qing_antichr_agitator
		}
		scope:qing_antichr_agitator = {
			# roster tail: home country, historical epithet, identity band, throne affinity.
			QING_roster_finalize = { nick = "NICKNAME_QING_ANTICHR_AGITATOR" }
			LOG_line = { sys = QING  msg = "missionary: anti-Christian agitator [this.GetName] takes the stage for [ROOT.GetTag] (sentiment fever)" }
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_missionary_spawn_agitator"  reason = "agitator already conjured (qing_spawned_antichr_agitator set)" }
	}
	LOG_exit = { sys = QING  fn = "QING_missionary_spawn_agitator"  result = OK }
}

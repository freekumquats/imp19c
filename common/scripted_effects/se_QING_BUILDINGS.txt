# the Wuyi/Huangshan export-tea works, the Songjiang cotton handicraft belt,
# and the Lianghuai/Changlu/Sichuan salt yards — so we place them directly
# rather than making the player re-build the empire's existing industry.
#
# add_building_level does NOT enforce the building `allow` gate (se_QING_MECHANICS

# SE_qing_armies / the SE_qing_navy_* squadrons — was moved to deferred events
# qing_force_setup.* per [#223/#224-fix]; this building setup remains inline.)
#
# NO BOM (se_*.txt convention; 232323 = ### header).
########################################################################

# Seed ONE production building at province $P$ (building $B$), guarded by
# ownership + a $G$ trade-good match so a re-mapped province can never mis-seed.
# Skips silently (logged) if the site fails its guard or already has the building.
QING_seed_building = {
	if = {
		limit = {
			exists = p:$P$
			p:$P$ = {
				owner = c:CHI
				trade_goods = $G$
				NOT = { has_building = $B$ }
			}
		}
		p:$P$ = {
			add_building_level = $B$
			LOG_line = { sys = QING  msg = "starting buildings: $B$ seeded at $NAME$ (p:$P$)" }
		}
	}
	else = {
		LOG_line = { sys = QING  msg = "starting buildings: SKIP $B$ at $NAME$ (p:$P$) — guard failed or already present" }
	}
}

# [bookmark-1763 #291] Seed a Board-of-Works infrastructure building (canal depot / river
# dike) at game start. Unlike QING_seed_building these are NOT tied to a trade good, so the
# guard is ownership-only. add_building_level bypasses the building `allow` tech gate (which
# only governs the player build menu). Idempotent (NOT has_building). Scope: any.
QING_seed_works_building = {
	if = {
		limit = {
			exists = p:$P$
			p:$P$ = {
				owner = c:CHI
				NOT = { has_building = $B$ }
			}
		}
		p:$P$ = {
			add_building_level = $B$
			LOG_line = { sys = QING  msg = "starting buildings: $B$ seeded at $NAME$ (p:$P$)" }
		}
	}
	else = {
		LOG_line = { sys = QING  msg = "starting buildings: SKIP $B$ at $NAME$ (p:$P$) — not CHI-owned or already present" }
	}
}

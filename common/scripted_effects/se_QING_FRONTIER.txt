########################################################################
# se_QING_FRONTIER.txt — [#346] FRONTIER-GARRISON OVERLAY: direct Qing military
# presence on Inner-Asian subject soil, and the War-Ministry ↔ Lifan-Yuan turf war.
#
# The 1815/1763 frontier banner garrisons (Ili 伊犁, Ürümqi 烏魯木齊, Kashgar 喀什噶爾,
# Mukden 盛京, Heilongjiang 黑龍江) sit in provinces owned by CHI's autonomous-
# governorship SUBJECTS (ILI/XNG/MKD/HLJ). The setup OOB deliberately raises them
# UNDER the subject (engine placement-validity, #338/B21) — but historically these were
# QING FRONTIER COMMANDS (將軍) answering to Beijing, not the local dependency. This
# overlay expresses that: a garrison COMMANDED BY A QING (c:CHI-employed) officer sitting
# on a subject's frontier province is a direct imperial military occupation, which:
#   • TIGHTENS integration of that dependency (the army is the spine of 改土歸流), and
#   • lowers the subject's loyalty/autonomy sentiment (occupation chafes), and
#   • steps on the LIFAN YUAN's indirect-rule remit (理藩院 governs by resident Amban,
#     NOT by bayonet) — a turf war between the Grand Minister of War (兵部) and the Grand
#     Director of the Lifan Yuan (理藩院) that this layer surfaces as an event.
#
# This LAYERS on the #347 detection idiom (every_army, unit_location, unit_owner) rather
# than re-architecting the setup OOB. Runs on the Qing governance pulse (CHI player).
# Scope: COUNTRY (ROOT = CHI). se_LOG sys = QING. Prefix = QING_frontier_.
########################################################################

# ======================================================================
# QING_fgar_scan — once per Qing pulse. Find every settled army COMMANDED BY a
# CHI officer that sits on a province owned by a CHI autonomous-governorship SUBJECT
# (not by CHI itself), and apply the occupation consequences to that subject. Bounded
# O(own armies). Throttled per subject by a timed var so the integration push +
# loyalty drop come at a measured cadence, not every pulse.
# ======================================================================
QING_fgar_scan = {
	if = {
		limit = { tag = CHI }
		LOG_line = { sys = QING  msg = "frontier-garrison overlay scan firing for" }
		every_army = {
			limit = {
				is_moving = no
				in_combat = no
				unit_size >= 3
				exists = commander
				commander = { is_alive = yes  employer = ROOT }   # a QING (Beijing) officer, not the subject's
				exists = unit_location
				unit_location = {
					exists = owner
					owner = { is_subject_of = ROOT }               # the ground belongs to a CHI subject
					owner = { is_subject_type = autonomous_governorship }
				}
			}
			# save the occupied dependency + the Qing commander via step-by-step scope
			# pushes (avoid a chained unit_location.owner deref in an effect).
			commander = { save_scope_as = qing_frontier_commander }
			unit_location = {
				owner = { save_scope_as = qing_frontier_subject }
			}
			ROOT = { QING_fgar_apply_occupation = yes }
		}
	}
}

# ----------------------------------------------------------------------
# QING_fgar_apply_occupation — the per-garrison consequence. Scope: COUNTRY (ROOT
# = CHI); expects scope:qing_frontier_subject (the occupied dependency) + scope:
# qing_frontier_commander (the Qing officer on its soil). Throttled per subject via a
# timed var on the subject so it fires ~twice a year, not every pulse.
# ----------------------------------------------------------------------
QING_fgar_apply_occupation = {
	if = {
		limit = {
			exists = scope:qing_frontier_subject
			scope:qing_frontier_subject = { NOT = { has_variable = qing_frontier_occupation_cd } }
		}
		scope:qing_frontier_subject = { set_variable = { name = qing_frontier_occupation_cd  days = 180 } }
		# mark the subject as garrisoned by Beijing (read by the War + Lifan Yuan panels).
		scope:qing_frontier_subject = {
			set_variable = { name = qing_frontier_garrisoned  value = 1  days = 200 }
		}
		# (1) TIGHTEN integration — the garrison is the spine of direct rule (改土歸流).
		# Reuse the proven integration verb (scope:target = the subject).
		scope:qing_frontier_subject = { save_scope_as = target }
		SUBJ_QING_advance_integration = { steps = 1 }
		# (2) OCCUPATION CHAFES — the dependency's loyalty toward Beijing dips (a bayonet
		# is not a bond). The subject's opinion of the overlord is the loyalty driver
		# (se_SUBJECT_QING model); nudge it down with a standing occupation modifier.
		scope:qing_frontier_subject = {
			if = {
				limit = { NOT = { has_country_modifier = qing_frontier_occupied } }
				add_country_modifier = { name = qing_frontier_occupied  duration = 200 }
			}
			add_opinion = { modifier = qing_frontier_occupation_opinion  target = ROOT }
		}
		# (3) LIFAN-YUAN TURF WAR — direct military occupation steps on the Court of
		# Colonial Affairs' indirect-rule remit. If the Lifan Yuan office is filled AND a
		# resident Amban sits in this very dependency, the resident and the garrison
		# commander clash over who governs; roll a friction event (throttled by the
		# same occupation cooldown above). The War Minister backs his general; the Lifan
		# Director backs his Amban.
		if = {
			limit = {
				has_variable = qing_office_lifanyuan_holder
				var:qing_office_lifanyuan_holder = { is_alive = yes  employer = ROOT }
				scope:qing_frontier_subject = {
					has_variable = qing_amban_here
					var:qing_amban_here = { is_alive = yes }
				}
				# only a real turf war when a War Minister ALSO sits (someone to back the army)
				has_variable = qing_office_war_holder
				var:qing_office_war_holder = { is_alive = yes  employer = ROOT }
				# don't spam: at most one turf-war event per realm per ~year
				NOT = { has_variable = qing_frontier_turfwar_cd }
			}
			set_variable = { name = qing_frontier_turfwar_cd  days = 365 }
			var:qing_office_war_holder = { save_scope_as = qing_frontier_war_min }
			var:qing_office_lifanyuan_holder = { save_scope_as = qing_frontier_lifan_dir }
			scope:qing_frontier_subject.var:qing_amban_here = { save_scope_as = qing_frontier_amban }
			trigger_event = { id = qing_frontier.1  days = { 5 20 } }
			LOG_line = { sys = QING  msg = "frontier turf war: War Ministry garrison vs Lifan Yuan Amban queued" }
		}
	}
}

# se_MARRIAGE.txt — [#276] DYNASTIC MARRIAGE DIPLOMACY.
#
# Royal marriage as a DIPLOMATIC instrument for the European / dynastic AI powers
# (and any player running such a realm). A marriage between two ruling houses binds
# the realms (a standing goodwill opinion, sometimes a formal alliance) and — where
# a junior house is left without a male heir — opens the road to DYNASTIC UNION and
# eventual inheritance, modelled on the existing royal_union subject type.
#
# SCOPE: an AI-autonomous dynastic layer for every eligible MONARCHY, driving itself while
# the player runs any realm (like the USA #93 / Japan #94 / Mexico #96 arcs). [#311 UPDATE]
# the feature is now FULL: eligibility gates (rank/religion/bloc), a dispensation/conversion
# BRIDGE for cross-confessional matches, dowry + claim transfer, betrothal + repudiation,
# alliance decay, contested-succession war-claims, loveless-marriage dysfunction, a bloc-weight
# indicator, and a PLAYER L4 GUI (MARRIAGE_actions.txt). The QING (CHI) are now ELIGIBLE — but a
# CHI<->Western match is hard-gated behind the bridge (there was no routine Sino-European marriage
# market), so the AI never forces one; only a deliberate player dispensation/conversion enables it.
#
# ORACLE GATE (all four §3 hooks cleared 2026-07-09, Terra-Indomita + Invictus):
#   [U-3] cross-court marry_character — PROVEN: c:TAG = { <child iterator> = {
#         save_scope_as = X } } then char = { marry_character = scope:X } (Invictus
#         me_bithynia.txt:173-229; ordered_child is the standard child iterator).
#   [U-2] royal_union FROM a marriage — PROVEN verb: FUNC_make_subject { type =
#         royal_union } (se_FUNC.txt:412; royal_union def 00_default.txt:914). The
#         LOGIC (junior house = the heirless one) is ours.
#   [U-1] auto-inheritance on shared-ruler death — engine auto-merge is UNPROVEN, so
#         NOT relied on. Modelled OURSELVES on on_character_death (is_ruler=yes),
#         transferring the junior realm's land with the PROVEN LAND_transfer_provinces
#         idiom (se_LAND.txt:348 — the only sanctioned land-transfer path). annex_country
#         is ABSENT from both oracles + the mod, so it is NOT used.
#   [U-4] AI willingness / diplomatic action — a marriage diplomatic-action payload
#         is UNPROVEN, so NOT used. AI-autonomous PULSE drives it (proven arc idiom).
#
# Per-partner bookkeeping uses the PROVEN variable_list idiom (add_to_variable_list /
# is_target_in_variable_list — exactly as #165 se_QING_SPHERE tracks its ring), NOT a
# dynamic name@scope var (that suffix syntax has zero precedent in this codebase).
#
# Storage
#   COUNTRY : qing_marriage_initialized (seed guard); marriage_bond_count (live bonds);
#             list marriage_partners (the realms we are wedded to); throttle
#             marriage_pulse_cooldown.
#   STATE/COUNTRY var marriage_union_junior : marks a royal_union WE created by marriage
#             (so L3 inheritance only fires on our own unions, not pre-existing ones).
#   Opinion : royal_marriage_opmod (standing goodwill between wedded houses).
#
# Cadence: MARRIAGE_pulse from yearly_country_pulse, self-throttled ~5 years, one
# partner attempt per pulse (a bounded random_country pick, NOT an O(n^2) scan).
# All logged sys = MARRIAGE, STATIC msgs only (#253). Prefix = MARRIAGE_.
#
# Sources (feasibility only, no content copied): standard 19thC dynastic-diplomacy
# history (the Coburg marriages, the "grandmother of Europe"); engine idioms above.
########################################################################

# ======================================================================
# MARRIAGE_init — seed the bond counter once. Scope: COUNTRY. Idempotent.
# ======================================================================
MARRIAGE_init = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_init" }
	if = {
		limit = { NOT = { has_variable = qing_marriage_initialized } }
		set_variable = { name = qing_marriage_initialized  value = 1 }
		set_variable = { name = marriage_bond_count  value = 0 }
	}
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_init"  result = OK }
}

# ======================================================================
# MARRIAGE_pulse — AI-autonomous driver. Scope: COUNTRY (a Western monarchy AI).
# Self-throttled ~5 years. Finds ONE partner realm not already wedded to us and
# forms a marriage pact; then, if a standing bond leaves a junior house heirless,
# escalates that bond to a dynastic royal_union. Bounded: a single random_country
# pick per pulse, so cost stays O(1) amortised even across many monarchies.
# ======================================================================
MARRIAGE_pulse = {
	if = {
		limit = { NOT = { has_variable = marriage_pulse_cooldown } }
		LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_pulse" }
		set_variable = { name = marriage_pulse_cooldown  days = 1825 }
		MARRIAGE_init = yes
		save_scope_as = marriage_home
		# --- L1: seek a fresh dynastic marriage, but only while we are not already
		# over-wed (a realm holds at most 3 standing dynastic bonds — beyond that the
		# houses are saturated and further weddings add nothing) AND our own house
		# still fields a marriageable child.
		if = {
			limit = {
				var:marriage_bond_count < 3
				current_ruler = { MARRIAGE_has_marriageable_child_trigger = yes }
			}
			# [#311] the pair-compatibility trigger reads the OTHER house as scope:target.
			# Faith/bloc compatibility is SYMMETRIC, so we save OUR OWN realm as scope:target and
			# evaluate the trigger from inside each candidate's scope (candidate-vs-home). save_scope_as
			# is an effect, so it is set HERE (not inside the random_country limit = trigger context).
			save_scope_as = target
			random_country = {
				limit = {
					NOT = { this = scope:marriage_home }
					MARRIAGE_eligible_realm_trigger = yes
					in_diplomatic_range = scope:marriage_home
					# not already wedded to us (proven variable_list membership test)
					NOT = {
						scope:marriage_home = {
							is_target_in_variable_list = { name = marriage_partners  target = prev }
						}
					}
					# the partner house must also field a marriageable child to wed
					current_ruler = { MARRIAGE_has_marriageable_child_trigger = yes }
					# [#311] the AUTONOMOUS pulse only makes COMPATIBLE matches (same Western bloc or
					# same faith). A cross-confessional / cross-bloc union (e.g. a Qing<->Western match)
					# needs a deliberate BRIDGE (dispensation/conversion) and is PLAYER-DRIVEN only, via
					# the marriage GUI — the AI does not force one (there was no routine Sino-European
					# marriage market). Evaluated candidate(this)-vs-home(scope:target, saved above).
					MARRIAGE_pair_compatible_trigger = yes
				}
				save_scope_as = marriage_partner_realm
			}
			if = {
				limit = { exists = scope:marriage_partner_realm }
				MARRIAGE_form_pact = { partner = scope:marriage_partner_realm  from_ruler = yes }
			}
		}
		# --- [#betrothal-rework] L1b: mature standing child betrothals (wed the pair once both are
		# adult) or dissolve them without penalty if a betrothed child has died.
		MARRIAGE_check_betrothals = yes
		# --- L2: escalate any standing bond whose junior house is now heirless.
		MARRIAGE_check_unions = yes
		# --- [#324] L2b: the FEMALE-LINE claim — a wedded house whose only heir is a
		# daughter its own agnatic law bars breeds a pressable inheritance claim for our
		# line (the #312 succession-law × marriage coupling; a cognatic realm lets her
		# inherit, so no claim). One grant per partner.
		MARRIAGE_check_heiress_claim = yes
		# --- [#311 hook #2] L2c: let aged dynastic alliances lapse (bond remains, alliance fades).
		MARRIAGE_check_decay = yes
		# --- [#311 hook #13] recompute our DYNASTIC BLOC WEIGHT (a simple numeric: the land this
		# bloodline's marriage network commands), so blocs can be ranked against each other.
		MARRIAGE_recompute_bloc_weight = yes
		LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_pulse"  result = OK }
	}
}

# ======================================================================
# [#betrothal-rework] MARRIAGE_apply_marriage_bond — the shared COUNTRY-level marriage payload:
# record the per-partner bond both ways, seal a formal alliance for a peace-time match, pay the
# dowry, and (for a deep bond) grant the dowry claim. Call in the SENIOR country scope; $partner$ =
# the junior COUNTRY. Extracted so BOTH a fresh marriage (MARRIAGE_form_pact) and a matured betrothal
# (MARRIAGE_mature_betrothal) apply an identical bond. Assumes the wedding (marry_character) already
# happened; the caller owns the char-specific loveless check.
# ======================================================================
MARRIAGE_apply_marriage_bond = {
	save_scope_as = bond_senior
	$partner$ = { save_scope_as = bond_junior }
	# per-partner bond via the proven variable_list idiom, both directions, so the pair never double-weds.
	scope:bond_senior = {
		MARRIAGE_init = yes
		if = {
			limit = { NOT = { is_target_in_variable_list = { name = marriage_partners  target = scope:bond_junior } } }
			add_to_variable_list = { name = marriage_partners  target = scope:bond_junior }
			change_variable = { name = marriage_bond_count  add = 1 }
			add_opinion = { modifier = royal_marriage_opmod  target = scope:bond_junior }
		}
	}
	scope:bond_junior = {
		MARRIAGE_init = yes
		if = {
			limit = { NOT = { is_target_in_variable_list = { name = marriage_partners  target = scope:bond_senior } } }
			add_to_variable_list = { name = marriage_partners  target = scope:bond_senior }
			change_variable = { name = marriage_bond_count  add = 1 }
			add_opinion = { modifier = royal_marriage_opmod  target = scope:bond_senior }
		}
	}
	# a strong dynastic match (not already allied, both at peace) seals a bond — BUT in Imperator
	# vanilla a GREAT POWER cannot hold ordinary alliances (engine-hardcoded rank restriction). So we
	# branch [user]: if NEITHER house is a great power, seal the real formal alliance (the Coburg
	# pattern); if EITHER is a great power, lay down the royal_marriage_entente modifier on both courts
	# instead — the diplomatic-warmth + mutual-defence bond that works around the no-alliance rule.
	# Both paths stamp the SAME self-expiring 30-yr timer + tracking list, so MARRIAGE_check_decay
	# lapses them uniformly when the generation passes.
	if = {
		# [#330 user] SUBJECT BRANCH — first, the overlord<->subject case. A marriage between a realm and
		# its OWN subject (or overlord) does NOT form an alliance/entente (a subject cannot ally its
		# overlord) — it TIGHTENS the dependency instead. TWO levers, matching how the engine + this mod
		# actually model subject loyalty (AMBAN/pilgrimage precedent, se_QING_AMBAN.txt):
		#   1. royal_marriage_subject_tie carries the SUBJECT-FACING loyalty_to_overlord key
		#      (marriage_modifiers.txt), so it must sit on the SUBJECT court — that is the key that
		#      raises THIS subject's loyalty toward ITS overlord. (The overlord-side alternative,
		#      subject_loyalty, is INERT on a leaf subject — the #314 B17 bug — so do NOT move this
		#      modifier onto the overlord. #330 review fix, corrected during the overnight merge.)
		#   2. per-subject loyalty is driven by the subject's OPINION of its overlord, so we ALSO bump
		#      the SUBJECT's opinion toward the overlord (this is what tightens THIS particular tie; the
		#      base bond's mutual royal_marriage_opmod already adds standing goodwill on top).
		# Both on the same self-expiring 30-yr timer as the alliance path so MARRIAGE_check_decay strips
		# them when the generation passes. Guard the modifier against stacking (add_country_modifier stacks).
		limit = {
			OR = {
				scope:bond_senior = { is_subject_of = scope:bond_junior }
				scope:bond_junior = { is_subject_of = scope:bond_senior }
			}
		}
		# identify overlord + subject, then apply the tie modifier to the SUBJECT (it carries the
		# subject-facing loyalty_to_overlord key) and bump the SUBJECT's opinion toward the overlord.
		if = {
			limit = { scope:bond_junior = { is_subject_of = scope:bond_senior } }
			# senior = overlord, junior = subject -> tie + opinion go on the junior (subject)
			if = {
				limit = { scope:bond_junior = { NOT = { has_country_modifier = royal_marriage_subject_tie } } }
				scope:bond_junior = { add_country_modifier = { name = royal_marriage_subject_tie  duration = -1 } }
			}
			scope:bond_junior = { add_opinion = { modifier = royal_marriage_opmod  target = scope:bond_senior } }
		}
		else = {
			# junior = overlord, senior = subject -> tie + opinion go on the senior (subject)
			if = {
				limit = { scope:bond_senior = { NOT = { has_country_modifier = royal_marriage_subject_tie } } }
				scope:bond_senior = { add_country_modifier = { name = royal_marriage_subject_tie  duration = -1 } }
			}
			scope:bond_senior = { add_opinion = { modifier = royal_marriage_opmod  target = scope:bond_junior } }
		}
		# same decay bookkeeping as the alliance/entente path so the tie lapses with the generation.
		scope:bond_senior = { set_variable = { name = marriage_alliance_timer  days = 10950 } }
		scope:bond_junior = { set_variable = { name = marriage_alliance_timer  days = 10950 } }
		scope:bond_senior = { add_to_variable_list = { name = marriage_alliance_partners  target = scope:bond_junior } }
		scope:bond_junior = { add_to_variable_list = { name = marriage_alliance_partners  target = scope:bond_senior } }
		LOG_line = { sys = MARRIAGE  msg = "pact sealed as an OVERLORD-SUBJECT dynastic tie (loyalty bond, no alliance)" }
	}
	else_if = {
		limit = {
			NOT = { scope:bond_senior = { alliance_with = scope:bond_junior } }
			scope:bond_senior = { NOT = { at_war = yes } }
			scope:bond_junior = { NOT = { at_war = yes } }
		}
		if = {
			# neither party a great power → a real formal alliance is allowed.
			limit = {
				scope:bond_senior = { NOT = { rank = great_power } }
				scope:bond_junior = { NOT = { rank = great_power } }
			}
			scope:bond_senior = { add_alliance = scope:bond_junior }
			LOG_line = { sys = MARRIAGE  msg = "pact sealed with a formal alliance (30-yr dynastic bond)" }
		}
		else = {
			# a great power is involved → the entente workaround (no formal alliance possible).
			# [#betrothal-review B19 fix] guard against STACKING — add_country_modifier stacks, so a
			# repeat match with the same great-power house would pile up duplicate entente modifiers.
			# Apply only when not already present (mirrors the guarded marriage_partners bond-record above).
			if = {
				limit = { scope:bond_senior = { NOT = { has_country_modifier = royal_marriage_entente } } }
				scope:bond_senior = { add_country_modifier = { name = royal_marriage_entente  duration = -1 } }
			}
			if = {
				limit = { scope:bond_junior = { NOT = { has_country_modifier = royal_marriage_entente } } }
				scope:bond_junior = { add_country_modifier = { name = royal_marriage_entente  duration = -1 } }
			}
			LOG_line = { sys = MARRIAGE  msg = "pact sealed with a dynastic ENTENTE (great power cannot ally; 30-yr bond)" }
		}
		# [#311 hook #2] the bond is DYNASTIC — it fades with the generation that made it: stamp a
		# self-expiring 30-yr timer + record it (both ways) so MARRIAGE_check_decay lapses it (breaking
		# the alliance OR stripping the entente) unless a fresh marriage renews it.
		scope:bond_senior = { set_variable = { name = marriage_alliance_timer  days = 10950 } }
		scope:bond_junior = { set_variable = { name = marriage_alliance_timer  days = 10950 } }
		scope:bond_senior = { add_to_variable_list = { name = marriage_alliance_partners  target = scope:bond_junior } }
		scope:bond_junior = { add_to_variable_list = { name = marriage_alliance_partners  target = scope:bond_senior } }
	}
	# [#311 hook #3] the DOWRY — a patrimonial transaction, not just a bond. Development capital flows
	# to the senior (the bride's portion; Catherine of Braganza / the Farnese portion). Once per pact.
	scope:bond_senior = { CURRENCY_grant_country_wealth = { thousands = 30 } }
	# a deep bond may carry a territorial claim in the contract (Burgundy→Maximilian).
	if = {
		limit = { scope:bond_senior = { has_variable = marriage_bond_count  var:marriage_bond_count >= 2 } }
		scope:bond_senior = { MARRIAGE_grant_dowry_claim = { from = scope:bond_junior } }
	}
}

# ======================================================================
# MARRIAGE_form_pact — wed a child of our house to a child of the partner house.
# Scope: COUNTRY (the senior/proposing house). $partner$ = the partner COUNTRY.
# Cross-court marriage is PROVEN (oracle U-3): save a marriageable child of the
# partner's ruler, then marry our ruler's marriageable child to it. Applies the
# standing goodwill opinion BOTH ways and, for a strong match, a formal alliance.
# ======================================================================
MARRIAGE_form_pact = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_form_pact (from_ruler=$from_ruler$)" }
	save_scope_as = pact_senior
	$partner$ = { save_scope_as = pact_junior }
	# [#313] $from_ruler$ = yes (default dynastic match, ruling family) | no (a lesser tie to a
	#   NON-ruling noble house of each realm: no union/inheritance implications — see the payload
	#   branch below). The selection idioms differ: ruling = current_ruler.ordered_child; other =
	#   ordered_character over the court's from_ruler_family=no adults. Both are proven (Invictus
	#   propose_marriage.txt gates on from_ruler_family; any_character/ordered_character are standard).
	#
	# [#313 FIX] GENDER ORIENTATION. The gender filter (marriage_filter_gender, read on the acting/
	#   senior country = ROOT/this) says which of OUR children weds: flag:daughter → OUR child is
	#   FEMALE and the PARTNER's is MALE; anything else (flag:son, or unset — the pulse/betrothal
	#   callers never set it) → OUR child is MALE and the partner's FEMALE (the historical default).
	#   Previously the pairing was hardcoded our-male/their-female, so flag:daughter silently no-opped
	#   (and a ruler with only daughters could never wed). We now pick our child of the filtered sex
	#   and the partner's child of the opposite sex. scope:marriage_bride = the PARTNER child,
	#   scope:marriage_groom = OUR child (names kept for the downstream payload; not literal sexes).
	if = {
		limit = { has_variable = marriage_filter_gender  var:marriage_filter_gender = flag:daughter }
		set_variable = { name = marriage_our_child_female  value = 1 }
	}
	else = {
		set_variable = { name = marriage_our_child_female  value = 0 }
	}
	# 1. the PARTNER child (opposite sex to ours) — eldest eligible (order_by = age, max = 1).
	if = {
		limit = { $from_ruler$ = yes }
		scope:pact_junior = {
			current_ruler = {
				ordered_child = {
					limit = {
						is_alive = yes  is_adult = yes  is_married = no  prisoner = no
						trigger_if = { limit = { scope:pact_senior = { var:marriage_our_child_female = 1 } }  is_female = no }
						trigger_else = { is_female = yes }
					}
					order_by = age
					max = 1
					save_scope_as = marriage_bride
				}
			}
		}
	}
	else = {
		scope:pact_junior = {
			ordered_character = {
				limit = {
					is_alive = yes  is_adult = yes  is_married = no  prisoner = no
					from_ruler_family = no
					NOT = { is_ruler = yes }
					trigger_if = { limit = { scope:pact_senior = { var:marriage_our_child_female = 1 } }  is_female = no }
					trigger_else = { is_female = yes }
				}
				order_by = age
				max = 1
				save_scope_as = marriage_bride
			}
		}
	}
	# 2. OUR child (of the filtered sex), married to the partner child. Guarded on the partner found.
	if = {
		limit = { exists = scope:marriage_bride  $from_ruler$ = yes }
		current_ruler = {
			ordered_child = {
				limit = {
					is_alive = yes  is_adult = yes  is_married = no  prisoner = no  NOT = { this = scope:marriage_bride }
					trigger_if = { limit = { scope:pact_senior = { var:marriage_our_child_female = 1 } }  is_female = yes }
					trigger_else = { is_male = yes }
				}
				order_by = age
				max = 1
				marry_character = scope:marriage_bride
				save_scope_as = marriage_groom
			}
		}
	}
	else_if = {
		limit = { exists = scope:marriage_bride }
		ordered_character = {
			limit = {
				is_alive = yes  is_adult = yes  is_married = no  prisoner = no
				from_ruler_family = no
				NOT = { is_ruler = yes }
				NOT = { this = scope:marriage_bride }
				trigger_if = { limit = { scope:pact_senior = { var:marriage_our_child_female = 1 } }  is_female = yes }
				trigger_else = { is_male = yes }
			}
			order_by = age
			max = 1
			marry_character = scope:marriage_bride
			save_scope_as = marriage_groom
		}
	}
	# 3. record the bond + diplomatic payload, but only if the wedding actually happened. The
	#    country-level payload (bond record, alliance, dowry) is the shared MARRIAGE_apply_marriage_bond
	#    helper (reused by betrothal maturation); the CHAR-specific loveless check stays here where the
	#    two wedded character scopes are available.
	if = {
		limit = {
			exists = scope:marriage_bride
			exists = scope:marriage_groom
		}
			# [#313] RULING-family match = full dynastic payload (bond record + alliance/entente + the union
			# road, since MARRIAGE_check_unions reads marriage_partners). OTHER-families match ($from_ruler=no)
			# = a LESSER tie: goodwill opinion only, NOT added to marriage_partners, so NO alliance/union/
			# inheritance can follow from it.
			if = {
				limit = { $from_ruler$ = yes }
				MARRIAGE_apply_marriage_bond = { partner = scope:pact_junior }
			}
			else = {
				scope:pact_senior = { add_opinion = { modifier = noble_marriage_opmod  target = scope:pact_junior } }
				scope:pact_junior = { add_opinion = { modifier = noble_marriage_opmod  target = scope:pact_senior } }
				LOG_line = { sys = MARRIAGE  msg = "lesser (non-ruling) marriage tie formed for" }
			}
			# [#330 user] FOREIGN-SPOUSE ORIGIN TAG — tag the incoming consort's homeland so the Grand
			# Council pulse (QING_council_foreign_spouse_check) can fire the homeland-lobby dilemma if she
			# later holds an office. Cheap + generic; only the Qing consume it. Applied for both ruling and
			# non-ruling matches. GUARD on the DEFAULT orientation (our son weds their daughter): only then
			# is scope:marriage_bride the PARTNER-court child who actually enters OUR house. In the daughter
			# orientation (marriage_our_child_female = 1) OUR daughter marries OUT and the "bride" scope is a
			# partner-court male who never joins our council — tagging him is dead data, so skip it.
			if = {
				limit = { NOT = { scope:pact_senior = { var:marriage_our_child_female = 1 } } }
				scope:marriage_bride = { set_variable = { name = marriage_origin_country  value = scope:pact_junior } }
			}
		# [#311 hook #10] a LOVELESS match: if the wedded pair are badly mismatched in temper
		# (a wide zeal gap — the fervent-vs-worldly friction the affinity engine already scores),
		# the union is unhappy and casts a shadow over the senior house's succession. Cheap read
		# on the two saved characters; stamps the dysfunction modifier on the senior realm.
		if = {
			limit = {
				exists = scope:marriage_groom
				exists = scope:marriage_bride
				OR = {
					AND = { scope:marriage_groom = { zeal >= 10 }  scope:marriage_bride = { zeal <= 3 } }
					AND = { scope:marriage_bride = { zeal >= 10 }  scope:marriage_groom = { zeal <= 3 } }
				}
			}
			scope:pact_senior = { MARRIAGE_apply_dysfunction = yes }
		}
		LOG_line = { sys = MARRIAGE  msg = "royal marriage concluded between two houses (dowry paid)" }
	}
	else = {
		LOG_fail = { sys = MARRIAGE  fn = "MARRIAGE_form_pact"  reason = "no marriageable pair found at wedding time" }
	}
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_form_pact"  result = OK }
}

# ======================================================================
# MARRIAGE_check_unions — L2. Scope: COUNTRY (senior house). For each realm we are
# wedded to, if THAT junior house is now heirless (its ruler has no living child to
# inherit) and it is a valid subordinate, bind it to us as a royal_union. Bounded to
# a single escalation per pulse (random_country over our wedded partners).
# ======================================================================
MARRIAGE_check_unions = {
	if = {
		limit = { var:marriage_bond_count > 0 }
		save_scope_as = union_senior
		random_country = {
			limit = {
				NOT = { this = scope:union_senior }
				# wedded to us (proven variable_list membership on the senior)
				scope:union_senior = {
					is_target_in_variable_list = { name = marriage_partners  target = prev }
				}
				# still an independent realm (not already someone's subject)
				is_subject = no
				exists = current_ruler
				# the junior house has failed — no living child to inherit at all.
				NOT = { current_ruler = { any_child = { is_alive = yes } } }
				# [#276 review-fix] the JUNIOR must not be larger than the senior. Without this
				# guard a small wedded realm could absorb a far more powerful heirless partner as
				# ITS royal_union (the crown passes to the bigger dynastic claim historically, not
				# the smaller) — a perverse minnow-swallows-whale union. num_of_cities is the
				# cheap available size proxy.
				num_of_cities <= scope:union_senior.num_of_cities
				# both still at peace with each other (a union is a settlement, not a conquest)
				NOT = { at_war = yes }
			}
			save_scope_as = union_junior
		}
		if = {
			limit = { exists = scope:union_junior }
			# form the personal union (proven verb; logic ours). The junior keeps its
			# own ruler until that ruler dies — inheritance is handled at death (L3).
			FUNC_make_subject = {
				overlord = scope:union_senior
				target = scope:union_junior
				type = royal_union
			}
			scope:union_junior = { set_variable = { name = marriage_union_junior  value = 1 } }
			LOG_line = { sys = MARRIAGE  msg = "heirless wedded house bound as a royal union" }
		}
	}
}

# ======================================================================
# [#324 / #312+#313 heiress-claim coupling] MARRIAGE_check_heiress_claim — L2b.
# Scope: COUNTRY (senior house). The FEMALE-LINE dimension of the dynastic game, and the
# concrete payoff of the #312 succession laws: where a wedded junior house has ONLY a
# DAUGHTER to inherit (a ruler with living daughters but no living son) AND that house's
# succession law BARS the female line (agnatic_succession_law), the heiress is
# disinherited at home — and if OUR house married into it, our line carries her claim.
# This is the Habsburg/Burgundy dynamic: Mary of Burgundy's hand brought the Burgundian
# inheritance to Maximilian precisely because the Salic bar made her a claim to be pressed,
# not a queen to be crowned. We grant our house a standing pressable claim on that realm's
# land, ONCE per partner (guarded by a per-target flag), so a female-line marriage can
# mature into a war of inheritance.
#
# The MIRROR case is handled by ABSENCE: if the junior house runs a female-PERMITTING law
# (cognatic — the #312 reform), the daughter inherits cleanly at home, so no disinheritance
# and no claim arises (the branch simply does not fire). Thus the succession LAW a realm
# adopts concretely changes which marriages breed pressable claims — exactly the coupling
# #312/#313 deferred. Bounded to one grant per pulse (random_country over our partners).
#   ROOT / this = the senior (our) house.
# ======================================================================
MARRIAGE_check_heiress_claim = {
	if = {
		limit = { var:marriage_bond_count > 0 }
		save_scope_as = heiress_senior
		random_country = {
			limit = {
				NOT = { this = scope:heiress_senior }
				# wedded to us (proven variable_list membership on the senior)
				scope:heiress_senior = {
					is_target_in_variable_list = { name = marriage_partners  target = prev }
				}
				exists = current_ruler
				# [#324 post-fix review] mirror MARRIAGE_check_unions' is_subject=no guard: never
				# grant an heiress claim on a realm already subordinated to us (our own royal_union
				# junior, or a third power's subject) — a bond stays in marriage_partners after a
				# union forms, so without this the daughter/no-son + agnatic state would breed a
				# redundant, semantically-wrong claim to press by war against a realm bound to us.
				is_subject = no
				# the junior house has a DAUGHTER but NO SON to inherit — the heiress case.
				current_ruler = {
					any_child = { is_alive = yes  is_female = yes }
					NOT = { any_child = { is_alive = yes  is_female = no } }
				}
				# ...and its law BARS her (agnatic). A cognatic realm lets her inherit -> no claim.
				has_law = agnatic_succession_law
				# not already granted a claim on this house's inheritance (once per partner).
				# [#324 review-fix] the marriage_heiress_claimed list lives on the SENIOR (it is
				# written on the senior at the grant below), so the membership test MUST re-enter
				# the senior's scope — exactly like the marriage_partners test above. Read at the
				# bare candidate scope it inspected the JUNIOR's (never-written) list and the NOT
				# always passed, leaking a duplicate + re-firing the claim every pulse.
				scope:heiress_senior = {
					NOT = { is_target_in_variable_list = { name = marriage_heiress_claimed  target = prev } }
				}
			}
			save_scope_as = heiress_junior
		}
		if = {
			limit = { exists = scope:heiress_junior }
			# record the grant so it fires once per partner, then press the claim on the
			# heiress's realm (the proven claim-grant path shared with the succession-war hook).
			add_to_variable_list = { name = marriage_heiress_claimed  target = scope:heiress_junior }
			MARRIAGE_press_succession_claim = { realm = scope:heiress_junior }
			LOG_line = { sys = MARRIAGE  msg = "female-line heiress barred by agnatic law -> our house presses her inheritance claim" }
		}
	}
}

# ======================================================================
# MARRIAGE_on_ruler_death_union — L3. Scope: CHARACTER (a ruler who just died).
# Called from on_character_death. If the dead ruler headed a royal_union junior we
# created by marriage (marriage_union_junior) AND that junior is now truly extinct
# (no living heir), the union INHERITS: the overlord takes the junior's land via the
# PROVEN LAND_transfer_provinces idiom (se_LAND.txt), NOT an unproven annex verb.
# Self-guards throughout.
# ======================================================================
MARRIAGE_on_ruler_death_union = {
	# only rulers of a marriage-made royal_union matter here.
	if = {
		limit = {
			is_ruler = yes
			exists = employer
			employer = {
				has_variable = marriage_union_junior
				is_subject = yes
				is_subject_type = royal_union
			}
		}
		employer = {
			save_scope_as = union_dying_realm
			# if the house is now truly extinct (no living heir), the union inherits.
			if = {
				limit = {
					NOT = { current_ruler = { any_child = { is_alive = yes } } }
					exists = overlord
				}
				overlord = { save_scope_as = union_inheritor }
				# [#311 hook #8] CONTESTED inheritance: if ANOTHER realm was ALSO wedded to the dying
				# house AND is a near-equal of the overlord (a rival dynastic claim of comparable weight),
				# the succession is DISPUTED — no clean inheritance; both press the claim by arms (the War
				# of the Spanish/Austrian/Bavarian Succession pattern). Set a guard so the clean land
				# transfer below is skipped, and grant claims to the overlord + the rival.
				set_variable = { name = union_contested  value = 0 }
				if = {
					limit = {
						any_country = {
							NOT = { this = scope:union_inheritor }
							NOT = { this = scope:union_dying_realm }
							exists = current_ruler
							is_target_in_variable_list = { name = marriage_partners  target = scope:union_dying_realm }
							num_of_cities >= scope:union_inheritor.num_of_cities
						}
					}
					set_variable = { name = union_contested  value = 1 }
					random_country = {
						limit = {
							NOT = { this = scope:union_inheritor }
							NOT = { this = scope:union_dying_realm }
							exists = current_ruler
							is_target_in_variable_list = { name = marriage_partners  target = scope:union_dying_realm }
							num_of_cities >= scope:union_inheritor.num_of_cities
						}
						save_scope_as = union_rival_claimant
					}
					scope:union_inheritor = { MARRIAGE_press_succession_claim = { realm = scope:union_dying_realm } }
					if = {
						limit = { exists = scope:union_rival_claimant }
						scope:union_rival_claimant = { MARRIAGE_press_succession_claim = { realm = scope:union_dying_realm } }
					}
					LOG_line = { sys = MARRIAGE  msg = "CONTESTED succession — a rival dynastic claim; pressed by arms, no clean inheritance" }
				}
				LOG_line = { sys = MARRIAGE  msg = "royal-union junior house extinct -> inheritance" }
				# gather the whole junior realm's land into a NAMED list (the idiom
				# LAND_transfer_provinces expects — target_provinces takes a plain list
				# built with add_to_list, cf. play_cede_candidates in se_DIPLOMACY), then
				# hand it to the overlord in one sanctioned transfer. LAND_transfer_provinces
				# is the ONLY correct path (se_LAND.txt): it re-homes the governorship vars
				# a raw engine ownership swap would strand.
				scope:union_dying_realm = {
					every_owned_province = {
						limit = { is_sea = no }
						add_to_list = marriage_inherit_provs
					}
				}
				if = {
					# [#311 hook #8] clean inheritance ONLY if the succession is uncontested; a
					# contested one (guard set above) is decided by war via the pressed claims, not
					# by an automatic land transfer.
					limit = {
						any_in_list = { list = marriage_inherit_provs  count >= 1 }
						var:union_contested = 0
					}
					scope:union_inheritor = {
						LAND_transfer_provinces = {
							target_provinces = marriage_inherit_provs
							grantee = scope:union_inheritor
						}
						LOG_line = { sys = MARRIAGE  msg = "royal-union realm inherited by overlord" }
					}
				}
				# [#276 review] defensive clear of the temp transfer list, matching the peer
				# idiom (se_DIPLOMACY.txt:1003 play_cede_candidates). on_character_death runs this
				# as a fresh effect so the list is empty per-invocation, but the explicit teardown
				# removes any doubt about stale carry-over.
				every_in_list = { list = marriage_inherit_provs  remove_from_list = marriage_inherit_provs }
				# the inherited house is gone: drop the standing bond from EVERY realm wedded
				# to it — not only the inheriting overlord. [#276 review-fix] the extinct junior
				# could have wed several houses (bond cap is 3); pruning only the overlord left
				# any THIRD-party partner with a phantom marriage_partners entry + inflated
				# marriage_bond_count forever (a dead realm blocking a live match slot). Iterate
				# the dying realm's OWN partner list (holds every realm wedded to it, overlord
				# included) and prune the reciprocal bond on each. The dying realm's list is torn
				# down with the realm, so it needs no self-cleanup.
				scope:union_dying_realm = {
					every_in_list = {
						variable = marriage_partners
						save_scope_as = union_ex_partner
						scope:union_ex_partner = {
							if = {
								limit = { is_target_in_variable_list = { name = marriage_partners  target = scope:union_dying_realm } }
								remove_list_variable = { name = marriage_partners  target = scope:union_dying_realm }
								change_variable = { name = marriage_bond_count  subtract = 1 }
							}
							# [#311 engine-review fix] also prune the created-alliance list, else the
							# dead realm leaves a stale marriage_alliance_partners entry that keeps the
							# decay guard's list non-empty forever (a phantom that never lapses).
							if = {
								limit = { is_target_in_variable_list = { name = marriage_alliance_partners  target = scope:union_dying_realm } }
								remove_list_variable = { name = marriage_alliance_partners  target = scope:union_dying_realm }
							}
							# [#324] and the heiress-claim once-guard, so the dead realm leaves no stale
							# entry (harmless, but keeps the list clean and lets a future bond re-trigger).
							if = {
								limit = { is_target_in_variable_list = { name = marriage_heiress_claimed  target = scope:union_dying_realm } }
								remove_list_variable = { name = marriage_heiress_claimed  target = scope:union_dying_realm }
							}
						}
					}
				}
			}
		}
	}
}

# ======================================================================
# [#311 hook #3] MARRIAGE_grant_dowry_claim — a strong match carries a territorial claim
# in the contract (the road by which marriages moved provinces: Burgundy→Maximilian). Call
# in the SENIOR country scope; $from$ = the junior partner COUNTRY. Grants the senior a
# pressable claim on ONE of the junior's provinces that BORDERS the senior (a claim you can
# actually reach). Guarded so it is a clean no-op if the junior has no adjacent province.
# ======================================================================
MARRIAGE_grant_dowry_claim = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_grant_dowry_claim" }
	save_scope_as = dowry_senior
	if = {
		limit = {
			$from$ = {
				any_owned_province = {
					is_sea = no
					NOT = { owner = scope:dowry_senior }
					any_neighbor_province = { owner = scope:dowry_senior }
				}
			}
		}
		$from$ = {
			random_owned_province = {
				limit = {
					is_sea = no
					NOT = { owner = scope:dowry_senior }
					any_neighbor_province = { owner = scope:dowry_senior }
				}
				save_scope_as = dowry_claim_prov
			}
		}
		if = {
			limit = { exists = scope:dowry_claim_prov }
			scope:dowry_senior = { add_claim = scope:dowry_claim_prov }
			LOG_line = { sys = MARRIAGE  msg = "dowry carried a border claim into the senior house" }
		}
	}
	else = {
		LOG_fail = { sys = MARRIAGE  fn = "MARRIAGE_grant_dowry_claim"  reason = "junior has no border province adjacent to senior; no dowry-claim" }
	}
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_grant_dowry_claim"  result = OK }
}

# ======================================================================
# [#311 hook #7] MARRIAGE_forge_bridge — establish a marriage BRIDGE with a house across the
# confessional/cultural line (a papal dispensation, or a bride's conversion — Catherine née
# Sophie of Anhalt-Zerbst converting to Orthodoxy, 1744). This is the ONLY way a cross-bloc /
# cross-faith pair becomes MARRIAGE_pair_compatible. Records the bond BOTH ways (bridges are
# mutual) via the proven variable_list idiom. Call in the PROPOSER country scope; $other$ =
# the partner COUNTRY. Player-driven (from the marriage GUI); costs are applied by the caller.
# ======================================================================
MARRIAGE_forge_bridge = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_forge_bridge" }
	MARRIAGE_init = yes
	save_scope_as = bridge_home
	if = {
		limit = { NOT = { is_target_in_variable_list = { name = marriage_bridge_partners  target = $other$ } } }
		add_to_variable_list = { name = marriage_bridge_partners  target = $other$ }
	}
	$other$ = {
		MARRIAGE_init = yes
		if = {
			limit = { NOT = { is_target_in_variable_list = { name = marriage_bridge_partners  target = scope:bridge_home } } }
			add_to_variable_list = { name = marriage_bridge_partners  target = scope:bridge_home }
		}
	}
	LOG_line = { sys = MARRIAGE  msg = "a marriage bridge (dispensation/conversion) was forged between two houses" }
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_forge_bridge"  result = OK }
}

# ======================================================================
# [#311 hook #15 / #betrothal-rework] MARRIAGE_betroth — a CHILD BETROTHAL: a binding pledge that
# two SPECIFIC children (who may be MINORS) will wed once BOTH reach adulthood. Historically the
# arranged childhood betrothal (Marie-Antoinette pledged years before her 1770 wedding). Unlike a
# marriage, this can be pledged for children not yet of age — that IS the point.
#
# Call in the PROPOSER country scope; $other$ = the partner COUNTRY. We pick one eligible child of
# each ruling house (opposite sex; the senior house supplies the GROOM, the partner the BRIDE — the
# same senior=groom convention MARRIAGE_form_pact uses) and bind them:
#   • each child gets a CHARACTER var betrothed_to_char pointing at the partner child (the proven TI
#     `set_variable value=scope:X days=N` idiom — a self-expiring cross-character reference), and a
#     betrothed_partner_country var, so the maturation checker + death handler can find the pair.
#   • the COUNTRY-level marriage_betrothed list + royal_betrothal_opmod goodwill are kept (they drive
#     the diplomacy UI / repudiation-insult exactly as before).
# If either house cannot field an eligible child the pledge is a no-op (logged). Maturation into an
# actual marriage is handled by MARRIAGE_check_betrothals (from the pulse); a child's death dissolves
# it WITHOUT penalty (an act of God), also in that checker.
# ======================================================================
MARRIAGE_betroth = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_betroth" }
	MARRIAGE_init = yes
	save_scope_as = betroth_home
	$other$ = { MARRIAGE_init = yes  save_scope_as = betroth_other }
	# 1. a betrothable child of the PARTNER house (the bride side): from the ruling family, alive,
	#    unwed, free, not already betrothed. MINORS ALLOWED (no is_adult gate) — the whole point.
	scope:betroth_other = {
		current_ruler = {
			ordered_child = {
				limit = {
					is_alive = yes
					is_married = no
					prisoner = no
					is_female = yes
					NOT = { has_variable = betrothed_to_char }
				}
				order_by = age
				max = 1
				save_scope_as = betroth_bride
			}
		}
	}
	# 2. a betrothable child of OUR house (the groom side), guarded on the bride being found.
	if = {
		limit = { exists = scope:betroth_bride }
		current_ruler = {
			ordered_child = {
				limit = {
					is_alive = yes
					is_married = no
					prisoner = no
					is_male = yes
					NOT = { has_variable = betrothed_to_char }
					NOT = { this = scope:betroth_bride }
				}
				order_by = age
				max = 1
				save_scope_as = betroth_groom
			}
		}
	}
	# 3. bind the pair + record the country-level pledge, only if BOTH children were found.
	if = {
		limit = {
			exists = scope:betroth_bride
			exists = scope:betroth_groom
		}
		# --- character-level bond: each child points at the other + at the partner COUNTRY.
		#     Self-expiring after 40 years (a betrothal that never matures lapses quietly).
		scope:betroth_groom = {
			set_variable = { name = betrothed_to_char        value = scope:betroth_bride  days = 14600 }
			set_variable = { name = betrothed_partner_country value = scope:betroth_other  days = 14600 }
		}
		scope:betroth_bride = {
			set_variable = { name = betrothed_to_char        value = scope:betroth_groom  days = 14600 }
			set_variable = { name = betrothed_partner_country value = scope:betroth_home   days = 14600 }
		}
		# --- [#betrothal-review B18 fix] RULER-INDEPENDENT anchor: record OUR betrothed child on a
		#     COUNTRY-level list so the pledge survives a SUCCESSION. Previously the child was re-found
		#     only via current_ruler.ordered_child — but a betrothal binds MINORS for years, so when the
		#     home ruler dies mid-wait the child is no longer current_ruler's child (it is a sibling of
		#     the new ruler, or IS the new ruler), the lookup found nothing, and the pledge was spuriously
		#     dissolved as if the child had died / could never mature. Each house stores its OWN betrothed
		#     child here; the checker/dissolve/repudiate resolve the child through this list (guarded
		#     is_alive), never through the ruler. betroth_home stores its groom; betroth_other its bride.
		scope:betroth_home  = { add_to_variable_list = { name = betrothed_children  target = scope:betroth_groom } }
		scope:betroth_other = { add_to_variable_list = { name = betrothed_children  target = scope:betroth_bride } }
		# --- country-level bond both ways: the diplomacy pledge + betrothal goodwill (as before).
		if = {
			limit = { NOT = { is_target_in_variable_list = { name = marriage_betrothed  target = scope:betroth_other } } }
			add_to_variable_list = { name = marriage_betrothed  target = scope:betroth_other }
			add_opinion = { modifier = royal_betrothal_opmod  target = scope:betroth_other }
		}
		scope:betroth_other = {
			if = {
				limit = { NOT = { is_target_in_variable_list = { name = marriage_betrothed  target = scope:betroth_home } } }
				add_to_variable_list = { name = marriage_betrothed  target = scope:betroth_home }
				add_opinion = { modifier = royal_betrothal_opmod  target = scope:betroth_home }
			}
		}
		LOG_line = { sys = MARRIAGE  msg = "a child betrothal was pledged (two specific children bound to wed at adulthood)" }
	}
	else = {
		LOG_fail = { sys = MARRIAGE  msg = "betrothal no-op: a house could not field an eligible child" }
	}
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_betroth"  result = OK }
}

# ======================================================================
# [#betrothal-rework] MARRIAGE_check_betrothals — mature or dissolve standing child betrothals.
# Called from MARRIAGE_pulse (country scope, the home realm). For each partner in our
# marriage_betrothed list, find the betrothed CHILD PAIR (via the betrothed_to_char character vars)
# and act:
#   • BOTH children alive + adult + unwed  → MARRY them (the pledge is kept: convert to a full
#     marriage bond via MARRIAGE_form_pact's payload), clear the betrothal cleanly.
#   • either child DEAD/absent            → dissolve the betrothal with NO penalty (an act of God,
#     per the design: "not if one child dies"); clear the country pledge + the surviving child's var.
#   • otherwise (still minors)            → leave the pledge standing.
# Bounded: iterates our own betrothal list (cap 3-ish) and, per entry, our ruler's children. O(1)
# amortised. We locate OUR betrothed child by matching its betrothed_partner_country to the partner.
# ======================================================================
MARRIAGE_check_betrothals = {
	if = {
		limit = { has_variable_list = marriage_betrothed }
		save_scope_as = betroth_check_home
		every_in_list = {
			variable = marriage_betrothed
			save_scope_as = betroth_check_partner
			# NOTE: inside this every_in_list the SCOPE is the listed PARTNER country. All HOME-side
			# operations must run inside scope:betroth_check_home explicitly.
			scope:betroth_check_home = {
				# [#betrothal-review B18 fix] clear any stale flag from a PRIOR partner iteration, so a
				# partner with no matching child below cannot inherit the previous partner's child.
				remove_variable = betroth_our_child_found
				# find OUR betrothed child pledged to THIS partner via the RULER-INDEPENDENT betrothed_children
				# list (survives succession — the child may no longer be current_ruler's offspring). Match on
				# the child's own betrothed_partner_country pointer + still alive.
				ordered_in_list = {
					variable = betrothed_children
					limit = {
						is_alive = yes
						has_variable = betrothed_partner_country
						var:betrothed_partner_country = scope:betroth_check_partner
					}
					order_by = age
					max = 1
					save_scope_as = betroth_our_child
					scope:betroth_check_home = { set_variable = betroth_our_child_found }
				}
			}
			# resolve the partner child from our child's own pointer (guard: our child may be gone).
			# All the resolution effects run in the HOME scope (this every_in_list scope is the partner).
			if = {
				limit = { scope:betroth_check_home = { has_variable = betroth_our_child_found } }
				if = {
					# --- our child's betrothed partner is DEAD/absent → dissolve, NO penalty.
					limit = { NOT = { exists = scope:betroth_our_child.var:betrothed_to_char } }
					scope:betroth_check_home = { MARRIAGE_dissolve_betrothal = { partner = scope:betroth_check_partner  reason = death } }
				}
				else = {
					scope:betroth_our_child.var:betrothed_to_char = { save_scope_as = betroth_their_child }
					if = {
						# --- BOTH alive + adult + unwed → the pledge matures into a marriage.
						limit = {
							scope:betroth_our_child = { is_alive = yes  is_adult = yes  is_married = no }
							scope:betroth_their_child = { is_alive = yes  is_adult = yes  is_married = no }
						}
						scope:betroth_check_home = { MARRIAGE_mature_betrothal = { partner = scope:betroth_check_partner } }
					}
					# else: at least one still a minor (or transiently unwed-blocked) → leave standing.
				}
			}
			else = {
				# our betrothed child is gone (died) → dissolve, NO penalty.
				scope:betroth_check_home = { MARRIAGE_dissolve_betrothal = { partner = scope:betroth_check_partner  reason = death } }
			}
		}
	}
}

# ======================================================================
# [#betrothal-rework] MARRIAGE_mature_betrothal — the pledge is KEPT: wed the two betrothed children
# (both now adult) and upgrade the country bond from betrothal to full marriage. Country scope (home);
# $partner$ = the partner COUNTRY. Requires scope:betroth_our_child + scope:betroth_their_child set by
# the caller (MARRIAGE_check_betrothals). Marries the specific pledged pair (NOT a fresh child pick),
# then applies the standing marriage payload via the shared helper.
# ======================================================================
MARRIAGE_mature_betrothal = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_mature_betrothal" }
	save_scope_as = pact_senior
	$partner$ = { save_scope_as = pact_junior }
	# wed the exact pledged children (bride = the female of the pair, for marry_character's opposite-sex
	# requirement — one of the two is female by construction of MARRIAGE_betroth).
	if = {
		limit = { scope:betroth_our_child = { is_female = yes } }
		scope:betroth_their_child = { marry_character = scope:betroth_our_child }
	}
	else = {
		scope:betroth_our_child = { marry_character = scope:betroth_their_child }
	}
	# upgrade the country bond: drop the betrothal pledge (the shared teardown clears the children's
	# betrothal vars + the betrothed_children list entries on BOTH courts, succession-safe), then lay
	# down the full marriage payload.
	MARRIAGE_clear_country_betrothal = { other = scope:pact_junior }
	MARRIAGE_apply_marriage_bond = { partner = scope:pact_junior }
	# [#330 user] FOREIGN-SPOUSE ORIGIN TAG — betroth_their_child (the partner court's child) weds INTO
	# our house, so tag them with their homeland (the junior court) for the Grand Council lobby pulse.
	scope:betroth_their_child = { set_variable = { name = marriage_origin_country  value = scope:pact_junior } }
	LOG_line = { sys = MARRIAGE  msg = "a betrothal MATURED into marriage as both children came of age" }
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_mature_betrothal"  result = OK }
}

# ======================================================================
# [#betrothal-rework] MARRIAGE_dissolve_betrothal — end a betrothal. $partner$ = the partner COUNTRY,
# $reason$ = death (no penalty — an act of God) or lapse. Clears the country pledge both ways and any
# surviving betrothed child's character vars. A pure teardown; NO jilted grievance (that belongs only
# to a WILFUL break — MARRIAGE_repudiate). Country scope (home).
# ======================================================================
MARRIAGE_dissolve_betrothal = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_dissolve_betrothal" }
	# [#betrothal-review B18 fix] the full teardown (country pledge both ways + betrothed_children list
	# entries + the children's char vars on both courts, matched by partner-country) is centralised in
	# MARRIAGE_clear_country_betrothal, which is succession-safe — no current_ruler.every_child (which
	# missed a child re-parented by a succession, leaving stale vars until the 40-yr self-expiry).
	MARRIAGE_clear_country_betrothal = { other = $partner$ }
	LOG_line = { sys = MARRIAGE  msg = "a betrothal dissolved ($reason$) — no grievance (not a wilful break)" }
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_dissolve_betrothal"  result = OK }
}

# ======================================================================
# [#betrothal-rework / B18 fix] MARRIAGE_clear_country_betrothal — the SINGLE teardown for a betrothal
# with $other$: clears the COUNTRY-level pledge (marriage_betrothed) both ways, AND the ruler-independent
# betrothed_children list entries + the pledged children's character vars on BOTH courts. Matched purely
# by the partner-country pointer, so it works regardless of any succession that re-parented the children.
# Shared by maturation, dissolution, and repudiation so every exit path cleans symmetrically. The child
# scopes are captured into a scratch list first because removing from a variable_list mid-iteration is
# unsafe. Country scope (home).
# ======================================================================
MARRIAGE_clear_country_betrothal = {
	save_scope_as = clearbetroth_home
	# --- OUR side: find each betrothed child pledged to $other$, clear its vars, drop it from the list.
	clear_variable_list = betroth_clear_scratch
	every_in_list = {
		variable = betrothed_children
		limit = {
			has_variable = betrothed_partner_country
			var:betrothed_partner_country = $other$
		}
		scope:clearbetroth_home = { add_to_variable_list = { name = betroth_clear_scratch  target = prev } }
	}
	every_in_list = {
		variable = betroth_clear_scratch
		save_scope_as = betroth_clear_child
		scope:betroth_clear_child = { remove_variable = betrothed_to_char  remove_variable = betrothed_partner_country }
		scope:clearbetroth_home = { remove_list_variable = { name = betrothed_children  target = scope:betroth_clear_child } }
	}
	clear_variable_list = betroth_clear_scratch
	# --- country-list pledge, our side.
	if = {
		limit = { is_target_in_variable_list = { name = marriage_betrothed  target = $other$ } }
		remove_list_variable = { name = marriage_betrothed  target = $other$ }
	}
	# --- THE PARTNER'S side: symmetric cleanup of its child pointed back at us + its country-list entry.
	$other$ = {
		clear_variable_list = betroth_clear_scratch
		every_in_list = {
			variable = betrothed_children
			limit = {
				has_variable = betrothed_partner_country
				var:betrothed_partner_country = scope:clearbetroth_home
			}
			prev = { add_to_variable_list = { name = betroth_clear_scratch  target = prev } }
		}
		every_in_list = {
			variable = betroth_clear_scratch
			save_scope_as = betroth_clear_child2
			scope:betroth_clear_child2 = { remove_variable = betrothed_to_char  remove_variable = betrothed_partner_country }
			prev = { remove_list_variable = { name = betrothed_children  target = scope:betroth_clear_child2 } }
		}
		clear_variable_list = betroth_clear_scratch
		if = {
			limit = { is_target_in_variable_list = { name = marriage_betrothed  target = scope:clearbetroth_home } }
			remove_list_variable = { name = marriage_betrothed  target = scope:clearbetroth_home }
		}
	}
}

# ======================================================================
# [#311 hook #5] MARRIAGE_repudiate — BREAK a standing betrothal: a grave diplomatic insult
# (the 1725 Louis XV / Spanish infanta affair, which needed Fleury's repair diplomacy). Call in
# the REPUDIATING country scope; $other$ = the jilted partner COUNTRY. Strips the betrothal bond
# both ways, lands a heavy opinion penalty + a lasting "jilted" grievance modifier on the injured
# house, and breaks any standing alliance between them. The repudiator takes a prestige hit.
# ======================================================================
MARRIAGE_repudiate = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_repudiate" }
	save_scope_as = repudiator
	# [#betrothal-review B18 fix] full teardown of the betrothal both ways (country pledge + the
	# betrothed_children list entries + the children's char vars on both courts) via the succession-safe
	# shared helper — no current_ruler.every_child (which missed a succession-re-parented child).
	MARRIAGE_clear_country_betrothal = { other = $other$ }
	# the jilted house's grievance: a deep opinion hit + a lasting resentment modifier on THEM
	# toward us, and the alliance (if any) is broken. This is the WILFUL-break penalty; a betrothal
	# ended by a child's DEATH goes through MARRIAGE_dissolve_betrothal instead and carries NO penalty.
	$other$ = {
		add_opinion = { modifier = royal_jilted_opmod  target = scope:repudiator }
		if = {
			limit = { alliance_with = scope:repudiator }
			break_alliance = scope:repudiator
			# [#311 engine-review fix] if that broken alliance was a marriage-created one, drop the
			# entry from the created-alliance list so decay never targets a dead alliance later.
			if = {
				limit = { is_target_in_variable_list = { name = marriage_alliance_partners  target = scope:repudiator } }
				remove_list_variable = { name = marriage_alliance_partners  target = scope:repudiator }
			}
		}
	}
	# the affront also costs the repudiator standing at the courts of Europe.
	current_ruler = { add_prestige = -30 }
	LOG_line = { sys = MARRIAGE  msg = "a betrothal was REPUDIATED — a grave insult to the jilted house" }
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_repudiate"  result = OK }
}

# ======================================================================
# [#311 hook #2] MARRIAGE_check_decay — let dynastic alliances FADE with the generation that
# made them. Called from the pulse. If our marriage-alliance timer has run out (the couple's
# generation has passed) AND the bond has not been renewed by a fresh marriage, the formal
# alliance lapses to mere goodwill — the historical decay of the Family Compacts across reigns.
# Country scope (the home realm). O(partners).
# ======================================================================
MARRIAGE_check_decay = {
	if = {
		# the timer variable is REMOVED by the engine when it expires (days= form), so its
		# ABSENCE — WHILE we still hold at least one marriage-CREATED alliance — is the decay signal.
		# [#311 engine-review fix] Gate on the dedicated marriage_alliance_partners list, NOT on
		# marriage_bond_count. bond_count>0 with no timer is a FALSE decay signal in the common case
		# where a couple wed while already allied (the alliance branch was skipped, so no timer was
		# ever set) — which previously made decay break that pre-existing, unrelated alliance every
		# pulse. Requiring a non-empty marriage_alliance_partners list means we only decay when we
		# genuinely created a marriage alliance whose timer has now run out.
		limit = {
			has_variable_list = marriage_alliance_partners
			NOT = { has_variable = marriage_alliance_timer }
		}
		save_scope_as = decay_home
		# lapse the alliance with each partner whose alliance WE created and are still allied to but
		# whose bond has aged out; the goodwill opinion + the marriage bond itself remain (the houses
		# are still kin), only the binding military alliance fades. Iterate the created-alliance list
		# so an independently-negotiated alliance with a wedded house is never touched. The shared
		# timer means the whole generation of marriage alliances lapses together (the intended
		# realm-aggregate Family-Compacts model), so we CLEAR the list AFTER the loop rather than
		# pruning per-entry (mutating a list mid-iteration is unsafe). A later re-marriage re-stamps
		# the timer and re-populates the list, starting a fresh generation cleanly.
		every_in_list = {
			variable = marriage_alliance_partners
			save_scope_as = decay_partner
			scope:decay_home = {
				# a lesser-pair bond is a real alliance → break it; a great-power bond is the
				# royal_marriage_entente modifier → strip it from both courts. Handle whichever applies.
				if = {
					limit = { alliance_with = scope:decay_partner }
					break_alliance = scope:decay_partner
					LOG_line = { sys = MARRIAGE  msg = "a dynastic alliance lapsed as its generation passed (bond remains)" }
				}
				if = {
					limit = { has_country_modifier = royal_marriage_entente }
					remove_country_modifier = royal_marriage_entente
				}
				# [#330 user] the overlord-subject dynastic tie also lapses with the generation.
				if = {
					limit = { has_country_modifier = royal_marriage_subject_tie }
					remove_country_modifier = royal_marriage_subject_tie
				}
			}
			scope:decay_partner = {
				if = {
					limit = { has_country_modifier = royal_marriage_entente }
					remove_country_modifier = royal_marriage_entente
				}
				if = {
					limit = { has_country_modifier = royal_marriage_subject_tie }
					remove_country_modifier = royal_marriage_subject_tie
				}
			}
		}
		scope:decay_home = { clear_variable_list = marriage_alliance_partners }
	}
}

# ======================================================================
# [#311 hook #10] MARRIAGE_apply_dysfunction — a LOVELESS/incompatible match is a dynastic
# liability (Joseph II's barren, unhappy second marriage; Frederick the Great's estrangement).
# Call in the country scope of a house whose ruler's marriage is incompatible; stamps a lasting
# court-discord modifier (a legitimacy + fertility drag). Guarded once. Used by the union/
# succession events when a match is judged incompatible (opposed traits / no heir produced).
# ======================================================================
MARRIAGE_apply_dysfunction = {
	LOG_enter = { sys = MARRIAGE  fn = "MARRIAGE_apply_dysfunction" }
	if = {
		limit = { NOT = { has_country_modifier = royal_marriage_dysfunction } }
		add_country_modifier = { name = royal_marriage_dysfunction  duration = -1 }
		LOG_line = { sys = MARRIAGE  msg = "a loveless dynastic marriage breeds court discord" }
	}
	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_apply_dysfunction"  result = OK }
}

# ======================================================================
# [#311 hook #8] MARRIAGE_press_succession_claim — when a wedded junior house dies out and the
# union inheritance is CONTESTED (a near-equal rival also has a marriage claim), grant the
# senior a pressable claim + a succession-war casus belli rather than a clean inheritance — the
# War of the Spanish/Austrian/Bavarian Succession pattern. Call in the SENIOR country scope;

MARRIAGE_press_succession_claim = {

	LOG_exit = { sys = MARRIAGE  fn = "MARRIAGE_press_succession_claim"  result = OK }
}

# ======================================================================
# [#311 hook #13, per user] DYNASTIC BLOC WEIGHT — a simple NUMERIC indicator of how much
# territory a bloodline's marriage network commands, comparable across houses (the Habsburg-
# vs-Bourbon-bloc measure), instead of a bespoke mapmode. Bloc weight = our own num_of_cities
# PLUS the num_of_cities of every realm we are directly wedded to (marriage_partners). Stored
# as qing_marriage_bloc_weight on the country + refreshed each pulse; the GUI/tooltip reads it
# and it can be compared to any other realm's same var to see who leads the dynastic game.
# Country scope (ROOT = the home realm). O(partners). NOT double-counting: a straight sum of
# the network's cities (a partner shared by two houses counts in each — bloc weight is a reach
# measure, not a partition).
# ======================================================================
MARRIAGE_recompute_bloc_weight = {
	MARRIAGE_init = yes
	save_scope_as = bloc_home
	# start with our own realm's cities.
	set_variable = { name = qing_marriage_bloc_weight  value = num_of_cities }
	# add each wedded partner's cities (guarded: a partner may have been annexed/removed).
	if = {
		limit = { has_variable_list = marriage_partners }
		every_in_list = {
			variable = marriage_partners
			limit = { exists = this  is_country = yes }
			# read the partner's city count into a scratch, then fold it into the home total.
			save_scope_as = bloc_partner
			set_variable = { name = bloc_partner_cities  value = num_of_cities }
			scope:bloc_home = {
				change_variable = { name = qing_marriage_bloc_weight  add = scope:bloc_partner.var:bloc_partner_cities }
			}
		}
	}
	LOG_line = { sys = MARRIAGE  msg = "dynastic bloc weight recomputed (own + wedded partners' cities) for" }
}

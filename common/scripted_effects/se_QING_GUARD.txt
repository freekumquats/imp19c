# emperor-power-base + sovereign-vitality mechanics in se_QING_COUNCIL.txt, unchanged) two

#   • a CORPS of 侍衛 guardsmen — real courtiers enrolled into the guard (marked
#     qing_is_imperial_guardsman), rebuilt into qing_guard_corps each pulse by
#     QING_ministry_recompute_perf_guard_commandant (se_QING_MINISTRY.txt);
#   • the BAYARA GUARD (巴牙喇) — a real elite legion the player musters at the capital
#     (a raise-once standing institution; readiness flag qing_bayara_guard_raised).

# The corps depth + Bayara readiness fold into qing_min_perf_guard_commandant, which folds

# ======================================================================
# QING_guard_raise_bayara — muster the elite Bayara Guard (巴牙喇) at the capital. A
# raise-once standing institution: 4 line battalions + 1 field-gun piece (the guard was
# an elite shock force, not an artillery train). Scope: COUNTRY (ROOT = CHI). Guarded on
# NOT already raised so it cannot be double-mustered; sets the readiness flag the perf
# compute reads. Mirrors QING_ili_raise_reconquest_legion (capital muster fallback).
# ======================================================================
QING_guard_raise_bayara = {
	LOG_enter = { sys = QING  fn = "QING_guard_raise_bayara" }
	if = {
		limit = {
			NOT = { has_variable = qing_bayara_guard_raised }
			exists = capital_scope
		}
		capital_scope = { state.governorship = { QING_guard_raise_bayara_at = { loc = ROOT.capital_scope } } }
		# mark the guard under arms — read by QING_ministry_recompute_perf_guard_commandant (+10).
		set_variable = { name = qing_bayara_guard_raised  value = 1 }
		LOG_line = { sys = QING  msg = "guard: the Bayara Guard is mustered at the capital for" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_guard_raise_bayara"  reason = "already raised or no capital to muster the Bayara Guard for" }
	}
	LOG_exit = { sys = QING  fn = "QING_guard_raise_bayara"  result = OK }
}

# Raise one Bayara Guard legion at $loc$ (a province scope). Called in governorship scope.
# An elite shock host: 4 line battalions + 1 field gun. Mirrors QING_ili_raise_reconquest_legion_at.
QING_guard_raise_bayara_at = {
	# $loc$ = province scope for the legion's muster point
	raise_legion = {
		create_unit = {
			name = "Bayara Guard"
			location = $loc$
			sub_unit = regular_infantry
			add_subunit = regular_infantry
			add_subunit = regular_infantry
			add_subunit = regular_infantry
			add_subunit = artillery
		}
	}
}

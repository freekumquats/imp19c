#   вҖў йҖ иҫҰиҷ• Imperial Workshops (under the е…§еӢҷеәң): jade, Jingdezhen imperial-kiln еҫЎзӘҜе»

# State (all on CHI):

# ======================================================================
# INIT вҖ” seed the patronage meter. Called once at game start (after QING_DECLINE_init
# so the era flag is available). The 1763 zenith begins mid-high (Qianlong's golden
# age); a later start begins lower. Idempotent. Scope: country (CHI).
# ======================================================================
QING_wenzhi_init = {
	LOG_enter = { sys = QING  fn = "QING_wenzhi_init" }
	if = {
		limit = { NOT = { has_variable = qing_wenzhi_patronage } }
		if = {
			limit = { has_variable = qing_high_qing_era }
			set_variable = { name = qing_wenzhi_patronage  value = 40 }
		}
		else = {
			set_variable = { name = qing_wenzhi_patronage  value = 20 }
		}
	}
	LOG_line = { sys = QING  msg = "wenzhi: patronage meter seeded for" }
	LOG_exit = { sys = QING  fn = "QING_wenzhi_init"  result = OK }
}

# ======================================================================
# QUARTERLY PULSE вҖ” called from QING_GOV_pulse. Patronage decays slowly (the glory of
# the court must be renewed by fresh commissions), and a HIGH standing radiates ambient
# legitimacy (a court famed for its splendour affirms the Mandate). Bounded [0, 100].
# Scope: country (CHI).
# ======================================================================
QING_wenzhi_pulse = {
	if = {
		limit = { has_variable = qing_wenzhi_patronage }
		# slow decay: -1/quarter (a still court's lustre fades)
		QING_DECLINE_nudge = { var = qing_wenzhi_patronage  amount = -1 }
		# ambient legitimacy while the court's patronage stands high (>= 60): the splendour
		# of the reign is itself an argument for the Mandate. Small, so it rewards sustained
		# patronage without being a dominant lever.
		if = {
			limit = { var:qing_wenzhi_patronage >= 60 }
			add_legitimacy = 1
		}
		# [MO-ENH #13] ZENITH MILESTONE (ж–ҮжІ»йјҺзӣӣ). When imperial patronage first crowns a golden age
		# (>= 80), fire a one-shot capstone beat mirroring the silver-reserve peak (qing_revenue.5): the
		# Hanlin + Southern Study scholars present a memorial praising the reign's ж–ҮжІ». Gated on the
		# shared court event slot (BT-28) + a one-shot flag, so it is a single memorable moment, never
		# spam. Makes the ambient >=60 legitimacy buff feel earned and visible at its apex.
		if = {
			limit = {
				var:qing_wenzhi_patronage >= 80
				NOT = { has_variable = qing_wenzhi_zenith_reached }
				NOT = { has_variable = qing_gc_event_slot_used }
			}
			set_variable = { name = qing_wenzhi_zenith_reached  value = 1 }
			set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [BT-28] claim court slot
			trigger_event = { id = qing_wenzhi.1  days = { 5 20 } }
			LOG_line = { sys = QING  msg = "wenzhi: imperial patronage reached its cultural zenith for" }
		}
	}
}

# ======================================================================
# ADVANCE вҖ” raise the patronage meter by $amount$, clamped [0, 100]. Scope: country (CHI).
# ======================================================================
QING_wenzhi_advance = {
	QING_wenzhi_init = yes
	QING_DECLINE_nudge = { var = qing_wenzhi_patronage  amount = $amount$ }
	LOG_line = { sys = QING  msg = "wenzhi: patronage advanced for" }
}

# ======================================================================
# INITIATIVE: FUND THE IMPERIAL WORKSHOPS (йҖ иҫҰиҷ•). The е…§еӢҷеәң's ateliers вҖ” jade,
# Jingdezhen porcelain, cloisonnГ© вҖ” funded from the privy purse (е…§её‘). Grants a lasting
# craft modifier (jade + porcelain, if not already held) and raises patronage. Spends
# treasury + a slice of the privy purse. Scope: country (CHI); called from the panel.
# ======================================================================
QING_wenzhi_fund_workshops = {
	LOG_enter = { sys = QING  fn = "QING_wenzhi_fund_workshops" }
	add_treasury = -40
	if = { limit = { exists = current_ruler }  current_ruler = { add_wealth = -32 } }   # [#424] privy purse (е…§её‘) = ruler wealth
	QING_wenzhi_advance = { amount = 12 }
	if = {
		limit = { NOT = { has_variable = qing_wenzhi_workshops_done } }
		set_variable = qing_wenzhi_workshops_done
		add_country_modifier = { name = qing_culture_imperial_jade  duration = -1 }
		add_country_modifier = { name = qing_culture_porcelain  duration = -1 }
		# [#437] CONCRETE: the йҖ иҫҰиҷ• ateliers are not only a modifier вҖ” raise a real Imperial Kiln
		# (еҫЎзӘҜе» , qing_porcelain_kiln_building) at a porcelain-producing province that lacks one (жҷҜеҫ·йҺ®
		# and its like), and lift that province's civilisation. add_building_level + add_civilization_value
		# are the proven province-scope idioms (ordered_owned_province / every_owned_province upstream).
		ordered_owned_province = {
			limit = {
				trade_goods = porcelain
				NOT = { has_building = qing_porcelain_kiln_building }
			}
			order_by = total_population
			max = 1
			add_building_level = qing_porcelain_kiln_building
			add_civilization_value = 5
		}
		LOG_line = { sys = QING  msg = "wenzhi: imperial workshops endowed вҖ” jade + porcelain renown, an Imperial Kiln raised for" }
	}
	else = {
		# already endowed вҖ” a fresh commission is a one-off glory, not a new standing modifier.
		current_ruler = { add_popularity = 5 }
		LOG_line = { sys = QING  msg = "wenzhi: a fresh workshop commission (already endowed) for" }
	}
	add_legitimacy = 2
	LOG_exit = { sys = QING  fn = "QING_wenzhi_fund_workshops"  result = OK }
}

# ======================================================================
# INITIATIVE: COMMISSION THE PAINTING ACADEMY (еҰӮж„ҸйӨЁ). The court atelier where even
# the Jesuit Castiglione (йғҺдё–еҜ§) serves, fusing Chinese and Western brush. Grants the
# court-painting modifier once + raises patronage + ruler glory. Scope: country (CHI).
# ======================================================================
QING_wenzhi_commission_painting = {
	LOG_enter = { sys = QING  fn = "QING_wenzhi_commission_painting" }
	add_treasury = -30
	QING_wenzhi_advance = { amount = 10 }
	if = {
		limit = { NOT = { has_country_modifier = qing_culture_court_painting } }
		add_country_modifier = { name = qing_culture_court_painting  duration = -1 }
		LOG_line = { sys = QING  msg = "wenzhi: the Ruyiguan painting academy is commissioned for" }
	}
	# [#437] CONCRETE: the еҰӮж„ҸйӨЁ is staffed by real court painters. Mint one вҖ” a learned, charismatic
	# artist of the emperor's household (marked qing_court_artist) вҖ” via the proven create_character roster
	# idiom (mirrors QING_exam_mint_scholar; #90 rule: no modifiers granted inside create_character, markers
	# set in a follow-up scope). CAPPED at a resident atelier of 5 living court artists (qing_court_artist_count),
	# self-healing so a court painter who dies frees a place вҖ” repeated commissions must not spawn unbounded
	# characters (save-bloat). The rebuild counts the LIVING marked artists first, then mints only if under cap.
	set_variable = { name = qing_court_artist_count  value = 0 }
	every_character = {
		limit = { has_variable = qing_court_artist  is_alive = yes  employer = ROOT }
		ROOT = { change_variable = { name = qing_court_artist_count  add = 1 } }
	}
	if = {
		limit = { var:qing_court_artist_count < 5 }
		create_character = {
			age = 30
			culture = root.primary_culture
			religion = root.religion
			add_charisma = 7
			add_finesse = 6
			add_zeal = 3
			add_martial = 1
			save_scope_as = qing_new_court_artist
		}
		scope:qing_new_court_artist = {
			set_home_country = ROOT
			set_variable = { name = qing_court_artist  value = 1 }
			add_popularity = 10
		}
	}
	current_ruler = { add_popularity = 8 }
	LOG_exit = { sys = QING  fn = "QING_wenzhi_commission_painting"  result = OK }
}

# ======================================================================
# INITIATIVE: LAUNCH THE COMPLETE LIBRARY (еӣӣеә«е…Ёжӣё). The greatest compilation of the
# age вҖ” and its shadow, the literary inquisition (ж–Үеӯ—зҚ„). Raises patronage, then hands
# off to the EXISTING qing_culture.5 event, which presents the compile-with-inquisition
# vs compile-without choice (do not duplicate that dilemma here). Guarded on the same
# one-shot the event uses (qing_culture_siku_done) so the library is founded only once.
# Scope: country (CHI). Draws on the Hanlin (#358) / Southern Study (#336) scholar corps.
# ======================================================================
QING_wenzhi_launch_siku = {
	LOG_enter = { sys = QING  fn = "QING_wenzhi_launch_siku" }
	# [#390 / T2 review-fix] convene only while no convening is already pending: qing_wenzhi_siku_pending
	# is the "dilemma open" guard (the button's is_shown requires it ABSENT). qing_culture.5's options
	# CLEAR pending вҖ” incl. the "cannot afford" DECLINE, which does NOT found the library вҖ” so the throne
	# may convene again later. That re-arm is the exploit surface: the former code granted the +15
	# patronage on EVERY convening, so declineвҶ’re-launchвҶ’decline farmed unbounded patronage (and, via the
	# >=60 ambient-legitimacy branch, a permanent legitimacy engine) before the library was ever founded.
	# Gate the +15 on a SEPARATE one-shot flag (qing_wenzhi_siku_patronised) that NO option clears, so the
	# stirring of the scholar corps pays its patronage exactly ONCE across all re-convenings.
	if = {
		limit = { NOT = { has_variable = qing_wenzhi_siku_pending } }
		set_variable = qing_wenzhi_siku_pending
		if = {
			limit = { NOT = { has_variable = qing_wenzhi_siku_patronised } }
			set_variable = qing_wenzhi_siku_patronised
			QING_wenzhi_advance = { amount = 15 }
			# [#437] CONCRETE: the greatest compilation of the age is a real advance in letters and
			# statecraft вҖ” grant civic research (the еӣӣеә«е…Ёжӣё gathered and systematised the empire's
			# scholarship). add_research { technology value } is the proven country-scope idiom
			# (Invictus 01_armenia.txt:1029). Once, gated by the same one-shot patronage flag.
			add_research = { technology = civic_tech  value = 25 }
		}
		# the scholar corps is stirred; hand off to the existing Siku dilemma event, which
		# self-guards on qing_culture_siku_done and applies the modifiers + the ж–Үеӯ—зҚ„ choice.
		trigger_event = { id = qing_culture.5  days = { 3 10 } }
		LOG_line = { sys = QING  msg = "wenzhi: the Complete Library (еӣӣеә«е…Ёжӣё) is convened for" }
	}
	LOG_exit = { sys = QING  fn = "QING_wenzhi_launch_siku"  result = OK }
}

# ======================================================================
# INITIATIVE: THE SOUTHERN INSPECTION TOUR (еҚ—е·Ў). The emperor progresses down the
# Jiangnan вҖ” a display of patronage and majesty that binds the wealthy south to the
# throne, at great cost to the treasury. Raises patronage + ruler glory + stability;
# a one-off grand occasion (no permanent modifier). Scope: country (CHI).
# ======================================================================
QING_wenzhi_southern_tour = {
	LOG_enter = { sys = QING  fn = "QING_wenzhi_southern_tour" }
	add_treasury = -60
	QING_wenzhi_advance = { amount = 14 }
	current_ruler = { add_popularity = 12 }
	add_legitimacy = 4
	add_stability = 1
	# [#437] CONCRETE: the progress down the Jiangnan is a real civilising splendour on the ground вҖ”
	# lift the civilisation of the wealthiest southern cities the tour passes through. The imperial
	# barge, the academies patronised, the works commissioned leave a mark. Cap at the 5 most-developed
	# owned provinces so a vast empire's tour is a Jiangnan progress, not a realm-wide sweep.
	ordered_owned_province = {
		limit = { has_city_status = yes }
		order_by = total_population
		max = 5
		add_civilization_value = 3
	}
	LOG_line = { sys = QING  msg = "wenzhi: the emperor makes a Southern Tour (еҚ—е·Ў) for" }
	LOG_exit = { sys = QING  fn = "QING_wenzhi_southern_tour"  result = OK }
}

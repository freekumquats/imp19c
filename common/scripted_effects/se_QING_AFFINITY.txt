# se_QING_AFFINITY.txt — CHARACTER COMPATIBILITY with the throne.
#
# The Grand Council / great-office / examination events don't treat courtiers as
# interchangeable counters: how a man gets on with the reigning emperor is scored
# from the WHOLE Imperator character model, exactly as the engine itself weighs
# characters (see the Characters wiki page):
#
#   • CHARISMA  — "the ability to charm and persuade others." A charismatic
#     official is easier to deal with and rises smoothly; a graceless one grates.
#   • ZEAL GAP  — zeal is the gift of inspiring faith. A wide gap in fervour
#     between sovereign and servant (a zealot under a worldly throne, or a
#     sceptic under a devout one) breeds friction; matched zeal binds.
#   • CULTURE   — a Manchu-culture courtier shares the conquest elite's world with
#     a Manchu throne; a Han official (or an alien one) stands at a remove. Match
#     draws them together, mismatch apart (the 滿漢 fault line, given teeth).
#   • RELIGION  — likewise a shared faith binds and a divergent one distances.
#   • FRIEND / RIVAL — the engine's own secondary relations: a friend of the
#     emperor is bound to him (+Loyalty in the base game), a rival set against him.
#   • KINSHIP   — a close relative of the imperial house is bound by blood (and is
#     the very stuff of the overmighty-kinsman danger).
#   • AGE & HEALTH — an aged or ailing official is a fading asset; the old grow
#     "increasingly susceptible to terminal illnesses," so their troubles — and
#     their threat — may resolve themselves.
#
# QING_char_affinity  — call inside a courtier's character scope (this = courtier).
#   Scores compatibility with root.current_ruler into qing_char_affinity (0..100,
#   50 = neutral) and logs it. A pure read — no side effects on the character.
# QING_char_bind      — call inside a courtier's scope AFTER QING_char_affinity.
#   Turns the score into consequences: a highly compatible man is bound to the
#   throne (loyalty + a friendship of the sovereign), an incompatible one is
#   alienated (loyalty loss + a rivalry). This is what "makes characters more or
#   less friendly" per the factors above.
#
# CHI-only, player-only suite. All logged to sys = QING.
#   ROOT = the Qing country (CHI, player); root.current_ruler = the emperor.
#
# Prefix = QING_char_
########################################################################

# ======================================================================
# SCORE — read every factor, store 0..100 in qing_char_affinity, log it.
# ======================================================================
QING_char_affinity = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = QING  fn = "QING_char_affinity" }
	set_variable = { name = qing_char_affinity  value = 50 }

	# --- CHARISMA: +1 per point of the courtier's charm above the average of 5.
	change_variable = { name = qing_char_affinity  add = charisma }
	change_variable = { name = qing_char_affinity  subtract = 5 }

	# --- ZEAL MISMATCH: subtract the ABSOLUTE gap in zeal from throne to servant.
	set_variable = { name = qing_aff_zeal_gap  value = zeal }
	change_variable = { name = qing_aff_zeal_gap  subtract = root.current_ruler.zeal }
	if = {
		limit = { var:qing_aff_zeal_gap < 0 }
		change_variable = { name = qing_aff_zeal_gap  multiply = -1 }
	}
	change_variable = { name = qing_char_affinity  subtract = var:qing_aff_zeal_gap }

	# --- CULTURE match / mismatch (the 滿漢 dyarchy at the personal level)
	if = {
		limit = { culture = root.current_ruler.culture }
		change_variable = { name = qing_char_affinity  add = 10 }
	}
	else = {
		change_variable = { name = qing_char_affinity  subtract = 8 }
	}

	# --- RELIGION match / mismatch
	if = {
		limit = { religion = root.current_ruler.religion }
		change_variable = { name = qing_char_affinity  add = 8 }
	}
	else = {
		change_variable = { name = qing_char_affinity  subtract = 8 }
	}

	# --- FRIENDSHIP / RIVALRY with the sovereign (engine secondary relations)
	if = {
		limit = { is_friend = root.current_ruler }
		change_variable = { name = qing_char_affinity  add = 15 }
	}
	if = {
		limit = { is_rival = root.current_ruler }
		change_variable = { name = qing_char_affinity  subtract = 20 }
	}

	# --- KINSHIP & FAMILY PRESTIGE: a close relative of the imperial house is
	# bound by blood, and a member of the reigning family more so — the great
	# families' standing (family prestige) is itself a courtier's stake in the
	# dynasty. A scion of a prestigious house is invested in the throne.
	if = {
		limit = { is_close_relative = root.current_ruler }
		change_variable = { name = qing_char_affinity  add = 8 }
	}
	if = {
		limit = { from_ruler_family = yes }
		change_variable = { name = qing_char_affinity  add = 8 }
	}
	# family prestige, banded (family.prestige is unbounded, so we don't add it raw)
	if = {
		limit = { family = { prestige >= 500 } }
		change_variable = { name = qing_char_affinity  add = 6 }
	}
	else_if = {
		limit = { family = { prestige >= 100 } }
		change_variable = { name = qing_char_affinity  add = 3 }
	}

	# --- STATESMANSHIP: the governing skills (finesse the administrator's gift,
	# martial the general's). A capable man serves the throne well and knows it —
	# a modest confidence bonus above the average pair (10 = 5+5).
	set_variable = { name = qing_aff_statecraft  value = finesse }
	change_variable = { name = qing_aff_statecraft  add = martial }
	change_variable = { name = qing_aff_statecraft  subtract = 10 }
	change_variable = { name = qing_aff_statecraft  divide = 2 }
	change_variable = { name = qing_char_affinity  add = var:qing_aff_statecraft }

	# --- STANDING: popularity (how the people see him) and prominence (his fame)
	# both raise a man's sense of his own worth to the state. Banded, not raw.
	if = { limit = { popularity >= 40 }   change_variable = { name = qing_char_affinity  add = 5 } }
	else_if = { limit = { popularity <= -20 } change_variable = { name = qing_char_affinity  subtract = 5 } }
	if = { limit = { prominence >= 60 }   change_variable = { name = qing_char_affinity  add = 4 } }

	# --- EXISTING LOYALTY feeds back: a man already loyal is congenial, a
	# disloyal one estranged (loyalty is a 0..100 state-loyalty read).
	if = { limit = { loyalty >= 65 }  change_variable = { name = qing_char_affinity  add = 6 } }
	else_if = { limit = { loyalty < 40 }  change_variable = { name = qing_char_affinity  subtract = 8 } }

	# --- VENALITY: a corrupt official is a self-server; an honest / just one a
	# pillar. Corruption also drags in step with the realm's corruption level.
	if = { limit = { has_trait = corrupt } change_variable = { name = qing_char_affinity  subtract = 8 } }
	if = { limit = { OR = { has_trait = honest  has_trait = just } } change_variable = { name = qing_char_affinity  add = 5 } }
	# corruption is ALSO a numeric character stat (0..100), distinct from the
	# corrupt trait: the higher a man's own venality, the more he serves himself.
	if = { limit = { corruption >= 40 } change_variable = { name = qing_char_affinity  subtract = 8 } }
	else_if = { limit = { corruption >= 15 } change_variable = { name = qing_char_affinity  subtract = 4 } }
	# a corrupt court makes even honest men warier of the throne — the realm's
	# corruption level (the decline counter) drags every courtier's affinity.
	if = {
		limit = { ROOT = { has_variable = qing_corruption_level  var:qing_corruption_level >= 60 } }
		change_variable = { name = qing_char_affinity  subtract = 5 }
	}

	# --- PERSONAL WEALTH & POWER BASE: the Heshen danger. A courtier rich in his
	# own right and thick with loyal retainers (power_base — holdings + veterans,
	# which the engine already docks 0.55 loyalty per unit) is powerful but LESS
	# dependent on the throne's favour, and so a warier, more dangerous servant.
	if = { limit = { wealth >= 300 }   change_variable = { name = qing_char_affinity  subtract = 5 } }
	if = { limit = { power_base >= 15 } change_variable = { name = qing_char_affinity  subtract = 6 } }
	else_if = { limit = { power_base >= 8 } change_variable = { name = qing_char_affinity  subtract = 3 } }

	# --- AGE & HEALTH: an ailing or aged official is a fading (less reliable) asset
	if = {
		limit = { health < 0.35 }
		change_variable = { name = qing_char_affinity  subtract = 10 }
	}
	if = {
		limit = { age >= 70 }
		change_variable = { name = qing_char_affinity  subtract = 6 }
	}

	# --- clamp 0..100
	if = { limit = { var:qing_char_affinity > 100 } set_variable = { name = qing_char_affinity  value = 100 } }
	if = { limit = { var:qing_char_affinity < 0 }   set_variable = { name = qing_char_affinity  value = 0 } }

	LOG_line = { sys = QING  msg = "char affinity: w/ throne = (zeal gap) for" }
	LOG_exit = { sys = QING  fn = "QING_char_affinity"  result = OK }
}

# ======================================================================
# APPLY — turn the score into loyalty + a friendship or rivalry.
# High affinity (>=70): the man is bound to the throne — loyalty and, at the top,
# a personal friendship of the sovereign. Low affinity (<=30): he is alienated —
# loyalty falls and, at the bottom, he becomes the emperor's rival. The mid band
# leaves the relations untouched. Auto-scores first if not yet scored.
# ======================================================================
QING_char_bind = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = QING  fn = "QING_char_bind" }
	if = {
		limit = { NOT = { has_variable = qing_char_affinity } }
		QING_char_affinity = yes
	}
	if = {
		limit = { var:qing_char_affinity >= 70 }
		add_loyalty = loyalty_qing_congenial
		# a genuinely congenial servant becomes a friend of the throne
		if = {
			limit = {
				var:qing_char_affinity >= 85
				NOT = { is_friend = root.current_ruler }
				NOT = { is_rival = root.current_ruler }
			}
			add_friend = root.current_ruler
			LOG_line = { sys = QING  msg = "char bind: befriended the throne (affinity high) for" }
		}
	}
	else_if = {
		limit = { var:qing_char_affinity <= 30 }
		add_loyalty = loyalty_qing_estranged
		# a deeply incompatible servant turns rival
		if = {
			limit = {
				var:qing_char_affinity <= 15
				NOT = { is_rival = root.current_ruler }
				NOT = { is_friend = root.current_ruler }
			}
			add_rival = root.current_ruler
			LOG_line = { sys = QING  msg = "char bind: set against the throne (affinity low) for" }
		}
	}
	LOG_line = { sys = QING  msg = "char bind applied: affinity= for" }
	LOG_exit = { sys = QING  fn = "QING_char_bind"  result = OK }
}

# ======================================================================
# MUTATORS — events don't only READ these factors, they can CHANGE them. A man
# can be sinicized, converted, cultivated into a more charismatic statesman, or
# have his fervour kindled or cooled. Each is call-site-agnostic (this = the
# character), logs the change, and re-scores affinity so the new state takes
# effect immediately. This is the loop the user asked for: the events are shaped
# by these factors AND reshape them.
# ======================================================================

# --- Assimilate a courtier to the imperial culture (漢化 / 滿化, per the throne's
# own culture): the sinicization of a Manchu, or a Han official taken into the
# banner establishment. Binds him closer to the conquest elite.
QING_char_assimilate = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = QING  fn = "QING_char_assimilate" }
	set_culture = root.current_ruler.culture
	LOG_line = { sys = QING  msg = "char mutate: assimilated to culture for" }
	QING_char_affinity = yes
	LOG_exit = { sys = QING  fn = "QING_char_assimilate"  result = OK }
}

	QING_char_affinity = yes
	LOG_exit = { sys = QING  fn = "QING_char_convert"  result = OK }
}

# --- Cultivate a courtier's ability. $stat$ = charisma|zeal|martial|finesse,
# $amount$ = signed points (accepts negatives — a disgrace or a wasting illness
# can DIMINISH a man). Used by the mentorship / office-seasoning / scandal events.
QING_char_cultivate = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = QING  fn = "QING_char_cultivate" }
	add_$stat$ = $amount$
	LOG_line = { sys = QING  msg = "char mutate: $stat$ += $amount$ for" }
	QING_char_affinity = yes
	LOG_exit = { sys = QING  fn = "QING_char_cultivate"  result = OK }
}

# --- Brand a courtier corrupt: adds the corrupt TRAIT (guarded) and drives up

	LOG_enter = { sys = QING  fn = "QING_char_taint" }

		limit = { NOT = { has_trait = corrupt } }

# ======================================================================
# PAIRWISE FRICTION — TRANSIENT, never stored per-pair.
# Two councillors' compatibility with EACH OTHER, computed on the fly at event
# time from the same character factors and written to ONE country-scope scratch
# variable (qing_pair_friction) that the calling event reads immediately and lets
# fall out of use. We deliberately do NOT persist an affinity between every pair
# of characters (that would be O(n^2) saved variables — the memory blow-up the
# user warned against); the council-infighting events resolve the clash in the
# moment and store nothing.
#
# Call at COUNTRY scope with two saved character scopes:
#   QING_pair_friction = { a = scope:feud_a  b = scope:feud_b }
# Result: var:qing_pair_friction (0..100; HIGH = they clash badly).
# ======================================================================
QING_pair_friction = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = QING  fn = "QING_pair_friction" }
	set_variable = { name = qing_pair_friction  value = 20 }

	# culture divide (the 滿漢 fault line between two ministers)
	if = {
		limit = { NOT = { $a$ = { culture = $b$.culture } } }
		change_variable = { name = qing_pair_friction  add = 25 }
	}
	# faith divide
	if = {
		limit = { NOT = { $a$ = { religion = $b$.religion } } }
		change_variable = { name = qing_pair_friction  add = 15 }
	}
	# a zeal gulf between them (fervent vs worldly)
	set_variable = { name = qing_pair_zeal_gap  value = $a$.zeal }
	change_variable = { name = qing_pair_zeal_gap  subtract = $b$.zeal }
	if = { limit = { var:qing_pair_zeal_gap < 0 }  change_variable = { name = qing_pair_zeal_gap  multiply = -1 } }
	change_variable = { name = qing_pair_friction  add = var:qing_pair_zeal_gap }
	# two proud, prominent men rub worse (rivalry of the eminent)
	if = {
		limit = { $a$ = { prominence >= 50 }  $b$ = { prominence >= 50 } }
		change_variable = { name = qing_pair_friction  add = 10 }
	}
	# venality breeds mutual distrust — the corrupt TRAIT...
	if = {
		limit = { OR = { $a$ = { has_trait = corrupt }  $b$ = { has_trait = corrupt } } }
		change_variable = { name = qing_pair_friction  add = 8 }
	}
	# ...and the numeric CORRUPTION stat too (a grasping minister sows distrust
	# among colleagues even before the trait crystallises; user 2026-07-07).
	if = {
		limit = { OR = { $a$ = { corruption >= 40 }  $b$ = { corruption >= 40 } } }
		change_variable = { name = qing_pair_friction  add = 8 }
	}
	else_if = {
		limit = { OR = { $a$ = { corruption >= 20 }  $b$ = { corruption >= 20 } } }
		change_variable = { name = qing_pair_friction  add = 4 }
	}
	# FRIENDS / RIVALS of each other (engine secondary relations): a friendship
	# smooths the council table; a standing rivalry poisons it (user 2026-07-07).
	if = {
		limit = { $a$ = { is_friend = $b$ } }
		change_variable = { name = qing_pair_friction  subtract = 15 }
	}
	if = {
		limit = { $a$ = { is_rival = $b$ } }
		change_variable = { name = qing_pair_friction  add = 20 }
	}
	# HIGHBORN vs LOWBORN: a great-house grandee and a low-born colleague grate
	# across the gulf of birth; two proud houses rival one another for precedence
	# (family prestige, banded — mirrors QING_char_affinity's prestige bands).
	if = {
		limit = {
			OR = {
				AND = { $a$ = { family = { prestige >= 300 } }  $b$ = { NOT = { family = { prestige >= 100 } } } }
				AND = { $b$ = { family = { prestige >= 300 } }  $a$ = { NOT = { family = { prestige >= 100 } } } }
			}
		}
		change_variable = { name = qing_pair_friction  add = 10 }
	}
	else_if = {
		limit = { $a$ = { family = { prestige >= 500 } }  $b$ = { family = { prestige >= 500 } } }
		change_variable = { name = qing_pair_friction  add = 8 }
	}
	# GENERATIONAL GULF: an aged grandee and a young upstart clash over precedence
	# and pace of change (age, banded-contrast — age>=N is the proven trigger form).
	if = {
		limit = {
			OR = {
				AND = { $a$ = { age >= 60 }  $b$ = { age <= 35 } }
				AND = { $b$ = { age >= 60 }  $a$ = { age <= 35 } }
			}
		}
		change_variable = { name = qing_pair_friction  add = 8 }
	}
	# WEALTH DISPARITY: a rich minister and a poor one distrust each other's means
	# and motives (wealth, banded-contrast).
	if = {
		limit = {
			OR = {
				AND = { $a$ = { wealth >= 200 }  $b$ = { wealth < 50 } }
				AND = { $b$ = { wealth >= 200 }  $a$ = { wealth < 50 } }
			}
		}
		change_variable = { name = qing_pair_friction  add = 6 }
	}
	# RELATIVE POPULARITY: when one minister markedly outshines the other in the
	# public eye, the eclipsed colleague resents it (popularity, banded-contrast).
	if = {
		limit = {
			OR = {
				AND = { $a$ = { popularity >= 40 }  $b$ = { popularity <= 0 } }
				AND = { $b$ = { popularity >= 40 }  $a$ = { popularity <= 0 } }
			}
		}
		change_variable = { name = qing_pair_friction  add = 6 }
	}
	# HEALTH: one hale and one failing — the ailing man clings to his post, the
	# vigorous one presses his advantage (health, banded-contrast).
	if = {
		limit = {
			OR = {
				AND = { $a$ = { health < 0.35 }  $b$ = { health >= 0.5 } }
				AND = { $b$ = { health < 0.35 }  $a$ = { health >= 0.5 } }
			}
		}
		change_variable = { name = qing_pair_friction  add = 6 }
	}
	# but kinship softens it
	if = {
		limit = { $a$ = { is_close_relative = $b$ } }
		change_variable = { name = qing_pair_friction  subtract = 15 }
	}

	if = { limit = { var:qing_pair_friction > 100 } set_variable = { name = qing_pair_friction  value = 100 } }
	if = { limit = { var:qing_pair_friction < 0 }   set_variable = { name = qing_pair_friction  value = 0 } }

	LOG_line = { sys = QING  msg = "pair friction: vs = for" }
	LOG_exit = { sys = QING  fn = "QING_pair_friction"  result = OK }
}

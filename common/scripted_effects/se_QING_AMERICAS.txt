# se_QING_AMERICAS.txt — [scripted-revolutions 2nd pass] texture effects for the
# NORTH AMERICAN OPPORTUNITY events (qing_americas_events.txt). The Napoleonic Wars and
# the War of 1812 fling the North American frontier open; a Qing that has already planted
# a Pacific foothold (the Great Pacific Enterprise colonisation tree — Alaska/Canada/
# California) may exploit it. The events OFFER the choices (the Qing is the player, D0);
# these helpers supply the mechanical texture.
#
#   ROOT / this = CHI (the Qing player). Prefix = QING_americas_. sys = QING.
#
# DESIGN — CLAIM & SPHERE, NOT CONQUEST. The Louisiana Purchase transferred France's
# PREEMPTION RIGHT (paper title to negotiate with / displace the Native nations), NOT
# de-facto control: the Mississippi-Missouri watershed was held by sovereign Native
# powers (Osage, Comanche, Lakota, Pawnee, Kiowa — tags OSG/CMC/LAK/PWN/… on the map)
# into the 1820s-30s (Hämäläinen, Lakota America / The Comanche Empire; DuVal, The Native
# Ground). So "buying Louisiana" here does NOT set_owned_by the interior. Instead it
#   (a) lays a Qing CLAIM on the watershed provinces (the preemption right made concrete —
#       the add_claim-in-loop idiom, proven se_DEJURE.txt:134), and
#   (b) draws the Plains Native nations under the Qing wing as vassal_tribe (Tribal Vassal)
#       subjects (FUNC_make_subject), which — being Qing subjects — auto-enter the existing
#       four-power sphere-of-influence ring (se_QING_SPHERE builds its ring from CHI's
#       subjects), so the trans-Mississippi west becomes a CONTESTED QING SPHERE, exactly
#       as India/Central Asia/the marches already are. No Native land is annexed.
# All FUNC_make_subject / add_claim / add_country_modifier calls are guarded so an absent
# or already-bound target is a clean no-op.
########################################################################

# ======================================================================
# QING_americas_claim_louisiana — lay the Qing preemption claim across the GREATER
# Louisiana watershed (the Great_Plains region + the Louisiana & Arkansas areas — far
# larger than the 14-province LSA tag, matching the real 1803 purchase). add_claim in a
# province loop is the proven idiom (se_DEJURE.txt). Provinces the Qing already holds are
# skipped. Scope: COUNTRY (CHI). Idempotent (adding an existing claim is a no-op).
# ======================================================================
QING_americas_claim_louisiana = {
	LOG_enter = { sys = QING  fn = "QING_americas_claim_louisiana" }
	save_scope_as = americas_claimant
	every_province = {
		limit = {
			is_sea = no
			OR = {
				is_in_region = Great_Plains
				is_in_area = Louisiana
				is_in_area = Arkansas
			}
			NOT = { owner = scope:americas_claimant }
		}
		add_claim = scope:americas_claimant
	}
	LOG_line = { sys = QING  msg = "Americas: Qing preemption claim laid across the greater Louisiana watershed for" }
	LOG_exit = { sys = QING  fn = "QING_americas_claim_louisiana"  result = OK }
}

# ======================================================================
# QING_americas_embrace_plains — draw the sovereign Plains Native nations under the Qing
# wing as vassal_tribe (Tribal Vassal) subjects (the historically-grounded suzerainty form for TRIBAL nations — a
# tributary tie, not displacement). Each such subject auto-enters the four-power sphere
# ring (se_QING_SPHERE), so the trans-Mississippi west becomes a contested Qing sphere.
# Scope: COUNTRY (CHI). Each embrace is guarded on the tag existing + not already ours.
# Tags: OSG Osage | CMC Comanche | LAK Lakota | DAK Dakota | PWN Pawnee | CHY Cheyenne |
#       ARP Arapaho | KIO (Kiowa — via CMC ally if no tag) | CDD Caddo | APA Apache | BLF Blackfoot.
# ======================================================================
QING_americas_embrace_plains = {
	LOG_enter = { sys = QING  fn = "QING_americas_embrace_plains" }
	QING_americas_embrace_one = { tag = OSG }
	QING_americas_embrace_one = { tag = CMC }
	QING_americas_embrace_one = { tag = LAK }
	QING_americas_embrace_one = { tag = DAK }
	QING_americas_embrace_one = { tag = PWN }
	QING_americas_embrace_one = { tag = CHY }
	QING_americas_embrace_one = { tag = ARP }
	QING_americas_embrace_one = { tag = CDD }
	QING_americas_embrace_one = { tag = APA }
	QING_americas_embrace_one = { tag = BLF }
	if = {
		limit = { NOT = { has_country_modifier = qing_americas_louisiana_claim } }
		add_country_modifier = { name = qing_americas_louisiana_claim  duration = -1 }
		# [CROSSWIRE 2026-07-21 arcs->domestic] a transoceanic sphere across the Pacific is only
		# thinkable for a modernizing power — the venture rewards and demands the self-strengthening
		# cause (steam hulls, telegraph, a blue-water navy). Inside the once-only modifier guard so it
		# advances once. Guarded on the meter's existence; qing_selfstr_progress is CHI-scope.
		if = {
			limit = { has_variable = qing_selfstr_progress }
			QING_selfstr_advance = { amount = 4 }
		}
	}
	LOG_line = { sys = QING  msg = "Americas: the Plains nations drawn into the Qing sphere as tributaries for" }
	LOG_exit = { sys = QING  fn = "QING_americas_embrace_plains"  result = OK }
}

# One Plains nation embraced as a vassal_tribe (Tribal Vassal). Scope: COUNTRY (CHI). $tag$ = the
# native country tag. Guarded so an absent or already-subject tag is a clean no-op.
QING_americas_embrace_one = {
	if = {
		limit = { exists = c:$tag$  NOT = { c:$tag$ = { is_subject_of = ROOT } } }
		# [1763-hist-fix 2026-07-17] The Plains nations are TRIBAL (waashat/earth_lodge) — the correct
		# form for a civilized power taking a tribal nation under its wing is vassal_tribe (Tribal Vassal),
		# NOT sinosphere_tributary (朝貢國, whose allow gate is Confucian/Buddhist East-Asia only). A tribal
		# vassal renders manpower + gains protection + civilizing growth — the historically-apt suzerainty.
		FUNC_make_subject = { overlord = ROOT  target = c:$tag$  type = vassal_tribe }
		LOG_line = { sys = QING  msg = "Americas: a Plains nation taken under the wing as a tribal vassal for" }
	}
}

# ======================================================================
# QING_americas_press_pacific_coast — press the Pacific-Northwest fur coast while the
# Anglo-Americans are at war and the Russian-American Company is stretched. Lays a claim on
# the Columbia mouth (Astoria, p:9250) — the American post the North West Company seized in
# 1813 — and grants the Pacific-trade foothold modifier. Scope: COUNTRY (CHI). Guarded.
# ======================================================================
QING_americas_press_pacific_coast = {
	LOG_enter = { sys = QING  fn = "QING_americas_press_pacific_coast" }
	if = { limit = { exists = p:9250  NOT = { owns = 9250 } }  add_claim = p:9250 }
	if = {
		limit = { NOT = { has_country_modifier = qing_col_pacific_trade } }
		add_country_modifier = { name = qing_col_pacific_trade  duration = -1 }
		# [CROSSWIRE 2026-07-21 arcs->domestic] a Pacific-coast foothold demands a modern navy to hold
		# it — the venture advances self-strengthening. Once, inside the foothold guard. CHI-scope.
		if = {
			limit = { has_variable = qing_selfstr_progress }
			QING_selfstr_advance = { amount = 3 }
		}
	}
	# contesting the Columbia irks Britain (the Hudson's Bay / North West Company) —
	# UNLESS the later Daoguang Doctrine has aligned the Qing with the American republic.
	if = {
		limit = { NOT = { has_country_modifier = qing_daoguang_doctrine } }
		QING_gp_provoke_britain = { severity = 8 }
	}
	LOG_line = { sys = QING  msg = "Americas: the Qing presses the Columbia fur coast for" }
	LOG_exit = { sys = QING  fn = "QING_americas_press_pacific_coast"  result = OK }
}

# ======================================================================
# [boot #18] QING_colonise_this_province — plant a Qing colony on the UNOWNED province this is
# scoped to (called from QING_colonise_frontier_button, which has already paid the price + verified
# the province is unowned/inhabitable/reachable). PROVINCE scope; scope:player = CHI. Transfers the
# province via set_owned_by (the proven in-mod province-transfer idiom, se_QING_BURMA:149).
#
# [B16] The SETTLERS are REAL EMIGRANTS drawn from a genuinely OVER-CAPACITY Qing province, not a
# pop conjured from thin air (the old create_state_pop = tribesmen). The empire's crowded heartland
# is exactly the pressure the overseas colony relieves (the whole 移民實邊 premise). Mirrors the
# proven se_QING_COLON nomad-in-gathering move-a-real-pop idiom (se_QING_COLON.txt:219-257):
#   • pick a random OWNED CHI province that is over its population capacity (random_owned_province,
#     NOT ordered_* — the oracle-warned unreliable sort, se_QING_COLON.txt:210) and holds a movable
#     commoner pop (not slaves/nobles — the same emigrant filter the migration system uses);
#   • move ONE such pop to the new colony (move_pop, the proven cross-province transfer);
#   • FALLBACK to create_state_pop = tribesmen only if no crowded source has a movable pop, so the
#     colony is never left empty.
# Guarded on still-unowned so a double-click / stale button cannot re-transfer owned land.
# ======================================================================
QING_colonise_this_province = {
	if = {
		limit = { has_owner = no }
		set_owned_by = scope:player
		save_scope_as = qing_colony_dest
		if = {
			# a real over-capacity Qing province with a movable commoner pop to send.
			limit = {
				scope:player = {
					any_owned_province = {
						MIGRATION_pop_overcapacity_overflow > 0
						any_pops_in_province = { NOT = { pop_type = slaves }  NOT = { pop_type = nobles } }
					}
				}
			}
			scope:player = {
				random_owned_province = {
					limit = {
						NOT = { this = scope:qing_colony_dest }
						MIGRATION_pop_overcapacity_overflow > 0
						any_pops_in_province = { NOT = { pop_type = slaves }  NOT = { pop_type = nobles } }
					}
					random_pops_in_province = {
						limit = { NOT = { pop_type = slaves }  NOT = { pop_type = nobles } }
						move_pop = scope:qing_colony_dest
					}
					LOG_line = { sys = QING  msg = "colonisation: real settlers emigrate from a crowded province to the new overseas colony for" }
				}
			}
		}
		else = {
			# no crowded source with a movable pop — deliver a starter host so the colony is not empty.
			create_state_pop = tribesmen
			LOG_line = { sys = QING  msg = "colonisation: no over-capacity source pop to move; seeded a starter colony pop (fallback) for" }
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_colonise_this_province"  reason = "province already owned; colony not planted (stale button)" }
	}
}

# se_QING_FACTION.txt — CONSERVATIVE vs REFORMIST factions of the Grand Council
# (#158, user request). A layer OVER the whole council: every seated office-
# holder AND the three dynastic figureheads (Emperor 皇帝, Crown Prince 皇太子,
# Empress Dowager 皇太后) carries a personal REFORM STANCE, scored from the same
# WHOLE-character model the affinity chart uses (traits, zeal, finesse, age,
# culture, venality — see se_QING_AFFINITY.txt). Those stances aggregate into two
# derived council meters:
#
#   qing_council_reform_lean   -100..+100  weighted-average stance of the council
#                              + the figureheads. <0 conservative, >0 reformist.
#                              The Emperor/Dowager/Prince are weighted heavily —
#                              they ARE the faction anchors (the Guangxu-vs-Cixi
#                              dynamic, 光緒 vs 慈禧, given teeth).
#   qing_council_polarization  headcount of the SMALLER opposing bloc: high only
#                              when reformists AND conservatives are BOTH well
#                              represented — i.e. a court split down the middle.
#   qing_council_polar_pen     the gridlock penalty derived from polarization,
#                              docked from council effectiveness in
#                              QING_council_recompute (a deadlocked court governs
#                              worse REGARDLESS of which side is winning — a united
#                              council, reformist OR conservative, functions).
#
# INTEGRATION (per the user's locked design decisions, 2026-07-07):
#   • DEADLOCK PENALTY: polar_pen is subtracted from qing_council_eff_target in
#     QING_council_recompute (se_QING_COUNCIL.txt), right beside the dynastic-
#     harmony coupling — the two "court politics" levers sit together.
#   • FEEDS THE REFORM METER: once per quarter QING_faction_feed_reform_balance
#     nudges the EXISTING country-level qing_reform_faction_balance (the #14 reform
#     end-state trajectory, pushed by qing_reform_missions.txt) toward the council's
#     lean — so WHO SITS ON THE COUNCIL actually drives the realm's reform fate.
#     Deliberately small (+/-1..2 per quarter) so it COMPLEMENTS, never clobbers,
#     the mission nudges (+5..+10 per reform task).
#
# All logged sys = QING. CHI-only. Prefix = QING_faction_ / QING_char_stance.
# Related: se_QING_AFFINITY.txt (the sister scorer — compatibility w/ throne),
# se_QING_COUNCIL.txt (the effectiveness lever this docks), se_QING_REFORM.txt
# (the reform end-states this feeds), se_QING_DYNASTY.txt (family beats that shove
# the figureheads' stance), events/qing_faction_events.txt (the showdown event).
########################################################################

# ======================================================================
# INIT — seed the derived meters for a clean GUI/log read before the first
# recompute. Idempotent. Country scope (ROOT = CHI). qing_reform_faction_balance
# is NOT seeded here (it is owned + seeded by the reform track).
# ======================================================================
QING_faction_init = {
	LOG_enter = { sys = QING  fn = "QING_faction_init" }
	if = { limit = { NOT = { has_variable = qing_council_reform_lean } }  set_variable = { name = qing_council_reform_lean  value = 0 } }
	if = { limit = { NOT = { has_variable = qing_council_polarization } } set_variable = { name = qing_council_polarization  value = 0 } }
	if = { limit = { NOT = { has_variable = qing_council_polar_weight } } set_variable = { name = qing_council_polar_weight  value = 0 } }
	if = { limit = { NOT = { has_variable = qing_council_polar_pen } }    set_variable = { name = qing_council_polar_pen  value = 0 } }
	LOG_exit = { sys = QING  fn = "QING_faction_init"  result = OK }
}

# ======================================================================
# PER-CHARACTER STANCE — call inside a character scope (this = the courtier /
# figurehead). Scores a REFORM POSITION into qing_char_stance (-100..+100;
# negative = conservative, positive = reformist), from the same whole-character
# model the affinity chart weighs. A pure read — no side effects on the character.
#   • reformer / traditionalist TRAITS are the primary poles (the engine's own
#     status traits, common/traits/00_imp19c.txt).
#   • ZEAL: fervour is the temper of the old orthodoxy — a zealot resists change,
#     a worldly man is readier for it.
#   • FINESSE: the administrator's gift; a capable statesman sees the case for
#     reform of the apparatus he runs.
#   • AGE: the young press for change, the aged grandees defend the ancestral forms.
#   • CULTURE: the Han literati carried much of the self-strengthening/reform
#     impulse; the Manchu conquest elite were the conservative anchor (滿漢, again).
#   • VENALITY: a corrupt official has a vested interest in the status quo.
#   • INTEGRITY: an honest/just man leans (mildly) to principled reform.
# ======================================================================
QING_char_stance = {
	LOG_enter = { sys = QING  fn = "QING_char_stance" }
	set_variable = { name = qing_char_stance  value = 0 }

	# --- TRAITS: the explicit poles ---------------------------------------------
	if = { limit = { has_trait = reformer }        change_variable = { name = qing_char_stance  add = 45 } }
	if = { limit = { has_trait = traditionalist }  change_variable = { name = qing_char_stance  subtract = 45 } }

	# --- ZEAL: fervent orthodoxy vs worldly pragmatism --------------------------
	if = { limit = { zeal >= 12 }      change_variable = { name = qing_char_stance  subtract = 12 } }
	else_if = { limit = { zeal >= 8 }  change_variable = { name = qing_char_stance  subtract = 6 } }
	else_if = { limit = { zeal <= 2 }  change_variable = { name = qing_char_stance  add = 6 } }

	# --- FINESSE: the reform-minded administrator -------------------------------
	if = { limit = { finesse >= 12 }     change_variable = { name = qing_char_stance  add = 8 } }
	else_if = { limit = { finesse >= 8 } change_variable = { name = qing_char_stance  add = 4 } }

	# --- AGE: young reformers vs aged grandees ----------------------------------
	if = { limit = { age <= 35 }       change_variable = { name = qing_char_stance  add = 12 } }
	else_if = { limit = { age >= 65 }  change_variable = { name = qing_char_stance  subtract = 12 } }
	else_if = { limit = { age >= 55 }  change_variable = { name = qing_char_stance  subtract = 6 } }

	# --- CULTURE: Han literati reformist-leaning, Manchu elite conservative ------
	if = {
		limit = { culture = root.current_ruler.culture }
		change_variable = { name = qing_char_stance  subtract = 5 }
	}
	else = {
		change_variable = { name = qing_char_stance  add = 8 }
	}

	# --- VENALITY: the corrupt defend the trough they feed at -------------------
	if = { limit = { has_trait = corrupt } change_variable = { name = qing_char_stance  subtract = 10 } }
	if = { limit = { corruption >= 40 }      change_variable = { name = qing_char_stance  subtract = 8 } }
	else_if = { limit = { corruption >= 20 } change_variable = { name = qing_char_stance  subtract = 4 } }

	# --- INTEGRITY: principled reform -------------------------------------------
	if = { limit = { OR = { has_trait = honest  has_trait = just } } change_variable = { name = qing_char_stance  add = 5 } }

	# --- [#308/user] PERSONAL DRIFT: a man's alignment is not fixed to his traits. This
	# persistent per-character component (qing_stance_drift, moved by QING_char_stance_shift)
	# lets an official gradually — or suddenly, at an event — shift where he stands, so the
	# office-name colours in the GC panel change over a reign. Folded in here so every read of
	# stance reflects the accumulated drift on top of the structural (trait/age/culture) base.
	if = { limit = { has_variable = qing_stance_drift }  change_variable = { name = qing_char_stance  add = var:qing_stance_drift } }

	# --- clamp -100..100
	if = { limit = { var:qing_char_stance > 100 }  set_variable = { name = qing_char_stance  value = 100 } }
	if = { limit = { var:qing_char_stance < -100 } set_variable = { name = qing_char_stance  value = -100 } }

	LOG_line = { sys = QING  msg = "char stance: = (>0 reformist) for" }
	LOG_exit = { sys = QING  fn = "QING_char_stance"  result = OK }
}

# ======================================================================
# [#308/user] SHIFT A CHARACTER's ALIGNMENT. Call inside a character scope. $amount$ signed
# (positive = drift reformist, negative = drift conservative). Moves the persistent
# qing_stance_drift component that QING_char_stance folds in, clamped so drift alone can flip
# a man but not send stance out of a sane band. Used by the occasional-shift pass below and
# available to events (a conversion, a disillusionment, a bandwagon to the winning side).
# ======================================================================
QING_char_stance_shift = {
	if = { limit = { NOT = { has_variable = qing_stance_drift } }  set_variable = { name = qing_stance_drift  value = 0 } }
	change_variable = { name = qing_stance_drift  add = $amount$ }
	if = { limit = { var:qing_stance_drift > 60 }  set_variable = { name = qing_stance_drift  value = 60 } }
	if = { limit = { var:qing_stance_drift < -60 } set_variable = { name = qing_stance_drift  value = -60 } }
	# re-score so the change takes effect (and the GC office-name colour updates) at once.
	QING_char_stance = yes
	# [logfix] a $amount$ macro inside a LOG/debug string triggers mass "unknown arguments"
	# load errors (the string is scanned for macro params at parse-time) — keep the message
	# STATIC. The signed shift is already observable via the qing_stance_drift var trace.
	LOG_line = { sys = QING  msg = "char stance shift applied to drift for" }
}

# ======================================================================
# [#308/user] OCCASIONAL ALIGNMENT SHIFTS across the council — the mechanic that makes the
# office-name colours change over time. Called ~yearly from QING_council_recompute (self-
# throttled there). Each seated member has a small chance to shift, USUALLY toward the court's
# prevailing lean (bandwagoning to the ascendant faction — the historical pull of a winning
# side), but sometimes AGAINST it (a contrarian, a man of conviction digging in). Reformer/
# traditionalist TRAIT-holders are anchored (they rarely move — their conviction is structural).
# Country scope (ROOT = CHI). O(offices).
# ======================================================================
QING_faction_drift_alignments = {
	LOG_enter = { sys = QING  fn = "QING_faction_drift_alignments" }
	if = {
		limit = { has_variable_list = qing_council_members }
		every_in_list = {
			variable = qing_council_members
			limit = {
				is_alive = yes
				employer = ROOT
				# the strongly-committed (explicit reformer/traditionalist) do not drift.
				NOT = { has_trait = reformer }
				NOT = { has_trait = traditionalist }
			}
			random = {
				chance = 20
				# usually bandwagon toward the council's prevailing lean; ~1 in 4, buck it.
				# random_list gives a true 75/25 split (two sibling random blocks would roll
				# independently — both or neither could fire).
				random_list = {
					75 = {
						if = {
							limit = { ROOT = { has_variable = qing_council_reform_lean  var:qing_council_reform_lean >= 0 } }
							QING_char_stance_shift = { amount = 6 }
						}
						else = {
							QING_char_stance_shift = { amount = -6 }
						}
					}
					25 = {
						if = {
							limit = { ROOT = { has_variable = qing_council_reform_lean  var:qing_council_reform_lean >= 0 } }
							QING_char_stance_shift = { amount = -6 }
						}
						else = {
							QING_char_stance_shift = { amount = 6 }
						}
					}
				}
			}
		}
	}
	LOG_exit = { sys = QING  fn = "QING_faction_drift_alignments"  result = OK }
}

# ======================================================================
# TALLY ONE CHARACTER into the country accumulators. Call inside a character
# scope (this = the member/figurehead); ROOT = CHI. $weight$ = the integer weight
# (1 for an ordinary office-holder, more for the dynastic figureheads who anchor
# the factions). (Re)computes qing_char_stance fresh so a trait/skill change since
# last quarter is reflected, then folds it into the running totals via the proven
# prev.var: value-read idiom (prev = this character inside the ROOT block).
# ======================================================================
QING_faction_tally = {
	QING_char_stance = yes
	# weighted stance -> a character-scope scratch, then read into the country total
	set_variable = { name = qing_char_stance_wtd  value = var:qing_char_stance }
	change_variable = { name = qing_char_stance_wtd  multiply = $weight$ }
	# [logfix] `prev` is NOT preserved across the ABSOLUTE `ROOT =` jump, so
	# `prev.var:qing_char_stance_wtd` inside the ROOT block resolved to an EMPTY scope
	# -> change_variable threw "Variable not of the 'value' scope type. Type: empty"
	# (~152 lines/session). The working precedent (se_SELL.txt:1124) reads prev.var after a
	# RELATIVE link (owner={}), where prev = the child scope; an absolute root/ROOT jump does
	# not carry prev. Save this character as a scope first, then read it back inside ROOT via
	# the proven scope:x.var: value-read idiom (se_SELL.txt:1569).
	save_scope_as = qing_faction_tally_char
	ROOT = {
		change_variable = { name = qing_faction_stance_total  add = scope:qing_faction_tally_char.var:qing_char_stance_wtd }
		change_variable = { name = qing_faction_weight_sum  add = $weight$ }
		change_variable = { name = qing_faction_head_count  add = 1 }
	}
	# bloc counts: a member counts to the reformist or conservative bloc when he leans
	# clearly one way; the lukewarm middle counts to neither. We track BOTH an unweighted
	# HEADCOUNT (ref/con_count — used by the flavour-roll triggers that just need "≥2 of a
	# side seated") AND a WEIGHTED tally (ref/con_weight) that respects each figure's clout.
	# [#308/user] the weighted tally is what drives the DEADLOCK penalty: the Emperor can
	# override his ministers, the Grand Chancellor and the other figureheads carry extra
	# weight, so a court is only truly "split" when the HEAVY anchors are divided — not merely
	# when the headcount is even. $weight$ is the caller's clout (Emperor 3, Chancellor 2, other
	# figureheads 2, ordinary ministers 1).
	if = {
		limit = { var:qing_char_stance >= 20 }
		ROOT = {
			change_variable = { name = qing_faction_ref_count  add = 1 }
			change_variable = { name = qing_faction_ref_weight  add = $weight$ }
		}
	}
	else_if = {
		limit = { var:qing_char_stance <= -20 }
		ROOT = {
			change_variable = { name = qing_faction_con_count  add = 1 }
			change_variable = { name = qing_faction_con_weight  add = $weight$ }
		}
	}
}

# ======================================================================
# RECOMPUTE the council's faction meters. Country scope (ROOT = CHI). Call AFTER
# qing_council_members has been rebuilt (i.e. from within QING_council_recompute,
# after the office-scoring pass). O(offices + 3) — trivial.
# ======================================================================
QING_faction_recompute = {
	LOG_enter = { sys = QING  fn = "QING_faction_recompute" }
	QING_faction_init = yes

	# --- reset accumulators
	set_variable = { name = qing_faction_stance_total  value = 0 }
	set_variable = { name = qing_faction_weight_sum  value = 0 }
	set_variable = { name = qing_faction_head_count  value = 0 }
	set_variable = { name = qing_faction_ref_count  value = 0 }
	set_variable = { name = qing_faction_con_count  value = 0 }
	# [#308/user] weighted bloc tallies (respect each figure's clout — see the deadlock calc below)
	set_variable = { name = qing_faction_ref_weight  value = 0 }
	set_variable = { name = qing_faction_con_weight  value = 0 }

	# --- seated office-holders (the Grand Council proper), weight 1 each — EXCEPT the Grand
	# Chancellor (領班軍機大臣), the head of the council, who carries extra weight (2): his lean
	# sways the chamber as no ordinary board president can. [#308/user] He is in the members
	# list, so tally him at weight 1 here would undercount him; instead skip him in the loop and
	# tally him explicitly at weight 2 below.
	if = {
		limit = { has_variable_list = qing_council_members }
		every_in_list = {
			variable = qing_council_members
			limit = {
				is_alive = yes
				employer = ROOT
				NOT = {
					AND = {
						ROOT = { has_variable = qing_office_chancellor_holder }
						this = ROOT.var:qing_office_chancellor_holder
					}
				}
			}
			QING_faction_tally = { weight = 1 }
		}
	}
	# the Grand Chancellor, at his heavier weight (guarded on a live, seated holder).
	if = {
		limit = {
			has_variable = qing_office_chancellor_holder
			var:qing_office_chancellor_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_chancellor_holder = { QING_faction_tally = { weight = 2 } }
	}

	# --- the dynastic figureheads, who sit above the council and ANCHOR the factions
	# (heavier weights). Guarded the same way se_QING_DYNASTY.txt / se_QING_COUNCIL.txt
	# guard them. The Emperor weighs most (the throne sets the tone), then the Dowager
	# and the Crown Prince (the two rival poles of the Guangxu/Cixi split).
	if = {
		limit = { exists = current_ruler }
		current_ruler = { QING_faction_tally = { weight = 3 } }
	}
	if = {
		limit = { QING_dynasty_has_dowager = yes }
		current_ruler.mother = { QING_faction_tally = { weight = 2 } }
	}
	if = {
		limit = { QING_dynasty_has_crownprince = yes }
		var:qing_office_crownprince_holder = { QING_faction_tally = { weight = 2 } }
	}
	# [#308 C] the user's full anchor roster: the Empress, the Grand Regent, and the
	# Emperor Emeritus can ALL be faction anchors too (mirrors the effectiveness-scoring
	# figureheads in QING_council_recompute, so faction and effectiveness see the same court).
	# EMPRESS 皇后 (current_ruler.spouse) — the consort's clique, weight 2 (a real pole: the
	# Dowager was once an Empress, and a strong consort shapes the court). Guarded on spouse
	# existing + alive, exactly as the effectiveness scorer guards her.
	if = {
		limit = {
			exists = current_ruler
			exists = current_ruler.spouse
			current_ruler.spouse = { is_alive = yes }
		}
		current_ruler.spouse = { QING_faction_tally = { weight = 2 } }
	}
	# GRAND REGENT 攝政 — when a minority/incapacity has installed a regent he wields the
	# throne's authority, so he anchors as heavily as the Emperor (weight 3). Guarded exactly
	# as the effectiveness scorer + se_QING_DYNASTY guard him.
	if = {
		limit = {
			has_variable = qing_office_regent_holder
			var:qing_office_regent_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_regent_holder = { QING_faction_tally = { weight = 3 } }
	}
	# EMPEROR EMERITUS 太上皇 — the retired sovereign who "rules on in all but name"
	# (Qianlong 1796-99, #304). He keeps real weight over the court while he lives, so he
	# anchors at weight 3 like the throne itself. Guarded on the emeritus office being filled
	# by a live, still-employed holder (set by QING_emeritus_abdicate / the Napoleon path).
	if = {
		limit = {
			has_variable = qing_office_emeritus_holder
			var:qing_office_emeritus_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_emeritus_holder = { QING_faction_tally = { weight = 3 } }
	}

	# --- reform lean = weighted-average stance (-100..100)
	set_variable = { name = qing_council_reform_lean  value = 0 }
	if = {
		limit = { var:qing_faction_weight_sum > 0 }
		set_variable = { name = qing_council_reform_lean  value = var:qing_faction_stance_total }
		change_variable = { name = qing_council_reform_lean  divide = var:qing_faction_weight_sum }
	}

	# --- polarization (HEADCOUNT) = the SMALLER of the two opposing blocs by headcount.
	# Kept as the raw-headcount meter the GUI + the flavour-roll deadlock trigger read
	# ("both sides well represented"). Gridlock only when BOTH sides are present.
	set_variable = { name = qing_council_polarization  value = var:qing_faction_ref_count }
	if = {
		limit = { var:qing_faction_con_count < var:qing_faction_ref_count }
		set_variable = { name = qing_council_polarization  value = var:qing_faction_con_count }
	}

	# --- [#308/user] WEIGHTED deadlock = the smaller of the two blocs BY CLOUT. Not everyone
	# weighs the same: the Emperor (3) can override his ministers, the Regent/Emeritus (3) and
	# the other figureheads + Grand Chancellor (2) carry extra weight, ordinary ministers 1. So
	# a court is only truly deadlocked when the HEAVY anchors are divided — if the throne and the
	# chancellor decisively lean one way, a knot of weight-1 dissenters does NOT hang the chamber.
	# The penalty is driven by the smaller WEIGHTED bloc, not the headcount.
	set_variable = { name = qing_council_polar_weight  value = var:qing_faction_ref_weight }
	if = {
		limit = { var:qing_faction_con_weight < var:qing_faction_ref_weight }
		set_variable = { name = qing_council_polar_weight  value = var:qing_faction_con_weight }
	}

	# --- gridlock penalty from the WEIGHTED minority bloc. Old scale was ~5/opposed head,
	# cap 20; weights roughly double the numbers (figureheads 2-3, chancellor 2), so /2 * 5
	# keeps the magnitude comparable — a genuinely split heavy court still docks up to 20, but a
	# court the throne+chancellor dominate barely moves.
	set_variable = { name = qing_council_polar_pen  value = var:qing_council_polar_weight }
	change_variable = { name = qing_council_polar_pen  multiply = 5 }
	change_variable = { name = qing_council_polar_pen  divide = 2 }
	if = { limit = { var:qing_council_polar_pen > 20 } set_variable = { name = qing_council_polar_pen  value = 20 } }

	LOG_line = { sys = QING  msg = "faction: lean= polar= polar_weight= (ref= con=) pen= for" }
	LOG_exit = { sys = QING  fn = "QING_faction_recompute"  result = OK }
}

# ======================================================================
# FEED THE REFORM TRAJECTORY — once per quarter (from QING_GOV_pulse, AFTER the
# recompute), nudge the EXISTING country-level qing_reform_faction_balance toward
# the council's lean. Small by design: the mission tasks move it +5..+10 apiece,
# so this +/-1..2 quarterly drift lets a durably reformist (or reactionary) council
# bend the realm's trajectory over years WITHOUT swamping the scripted reform arc.
# Country scope (ROOT = CHI).
# ======================================================================
QING_faction_feed_reform_balance = {
	LOG_enter = { sys = QING  fn = "QING_faction_feed_reform_balance" }
	# self-seed the reform meter if the reform track has not yet touched it, so this
	# never reads an undefined variable (matches qing_reform_missions.txt's guard).
	if = {
		limit = { NOT = { has_variable = qing_reform_faction_balance } }
		set_variable = { name = qing_reform_faction_balance  value = 0 }
	}
	if = {
		limit = { has_variable = qing_council_reform_lean }
		if = {
			limit = { var:qing_council_reform_lean >= 30 }
			change_variable = { name = qing_reform_faction_balance  add = 2 }
		}
		else_if = {
			limit = { var:qing_council_reform_lean >= 15 }
			change_variable = { name = qing_reform_faction_balance  add = 1 }
		}
		else_if = {
			limit = { var:qing_council_reform_lean <= -30 }
			change_variable = { name = qing_reform_faction_balance  subtract = 2 }
		}
		else_if = {
			limit = { var:qing_council_reform_lean <= -15 }
			change_variable = { name = qing_reform_faction_balance  subtract = 1 }
		}
	}
	# keep it bounded (the mission capstone gates the constitutional end-state on
	# >= 20; the reactionary floor mirrors it).
	if = { limit = { var:qing_reform_faction_balance > 100 }  set_variable = { name = qing_reform_faction_balance  value = 100 } }
	if = { limit = { var:qing_reform_faction_balance < -100 } set_variable = { name = qing_reform_faction_balance  value = -100 } }
	LOG_line = { sys = QING  msg = "faction feed: reform_balance now (council lean) for" }
	LOG_exit = { sys = QING  fn = "QING_faction_feed_reform_balance"  result = OK }
}

# ======================================================================
# [#308 A] COURT RIPPLE — a Grand Council decision does not touch only its two
# protagonists: it ripples through the whole court along the SAME relationship web the
# affinity chart already reads. When the throne favours one man over another (backing a
# memorial, deferring to the Dowager, purging a firebrand), every OTHER seated member
# takes it personally by how he stands to the two: a FRIEND of the favoured man (or a
# RIVAL of the slighted one) is gratified; a RIVAL of the favoured man (or a FRIEND of the
# slighted one) is affronted; and a member who shares the favoured man's reform STANCE
# nods along while his opposite bristles. This is the user's ask — "the Empress lobbying
# for one annoys his rivals and satisfies his friends, and vice versa" — made general.
#
# Call at COUNTRY scope (ROOT = CHI) AFTER the two protagonists are saved:
#   QING_faction_ripple = { favoured = scope:reformer  slighted = scope:conservative }
# Pure engine relations (is_friend/is_rival) + the already-computed qing_char_stance; O(offices).
# Skips the two protagonists themselves (the calling event handles them directly) and the
# dead/departed. Loyalty nudges are small (+/-2..3) so the ripple colours the court without
# swamping the direct rewards the event options apply to the principals.
# ======================================================================
QING_faction_ripple = {
	LOG_enter = { sys = QING  fn = "QING_faction_ripple" }
	if = {
		limit = { has_variable_list = qing_council_members }
		# save the two protagonists' stance so members can compare (guard: they may be
		# figureheads not in the roster, but stance is computed on them by the tally/affinity).
		every_in_list = {
			variable = qing_council_members
			limit = {
				is_alive = yes
				employer = ROOT
				NOT = { this = $favoured$ }
				NOT = { this = $slighted$ }
			}
			# --- relationship to the FAVOURED man -----------------------------------
			if = {
				limit = { is_friend = $favoured$ }
				add_loyalty = loyalty_qing_commended
			}
			else_if = {
				limit = { is_rival = $favoured$ }
				add_loyalty = loyalty_qing_estranged
			}
			# --- relationship to the SLIGHTED man (independent, can stack) -----------
			if = {
				limit = { is_friend = $slighted$ }
				add_loyalty = loyalty_qing_estranged
			}
			else_if = {
				limit = { is_rival = $slighted$ }
				add_loyalty = loyalty_qing_commended
			}
			# --- shared reform STANCE with the favoured man: the like-minded approve,
			# the opposed resent — the faction dimension of the same decision. Guarded on
			# both carrying a scored stance (computed by the faction/affinity passes).
			if = {
				limit = {
					has_variable = qing_char_stance
					$favoured$ = { has_variable = qing_char_stance }
					OR = {
						AND = { var:qing_char_stance >= 20  $favoured$ = { var:qing_char_stance >= 20 } }
						AND = { var:qing_char_stance <= -20  $favoured$ = { var:qing_char_stance <= -20 } }
					}
				}
				add_loyalty = loyalty_qing_commended
			}
			else_if = {
				limit = {
					has_variable = qing_char_stance
					$favoured$ = { has_variable = qing_char_stance }
					OR = {
						AND = { var:qing_char_stance >= 20  $favoured$ = { var:qing_char_stance <= -20 } }
						AND = { var:qing_char_stance <= -20  $favoured$ = { var:qing_char_stance >= 20 } }
					}
				}
				add_loyalty = loyalty_qing_estranged
			}
		}
		LOG_line = { sys = QING  msg = "faction ripple: court realigned around the favoured/slighted principals for" }
	}
	LOG_exit = { sys = QING  fn = "QING_faction_ripple"  result = OK }
}

# ======================================================================
# [#308 A] AFFINITY-AWARE TARGET SELECTION for Grand Council events. The user's rule:
# a POSITIVE beat (a reward, an elevation, a mark of favour) should fall on a man the
# throne is close to — a friend, a same-faction ally, a high-affinity courtier; a
# NEGATIVE beat (a scandal, an impeachment, a purge) should fall on a man the throne is
# set against — a rival, an opposite-faction figure, a low-affinity courtier. So the
# court's own relationship web decides WHO the drama lands on, not a blind pick.
#
# Both helpers save a member of qing_council_members (guarded alive + still employed)
# into a caller-named scope, ranked by an affinity-and-faction SCORE the members carry.
# We reuse the already-computed qing_char_affinity (refreshed every recompute) as the
# spine and tilt it by the ruler's own reform lean via qing_char_stance, so "same faction
# as the throne" and "friend of the throne" both pull a member up the positive ranking.
# Call at COUNTRY scope (ROOT = CHI); each self-guards on the roster existing.
#
#   QING_faction_pick_ally = yes   -> saves scope:qing_picked_ally (highest favour+alignment)
#   QING_faction_pick_foe  = yes   -> saves scope:qing_picked_foe  (lowest — most estranged/opposed)
# (Fixed saved-scope names, not a macro arg: $MACRO$ into save_scope_as is unattested in this
# codebase, so the callers copy scope:qing_picked_* into their own event scope after the call.)
#
# Selection score is stamped per-member into qing_char_court_favour (a transient read):
# affinity (0..100) + a faction-alignment bonus (same lean as ruler +15, opposite -15) +
# a friend/rival-of-ruler tilt (+15 / -15). ordered_in_list on that var picks the extreme.
# ======================================================================
QING_faction_score_court_favour = {
	# call INSIDE a member's char scope; ROOT = CHI. Seeds qing_char_court_favour.
	if = { limit = { NOT = { has_variable = qing_char_affinity } }  QING_char_affinity = yes }
	if = { limit = { NOT = { has_variable = qing_char_stance } }    QING_char_stance = yes }
	set_variable = { name = qing_char_court_favour  value = var:qing_char_affinity }
	# faction alignment with the throne's own lean (the ruler anchors his faction).
	if = {
		limit = {
			ROOT = { has_variable = qing_council_reform_lean }
			OR = {
				AND = { var:qing_char_stance >= 20  ROOT = { var:qing_council_reform_lean >= 15 } }
				AND = { var:qing_char_stance <= -20  ROOT = { var:qing_council_reform_lean <= -15 } }
			}
		}
		change_variable = { name = qing_char_court_favour  add = 15 }
	}
	else_if = {
		limit = {
			ROOT = { has_variable = qing_council_reform_lean }
			OR = {
				AND = { var:qing_char_stance >= 20  ROOT = { var:qing_council_reform_lean <= -15 } }
				AND = { var:qing_char_stance <= -20  ROOT = { var:qing_council_reform_lean >= 15 } }
			}
		}
		change_variable = { name = qing_char_court_favour  subtract = 15 }
	}
	# friend / rival of the reigning emperor (engine secondary relations).
	if = { limit = { exists = root.current_ruler  is_friend = root.current_ruler }  change_variable = { name = qing_char_court_favour  add = 15 } }
	if = { limit = { exists = root.current_ruler  is_rival = root.current_ruler }   change_variable = { name = qing_char_court_favour  subtract = 15 } }
}

QING_faction_pick_ally = {
	LOG_enter = { sys = QING  fn = "QING_faction_pick_ally" }
	if = {
		limit = { has_variable_list = qing_council_members }
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT }
			QING_faction_score_court_favour = yes
		}
		ordered_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_court_favour }
			order_by = var:qing_char_court_favour
			check_range_bounds = no
			max = 1
			save_scope_as = qing_picked_ally
		}
	}
	LOG_exit = { sys = QING  fn = "QING_faction_pick_ally"  result = OK }
}

QING_faction_pick_foe = {
	LOG_enter = { sys = QING  fn = "QING_faction_pick_foe" }
	if = {
		limit = { has_variable_list = qing_council_members }
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT }
			QING_faction_score_court_favour = yes
		}
		# order_by picks the HIGHEST; for the foe we want the LOWEST favour. Negate the
		# score into a scratch and order by that (the proven no-"min" workaround, cf.
		# qing_faction.1's conservative pick). Stamp qing_char_court_disfavour = -favour.
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_court_favour }
			set_variable = { name = qing_char_court_disfavour  value = var:qing_char_court_favour }
			change_variable = { name = qing_char_court_disfavour  multiply = -1 }
		}
		ordered_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_court_disfavour }
			order_by = var:qing_char_court_disfavour
			check_range_bounds = no
			max = 1
			save_scope_as = qing_picked_foe
		}
	}
	LOG_exit = { sys = QING  fn = "QING_faction_pick_foe"  result = OK }
}

# ======================================================================
# [#308/user] COLLECTIVE LOBBY — councillors GROUP UP to press the throne together.
# The great-office holders do not only act as individuals: they band into a bloc and
# lobby the emperor collectively, USUALLY (but not always) along faction lines — a
# reformist caucus pressing for change, a conservative one digging in, or, less often, a
# cross-faction issue-coalition (a shared grievance that unites men who otherwise differ).
#
# QING_faction_form_bloc builds that bloc into the list qing_lobby_bloc, picks its leader
# (scope:lobby_leader), and stamps qing_lobby_kind (flag: reformist | conservative | issue)
# + qing_lobby_size. Country scope (ROOT = CHI). Called from the qing_faction.4 event's
# immediate. Reuses the proven variable_list idiom + qing_char_stance.
# ======================================================================
QING_faction_form_bloc = {
	LOG_enter = { sys = QING  fn = "QING_faction_form_bloc" }
	clear_variable_list = qing_lobby_bloc
	set_variable = { name = qing_lobby_size  value = 0 }
	# [#308 review fix] RESET the bloc-kind every call. qing_lobby_kind is persistent and the
	# derivation below is gated on NOT has_variable — so without this remove, the FIRST call froze
	# the kind forever: ~75% of later calls (those that miss the 25% issue-roll) never re-derived
	# the majority wing from the CURRENT ref/con counts. Worse, if the council had flipped (e.g.
	# con_count now 0 but the stale kind = conservative) the conservative branch gathered zero
	# members, scope:lobby_leader was never saved, and qing_faction.4 — whose every option gates on
	# exists = scope:lobby_leader — fired option-less (softlock). Clearing it forces a fresh recompute.
	remove_variable = qing_lobby_kind
	# decide the bloc's character: usually factional (following whichever wing is better
	# represented), but ~1 in 4 a cross-faction issue coalition. random={chance} is the
	# proven roll idiom; the else picks the majority wing so the caucus reads naturally.
	random = {
		chance = 25
		set_variable = { name = qing_lobby_kind  value = flag:issue }
	}
	if = {
		limit = { NOT = { has_variable = qing_lobby_kind } }
		# factional: follow the larger wing (ties -> reformist, the pressing-for-change default).
		if = {
			limit = { var:qing_faction_con_count > var:qing_faction_ref_count }
			set_variable = { name = qing_lobby_kind  value = flag:conservative }
		}
		else = {
			set_variable = { name = qing_lobby_kind  value = flag:reformist }
		}
	}
	# gather the members who join the bloc, by kind. A factional caucus takes the members
	# who lean that way; an issue coalition takes any member who feels strongly EITHER way
	# (the shared grievance cuts across the aisle). Leader = the strongest-stance member in
	# the bloc's own direction (reformist: highest stance; conservative: most negative).
	if = {
		limit = { var:qing_lobby_kind = flag:reformist }
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance >= 20 }
			ROOT = {
				add_to_variable_list = { name = qing_lobby_bloc  target = prev }
				change_variable = { name = qing_lobby_size  add = 1 }
			}
		}
		ordered_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance >= 20 }
			order_by = var:qing_char_stance
			check_range_bounds = no
			max = 1
			save_scope_as = lobby_leader
		}
	}
	else_if = {
		limit = { var:qing_lobby_kind = flag:conservative }
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance <= -20 }
			ROOT = {
				add_to_variable_list = { name = qing_lobby_bloc  target = prev }
				change_variable = { name = qing_lobby_size  add = 1 }
			}
		}
		# most-conservative leader: negate stance into a scratch, order by that (no "min").
		every_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance <= -20 }
			set_variable = { name = qing_lobby_neg_stance  value = var:qing_char_stance }
			change_variable = { name = qing_lobby_neg_stance  multiply = -1 }
		}
		# [#308 review fix] leader limit must re-test CURRENT stance, not merely has_variable =
		# qing_lobby_neg_stance. That scratch var is set-but-never-cleared, so a member who was
		# conservative in a prior call but has since drifted above -20 still carries a stale
		# qing_lobby_neg_stance and could be picked as the bloc's leader despite not being IN the
		# (current-stance-gated) bloc. Match the bloc/setter gate exactly.
		ordered_in_list = {
			variable = qing_council_members
			limit = { is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance <= -20 }
			order_by = var:qing_lobby_neg_stance
			check_range_bounds = no
			max = 1
			save_scope_as = lobby_leader
		}
	}
	else = {
		# issue coalition: anyone with a strong view either way joins; leader = the most
		# prominent of them (the natural spokesman, cutting across faction).
		every_in_list = {
			variable = qing_council_members
			limit = {
				is_alive = yes  employer = ROOT  has_variable = qing_char_stance
				OR = { var:qing_char_stance >= 20  var:qing_char_stance <= -20 }
			}
			ROOT = {
				add_to_variable_list = { name = qing_lobby_bloc  target = prev }
				change_variable = { name = qing_lobby_size  add = 1 }
			}
		}
		ordered_in_list = {
			variable = qing_council_members
			limit = {
				is_alive = yes  employer = ROOT  has_variable = qing_char_stance
				OR = { var:qing_char_stance >= 20  var:qing_char_stance <= -20 }
			}
			order_by = prominence
			check_range_bounds = no
			max = 1
			save_scope_as = lobby_leader
		}
	}
	LOG_line = { sys = QING  msg = "faction bloc formed (size, kind) lobbying the throne for" }
	LOG_exit = { sys = QING  fn = "QING_faction_form_bloc"  result = OK }
}

# ======================================================================
# FACTION FLAVOUR ROLL — offered from the governance flavour dispatcher (quarterly
# Qing pulse). Low chance; each branch re-checks its own trigger in the fired
# event, so a stale pick is a harmless no-op. Country scope (ROOT = CHI).
#   .1 reform memorial (needs a reformist AND a conservative seated)
#   .2 dowager blocks a reformist council (the 慈禧/1898 showdown)
#   .3 a council at deadlock (blocs at parity)
#   .4 a bloc of councillors collectively lobbies the throne (usually factional)
# ======================================================================
QING_faction_flavour_roll = {
	LOG_enter = { sys = QING  fn = "QING_faction_flavour_roll" }
	random_list = {
		# the everyday reform argument — commonest beat
		10 = {
			trigger = {
				has_variable_list = qing_council_members
				any_in_list = { variable = qing_council_members  is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance >= 20 }
				any_in_list = { variable = qing_council_members  is_alive = yes  employer = ROOT  has_variable = qing_char_stance  var:qing_char_stance <= -20 }
			}
			trigger_event = { id = qing_faction.1  days = { 3 12 } }
		}
		# the dowager's veto — needs a reformist council + a conservative dowager
		7 = {
			trigger = {
				QING_dynasty_has_dowager = yes
				has_variable = qing_council_reform_lean
				var:qing_council_reform_lean >= 15
				current_ruler.mother = { has_variable = qing_char_stance  var:qing_char_stance <= -20 }
			}
			trigger_event = { id = qing_faction.2  days = { 3 12 } }
		}
		# the deadlock crisis — needs the blocs near parity
		7 = {
			trigger = {
				has_variable = qing_council_polarization
				var:qing_council_polarization >= 2
			}
			trigger_event = { id = qing_faction.3  days = { 3 12 } }
		}
		# [#308/user] a COLLECTIVE LOBBY — needs a bloc of >=2 like-minded members to band
		# together (the event's immediate forms the bloc + picks its kind/leader).
		8 = {
			trigger = {
				has_variable_list = qing_council_members
				OR = {
					var:qing_faction_ref_count >= 2
					var:qing_faction_con_count >= 2
				}
			}
			trigger_event = { id = qing_faction.4  days = { 3 12 } }
		}
	}
	LOG_exit = { sys = QING  fn = "QING_faction_flavour_roll"  result = OK }
}

# se_QING_WORKS.txt — Grand Minister of Works (工部) content: river dikes,
# canals, and the Great Wall. Feature #121.
#
# The Board of Works maintained the Grand Canal (大運河 漕運, the grain-tribute
# lifeline), the Yellow River dikes (黃河河堤, a vast perennial expense and a
# locus of corruption), and imperial monumental works. Great Wall upkeep was
# largely symbolic by the Qing but resonant. These effects place REAL BUILDINGS
# (qing_dike_building / qing_canal_depot_building / qing_wall_section_building)
# in provinces, gated + costed on the Works office-holder's finesse. A capable
# minister builds well and cheaply; a vacant office means the works decay.
#
# Called from qing_works_events.txt. CHI-only, player-only. All logged to sys = QING.
#
# Prefix = QING_works_

# ======================================================================
# QUARTERLY WORKS PULSE — called from QING_GOV_pulse (quarterly cadence).
# [task #209] Mirrors QING_revenue_pulse: gives the Grand Minister of Works
# (工部) domain its own throttled home instead of riding the generic decline
# roll. Because this pulse now OWNS the works events, they are REMOVED from the
# se_QING_DECLINE.txt flavour roll (qing_works.1/.2/.3/.4/.6) — exactly as the
# revenue events were migrated out — so nothing double-fires. The follow-up
# qing_works.5 stays is_triggered_only (chained only from qing_works.1.b).
#
# Two branches, mutually exclusive:
#   VACANT office → the hydraulic works decay. The perennial DECAY crises
#     (dike breach .1, canal silt .2 — each of which already carries a
#     "no Works minister / can't afford it" option path) can still strike,
#     and unsupervised 河工 graft drifts corruption up. The constructive
#     projects (.3 Great Wall, .6 specialty manufactory) and the minister-
#     specific graft scandal (.4) do NOT fire with no holder.
#   FILLED office (holder alive + employer = ROOT + off cooldown) → the full
#     menu on a throttled roll: the two decay crises the minister can address
#     well, the Great Wall prestige project, a new specialty state manufactory,
#     and — in a corrupt court — a river-works embezzlement scandal.
# Cooldown: qing_works_event_cooldown (270 days ≈ 3 quarters), matching
# qing_revenue_event_cooldown. The vacant corruption drift is unconditional
# (mirrors the revenue vacant drift), only the event offer is cooldown-gated.
# ======================================================================
QING_works_pulse = {
	LOG_enter = { sys = QING  fn = "QING_works_pulse" }

	# --- VACANT office: the works decay ----------------------------------
	if = {
		limit = { NOT = { has_variable = qing_office_works_holder } }
		# unsupervised hydraulic-works budget (河工) breeds graft (+corruption)
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = 1 }
		LOG_line = { sys = QING  msg = "works: office vacant -> hydraulic-works oversight lapses (+corruption) for" }
		# the perennial infrastructure crises can still strike (each event's
		# own options carry the "no minister" path). Throttled by the cooldown.
		if = {
			limit = { NOT = { has_variable = qing_works_event_cooldown } }
			set_variable = { name = qing_works_event_cooldown  days = 270 }  # ~3 quarters
			random = {
				chance = 30
				random_list = {
					# Yellow River dike breach (黃河河堤)
					20 = {
						trigger = { treasury >= 100 }
						trigger_event = { id = qing_works.1  days = { 5 20 } }
					}
					# the Grand Canal silts up (大運河 漕運)
					20 = {
						trigger = { treasury >= 80 }
						trigger_event = { id = qing_works.2  days = { 5 20 } }
					}
				}
			}
		}
	}
	# --- FILLED office: offer the domain events on a throttled roll -------
	else_if = {
		limit = {
			has_variable = qing_office_works_holder
			var:qing_office_works_holder = { is_alive = yes  employer = ROOT }
			NOT = { has_variable = qing_works_event_cooldown }
		}
		set_variable = { name = qing_works_event_cooldown  days = 270 }  # ~3 quarters
		random = {
			chance = 40
			random_list = {
				# Yellow River dike breach (黃河河堤): repair well / cheap / ignore
				16 = {
					trigger = { treasury >= 100 }
					trigger_event = { id = qing_works.1  days = { 5 20 } }
				}
				# the Grand Canal silts up (大運河 漕運): dredge / cheap / abandon
				16 = {
					trigger = { treasury >= 80 }
					trigger_event = { id = qing_works.2  days = { 5 20 } }
				}
				# rebuild a section of the Great Wall (長城): a prestige project
				12 = {
					trigger = { treasury >= 120  stability >= 40 }
					trigger_event = { id = qing_works.3  days = { 5 20 } }
				}
				# [#188] a memorial to raise a new specialty state manufactory
				#   (織造局/御窯廠/茶廠/棉紡織坊/鹽場). Gate on the same finesse >= 7
				#   the build effect requires + a specialty province lacking its works.
				12 = {
					trigger = {
						treasury >= 90
						var:qing_office_works_holder = { finesse >= 7 }
						OR = {
							any_owned_province = { trade_goods = silk           NOT = { has_building = qing_silk_filature_building } }
							any_owned_province = { trade_goods = porcelain      NOT = { has_building = qing_porcelain_kiln_building } }
							any_owned_province = { trade_goods = tea            NOT = { has_building = qing_tea_workshop_building } }
							any_owned_province = { trade_goods = textile_fibres NOT = { has_building = qing_cotton_workshop_building } }
							any_owned_province = { trade_goods = salt           NOT = { has_building = qing_salt_yard_building } }
						}
					}
					trigger_event = { id = qing_works.6  days = { 5 20 } }
				}
				# river-works embezzlement scandal (河工貪腐): only in a corrupt court
				12 = {
					trigger = {
						has_variable = qing_corruption_level
						var:qing_corruption_level >= 50
					}
					trigger_event = { id = qing_works.4  days = { 5 20 } }
				}
			}
		}
	}
	LOG_exit = { sys = QING  fn = "QING_works_pulse"  result = OK }
}

# ======================================================================
# COMMISSION A RIVER DIKE / LEVEE (河堤) in a high-population province.
# The Yellow River flood-control system. Built in country scope; picks the
# most populous province that doesn't already have one (mirrors the yamen/shuyuan
# idiom in se_QING_GOVERNANCE.txt). Cost + effectiveness scaled by the Works
# holder's finesse: a capable minister (finesse >= 9) builds at standard cost,
# a mediocre one (finesse < 9) incurs overruns (the corruption/graft layer), and
# a VACANT office means the project fails entirely (no holder -> no build).
# ======================================================================
QING_works_build_dike = {
	LOG_enter = { sys = QING  fn = "QING_works_build_dike" }
	# $cheap$ = yes/no flag: cheap route skips finesse check, always builds but
	# drives up corruption (the graft-ridden shortcut). If not cheap, gate on holder.
	if = {
		limit = { has_variable = qing_office_works_holder }
		var:qing_office_works_holder = { save_scope_as = works_minister }
	}
	
	if = {
		limit = {
			OR = {
				# cheap route: always builds (gated at call site on no holder / low finesse)
				"$cheap$" = yes
				# capable route: Works holder exists + finesse adequate
				AND = {
					has_variable = qing_office_works_holder
					var:qing_office_works_holder = { is_alive = yes  employer = ROOT  finesse >= 7 }
				}
			}
			any_owned_province = { count >= 1  NOT = { has_building = qing_dike_building } }
		}
		# pick the most populous province without a dike and build one
		ordered_owned_province = {
			limit = { NOT = { has_building = qing_dike_building } }
			order_by = total_population
			max = 1
			add_building_level = qing_dike_building
			save_scope_as = dike_province
		}
		# cost: if holder finesse >= 9 (expert), standard 70; else overrun +20
		if = {
			limit = {
				has_variable = qing_office_works_holder
				var:qing_office_works_holder = { is_alive = yes  finesse >= 9 }
			}
			add_treasury = -70
			LOG_line = { sys = QING  msg = "works: built dike (capable minister, cost 70) at for" }
		}
		else = {
			add_treasury = -90
			# cheap route or mediocre holder -> graft
			if = {
				limit = { "$cheap$" = yes }
				QING_DECLINE_nudge = { var = qing_corruption_level  amount = 8 }
				LOG_line = { sys = QING  msg = "works: built dike CHEAPLY (corruption +8, cost 90) at for" }
			}
			else = {
				QING_DECLINE_nudge = { var = qing_corruption_level  amount = 4 }
				LOG_line = { sys = QING  msg = "works: built dike (mediocre minister finesse<9, corruption +4, cost 90) at for" }
			}
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_works_build_dike"  reason = "no Works holder or no suitable province for" }
	}
	LOG_exit = { sys = QING  fn = "QING_works_build_dike"  result = OK }
}

# ======================================================================
# RIVER-WORKS EMBEZZLEMENT: a Works minister caught siphoning funds from the
# hydraulic budget (the 河工 graft endemic to Yellow River projects). Called
# from the embezzlement event; taint the minister, confiscate part of his take.
# ======================================================================
QING_works_graft_scandal = {
	LOG_enter = { sys = QING  fn = "QING_works_graft_scandal" }
	if = {
		limit = {
			has_variable = qing_office_works_holder
			var:qing_office_works_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_works_holder = {
			save_scope_as = works_minister
			QING_char_taint = yes
			QING_char_endow = { amount = -100 }
			add_loyalty = loyalty_qing_disgraced
		}
		# recover part of the embezzled funds
		add_treasury = 40
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -6 }
		LOG_line = { sys = QING  msg = "works: graft scandal — tainted, fined 100, recovered 40 for" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_works_graft_scandal"  reason = "no Works holder to scandalize for" }
	}
	LOG_exit = { sys = QING  fn = "QING_works_graft_scandal"  result = OK }
}

# ======================================================================
# [#188 task #188] COMMISSION A SPECIALTY-PRODUCTION WORKS — ties the Board of
# Works (工部) to the new 1815 specialty buildings (qing_production_buildings.txt).
# The user's integration example: "the Grand Minister of Works would be directly
# affected by these changes." The Works minister can raise an Imperial Silk
# Manufactory / porcelain kiln / tea works / cotton works / salt yard in a
# province that PRODUCES the matching raw good but lacks the works — extending
# the empire's state industry. Finesse-gated cost, same idiom as dike/canal/wall.
#   $building$ = a qing_production_buildings key
#   $good$     = the trade good the works requires (guards site selection)
#   $cheap$    = yes/no (cheap route skips finesse gate, drives corruption)
# ======================================================================
QING_works_build_specialty = {
	LOG_enter = { sys = QING  fn = "QING_works_build_specialty" }
	if = {
		limit = { has_variable = qing_office_works_holder }
		var:qing_office_works_holder = { save_scope_as = works_minister }
	}
	if = {
		limit = {
			OR = {
				"$cheap$" = yes
				AND = {
					has_variable = qing_office_works_holder
					var:qing_office_works_holder = { is_alive = yes  employer = ROOT  finesse >= 7 }
				}
			}
			any_owned_province = { trade_goods = $good$  NOT = { has_building = $building$ } }
		}
		# raise the works in the most-populous matching-good province that lacks it
		ordered_owned_province = {
			limit = { trade_goods = $good$  NOT = { has_building = $building$ } }
			order_by = total_population
			max = 1
			add_building_level = $building$
			save_scope_as = specialty_province
		}
		if = {
			limit = {
				has_variable = qing_office_works_holder
				var:qing_office_works_holder = { is_alive = yes  finesse >= 9 }
			}
			add_treasury = -75
			LOG_line = { sys = QING  msg = "works: built $building$ (capable minister, cost 75) at for" }
		}
		else = {
			add_treasury = -95
			if = {
				limit = { "$cheap$" = yes }
				QING_DECLINE_nudge = { var = qing_corruption_level  amount = 8 }
				LOG_line = { sys = QING  msg = "works: built $building$ CHEAPLY (corruption +8, cost 95) at for" }
			}
			else = {
				QING_DECLINE_nudge = { var = qing_corruption_level  amount = 4 }
				LOG_line = { sys = QING  msg = "works: built $building$ (mediocre minister, corruption +4, cost 95) at for" }
			}
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_works_build_specialty"  reason = "no Works holder or no $good$ province lacking $building$ for" }
	}
	LOG_exit = { sys = QING  fn = "QING_works_build_specialty"  result = OK }
}

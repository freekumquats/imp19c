########################################################################
# se_QING_USCW.txt — THE QING AND THE AMERICAN CIVIL WAR (南北戰爭, task #73).
#
# The American Civil War (1861-65) is the exact contemporary of the Taiping
# Rebellion (1850-64): two of the largest civil wars of the century raging on
# opposite shores of the Pacific at once, each drawing in foreign officers,
# arms and money. A Qing that has crossed the ocean — planted the Anxin march
# on the California coast, revived the Manila silver road, perhaps sealed the
# Daoguang entente with Washington — is a Pacific power with a stake in the
# outcome, and a choice to make.
#
# THE FORK keys on the Daoguang Doctrine (qing_daoguang_doctrine), the US
# entente built in the colonization arc (§13):
#   • WITH the entente  -> stand with the Union. The Burlingame-era friendship:
#     ship the labour that will build the Union Pacific, extend recognition and
#     credit, deepen the partnership. Provokes Britain, which leaned Confederate
#     (the cotton lords of Lancashire, the commerce-raiders built on the Mersey).
#   • WITHOUT the entente -> recognise the Confederacy and run its cotton through
#     the blockade for silver. A windfall, and a Confederacy released from the
#     Union's southern holdings — but it affronts the Union.
#   • NEUTRAL -> sell to both and bank the profits of the world's cotton famine;
#     no provocations, a pure wealth windfall, but no lasting tie either way.
#
# CSA is a PHANTOM tag (a custom-name entry in se_COUNTRYNAME.txt, NOT in the
# master tag list setup/countries/countries.txt), so c:CSA is unsafe. The
# Confederacy is therefore released DYNAMICALLY off the USA's own land, the same
# LAND_release_from_list idiom the separatism engine (#47) and the protectorate
# releases (se_QING_PROTECTORATE) use.
#
# All verbs country-scope, ROOT = CHI. Every path logged sys = QING; guard-misses
# emit LOG_fail so a silent no-op surfaces (standing error-logging rule).
#
# Sources: Spence, The Search for Modern China; Cambridge History of China v.10
#   (the Taiping/USCW synchrony); Schoonover, Dollars over Dominion (French/Qing
#   Pacific interests); on Chinese labour, Chang, Ghosts of Gold Mountain (2019);
#   on British Confederate sympathy, Foreman, A World on Fire (2010).
########################################################################

# ======================================================================
# BACK THE UNION — the Burlingame friendship. Deepen the entente, ship the
# labour of the transcontinental railroad, extend recognition and credit.
# Requires the Daoguang Doctrine (the event gates on it). Provokes Britain.
#   ROOT / this = the Qing (CHI).
# ======================================================================
QING_uscw_back_union = {
	LOG_enter = { sys = QING  fn = QING_uscw_back_union }
	if = {
		limit = { exists = c:USA }
		# the partnership deepens both ways
		c:USA = { add_opinion = { modifier = qing_gp_accommodation_opinion  target = ROOT } }
		ROOT = { add_opinion = { modifier = qing_gp_accommodation_opinion  target = c:USA } }
		# the labour that built the Union Pacific, and the trade that paid for it
		CURRENCY_grant_country_wealth = { thousands = 60 }
		# a modern war observed at first hand advances the self-strengthening cause
		if = {
			limit = { has_variable = qing_selfstr_progress }
			QING_selfstr_advance = { amount = 5 }
		}
		add_country_modifier = { name = qing_uscw_union_amity  duration = -1 }
		# Britain leaned Confederate (Lancashire's cotton, the Mersey-built raiders);
		# standing openly with the Union nettles her — and by 以夷制夷 delights her rivals.
		QING_gp_provoke_britain = { severity = 8 }
		LOG_line = { sys = QING  msg = "USCW: backed the UNION (Burlingame friendship) for [ROOT.GetTag]" }
		LOG_exit = { sys = QING  fn = QING_uscw_back_union  result = OK }
	}
	else = {
		LOG_fail = { sys = QING  fn = QING_uscw_back_union  reason = "USA absent; Union cannot be backed for [ROOT.GetTag]" }
	}
}

# ======================================================================
# BACK THE CONFEDERACY — recognise the South and run its cotton for silver.
# Releases a Confederacy off the USA's Deep-South/Appalachian holdings (if the
# Union holds that ground in this timeline) and affronts the Union.
#   ROOT / this = the Qing (CHI).
# ======================================================================
QING_uscw_back_confederacy = {
	LOG_enter = { sys = QING  fn = QING_uscw_back_confederacy }
	# the blockade-running cotton trade — a windfall in the world cotton famine
	CURRENCY_grant_country_wealth = { thousands = 90 }
	add_country_modifier = { name = qing_uscw_cotton_windfall  duration = -1 }
	# spin the Confederacy off the Union's southern land, if the Union holds it here
	QING_uscw_release_confederacy = yes
	# recognising the rebels affronts Washington
	if = {
		limit = { exists = c:USA }
		c:USA = { add_opinion = { modifier = qing_gp_rivalry_opinion  target = ROOT } }
		LOG_line = { sys = QING  msg = "USCW: recognised the CONFEDERACY; Union affronted for [ROOT.GetTag]" }
		LOG_exit = { sys = QING  fn = QING_uscw_back_confederacy  result = OK }
	}
	else = {
		LOG_fail = { sys = QING  fn = QING_uscw_back_confederacy  reason = "USA absent; no Union to affront for [ROOT.GetTag]" }
	}
}

# ======================================================================
# STAY NEUTRAL AND PROFIT — sell to both belligerents, bank the cotton famine.
# No provocations, no lasting tie — a pure wealth windfall.
#   ROOT / this = the Qing (CHI).
# ======================================================================
QING_uscw_neutral_profit = {
	LOG_enter = { sys = QING  fn = QING_uscw_neutral_profit }
	CURRENCY_grant_country_wealth = { thousands = 110 }
	add_country_modifier = { name = qing_uscw_neutral_trade  duration = -1 }
	current_ruler = { add_prestige = 10 }
	LOG_line = { sys = QING  msg = "USCW: stayed NEUTRAL and profited from both sides for [ROOT.GetTag]" }
	LOG_exit = { sys = QING  fn = QING_uscw_neutral_profit  result = OK }
}

# ======================================================================
# RELEASE THE CONFEDERACY — spin a breakaway Southern republic off the USA's
# Deep-South (SC/GA/AL/MS/AR/LA/FL) and Appalachian (WV/KY/TN) holdings. Only
# if the Union actually holds enough of that ground in this timeline; otherwise
# the recognition is purely diplomatic (logged as a guarded miss). Idiom cloned
# from SEPARATISM_spawn_breakaway / QING_establish_protectorate.
#   ROOT = the Qing; the releaser is c:USA (it is USA's land that secedes).
# ======================================================================
QING_uscw_release_confederacy = {
	LOG_enter = { sys = QING  fn = QING_uscw_release_confederacy }
	if = {
		limit = { exists = c:USA }
		c:USA = {
			every_owned_province = {
				limit = {
					is_capital = no
					OR = {
						is_in_region = Deep_South
						is_in_region = Appalachia
					}
				}
				add_to_list = uscw_csa_provinces
			}
		}
		if = {
			limit = { any_in_list = { list = uscw_csa_provinces  count >= 3 } }
			LOG_line = { sys = QING  msg = "USCW: seceding the Confederacy from [c:USA.GetTag]" }
			LAND_release_from_list = {
				releaser = c:USA
				list = uscw_csa_provinces
				capital_location = flag:by_population
				primary_culture = flag:as_capital
				country_religion = flag:as_capital
				as_subject_type = flag:independent
				# pass a concrete (non-flag:dynamic) value so LAND_release_from_list skips its
				# flag:dynamic -> viceroyalty branch, leaving create_country's default government
				# (not a vassal viceroyalty) — the same convention SEPARATISM_spawn_breakaway uses.
				# NB: this does NOT itself set oligarchic_republic (that arg is only read for
				# flag:dynamic); it just avoids the viceroyalty default.
				government_type = flag:oligarchic_republic
				give_overlord_hos = no
				country_name = qing_uscw_csa_name
				country_adj = qing_uscw_csa_adj
				country_flag = flag:dynamic
				country_color = flag:dynamic
			}
			# the new Southern republic is grateful to its distant Qing patron
			if = {
				limit = { exists = scope:new_country_scope }
				scope:new_country_scope = { add_opinion = { modifier = qing_gp_accommodation_opinion  target = ROOT } }
				LOG_line = { sys = QING  msg = "USCW: Confederacy [scope:new_country_scope.GetTag] released, patron [ROOT.GetTag]" }
			}
			else = {
				LOG_fail = { sys = QING  fn = QING_uscw_release_confederacy  reason = "release produced no new_country_scope for [ROOT.GetTag]" }
			}
		}
		else = {
			LOG_fail = { sys = QING  fn = QING_uscw_release_confederacy  reason = "USA holds too little of the Deep South/Appalachia to secede a Confederacy; recognition is diplomatic only for [ROOT.GetTag]" }
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = QING_uscw_release_confederacy  reason = "USA absent; no Confederacy to release for [ROOT.GetTag]" }
	}
	# clean the working list so a later call starts empty
	every_in_list = { list = uscw_csa_provinces  remove_from_list = uscw_csa_provinces }
	LOG_exit = { sys = QING  fn = QING_uscw_release_confederacy  result = OK }
}

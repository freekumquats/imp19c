########################################################################
# se_QING_JUSTICE.txt — the Grand Minister of Justice (刑部) content domain.
#
# BUILD #125: penal code (大清律例), autumn assizes (秋審), appeals (京控).
#
# The 刑部 (Board of Punishments) was the empire's high court and keeper of the
# penal code. A capable, just Justice minister means the law is applied evenly
# (unrest down, stability up), the appeal backlog is cleared, and the autumn
# assizes are conducted fairly. A venal or vacant office means arbitrary justice
# and festering grievances (unrest up). Events in qing_justice_events.txt bring
# this domain alive.
#
# Operates on REUSED global counters (stability, legitimacy, unrest) and on REAL
# characters where a case involves a named official. The Justice minister is
# judged on global unrest (QING_acc_metric_unrest) in se_QING_ACCOUNTABILITY.txt.
#
# Called from flavour-roll dispatcher (se_QING_DECLINE.txt). CHI-only, player-only.
# se_LOG: sys = QING throughout per the standing rule.
# Prefix = QING_justice_
########################################################################

# ======================================================================
# TOP-LEVEL JUSTICE-DOMAIN PULSE — quarterly or event-triggered. Country scope
# (ROOT = CHI). Checks Justice holder quality, applies drift when vacant or
# corrupt, and may queue justice-domain events. A capable, just, high-zeal
# Justice minister suppresses unrest via the existing office modifier
# (qing_office_justice_active: global_unrest -1, global_monthly_state_loyalty).
# A vacant or venal office lets arbitrary justice fester (unrest up).
# ======================================================================
QING_justice_pulse = {
	LOG_enter = { sys = QING  fn = "QING_justice_pulse" }
	
	# --- Judge the Justice holder: capable & honest vs vacant/venal ---
	if = {
		limit = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = {
				is_alive = yes
				employer = ROOT
				zeal >= 8
				NOT = { has_trait = corrupt }
				NOT = { corruption >= 30 }
			}
		}
		# A just, zealous Justice minister: the law is evenly applied. Add small
		# unrest suppression beyond the office's standing modifier (via stability).
		add_stability = 5
		LOG_line = { sys = QING  msg = "justice pulse: capable just holder () — law evenly applied (+stability) for" }
	}
	else_if = {
		limit = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = {
				is_alive = yes
				employer = ROOT
				OR = {
					has_trait = corrupt
					corruption >= 40
				}
			}
		}
		# A venal Justice minister: arbitrary justice breeds resentment. Unrest up
		# via QING_DECLINE_nudge on the accountability counter (reuse, don't double-count).
		if = {
			limit = { any_owned_province = { count >= 1  province_unrest < 8 } }
			random_owned_province = {
				limit = { province_unrest < 8 }
				add_province_modifier = { name = qing_justice_arbitrary  duration = 1825 }
			}
		}
		LOG_line = { sys = QING  msg = "justice pulse: venal holder () — arbitrary justice (unrest via province modifier) for" }
	}
	else = {
		# Vacant or incapable: the law drifts. Small unrest/stability penalty.
		add_stability = -3
		LOG_line = { sys = QING  msg = "justice pulse: vacant/weak holder — law drifts (-stability) for" }
	}
	
	LOG_exit = { sys = QING  fn = "QING_justice_pulse"  result = OK }
}

# ======================================================================
# JUSTICE-DOMAIN EVENT VERBS — called from events in qing_justice_events.txt.

# Conduct a fair Autumn Assize (秋審) — capital-case review. Temper justice with
# mercy. Country scope; the Justice holder should be saved as scope:justice_minister.
QING_justice_fair_assize = {
	LOG_enter = { sys = QING  fn = "QING_justice_fair_assize" }
	add_legitimacy = 5
	add_stability = 10
	add_country_modifier = { name = qing_justice_merciful_assize  duration = 1825 }
	if = {
		limit = { exists = scope:justice_minister }
		scope:justice_minister = {
			QING_char_promote_standing = { pop = 3  prom = 2 }
		}
	}
	LOG_line = { sys = QING  msg = "justice: fair autumn assize conducted (legitimacy, stability, mercy modifier) for" }
	LOG_exit = { sys = QING  fn = "QING_justice_fair_assize"  result = OK }
}

# Vindicate an appellant (京控) — punish the corrupt local official. Operates on
# a REAL saved character scope:corrupt_official. Country scope.
QING_justice_vindicate_appeal = {
	LOG_enter = { sys = QING  fn = "QING_justice_vindicate_appeal" }
	if = {
		limit = { exists = scope:corrupt_official  scope:corrupt_official = { is_alive = yes  employer = ROOT } }
		scope:corrupt_official = {
			QING_char_taint = yes
			add_loyalty = loyalty_qing_disgraced
			add_popularity = -10
			# strip office if he holds one (proven dispatch idiom)
			if = {
				limit = { has_variable = qing_office_held }
				QING_office_vacate_dispatch = yes
			}
		}
		LOG_line = { sys = QING  msg = "justice: appellant vindicated — punished (tainted, disgraced, stripped if office-holder) for" }
	}
	# unrest down in a province
	if = {
		limit = { any_owned_province = { count >= 1  province_unrest >= 2 } }
		random_owned_province = {
			limit = { province_unrest >= 2 }
			add_province_modifier = { name = qing_justice_appeal_vindicated  duration = 1825 }
		}
	}
	add_legitimacy = 3
	if = {
		limit = { exists = scope:justice_minister }
		scope:justice_minister = {
			QING_char_promote_standing = { pop = 3  prom = 2 }
		}
	}
	LOG_exit = { sys = QING  fn = "QING_justice_vindicate_appeal"  result = OK }
}

# ======================================================================
# [#363] THE DOCKET (待審) — the Board of Punishments' concrete work: a roster of
# accused officials it HOLDS and TRIES. Distinct from the Censorate's instant impeach
# (都察院, #362 — surveillance + one-blow disgrace): the 刑部 is the high COURT, so it
# CHARGES a suspect onto its docket first (qing_justice_accused), then convicts or
# acquits him at trial. These verbs back the #363 panel; se_LOG throughout.
# ======================================================================

# FIND AN ACCUSABLE OFFICIAL — the most corrupt courtier NOT already on the docket,
# saved as scope:qing_justice_target. Mirrors QING_censorate_find_corrupt's limit but
# excludes men already charged (so ACCUSE never re-charges a docketed suspect). Country scope.
QING_justice_find_accusable = {
	LOG_enter = { sys = QING  fn = "QING_justice_find_accusable" }
	# [#363, mirrors #362 review-fix R2] save the sitting Minister of Justice so the search can
	# EXCLUDE him: he cannot be charged onto his OWN docket (that would double-role him — scored
	# as the presiding minister in perf term (a) AND tallied in the docket-backlog term (c)). A
	# corrupt Justice minister is dealt with by removing him from office, not by self-indictment.
	# `this = scope:X` char-equality is the proven exclusion idiom (se_QING_COLON.txt:185).
	if = {
		limit = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_justice_holder = { save_scope_as = qing_justice_sitting_minister }
	}
	ordered_character = {
		limit = {
			employer = ROOT
			is_ruler = no
			is_adult = yes
			is_alive = yes
			NOT = { has_variable = qing_rebel_foil }
			NOT = { has_variable = qing_justice_accused }
			# not the sitting minister himself (only checked when one is seated).
			NOT = {
				AND = {
					exists = scope:qing_justice_sitting_minister
					this = scope:qing_justice_sitting_minister
				}
			}
			OR = {
				corruption >= 30
				has_trait = corrupt
				loyalty < 40
			}
		}
		order_by = corruption
		check_range_bounds = no
		max = 1
		save_scope_as = qing_justice_target
	}
	if = {
		limit = { exists = scope:qing_justice_target }
		LOG_line = { sys = QING  msg = "justice: an accusable official found for the docket for" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_justice_find_accusable"  reason = "no accusable official found for" }
	}
	LOG_exit = { sys = QING  fn = "QING_justice_find_accusable"  result = OK }
}

# CONVICT AN ACCUSED (定罪) — the trial goes against the accused. Disgrace + strip
# office + reduce corruption, and (the concrete 刑部 penalty the design calls for) condemn
# the convict's household to PENAL SERVITUDE (流刑/籍沒): a pop in the realm's capital is
# reduced to the slave class. Scope: the accused CHARACTER (ROOT); scope:player = CHI.
# Uses only proven idioms: QING_char_taint, QING_office_vacate_dispatch, and the
# set_pop_type = slaves conversion (naval_raiding.txt:174). Unmarks the docket flag.
QING_justice_convict_accused = {
	LOG_enter = { sys = QING  fn = "QING_justice_convict_accused" }
	# the code falls on the convict: tainted (corrupt trait + corruption up), disgraced,
	# and stripped of any great office he held.
	QING_char_taint = yes
	add_loyalty = loyalty_qing_disgraced
	add_popularity = -20
	if = { limit = { has_variable = qing_office_held }  QING_office_vacate_dispatch = yes }
	remove_variable = qing_justice_accused
	# PENAL SERVITUDE (籍沒): the convict's household reduced to bond-servitude — a non-slave
	# pop in the capital province is condemned to the slave class. Guarded so it no-ops when
	# the capital has no eligible pop (never errors). Concrete slaves-class hook per the design.
	scope:player = {
		capital_scope = {
			if = {
				limit = { any_pops_in_province = { NOT = { pop_type = slaves } } }
				random_pops_in_province = {
					limit = { NOT = { pop_type = slaves } }
					set_pop_type = slaves
				}
			}
		}
		# a conviction cleans a little rot out of the system.
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -5 }
	}
	LOG_line = { sys = QING  msg = "justice: an accused convicted — disgraced, stripped, household to penal servitude for" }
	LOG_exit = { sys = QING  fn = "QING_justice_convict_accused"  result = OK }
}

# ACQUIT AN ACCUSED (開釋) — the trial clears him. Struck from the docket; a wrongly-held
# man's loyalty is soothed a touch. Scope: the accused CHARACTER (ROOT); scope:player = CHI.
QING_justice_acquit_accused = {
	LOG_enter = { sys = QING  fn = "QING_justice_acquit_accused" }
	remove_variable = qing_justice_accused
	add_loyalty = loyalty_qing_commended
	LOG_line = { sys = QING  msg = "justice: an accused acquitted and struck from the docket for" }
	LOG_exit = { sys = QING  fn = "QING_justice_acquit_accused"  result = OK }
}

# Protect the official (京控 denied) — shield the magistrate, suppress the appeal.
QING_justice_deny_appeal = {
	LOG_enter = { sys = QING  fn = "QING_justice_deny_appeal" }
	if = {
		limit = { any_owned_province = { count >= 1  province_unrest < 8 } }
		random_owned_province = {
			limit = { province_unrest < 8 }
			add_province_modifier = { name = qing_justice_appeal_denied  duration = 1825 }
		}
	}
	add_tyranny = 2
	if = {
		limit = { exists = scope:corrupt_official  scope:corrupt_official = { is_alive = yes  employer = ROOT } }
		scope:corrupt_official = {
			add_loyalty = loyalty_qing_commended
		}
	}
	LOG_line = { sys = QING  msg = "justice: appeal denied — official protected (unrest up, tyranny) for" }
	LOG_exit = { sys = QING  fn = "QING_justice_deny_appeal"  result = OK }
}

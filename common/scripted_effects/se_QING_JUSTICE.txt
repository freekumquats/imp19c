# se_QING_JUSTICE.txt — the Grand Minister of Justice (刑部) content domain.
#
# BUILD #125: penal code (大清律例), autumn assizes (秋審), appeals (京控).
#
# The 刑部 (Board of Punishments) was the empire's high court and keeper of the
# penal code. A capable, just Justice minister means the law is applied evenly
# (unrest down, stability up), the appeal backlog is cleared, and the autumn
# assizes are conducted fairly. A venal or vacant office means arbitrary justice
# and festering grievances (unrest up). Events in qing_justice_events.txt bring
# this domain alive.
#
# Operates on REUSED global counters (stability, legitimacy, unrest) and on REAL
# characters where a case involves a named official. The Justice minister is
# judged on global unrest (QING_acc_metric_unrest) in se_QING_ACCOUNTABILITY.txt.
#
# Called from flavour-roll dispatcher (se_QING_DECLINE.txt). CHI-only, player-only.
# se_LOG: sys = QING throughout per the standing rule.
# Prefix = QING_justice_
########################################################################

# ======================================================================
# TOP-LEVEL JUSTICE-DOMAIN PULSE — quarterly or event-triggered. Country scope
# (ROOT = CHI). Checks Justice holder quality, applies drift when vacant or
# corrupt, and may queue justice-domain events. A capable, just, high-zeal
# Justice minister suppresses unrest via the existing office modifier
# (qing_office_justice_active: global_unrest -1, global_monthly_state_loyalty).
# A vacant or venal office lets arbitrary justice fester (unrest up).
# ======================================================================
QING_justice_pulse = {
	LOG_enter = { sys = QING  fn = "QING_justice_pulse" }
	
	# --- Judge the Justice holder: capable & honest vs vacant/venal ---
	if = {
		limit = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = {
				is_alive = yes
				employer = ROOT
				zeal >= 8
				NOT = { has_trait = corrupt }
				NOT = { corruption >= 30 }
			}
		}
		# A just, zealous Justice minister: the law is evenly applied. Add small
		# unrest suppression beyond the office's standing modifier (via stability).
		add_stability = 5
		LOG_line = { sys = QING  msg = "justice pulse: capable just holder () — law evenly applied (+stability) for" }
	}
	else_if = {
		limit = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = {
				is_alive = yes
				employer = ROOT
				OR = {
					has_trait = corrupt
					corruption >= 40
				}
			}
		}
		# A venal Justice minister: arbitrary justice breeds resentment. Unrest up
		# via QING_DECLINE_nudge on the accountability counter (reuse, don't double-count).
		if = {
			limit = { any_owned_province = { count >= 1  province_unrest < 8 } }
			random_owned_province = {
				limit = { province_unrest < 8 }
				add_province_modifier = { name = qing_justice_arbitrary  duration = 1825 }
			}
		}
		LOG_line = { sys = QING  msg = "justice pulse: venal holder () — arbitrary justice (unrest via province modifier) for" }
	}
	else = {
		# Vacant or incapable: the law drifts. Small unrest/stability penalty.
		add_stability = -3
		LOG_line = { sys = QING  msg = "justice pulse: vacant/weak holder — law drifts (-stability) for" }
	}
	
	LOG_exit = { sys = QING  fn = "QING_justice_pulse"  result = OK }
}

# Conduct a fair Autumn Assize (秋審) — capital-case review. Temper justice with
# mercy. Country scope; the Justice holder should be saved as scope:justice_minister.
QING_justice_fair_assize = {
	LOG_enter = { sys = QING  fn = "QING_justice_fair_assize" }
	add_legitimacy = 5
	add_stability = 10
	add_country_modifier = { name = qing_justice_merciful_assize  duration = 1825 }
	if = {
		limit = { exists = scope:justice_minister }
		scope:justice_minister = {
			QING_char_promote_standing = { pop = 3  prom = 2 }
		}
	}
	LOG_line = { sys = QING  msg = "justice: fair autumn assize conducted (legitimacy, stability, mercy modifier) for" }
	LOG_exit = { sys = QING  fn = "QING_justice_fair_assize"  result = OK }
}

# Vindicate an appellant (京控) — punish the corrupt local official. Operates on
# a REAL saved character scope:corrupt_official. Country scope.
QING_justice_vindicate_appeal = {
	LOG_enter = { sys = QING  fn = "QING_justice_vindicate_appeal" }
	if = {
		limit = { exists = scope:corrupt_official  scope:corrupt_official = { is_alive = yes  employer = ROOT } }
		scope:corrupt_official = {
			QING_char_taint = yes
			add_loyalty = loyalty_qing_disgraced
			add_popularity = -10
			# strip office if he holds one (proven dispatch idiom)
			if = {
				limit = { has_variable = qing_office_held }
				QING_office_vacate_dispatch = yes
			}
		}
		LOG_line = { sys = QING  msg = "justice: appellant vindicated — punished (tainted, disgraced, stripped if office-holder) for" }
	}
	# unrest down in a province
	if = {
		limit = { any_owned_province = { count >= 1  province_unrest >= 2 } }
		random_owned_province = {
			limit = { province_unrest >= 2 }
			add_province_modifier = { name = qing_justice_appeal_vindicated  duration = 1825 }
		}
	}
	add_legitimacy = 3
	if = {
		limit = { exists = scope:justice_minister }
		scope:justice_minister = {
			QING_char_promote_standing = { pop = 3  prom = 2 }
		}
	}
	LOG_exit = { sys = QING  fn = "QING_justice_vindicate_appeal"  result = OK }
}

# ======================================================================
# [#363] THE DOCKET (待審) — the Board of Punishments' concrete work: a roster of
# accused officials it HOLDS and TRIES. Distinct from the Censorate's instant impeach
# (都察院, #362 — surveillance + one-blow disgrace): the 刑部 is the high COURT, so it
# CHARGES a suspect onto its docket first (qing_justice_accused), then convicts or
# acquits him at trial. These verbs back the #363 panel; se_LOG throughout.
# ======================================================================

# FIND AN ACCUSABLE OFFICIAL — the most corrupt courtier NOT already on the docket,
# saved as scope:qing_justice_target. Mirrors QING_censorate_find_corrupt's limit but
# excludes men already charged (so ACCUSE never re-charges a docketed suspect). Country scope.
QING_justice_find_accusable = {
	LOG_enter = { sys = QING  fn = "QING_justice_find_accusable" }
	# [#14 boot] the old sitting-Justice-minister save+exclude is gone: STRIP-THEN-ACCUSE relieves any
	# office-holder (including the Justice minister) of his post on charge, so he is never double-roled.
	# NOTE this effect + its qing_justice.5 auto-pick trampoline are RETAINED-BUT-UNUSED (the panel now
	# opens the QING_justice_refresh_accusable picker); kept as reference for the shared accusable limit.
	ordered_character = {
		limit = {
			employer = ROOT
			is_ruler = no
			is_adult = yes
			is_alive = yes
			NOT = { has_variable = qing_rebel_foil }
			NOT = { has_variable = qing_justice_accused }
			# [#14 boot] STRIP-THEN-ACCUSE: office-holders, commanders, and governors are NO LONGER
			# exempt — they are RELIEVED of the shielding position on charge (QING_justice_strip_for_trial,
			# fired from qing_justice.8) and then face trial as private men. So the old blanket exclusions
			# (NOT={has_variable=qing_office_held}, is_general/is_admiral/is_governor = no) are dropped: a
			# corrupt minister/commander/governor IS chargeable now. The sitting Justice minister is not
			# specially excluded any more either — charging him strips him first, so he is never double-
			# roled (scored as presiding minister AND docket backlog) because he is relieved atomically
			# before the accused mark is set.
			# [#14 review-fix] the SITTING Grand Minister of Justice cannot be charged onto his OWN docket:
			# stripping him would vacate the Justice seat, and the convict/acquit levers both require a
			# seated minister — so a self-charge would strand him un-triable until a successor is named. A
			# corrupt Justice minister is dealt with by relieving him of office first, then charging him as a
			# private man. Character-local (qing_office_held = flag:justice, the guarded var-compare idiom
			# proven at :190) so it stays byte-identical across all four accusable sites. OTHER office-holders
			# stay chargeable (they are stripped on charge, #14) — only the presiding judge is barred.
			NOT = { AND = { has_variable = qing_office_held  var:qing_office_held = flag:justice } }
			# [#419] BUT the vanilla `imprison` effect (run at CONVICT) refuses the heir / co-ruler /
			# family & clan heads — birth/imperial-house positions that CANNOT be stripped. Keep excluding
			# them so ACCUSE never docks a man CONVICT then cannot actually jail. is_ruler is excluded above.
			is_co_ruler = no
			is_primary_heir = no
			is_head_of_family = no
			is_clan_chief = no
			# [#419] BROADENED the accusable pool: the 刑部 tried the whole span of official misconduct
			# (贓/酷/怠/叛 — graft, cruelty, dereliction, disloyalty), not only outright corruption. Was
			# only { corruption>=30 OR corrupt OR loyalty<40 }; now also catches a moderately-corrupt
			# official (>=20), the cruel/arbitrary magistrate (酷吏 — a real 刑部 charge), and the openly
			# disloyal (loyalty<50). Widens who the docket can reach without making everyone chargeable.
			OR = {
				corruption >= 20
				has_trait = corrupt
				has_trait = cruel
				has_trait = arbitrary
				loyalty < 50
			}
		}
		order_by = corruption
		check_range_bounds = no
		max = 1
		save_scope_as = qing_justice_target
	}
	if = {
		limit = { exists = scope:qing_justice_target }
		LOG_line = { sys = QING  msg = "justice: an accusable official found for the docket for" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "QING_justice_find_accusable"  reason = "no accusable official found for" }
	}
	LOG_exit = { sys = QING  fn = "QING_justice_find_accusable"  result = OK }
}

# ======================================================================
# [#6] QING_justice_refresh_accusable — build the qing_justice_accusable roster
# (the ACCUSE picker's datamodel) sorted MOST-CORRUPT-FIRST. Country scope (CHI).
#
# Replaces the old auto-pick-the-single-worst-man behaviour: the panel's ACCUSE lever
# now opens a picker window listing every chargeable official (like the Grand Council
# Appoint picker), so the player chooses WHOM to charge. The limit is byte-identical to
# QING_justice_find_accusable's ordered_character limit (every eligibility bar the docket
# and the vanilla imprison effect impose) so a picked man always reaches the docket and
# is always convictable — no dead rows. order_by = corruption, worst first.
#
# ordered_character SORTING inline is proven safe from a scripted_EFFECT called via a
# scripted_gui verb (the appoint picker's QING_council_refresh_candidates does exactly
# this and boots) — the crash class is inlining a sorting iterator into a compiled BUTTON
# effect chain, which we avoid by putting the sort here and firing this from the verb.
# ======================================================================
QING_justice_refresh_accusable = {
	LOG_enter = { sys = QING  fn = "QING_justice_refresh_accusable" }
	clear_variable_list = qing_justice_accusable
	ordered_character = {
		limit = {
			employer = ROOT
			is_ruler = no
			is_adult = yes
			is_alive = yes
			NOT = { has_variable = qing_rebel_foil }
			NOT = { has_variable = qing_justice_accused }
			# [#14 boot] STRIP-THEN-ACCUSE: office-holders/commanders/governors are chargeable now — they
			# are relieved of the shielding position on charge (QING_justice_strip_for_trial via qing_justice.8),
			# so the old NOT={has_variable=qing_office_held} + is_general/is_admiral/is_governor=no exclusions
			# are dropped. Keep only imprison's un-strippable bars (heir/co-ruler/family & clan heads); is_ruler
			# excluded above. MUST stay byte-identical to QING_justice_find_accusable + the panel gate.
			# [#14 review-fix] the sitting Justice minister cannot try himself (byte-identical to find_accusable).
			NOT = { AND = { has_variable = qing_office_held  var:qing_office_held = flag:justice } }
			is_co_ruler = no
			is_primary_heir = no
			is_head_of_family = no
			is_clan_chief = no
			# the broadened misconduct pool (贓/酷/怠/叛) — byte-identical to find_accusable.
			OR = {
				corruption >= 20
				has_trait = corrupt
				has_trait = cruel
				has_trait = arbitrary
				loyalty < 50
			}
		}
		order_by = corruption
		check_range_bounds = no
		max = 12   # show the dozen most venal — mirrors QING_COUNCIL_CANDIDATE_CAP
		# the list is COUNTRY-held: add through ROOT with the courtier as prev (appoint idiom).
		ROOT = { add_to_variable_list = { name = qing_justice_accusable  target = prev } }
	}
	LOG_line = { sys = QING  msg = "justice: accusable-official picker roster refreshed for" }
	LOG_exit = { sys = QING  fn = "QING_justice_refresh_accusable"  result = OK }
}

# ======================================================================
# [#14 boot] STRIP THE SHIELDING POSITION FOR TRIAL (奪職候審) — a charged official is RELIEVED
# of any great office / command / governorship the moment he is docketed, so that (a) he faces
# trial as a private man the way the historic 刑部 required a suspect be stripped of rank before
# indictment, and (b) the eventual CONVICT can actually jail him (the vanilla `imprison` engine
# effect refuses serving commanders/governors — QING_justice_convict_accused's #419 note). Birth /
# imperial-house positions (heir, co-ruler, family & clan heads) are NOT strippable and are excluded
# upstream by the accusable limit, so this only ever relieves an APPOINTED post. Scope: the accused
# CHARACTER (ROOT). Uses only proven char-scope idioms: QING_office_vacate_dispatch (censorate
# impeach-uphold, se_QING_CENSORATE.txt:143), remove_command (annexation.txt:546), remove_as_governor
# (office_eligibility_events.txt:186). No sorting iterator / imprison / create_character here.
# ======================================================================
QING_justice_strip_for_trial = {
	LOG_enter = { sys = QING  fn = "QING_justice_strip_for_trial" }
	# relieve of a Grand Council great office (dispatches to the correct QING_office_vacate branch).
	if = { limit = { has_variable = qing_office_held }  QING_office_vacate_dispatch = yes }
	# relieve of army/navy command so the convict is jailable (imprison refuses serving commanders).
	if = { limit = { OR = { is_general = yes  is_admiral = yes } }  remove_command = yes }
	# relieve of any governorship (imprison likewise refuses serving governors).
	if = { limit = { is_governor = yes }  remove_as_governor = yes }
	LOG_line = { sys = QING  msg = "justice: an accused stripped of office/command/governorship for trial for" }
	LOG_exit = { sys = QING  fn = "QING_justice_strip_for_trial"  result = OK }
}


# CONVICT AN ACCUSED (定罪) — the trial goes against the accused. Disgrace + strip office + reduce
# corruption, and (the concrete 刑部 penalty) IMPRISON the convict via the ENGINE's own prison mechanic
# (下獄). Scope: the accused CHARACTER (ROOT); scope:player = CHI. Uses only proven idioms: QING_char_taint,
# QING_office_vacate_dispatch, and the vanilla `imprison = { target = <char> }` run at country scope (TI
# imprison.txt:84 — scope:actor = { imprison = { target = scope:target } }). [#419] REPLACES the former
# set_pop_type = slaves household-enslavement hook (the design pivoted off the slave-class mechanic): a
# convicted official is now JAILED as a real prisoner of the throne, not modelled by enslaving a capital pop.
QING_justice_convict_accused = {
	LOG_enter = { sys = QING  fn = "QING_justice_convict_accused" }
	# the code falls on the convict: tainted (corrupt trait + corruption up), disgraced,
	# and stripped of any great office he held.
	QING_char_taint = yes
	add_loyalty = loyalty_qing_disgraced
	add_popularity = -20
	# [#14 review-fix] re-strip office/command/governorship at conviction too, not just at charge: a
	# docketed man RE-APPOINTED to command or a governorship between charge and conviction would otherwise
	# be refused by the vanilla `imprison` below (it bars serving commanders/governors), leaving him
	# disgraced-but-unjailed. Reuse the same strip helper (idempotent — its ifs no-op if he holds nothing).
	QING_justice_strip_for_trial = yes
	remove_variable = qing_justice_accused
	# [#419] IMPRISONMENT (下獄): the convict is jailed by the throne through the engine's OWN prison
	# mechanic. `imprison` runs at COUNTRY scope with target = the character (TI imprison.txt:84); guarded
	# on `prisoner = no` (the proven already-imprisoned test, imprison.txt:16/25) so it never errors on a
	# re-conviction. ROOT = the convict, scope:player = CHI.
	if = {
		limit = { prisoner = no }
		save_scope_as = qing_justice_convict
		scope:player = { imprison = { target = scope:qing_justice_convict } }
	}
	scope:player = {
		# a conviction cleans a little rot out of the system.
		QING_DECLINE_nudge = { var = qing_corruption_level  amount = -5 }
	}
	LOG_line = { sys = QING  msg = "justice: an accused convicted — disgraced, stripped of office, and imprisoned for" }
	LOG_exit = { sys = QING  fn = "QING_justice_convict_accused"  result = OK }
}

# ACQUIT AN ACCUSED (開釋) — the trial clears him. Struck from the docket; a wrongly-held
# man's loyalty is soothed a touch. Scope: the accused CHARACTER (ROOT); scope:player = CHI.
QING_justice_acquit_accused = {
	LOG_enter = { sys = QING  fn = "QING_justice_acquit_accused" }
	remove_variable = qing_justice_accused
	add_loyalty = loyalty_qing_commended
	LOG_line = { sys = QING  msg = "justice: an accused acquitted and struck from the docket for" }
	LOG_exit = { sys = QING  fn = "QING_justice_acquit_accused"  result = OK }
}

# Protect the official (京控 denied) — shield the magistrate, suppress the appeal.
QING_justice_deny_appeal = {
	LOG_enter = { sys = QING  fn = "QING_justice_deny_appeal" }
	if = {
		limit = { any_owned_province = { count >= 1  province_unrest < 8 } }
		random_owned_province = {
			limit = { province_unrest < 8 }
			add_province_modifier = { name = qing_justice_appeal_denied  duration = 1825 }
		}
	}
	add_tyranny = 2
	if = {
		limit = { exists = scope:corrupt_official  scope:corrupt_official = { is_alive = yes  employer = ROOT } }
		scope:corrupt_official = {
			add_loyalty = loyalty_qing_commended
		}
	}
	LOG_line = { sys = QING  msg = "justice: appeal denied — official protected (unrest up, tyranny) for" }
	LOG_exit = { sys = QING  fn = "QING_justice_deny_appeal"  result = OK }
}

# se_QING_SPHERE.txt — [#165] Four-power dynamic sphere-of-influence.
#
# The 19th-century Great Game, modelled as a running contest for the states of
# the Qing's contested MARCHES. FOUR powers push influence into each frontier
# state: China (CHI) itself, Britain (GBR), France (FRA), Russia (RUS). Each
# contested state carries four influence scores (0..100) and a dominant holder;
# the balance shifts as each power pushes and every score decays, with the
# historical entrenchment of the European powers ramping their push up over the
# century.
#
# DESIGN — PUSH sources + PULL diffusion. [#307] The PULL model (a state reading a
# NEIGHBOURING state's numeric influence var mid-iteration) is now PROVEN: the
# corrected SPHERE probe (develop c38b3f38) returned "PULL VERDICT = SUPPORTED" from
# a genuine script trigger (var:sphere_pull_sum > 0) — a state CAN read a neighbour's
# numeric var both as a trigger (var:X >= n) and an effect (change_variable add =
# scope:nbr.var:X). See memory imp19c-sphere-idioms-oracle (RESOLVED). This SUPERSEDES
# the earlier "PUSH-only, PULL is UNPROVEN" fallback note.
#
# The contest now has TWO terms:
#   • PUSH (source): each power injects influence into states it can reach by PROVEN
#     property reads — China on its/subjects' soil + the immediate frontier, each
#     European power in its theatre (the QING_gp_sphere_is_* classifiers), ramped over
#     the century. These are the SOURCES that originate influence.
#   • PULL (diffusion): influence then BLEEDS between adjacent contested marches — a
#     state pulls each power's influence toward its neighbours' (a power dominant in one
#     march radiates into the next), via the proven cross-state numeric read. To keep
#     the bleed order-independent, each ring state's four scores are SNAPSHOT (into
#     *_influence_snap) before any tick, and the bleed reads neighbours' snapshots (not
#     the live values being mutated this pass).
# All scores decay yearly; the dominant holder is a plain 4-way max over four FIXED-name
# vars — no dynamic ordered_in_list deref.
#
# The dominant holder is stored as a NUMERIC CODE (proven numeric equality;
# flag-var-to-flag-var equality has no precedent in this codebase and is avoided):
#   qing_sphere_dominant / _dominant_prev : 0 neutral | 1 china | 2 britain |
#                                           3 france  | 4 russia.
#
# Storage
#   STATE   : china_influence / britain_influence / france_influence /
#             russia_influence (0..100); qing_sphere_dominant (+_prev, codes
#             above); marker qing_sphere_contested.
#   COUNTRY : list qing_sphere_states (the contested ring, rebuilt each pulse);
#             counters qing_sphere_china_count / _britain_count / _france_count /
#             _russia_count / _neutral_count (for GUI + GP tie-in); throttle
#             qing_sphere_cooldown (yearly).
#
# GP tie-in — when a rival flips a state away from China, that rival's existing
# qing_gp_tension_<power> counter is nudged up (a provocation); when China wins
# a state back from a rival, that rival's tension is eased. Reuses the proven
# QING_DECLINE_nudge + qing_gp_tension_* counters (se_QING_DIPLO.txt).
#
# Seeded once by QING_sphere_init (on_game_initialized, CHI), evolved quarterly
# by QING_sphere_pulse (qing_mechanics_pulse_on_action; self-throttled yearly).
# CHI player-only. All logged sys = QING with STATIC msgs (#253).
########################################################################

# ======================================================================
# QING_sphere_init — seed the contested ring and its starting balance ONCE.
# Scope: COUNTRY (CHI). Idempotent (guards on qing_sphere_initialized).
# ======================================================================
QING_sphere_init = {
	LOG_enter = { sys = QING  fn = "QING_sphere_init" }
	if = {
		limit = { NOT = { has_variable = qing_sphere_initialized } }
		set_variable = { name = qing_sphere_initialized  value = 1 }
		QING_sphere_reset_counters = yes
		# Build the contested ring, then give each state a historically-flavoured
		# opening balance (the frontier is not blank at 1815 — Britain already
		# sits on India, Russia on Siberia, China on its own tributary marches).
		QING_sphere_build_ring = yes
		every_in_list = {
			variable = qing_sphere_states	# [#165 review-fix] variable-list built with add_to_variable_list MUST be iterated with variable=, not list= (script-list namespace) — the list= form iterated an empty phantom list and made the whole feature a silent no-op
			QING_sphere_seed_state = yes
		}
	}
	LOG_exit = { sys = QING  fn = "QING_sphere_init"  result = OK }
}

QING_sphere_reset_counters = {
	# Scope: COUNTRY. Zero the per-power contested-state tallies.
	set_variable = { name = qing_sphere_china_count    value = 0 }
	set_variable = { name = qing_sphere_britain_count  value = 0 }
	set_variable = { name = qing_sphere_france_count   value = 0 }
	set_variable = { name = qing_sphere_russia_count   value = 0 }
	set_variable = { name = qing_sphere_neutral_count  value = 0 }
}

# ======================================================================
# QING_sphere_build_ring — (re)build the list of contested frontier states.
# Scope: COUNTRY (CHI). Cleared and rebuilt each call so the ring tracks the
# live border as conquest/loss moves it. State influence vars live ON the state,
# so rebuilding the LIST never discards a state's accumulated balance.
#
# The ring = every land state just OUTSIDE the empire (a non-CHI-owned land
# neighbour of a CHI province) PLUS every state held by a Qing subject (the
# tributary marches are China's own contested sphere). Bounded by border size.
# ======================================================================
QING_sphere_build_ring = {
	save_scope_as = sphere_owner
	clear_variable_list = qing_sphere_states
	# Frontier ring: land neighbours we do not own ourselves.
	every_owned_province = {
		limit = { is_sea = no }
		every_neighbor_province = {
			limit = {
				is_sea = no
				exists = owner
				NOT = { owner = scope:sphere_owner }
				# [logfix] total_population is a PROVINCE trigger, invalid read directly in STATE
				# scope (error.log: "total_population trigger [ Wrong scope: state, expected
				# province ]"). Reach it via any_state_province (proven, invest_in_state_buttons.txt:149).
				state = { any_state_province = { total_population > 0 } }
			}
			state = {
				if = {
					limit = {
						NOT = {
							scope:sphere_owner = {
								is_target_in_variable_list = {
									name = qing_sphere_states
									target = prev
								}
							}
						}
					}
					save_scope_as = ring_state
					scope:sphere_owner = {
						add_to_variable_list = {
							name = qing_sphere_states
							target = scope:ring_state
						}
					}
				}
			}
		}
	}
	# Tributary marches: every state a Qing subject holds is contested China-side.
	# [#280 review-fix] every_subject is NON-recursive (DIRECT subjects only), so the nested
	# chains the 1815 baseline ships (CHI->ILI->XNG, CHI->TIB->LTG/BTG, ILI->KML) left the
	# sub-subjects' states OUT of the contested ring. Add a second every_subject level so a
	# subject's OWN subjects' states seed too. QING_sphere_add_ring_state dedups, so a state
	# reachable by two paths is added once. Two levels cover every chain that ships.
	# [logfix] total_population is a PROVINCE trigger — invalid in the STATE scope of
	# every_country_state. Reach it via any_state_province (proven, invest_in_state_buttons.txt:149).
	every_subject = {
		every_country_state = {
			limit = { any_state_province = { total_population > 0 } }
			QING_sphere_add_ring_state = yes
		}
		every_subject = {
			every_country_state = {
				limit = { any_state_province = { total_population > 0 } }
				QING_sphere_add_ring_state = yes
			}
		}
	}
	LOG_line = { sys = QING  msg = "sphere: contested ring rebuilt" }
}

# ======================================================================
# QING_sphere_add_ring_state — [#280] dedup add helper: add THIS state (prev)
# to the contested ring unless the owner already listed it. Shared by the direct-
# subject and sub-subject walks so a state on two subject paths is never double-listed.
# Scope: STATE.
# ======================================================================
QING_sphere_add_ring_state = {
	if = {
		limit = {
			NOT = {
				scope:sphere_owner = {
					is_target_in_variable_list = {
						name = qing_sphere_states
						target = prev
					}
				}
			}
		}
		save_scope_as = ring_add_state
		scope:sphere_owner = {
			add_to_variable_list = {
				name = qing_sphere_states
				target = scope:ring_add_state
			}
		}
	}
}

# ======================================================================
# QING_sphere_ensure_vars — init the four influence scores on a state if unset.
# Scope: STATE.
# ======================================================================
QING_sphere_ensure_vars = {
	if = { limit = { NOT = { has_variable = china_influence } }   set_variable = { name = china_influence    value = 0 } }
	if = { limit = { NOT = { has_variable = britain_influence } } set_variable = { name = britain_influence  value = 0 } }
	if = { limit = { NOT = { has_variable = france_influence } }  set_variable = { name = france_influence   value = 0 } }
	if = { limit = { NOT = { has_variable = russia_influence } }  set_variable = { name = russia_influence   value = 0 } }
	# [#307 PULL] per-power SNAPSHOT of this state's scores, stamped once per pulse before any
	# tick so the neighbour-bleed reads a stable value (not one being mutated mid-iteration).
	if = { limit = { NOT = { has_variable = china_influence_snap } }   set_variable = { name = china_influence_snap    value = 0 } }
	if = { limit = { NOT = { has_variable = britain_influence_snap } } set_variable = { name = britain_influence_snap  value = 0 } }
	if = { limit = { NOT = { has_variable = france_influence_snap } }  set_variable = { name = france_influence_snap   value = 0 } }
	if = { limit = { NOT = { has_variable = russia_influence_snap } }  set_variable = { name = russia_influence_snap   value = 0 } }
	if = { limit = { NOT = { has_variable = qing_sphere_dominant } }      set_variable = { name = qing_sphere_dominant       value = 0 } }
	if = { limit = { NOT = { has_variable = qing_sphere_dominant_prev } } set_variable = { name = qing_sphere_dominant_prev  value = 0 } }
	set_variable = { name = qing_sphere_contested  value = 1 }
}

QING_sphere_clamp_state = {
	# Scope: STATE. Keep every score in 0..100.
	if = { limit = { var:china_influence   > 100 } set_variable = { name = china_influence    value = 100 } }
	if = { limit = { var:china_influence   < 0 }   set_variable = { name = china_influence    value = 0 } }
	if = { limit = { var:britain_influence > 100 } set_variable = { name = britain_influence  value = 100 } }
	if = { limit = { var:britain_influence < 0 }   set_variable = { name = britain_influence  value = 0 } }
	if = { limit = { var:france_influence  > 100 } set_variable = { name = france_influence   value = 100 } }
	if = { limit = { var:france_influence  < 0 }   set_variable = { name = france_influence   value = 0 } }
	if = { limit = { var:russia_influence  > 100 } set_variable = { name = russia_influence   value = 100 } }
	if = { limit = { var:russia_influence  < 0 }   set_variable = { name = russia_influence   value = 0 } }
}

# ======================================================================
# QING_sphere_seed_state — opening balance for a contested state at 1815.
# Scope: STATE. Classifies the state by the proven province-scope sphere
# classifiers and gives the owning power a head start; China leads on its own
# and its subjects' soil, and holds a lesser presence on the immediate frontier.
# ======================================================================
QING_sphere_seed_state = {
	QING_sphere_ensure_vars = yes
	# --- China's own baseline: strong on its/subjects' soil, modest on the ring.
	if = {
		limit = {
			OR = {
				owner = scope:sphere_owner
				owner = { is_subject_of = scope:sphere_owner }
				# [#280 review-fix] nested subject soil is ALSO home: a sub-subject (XNG under ILI under
				# CHI, KML under ILI) is not a DIRECT is_subject_of the Qing, so without this its home
				# marches were mis-scored as frontier (capped 55, flippable). Walk one overlord link up:
				# if the owner's overlord is our subject, the land is inside the Qing subject tree.
				# overlord{is_subject_of} is a proven idiom (000_GOVERNMENT_custom_loc.txt).
				owner = { overlord = { is_subject_of = scope:sphere_owner } }
			}
		}
		set_variable = { name = china_influence  value = 60 }
	}
	else = {
		set_variable = { name = china_influence  value = 25 }
	}
	# --- European baselines by theatre (spheres are non-overlapping by invariant).
	if = {
		limit = { any_state_province = { QING_gp_sphere_is_britain = yes } }
		set_variable = { name = britain_influence  value = 45 }
	}
	else_if = {
		limit = { any_state_province = { QING_gp_sphere_is_france = yes } }
		set_variable = { name = france_influence  value = 40 }
	}
	else_if = {
		limit = { any_state_province = { QING_gp_sphere_is_russia = yes } }
		set_variable = { name = russia_influence  value = 45 }
	}
	QING_sphere_clamp_state = yes
	QING_sphere_recompute_dominant = yes
	# [#280 review-fix] The former "prime qing_sphere_dominant_prev" line here was DEAD:
	# QING_sphere_tick_state unconditionally re-sets qing_sphere_dominant_prev at its top
	# (before the only reader, the flip check) every pulse, so the seed-time prime never
	# survived to be read. Removed. qing_sphere_dominant_prev is still initialised in
	# QING_sphere_ensure_vars (guards the very first tick's flip compare).
}

# ======================================================================
# QING_sphere_recompute_dominant — 4-way max over the four FIXED-name scores.
# Scope: STATE. Writes the NUMERIC holder code (1 china | 2 britain | 3 france
# | 4 russia | 0 neutral). Ties break China > Britain > France > Russia
# (home-field first, so only a STRICTLY greater rival displaces the incumbent).
# A state whose top score is below the contest floor (20) reads as neutral.
# ======================================================================
QING_sphere_recompute_dominant = {
	set_local_variable = { name = top_val  value = var:china_influence }
	set_variable = { name = qing_sphere_dominant  value = 1 }
	if = {
		limit = { var:britain_influence > local_var:top_val }
		set_local_variable = { name = top_val  value = var:britain_influence }
		set_variable = { name = qing_sphere_dominant  value = 2 }
	}
	if = {
		limit = { var:france_influence > local_var:top_val }
		set_local_variable = { name = top_val  value = var:france_influence }
		set_variable = { name = qing_sphere_dominant  value = 3 }
	}
	if = {
		limit = { var:russia_influence > local_var:top_val }
		set_local_variable = { name = top_val  value = var:russia_influence }
		set_variable = { name = qing_sphere_dominant  value = 4 }
	}
	# Contest floor: nobody holds a march until someone crosses 20.
	if = {
		limit = { local_var:top_val < 20 }
		set_variable = { name = qing_sphere_dominant  value = 0 }
	}
}

# ======================================================================
# QING_sphere_snapshot_state — [#307 PULL] freeze this state's four live scores into
# their *_influence_snap vars. Called on EVERY ring state before ANY tick, so the
# neighbour-bleed (which reads snapshots) is order-independent — a state bled early in
# the pass and one bled late both read the SAME pre-pass neighbour values. Scope: STATE.
# ======================================================================
QING_sphere_snapshot_state = {
	QING_sphere_ensure_vars = yes
	set_variable = { name = china_influence_snap    value = var:china_influence }
	set_variable = { name = britain_influence_snap  value = var:britain_influence }
	set_variable = { name = france_influence_snap   value = var:france_influence }
	set_variable = { name = russia_influence_snap   value = var:russia_influence }
}

# ======================================================================
# QING_sphere_bleed_state — [#307 PULL] neighbour diffusion. Each power's influence
# on THIS state is pulled a little toward the HIGHEST value that power holds on an
# adjacent CONTESTED march (a power dominant next door radiates in). Reads neighbours'
# *_influence_snap (the frozen pre-pass value — the proven cross-state numeric read:
# var:X as a trigger + scope:nbr.var:X in an effect), so the bleed is order-independent
# and cannot feed back on itself within a pass. Scope: STATE. Runs AFTER snapshot, BEFORE
# the source push/decay tick, so the tick's clamp/cap/recompute see the bled values.
#
# For each power: find the max neighbour snapshot (only among contested marches — those
# carry qing_sphere_contested); if it EXCEEDS this state's own snapshot, pull a fraction
# (1/4 of the gap, min 0) toward it. A power weak everywhere nearby bleeds nothing.
# ======================================================================
QING_sphere_bleed_state = {
	save_scope_as = bleed_src_state
	QING_sphere_bleed_one_power = { power = china }
	QING_sphere_bleed_one_power = { power = britain }
	QING_sphere_bleed_one_power = { power = france }
	QING_sphere_bleed_one_power = { power = russia }
}

# One power's neighbour-bleed. Scope: STATE (= scope:bleed_src_state, set by the caller).
# $power$ = china|britain|france|russia. Accumulates the max neighbour snapshot on a
# persistent temp var on the SOURCE state (so the nested province/neighbour loop writes
# back to a stable saved scope, not a transient one), then pulls 1/4 the positive gap.
QING_sphere_bleed_one_power = {
	# Running max of neighbour snapshots, on a PERSISTENT temp var on the source state
	# (persistent-var arithmetic is the proven idiom; local_var arithmetic with var:
	# operands is not). Cleared at the end of this call so no state carries it between pulses.
	set_variable = { name = sphere_bleed_nbrmax  value = 0 }
	# Walk the source state's provinces -> their land neighbours -> the neighbour's state,
	# and raise sphere_bleed_nbrmax to the strongest contested-neighbour snapshot for this power.
	every_state_province = {
		every_neighbor_province = {
			limit = { is_sea = no  exists = state }
			state = {
				if = {
					limit = {
						NOT = { this = scope:bleed_src_state }
						has_variable = qing_sphere_contested
						var:$power$_influence_snap > scope:bleed_src_state.var:sphere_bleed_nbrmax
					}
					# raise the running max on the SOURCE state (proven cross-state numeric read).
					save_scope_as = bleed_nbr_state
					scope:bleed_src_state = {
						set_variable = { name = sphere_bleed_nbrmax  value = scope:bleed_nbr_state.var:$power$_influence_snap }
					}
				}
			}
		}
	}
	# Pull toward the strongest neighbour if it out-influences our own snapshot: close 1/4 the gap.
	if = {
		limit = { var:sphere_bleed_nbrmax > var:$power$_influence_snap }
		# gap = nbrmax - own snapshot; bleed = gap / 4 (persistent-var arithmetic, proven).
		set_variable = { name = sphere_bleed_gap  value = var:sphere_bleed_nbrmax }
		change_variable = { name = sphere_bleed_gap  subtract = var:$power$_influence_snap }
		change_variable = { name = sphere_bleed_gap  divide = 4 }
		change_variable = { name = $power$_influence  add = var:sphere_bleed_gap }
		remove_variable = sphere_bleed_gap
	}
	remove_variable = sphere_bleed_nbrmax
}

# ======================================================================
# QING_sphere_pulse — quarterly driver (self-throttled to ~yearly). Rebuilds the
# ring, evolves every contested state, tallies the balance, and fires GP tension
# on any state that changed hands. Scope: COUNTRY (CHI).
# ======================================================================
QING_sphere_pulse = {
	if = {
		limit = { NOT = { has_variable = qing_sphere_cooldown } }
		LOG_enter = { sys = QING  fn = "QING_sphere_pulse" }
		set_variable = { name = qing_sphere_cooldown  days = 365 }
		save_scope_as = sphere_owner
		QING_sphere_reset_counters = yes
		QING_sphere_build_ring = yes
		# [#307 PULL] Pass 1: SNAPSHOT every ring state's scores BEFORE any tick, so the
		# neighbour-bleed in pass 2 reads frozen pre-pass values (order-independent).
		every_in_list = {
			variable = qing_sphere_states
			QING_sphere_snapshot_state = yes
		}
		# [#307 PULL] Pass 2: bleed (neighbour diffusion, reads snapshots) then tick (source
		# push/decay/clamp/cap/recompute) each state, and tally the freshly-recomputed holder.
		every_in_list = {
			variable = qing_sphere_states	# [#165 review-fix] see QING_sphere_init: iterate the variable-list with variable=, not list=
			QING_sphere_bleed_state = yes
			QING_sphere_tick_state = yes
			# Tally the (freshly recomputed) dominant holder into CHI's counters.
			if = { limit = { var:qing_sphere_dominant = 1 } scope:sphere_owner = { change_variable = { name = qing_sphere_china_count    add = 1 } } }
			else_if = { limit = { var:qing_sphere_dominant = 2 } scope:sphere_owner = { change_variable = { name = qing_sphere_britain_count  add = 1 } } }
			else_if = { limit = { var:qing_sphere_dominant = 3 } scope:sphere_owner = { change_variable = { name = qing_sphere_france_count   add = 1 } } }
			else_if = { limit = { var:qing_sphere_dominant = 4 } scope:sphere_owner = { change_variable = { name = qing_sphere_russia_count   add = 1 } } }
			else = { scope:sphere_owner = { change_variable = { name = qing_sphere_neutral_count  add = 1 } } }
		}
		LOG_exit = { sys = QING  fn = "QING_sphere_pulse"  result = OK }
	}
}

# ======================================================================
# QING_sphere_tick_state — one state's yearly evolution. Scope: STATE.
#   1. decay every score (influence must be maintained);
#   2. China pushes where it can reach (own/subject soil, or a CHI-owned
#      neighbour province);
#   3. the theatre's European power pushes, scaled by historical entrenchment
#      (weak before its soft date, full after its hard date);
#   4. recompute the dominant holder and fire GP tension on any change of hands.
# ======================================================================
QING_sphere_tick_state = {
	QING_sphere_ensure_vars = yes
	# Remember who held it before this tick (for flip detection).
	set_variable = { name = qing_sphere_dominant_prev  value = var:qing_sphere_dominant }

	# 1. Yearly decay toward zero.
	change_variable = { name = china_influence    subtract = 3 }
	change_variable = { name = britain_influence  subtract = 3 }
	change_variable = { name = france_influence   subtract = 3 }
	change_variable = { name = russia_influence   subtract = 3 }

	# 2. China's push: strong on its/subjects' soil, lighter across the border.
	#    is_home marks a march that is China's own / a subject's soil — there China
	#    pushes hard (+11, net +8 after decay) and holds without limit. On a mere
	#    FRONTIER march (a foreign neighbour China only reaches across the border) it
	#    pushes lighter AND is ceiling-capped below a fully-entrenched rival (see the
	#    frontier-cap step after the clamp) so the European ramp can flip its sphere —
	#    [#275 sphere-review-fix] without the cap China's steady +7 saturated EVERY
	#    frontier march at 100 and the China-first tie-break held them forever,
	#    collapsing the four-power contest to permanent China dominance.
	set_local_variable = { name = is_home  value = 0 }
	if = {
		limit = {
			OR = {
				owner = scope:sphere_owner
				owner = { is_subject_of = scope:sphere_owner }
				# [#280 review-fix] nested subject soil is ALSO home: a sub-subject (XNG under ILI under
				# CHI, KML under ILI) is not a DIRECT is_subject_of the Qing, so without this its home
				# marches were mis-scored as frontier (capped 55, flippable). Walk one overlord link up:
				# if the owner's overlord is our subject, the land is inside the Qing subject tree.
				# overlord{is_subject_of} is a proven idiom (000_GOVERNMENT_custom_loc.txt).
				owner = { overlord = { is_subject_of = scope:sphere_owner } }
			}
		}
		set_local_variable = { name = is_home  value = 1 }
		change_variable = { name = china_influence  add = 11 }
	}
	else_if = {
		limit = { any_state_province = { any_neighbor_province = { owner = scope:sphere_owner } } }
		change_variable = { name = china_influence  add = 7 }
	}

	# 3. The theatre power's push, ramped over the century (the Great Game
	#    intensifies: nominal reach early, full entrenchment late).
	if = {
		limit = { any_state_province = { QING_gp_sphere_is_britain = yes } }
		QING_sphere_gp_push = { power = britain  soft = 1826.1.1  hard = 1852.1.1 }
	}
	else_if = {
		limit = { any_state_province = { QING_gp_sphere_is_france = yes } }
		QING_sphere_gp_push = { power = france  soft = 1858.1.1  hard = 1885.1.1 }
	}
	else_if = {
		limit = { any_state_province = { QING_gp_sphere_is_russia = yes } }
		QING_sphere_gp_push = { power = russia  soft = 1858.1.1  hard = 1885.1.1 }
	}

	QING_sphere_clamp_state = yes
	# [#275 sphere-review-fix] FRONTIER CAP: on a non-home march China's influence is
	# held at 55 — enough to lead early (rivals seed 40-45 and start at nominal push),
	# but below the 100 a fully-entrenched European power climbs to after its hard date,
	# so the historical ramp genuinely flips the march away. Home/subject marches are
	# uncapped (China's own sphere holds). Must run AFTER the 0..100 clamp.
	if = {
		limit = { local_var:is_home = 0  var:china_influence > 55 }
		set_variable = { name = china_influence  value = 55 }
	}
	QING_sphere_recompute_dominant = yes

	# 4. Fire GP tension on a change of hands (only China<->rival transitions).
	if = {
		limit = { NOT = { var:qing_sphere_dominant = var:qing_sphere_dominant_prev } }
		QING_sphere_flip_reaction = yes
	}
}

# ======================================================================
# QING_sphere_gp_push — add the theatre power's influence, scaled by date.
# Scope: STATE. $power$ = britain|france|russia; $soft$/$hard$ = the historical
# window over which that power's grip firmed up (mirrors QING_gp_grip_grace).
# ======================================================================
QING_sphere_gp_push = {
	if = {
		limit = { current_date >= $hard$ }
		change_variable = { name = $power$_influence  add = 12 }
	}
	else_if = {
		limit = { current_date >= $soft$ }
		change_variable = { name = $power$_influence  add = 7 }
	}
	else = {
		change_variable = { name = $power$_influence  add = 3 }
	}
}

# ======================================================================
# QING_sphere_flip_reaction — a state changed hands this tick. Scope: STATE.
# China losing a march to a rival provokes that rival's tension; China winning
# one back eases it. China<->neutral and rival<->rival shifts are ignored (no
# direct Sino-foreign provocation). Uses scope:sphere_owner (set in the pulse).
# Codes: 1 china | 2 britain | 3 france | 4 russia.
# ======================================================================
QING_sphere_flip_reaction = {
	# China LOST this state to a European rival -> that rival's tension rises.
	if = {
		limit = { var:qing_sphere_dominant_prev = 1 }
		if = { limit = { var:qing_sphere_dominant = 2 } scope:sphere_owner = { QING_DECLINE_nudge = { var = qing_gp_tension_britain  amount = 3 } } }
		else_if = { limit = { var:qing_sphere_dominant = 3 } scope:sphere_owner = { QING_DECLINE_nudge = { var = qing_gp_tension_france   amount = 3 } } }
		else_if = { limit = { var:qing_sphere_dominant = 4 } scope:sphere_owner = { QING_DECLINE_nudge = { var = qing_gp_tension_russia  amount = 3 } } }
		LOG_line = { sys = QING  msg = "sphere: China lost a march to a rival" }
	}
	# China WON this state back from a European rival -> that rival's tension eases.
	else_if = {
		limit = { var:qing_sphere_dominant = 1 }
		if = { limit = { var:qing_sphere_dominant_prev = 2 } scope:sphere_owner = { QING_DECLINE_nudge = { var = qing_gp_tension_britain  amount = -2 } } }
		else_if = { limit = { var:qing_sphere_dominant_prev = 3 } scope:sphere_owner = { QING_DECLINE_nudge = { var = qing_gp_tension_france   amount = -2 } } }
		else_if = { limit = { var:qing_sphere_dominant_prev = 4 } scope:sphere_owner = { QING_DECLINE_nudge = { var = qing_gp_tension_russia  amount = -2 } } }
		LOG_line = { sys = QING  msg = "sphere: China won a march back from a rival" }
	}
}

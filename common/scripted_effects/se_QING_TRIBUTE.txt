########################################################################
# se_QING_TRIBUTE.txt — [#322 / overnight-build-#2] THE TRIBUTE-EMBASSY RHYTHM (朝貢體系).
#
# The mod already models the tributary order two ways: the abstract suzerain-prestige
# meter (qing_suzerain_prestige, se_QING_VASSAL.txt) and the DEFENSIVE crises (defend/
# abandon a vassal under foreign pressure). What was MISSING — and what this module
# adds, per the concrete-over-abstract rule — is the POSITIVE ongoing RHYTHM of the
# tributary system at its zenith: the periodic tribute EMBASSIES the vassals sent to
# Peking (the Korean 燕行使, the Vietnamese and Ryukyu missions), each a concrete ENVOY
# CHARACTER who arrives, presents tribute (a trade profit to the throne), and is
# received — raising or, if snubbed, straining the suzerain's prestige. This turns the
# tributary order from a defensive-only meter into a living court ceremony.
#
# CONCRETE OBJECTS (not a counter):
#   • A real ENVOY create_character per mission (custom identity via add_nickname, the
#     roster idiom), scoped into the arrival event so the player receives a PERSON.
#   • The REAL subject tags at the 1763/1815 start (KOR / VIE / NEP / RYU / TNN — the
#     dependency=tributary rows in setup/main/00_default.txt) drive it: a mission only
#     comes from a tag that STILL exists AND is STILL our subject, so a lost tributary
#     stops sending (concrete tie to the actual subject graph, not a hardcoded list).
#
# SCHEDULER (oracle-probe fallback baked in): rather than N self-requeuing dated
# events (event-queue bloat over a long game), ONE shared check runs on the quarterly
# mechanics pulse (QING_tribute_schedule_missions), iterating our tributary subjects
# and firing an arrival for any whose per-vassal cooldown has lapsed. O(subjects),
# cheap, no queue growth. Cadence differs by vassal (Korea most frequent, historically).
#
# WIRED:
#   • QING_tribute_init — seed (idempotent), at game-start init.
#   • QING_tribute_schedule_missions — quarterly, from qing_mechanics_pulse_on_action.
#   • qing_tribute.1 — a tribute embassy arrives (the arrival event; envoy + choices).
#   • QING_tribute_prestige_from_missions — fold recent-mission tempo into the
#     suzerain-prestige drift (a vigorous embassy calendar sustains the order).
#
# CHI-only, player-facing (the pulse that drives it is CHI + is_ai=no gated). All
# logged sys = QING, STATIC msgs only (#253). no-BOM/LF. Prefix = QING_tribute_.
#
#   ROOT / this = the Qing country (CHI, player); root.current_ruler = the emperor.
#
# Sources (feasibility only): John K. Fairbank (ed.), The Chinese World Order (1968);
#   Takeshi Hamashita, China, East Asia and the Global Economy (2008); Kirk Larsen,
#   Tradition, Treaties, and Trade (2008). Engine idioms cited inline.
########################################################################

# ======================================================================
# QING_tribute_init — seed the embassy bookkeeping once. Scope: COUNTRY. Idempotent.
# Per-vassal last-mission cooldowns are stamped lazily on first schedule, so the only
# seed here is the missions-this-decade tempo counter the prestige drift reads.
# ======================================================================
QING_tribute_init = {
	LOG_enter = { sys = QING  fn = "QING_tribute_init" }
	if = {
		limit = { NOT = { has_variable = qing_tribute_initialized } }
		set_variable = { name = qing_tribute_initialized  value = 1 }
		set_variable = { name = qing_tribute_missions_recent  value = 0 }
	}
	LOG_exit = { sys = QING  fn = "QING_tribute_init"  result = OK }
}

# ======================================================================
# QING_tribute_schedule_missions — Scope: COUNTRY (CHI). The single shared scheduler,
# run quarterly. For each of our tributary subjects whose per-vassal cooldown has
# lapsed, mint an envoy and fire the arrival. Bounded to ONE mission per pulse (the
# first eligible subject) so several tributaries never dogpile the player in one
# quarter — the others come on later pulses. Cooldown cadence ~ historical tempo.
# ======================================================================
QING_tribute_schedule_missions = {
	if = {
		limit = {
			tag = CHI
			# [BT-28] external event slot: at most one tribute embassy per pulse; shares the single external
			# slot so tribute never dogpiles other external beats. Court events use a separate slot (qing_gc_event_slot_used).
			NOT = { has_variable = qing_ext_event_slot_used }
			any_subject = { QING_tribute_is_tributary_trigger = yes }
		}
		LOG_enter = { sys = QING  fn = "QING_tribute_schedule_missions" }
		QING_tribute_init = yes
		# [#322 review-fix — CRITICAL] pick ONE due tributary and run ALL mission effects
		# INSIDE random_subject, so a no-match pulse (every tributary still on cooldown) is
		# a clean no-op. The previous structure saved the scope inside random_subject but ran
		# the mint/fire OUTSIDE it guarded only on exists=scope — and a saved scope PERSISTS
		# across pulses, so once all vassals were on cooldown the stale scope re-fired a bogus
		# embassy + leaked an envoy every quarter. Keeping the effects inside is the proven
		# AMBAN idiom (se_QING_AMBAN.txt:164-185): no match -> the block never executes.
		# The per-subject cooldown var lives on the SUBJECT (this), so it is per-vassal by
		# construction and re-gates correctly once the days= timer lapses.
		random_subject = {
			limit = {
				QING_tribute_is_tributary_trigger = yes
				NOT = { has_variable = qing_tribute_cooldown }
			}
			save_scope_as = trib_sender
			# [#322 deferred] PER-VASSAL CADENCE. The tributaries did NOT send on one clock:
			# Korea (the model tributary, the 燕行使) sent most often — several missions a
			# decade — while the lesser courts (Vietnam, Nepal, Ryukyu, Taiwan-aborigine)
			# came more rarely. Stamp the cooldown by the sender's tag so Korea's rhythm is
			# tighter (~3y) and the others slower (~6y), keeping the historical asymmetry.
			# The var lives on the SUBJECT (this), so it is per-vassal by construction.
			QING_tribute_stamp_cadence = yes
			ROOT = { set_variable = { name = qing_ext_event_slot_used  value = 1 } }  # [BT-28] claim external slot on the fire path
			# hop back to the overlord (ROOT = CHI is unchanged inside random_subject) to mint
			# the envoy in Peking's court and fire the arrival country-event on the Qing.
			ROOT = {
				QING_tribute_mint_envoy = { sender = scope:trib_sender }
				# [BT-7 logfix] persist both mission scopes as country vars so the DELAYED
				# arrival (5-20d later) can re-save them in its own immediate — a raw saved
				# scope can go stale across the delay window, so scope:trib_sender/trib_envoy
				# rendered ERROR:[scope:...GetName] in qing_tribute.1. Same class as BT-1.
				set_variable = { name = qing_trib1_sender  value = scope:trib_sender }
				set_variable = { name = qing_trib1_envoy   value = scope:trib_envoy }
				trigger_event = { id = qing_tribute.1  days = { 5 20 } }
				LOG_line = { sys = QING  msg = "tribute: an embassy is dispatched to Peking for" }
			}
		}
		LOG_exit = { sys = QING  fn = "QING_tribute_schedule_missions"  result = OK }
	}
}

# ======================================================================
# QING_tribute_expire_cooldowns — Scope: COUNTRY (CHI). Companion to the scheduler:
# tick DOWN nothing (the engine expires the days= timer itself), but a subject that
# LEAVES the tributary fold keeps a stale cooldown var forever — harmless (it just
# never sends again while non-subject), and re-subjugation clears it here so the
# rhythm resumes. Cheap; runs on the same pulse. Kept separate for clarity.
# ======================================================================
# (No body needed — the days= cooldown auto-expires and the schedule re-gates on
#  QING_tribute_is_tributary_trigger every pulse, so a freed-then-resubjugated vassal
#  simply resumes once its timer lapses. Documented here so the absence is intentional.)

# ======================================================================
# QING_tribute_mint_envoy — Scope: COUNTRY (CHI). $sender$ = the tributary COUNTRY.
# Create the mission's envoy as a real character in the Qing court and scope him for
# the arrival event. Culture follows the sender's primary culture so a Korean mission
# looks Korean, a Vietnamese one Vietnamese (the proven scope-culture read). Marked so
# he is a minor diplomatic figure, not an officeholder; QING_tribute_dismiss_envoy sends
# him back home when the audience ends (both options of qing_tribute.1) so envoys never
# accumulate in the Qing court. He holds no list membership, so a death leaves no orphan
# reference (no on_character_death hook needed, unlike the amban/pilgrim/scholar features).
# ======================================================================
QING_tribute_mint_envoy = {
	$sender$ = { save_scope_as = trib_envoy_home }
	create_character = {
		age = 45
		culture = scope:trib_envoy_home.culture
		religion = scope:trib_envoy_home.religion
		add_finesse = 6
		add_charisma = 6
		add_martial = 1
		add_zeal = 3
		save_scope_as = trib_envoy
	}
	scope:trib_envoy = {
		set_home_country = ROOT
		set_as_minor_character = THIS
		set_variable = { name = qing_is_tribute_envoy  value = 1 }
		# record who sent him, so the dismissal can send him back home.
		set_variable = { name = qing_tribute_envoy_home  value = scope:trib_envoy_home }
	}
	LOG_line = { sys = QING  msg = "tribute: minted an embassy envoy for" }
}

# ======================================================================
# QING_tribute_dismiss_envoy — [#322 review-fix] Scope: COUNTRY (CHI). The mission is
# received (either option of qing_tribute.1); the envoy's business is done, so send him
# HOME to the tributary that sent him rather than let him accumulate in the Qing court
# forever (each mission otherwise left a permanent minor character — a save/court-list
# leak). Self-guards on the scope. He keeps his marker so the on_death hook (below)
# still cleans any dangling reference if he dies abroad.
# ======================================================================
QING_tribute_dismiss_envoy = {
	if = {
		limit = { exists = scope:trib_envoy }
		scope:trib_envoy = {
			if = {
				limit = { has_variable = qing_tribute_envoy_home  exists = var:qing_tribute_envoy_home }
				set_home_country = var:qing_tribute_envoy_home
				LOG_line = { sys = QING  msg = "tribute: envoy dismissed home after audience for" }
			}
		}
	}
}

# ======================================================================
# QING_tribute_receive — Scope: COUNTRY (CHI). $sender$ = the tributary. The payload of
# a mission's TRIBUTE-TRADE PROFIT (a treasury haul scaled by the sender's income, the
# historical asymmetric-gift trade) + the mission tempo counter. [#322 review-fix] the
# suzerain-prestige change is NO LONGER applied here — it belongs entirely to the event
# option (a full audience firms it, a curt dismissal strains it), so this shared helper
# stays prestige-neutral and the option's own nudge is not silently cancelled.
# ======================================================================
QING_tribute_receive = {
	LOG_enter = { sys = QING  fn = "QING_tribute_receive" }
	# tribute-trade profit: a quarter's worth of ~25% of the sender's monthly income,
	# clamped so a poor tributary cannot pay what it lacks. has_monthly_income is the
	# proven country-income read (00_event_values.txt), as used by the B17 collector.
	if = {
		limit = { exists = $sender$ }
		# [logfix R3 2026-07-17] Do the SET, the guarded READ, and the REMOVE all INSIDE a single
		# $sender$ scope block. The prior R2 shape set qing_trib_gift inside `$sender$ = { }` but then
		# read it from a SEPARATE sibling `if` via `$sender$ = { var:qing_trib_gift > 0 }` / a
		# `$sender$.var:qing_trib_gift` treasury add — and that cross-block re-entry read the var
		# before/without the set landing in the same evaluation, flooding error.log 139x/boot
		# ("Failed to fetch variable 'qing_trib_gift' ... not being set" + "Event target link 'var'
		# returned an unset scope" + "Invalid left side during comparison 'var'"). Collapsing to one
		# scope block, and guarding the read with has_variable, makes the read unconditionally safe.
		# The profit is added to ROOT (the Qing) — captured into a temp before leaving $sender$ scope.
		$sender$ = {
			set_variable = { name = qing_trib_gift  value = qing_tribute_gift_svalue }
			if = {
				limit = { has_variable = qing_trib_gift  var:qing_trib_gift > 0 }
				save_temporary_scope_as = trib_gift_payer
				ROOT = { add_treasury = scope:trib_gift_payer.var:qing_trib_gift }
			}
			remove_variable = qing_trib_gift
		}
	}
	change_variable = { name = qing_tribute_missions_recent  add = 1 }
	LOG_exit = { sys = QING  fn = "QING_tribute_receive"  result = OK }
}

# ======================================================================
# QING_tribute_prestige_from_missions — Scope: COUNTRY (CHI). Fold the recent-mission
# tempo into the suzerain-prestige drift and decay the tempo counter, so a vigorous
# embassy calendar sustains the order and a lapse lets it slip. Runs on the pulse
# after the schedule. Self-guards on the counter.
# ======================================================================
QING_tribute_prestige_from_missions = {
	if = {
		limit = { has_variable = qing_tribute_missions_recent }
		# a busy calendar (>=3 recent missions) affirms the order; a silent one lets it fade.
		if = {
			limit = { var:qing_tribute_missions_recent >= 3 }
			QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = 1 }
		}
		else_if = {
			limit = { var:qing_tribute_missions_recent = 0 }
			QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -1 }
		}
		# decay the tempo counter toward zero (one step per pulse) so it reflects a
		# ROLLING window, not a lifetime total.
		if = { limit = { var:qing_tribute_missions_recent > 0 }  change_variable = { name = qing_tribute_missions_recent  subtract = 1 } }
	}
}

# ======================================================================
# [#322 deferred] QING_tribute_stamp_cadence — Scope: the SUBJECT tributary (this),
# called inside the scheduler's random_subject. Stamp the per-vassal send cooldown by
# tag: Korea's embassy calendar was the densest (the 燕行使 came several times a decade),
# so it re-gates fastest; the lesser courts send more rarely. All still a single band
# each (boot-safe, no arithmetic). A tributary with no special case gets the middle
# cadence. The var is per-subject by construction (stamped on `this`).
# ======================================================================
QING_tribute_stamp_cadence = {
	if = {
		# Korea — the model tributary, most frequent missions (~3y).
		limit = { tag = KOR }
		set_variable = { name = qing_tribute_cooldown  days = 1095 }
	}
	else_if = {
		# Vietnam / Ryukyu — regular but less frequent (~5y).
		limit = { OR = { tag = VIE  tag = RYU } }
		set_variable = { name = qing_tribute_cooldown  days = 1825 }
	}
	else = {
		# the outer courts (Nepal, Taiwan-aborigine, any other) — the rarest (~6y).
		set_variable = { name = qing_tribute_cooldown  days = 2190 }
	}
}

# ======================================================================
# [#322 deferred] QING_tribute_invest_new_ruler — Scope: COUNTRY (CHI), called from the
# tributary-side on_ruler_change hook. THE INVESTITURE (冊封): when a tributary's throne
# passed to a new ruler, the Qing court confirmed him with a patent of investiture and a
# seal — the ritual that RENEWED the tributary bond each reign. $vassal$ = the tributary
# COUNTRY (its new ruler already seated). Fires the investiture dilemma country-event on
# the Qing so the player chooses whether to invest promptly (affirms the order), extract a
# price, or withhold recognition (a snub that strains the tie). Self-guards on the tie
# still holding. The event carries the vassal via a saved scope.
# ======================================================================
QING_tribute_invest_new_ruler = {
	# this = the Qing overlord (CHI). Save it as an explicit scope so the tributary test
	# does NOT depend on ROOT (this helper is reached from the tributary-side on_ruler_change,
	# where ROOT is the TRIBUTARY, not CHI — so QING_tribute_is_tributary_trigger's `is_subject_of
	# = ROOT` would be wrong here). We test the tie against the saved overlord scope instead.
	save_scope_as = trib_overlord
	if = {
		limit = {
			tag = CHI
			exists = $vassal$
			$vassal$ = {
				is_subject_of = scope:trib_overlord
				OR = { is_subject_type = tributary  is_subject_type = sinosphere_tributary }   # [1763-hist-fix] ritual tributaries get investiture too
				exists = current_ruler
			}
		}
		LOG_enter = { sys = QING  fn = "QING_tribute_invest_new_ruler" }
		# [T2 review-fix / BT-7 idiom] persist the vassal as a var — the raw saved scope can go stale
		# across the 10-40d trigger_event delay (.2 re-saves it in immediate). ROOT here = CHI (the
		# overlord hop), so the var lives on the country that fires + reads .2.
		set_variable = { name = qing_trib2_vassal  value = $vassal$ }
		$vassal$ = { save_scope_as = trib_invest_vassal }
		trigger_event = { id = qing_tribute.2  days = { 10 40 } }
		LOG_line = { sys = QING  msg = "tribute: a tributary's new ruler awaits investiture for" }
		LOG_exit = { sys = QING  fn = "QING_tribute_invest_new_ruler"  result = OK }
	}
}

# ======================================================================
# [#322 deferred] QING_tribute_breakaway_check — Scope: COUNTRY (CHI). Run on the quarterly
# pulse. When suzerain-prestige has fallen LOW, the tributary order frays and a restless
# vassal may test the bond. Picks one tributary and fires the breakaway-crisis dilemma
# (throne must overawe it, buy it back, or let it drift), throttled on its own cooldown so
# it is a rare crisis, not spam. Self-guards on having a tributary AND low prestige, so it
# is a clean no-op at the High Qing zenith (prestige seeds at 70; the crisis needs <35).
# ======================================================================
QING_tribute_breakaway_check = {
	if = {
		limit = {
			tag = CHI
			has_variable = qing_suzerain_prestige
			var:qing_suzerain_prestige < 35
			NOT = { has_variable = qing_tribute_breakaway_cooldown }
			any_subject = { QING_tribute_is_tributary_trigger = yes }
		}
		LOG_enter = { sys = QING  fn = "QING_tribute_breakaway_check" }
		set_variable = { name = qing_tribute_breakaway_cooldown  days = 1095 }   # ~3y throttle
		random_subject = {
			limit = { QING_tribute_is_tributary_trigger = yes }
			save_scope_as = trib_restless
			ROOT = {
				# [T2 review-fix / BT-7 idiom] persist the tributary as a var on CHI — the raw saved scope
				# goes stale across the 5-20d delay; .3 re-saves it in immediate.
				set_variable = { name = qing_trib3_restless  value = scope:trib_restless }
				trigger_event = { id = qing_tribute.3  days = { 5 20 } }
				LOG_line = { sys = QING  msg = "tribute: a restless tributary tests the bond (low suzerain prestige) for" }
			}
		}
		LOG_exit = { sys = QING  fn = "QING_tribute_breakaway_check"  result = OK }
	}
}

# ======================================================================
# [#322 deferred] QING_tribute_gp_interference_check — Scope: COUNTRY (CHI). Run on the
# quarterly pulse. As the Western powers pressed into East Asia, a great power whose China
# tension has risen may meddle in a Qing tributary (the historical arc: France in Vietnam,
# Britain in Burma/Nepal's marches, Japan later in Korea/Ryukyu). Picks the highest-tension
# power and fires the interference dilemma on one tributary. Throttled + self-guards on a
# tributary existing AND some GP tension having built, so it is inert at the isolated zenith.
# ======================================================================
QING_tribute_gp_interference_check = {
	if = {
		limit = {
			tag = CHI
			NOT = { has_variable = qing_tribute_gp_cooldown }
			any_subject = { QING_tribute_is_tributary_trigger = yes }
			OR = {
				var:qing_gp_tension_britain >= 40
				var:qing_gp_tension_france >= 40
				var:qing_gp_tension_russia >= 40
			}
		}
		LOG_enter = { sys = QING  fn = "QING_tribute_gp_interference_check" }
		set_variable = { name = qing_tribute_gp_cooldown  days = 1460 }   # ~4y throttle
		# name the meddling power = the highest China-tension of the three (proven ordered
		# read cf. the sphere max-pick; here a simple cascade keeps it boot-safe).
		QING_tribute_pick_meddler = yes
		random_subject = {
			limit = { QING_tribute_is_tributary_trigger = yes }
			save_scope_as = trib_meddled
			ROOT = {
				# [T2 review-fix / BT-7 idiom] persist the tributary as a var on CHI — the raw saved scope
				# goes stale across the 5-20d delay; .4 re-saves it in immediate.
				set_variable = { name = qing_trib4_meddled  value = scope:trib_meddled }
				trigger_event = { id = qing_tribute.4  days = { 5 20 } }
				LOG_line = { sys = QING  msg = "tribute: a great power meddles in a Qing tributary for" }
			}
		}
		LOG_exit = { sys = QING  fn = "QING_tribute_gp_interference_check"  result = OK }
	}
}

# ======================================================================
# [#322 deferred] QING_tribute_pick_meddler — Scope: COUNTRY (CHI). Store the meddling
# power's identity as a flag var (qing_tribute_meddler = flag:britain|france|russia) for
# the interference event to read + display. Highest tension wins; ties resolve
# Britain > France > Russia (the order of Western pressure on the Qing sphere).
# ======================================================================
QING_tribute_pick_meddler = {
	if = {
		limit = {
			var:qing_gp_tension_britain >= qing_gp_tension_france_cmpsvalue
			var:qing_gp_tension_britain >= qing_gp_tension_russia_cmpsvalue
		}
		set_variable = { name = qing_tribute_meddler  value = flag:britain }
	}
	else_if = {
		limit = { var:qing_gp_tension_france >= qing_gp_tension_russia_cmpsvalue }
		set_variable = { name = qing_tribute_meddler  value = flag:france }
	}
	else = {
		set_variable = { name = qing_tribute_meddler  value = flag:russia }
	}
}

# ======================================================================
# [BT-11] QING_tribute_default_check — Scope: COUNTRY (CHI). Run on the quarterly pulse.
# Reads the qing_tribute_defaulted marker that QING_subject_collect_tribute stamps on any
# subject whose treasury could not cover its owed tribute this quarter (a real default,
# not the customary light rendering). Routes the shortfall through the LIFAN YUAN (理藩院):
#   • dings the tributary-order meter (qing_suzerain_prestige) — an unanswered default
#     erodes the standing of the whole order;
#   • if a resident amban supervises the defaulter, marks him ineffective (the resident is
#     the throne's man on the ground and shares the blame for a dependency slipping);
# then fires the PUNISH/FORGIVE dilemma (qing_tribute.5) on ONE defaulting subject. Honours
# the BT-28 external event slot (counts as the quarter's one external/tribute event) and its
# own ~2y cooldown, so it never stacks with the embassy scheduler (which also claims the slot).
# (The breakaway/interference checks predate BT-28 and do not claim the slot, so they are not
# mutually exclusive with this one; collisions are rare given the 2-4y cooldowns.) Self-guards
# on a defaulter existing, a clean no-op at the solvent High Qing zenith.
# ======================================================================
QING_tribute_default_check = {
	if = {
		limit = {
			tag = CHI
			NOT = { has_variable = qing_ext_event_slot_used }
			NOT = { has_variable = qing_tribute_default_cooldown }
			any_subject = {
				QING_tribute_is_tributary_trigger = yes
				has_variable = qing_tribute_defaulted
			}
		}
		LOG_enter = { sys = QING  fn = "QING_tribute_default_check" }
		set_variable = { name = qing_tribute_default_cooldown  days = 730 }   # ~2y throttle
		# route the default through the tributary order (the whole order's standing slips)
		QING_DECLINE_nudge = { var = qing_suzerain_prestige  amount = -4 }
		random_subject = {
			limit = {
				QING_tribute_is_tributary_trigger = yes
				has_variable = qing_tribute_defaulted
			}
			save_scope_as = trib_defaulter
			# clear the marker now so a single default is judged once (the dilemma resolves it)
			remove_variable = qing_tribute_defaulted
			# if a resident amban supervises this dependency, he shares the blame — mark him
			# ineffective in the throne's eyes (the same opinion the amban evaluator uses for a
			# resident who lets a dependency slip). Guarded on a live posted amban being present.
			if = {
				limit = {

# se_QING_REVENUE.txt — the Grand Minister of Revenue (戶部) domain.
#
# The 戶部 (Ministry of Revenue) held the imperial fisc: the census, land tax,
# the salt gabelle (鹽政 — a vast revenue source riddled with patronage), and
# the ever-normal granaries (常平倉, grain reserves for price stabilisation).
# The 19thC silver crisis (銀荒/銀貴錢賤 — silver draining out via opium, copper-
# cash inflation ruining taxpayers) was THE fiscal emergency of the era.
#
# This suite reuses the EXISTING granary building (add_building_level on
# qing_granary_building in Yellow River provinces) and the EXISTING currency-
# stress counter (qing_currency_stress, bands at 30/60). It does NOT duplicate
# parallel state — the Revenue holder acts ON the concrete objects and counters
# already in the decline engine.
#
# Coupling: a capable, honest, high-finesse Revenue minister keeps the salt
# gabelle honest (income buff), stocks the granaries (unrest control via the
# concrete building), and can act against the silver crisis (nudge currency
# stress down). A venal or vacant office lets salt graft and currency stress
# worsen. Uses QING_pair_friction for governor/tax-commissioner charge coupling.
#
# State (all on CHI):
#   qing_office_revenue_holder    the character (se_QING_COUNCIL.txt)
#   qing_salt_gabelle_reformed     0/1 flag: has the salt monopoly been cleaned up
#   qing_revenue_event_cooldown    days: throttle the revenue events
#
# Character-scope marks (on the Revenue holder):
#   (reuses qing_office_held from se_QING_COUNCIL.txt)
#
# Prefix = QING_revenue_ / sys = QING

# ======================================================================
# [#372 / #425] SILVER-RESERVE CONSTANTS + INIT. The 戶部銀庫 (Board of Revenue central silver
# treasury) held a CONCRETE, well-documented balance across the Qing: ~62M taels at the 1763
# zenith, climbing to its all-time peak of ~81.8M taels in 乾隆42 (1777) under sound
# administration, then draining in the decline (~69M by 1795, ~28M by 1820 as the Heshen fortune
# was seized and the White Lotus war bled it).
#
# [#425 RETIREMENT] This balance is NO LONGER a bespoke counter. It IS the engine's currency-
# backing reserve, silver_reserve_size (se_CURRENCY.txt: seeded per-tag for CHI, date-gated to
# 62000/46140; read into currency power at :1913). The old qing_silver_reserve var (in 萬兩) is
# GONE — driving silver_reserve_size directly means silver inflow/outflow actually moves the
# currency backing (the real 銀荒 dynamic). Counted in 千兩 (thousand taels) = the engine unit,
# so all #372 figures are ×10 vs the retired counter (6200萬兩 → 62000千兩, peak 8182 → 81820).
#
# INIT now only seeds the PEAK CONSTANT the accumulation targets (the reserve itself is seeded
# by the engine). qing_silver_reserve_peaked marks the one-shot milestone event. Sources in
# memory imp19c-silver-reserve-figures. Scope: country (CHI).
# ======================================================================
QING_revenue_init = {
	LOG_enter = { sys = QING  fn = "QING_revenue_init" }
	# the historic all-time high the accumulation targets: 8182萬兩 (乾隆42/1777) = 81820 千兩.
	if = { limit = { NOT = { has_variable = qing_silver_reserve_peak } }   set_variable = { name = qing_silver_reserve_peak  value = 81820 } }
	LOG_line = { sys = QING  msg = "revenue: silver reserve peak constant seeded (千兩); reserve itself = engine silver_reserve_size for" }
	LOG_exit = { sys = QING  fn = "QING_revenue_init"  result = OK }
}

# ======================================================================
# [#372 / #425] SILVER-RESERVE DRIFT — called each quarter from QING_revenue_pulse. Drives the
# engine's REAL currency-backing reserve, silver_reserve_size (千兩), DIRECTLY — so the silver
# the throne accumulates or bleeds actually moves the currency backing (se_CURRENCY.txt:1913),
# the true 銀荒 dynamic. During the High Qing era, a seated + capable Revenue minister (戶部尚書)
# accumulates the treasury toward its historic ~81.8M-tael peak; a middling minister climbs
# slowly; a vacant/venal office or the post-Heshen decline era draws it DOWN. A high silver-
# crisis reading (qing_currency_stress) also bleeds the reserve (the 銀荒 drain). Bounded
# [0, peak]. Fires the peak-milestone event (qing_revenue.5) ONCE when the treasury first nears
# its all-time high. All figures ×10 vs the retired 萬兩 counter (千兩 units). Scope: country (CHI).
# ======================================================================
QING_revenue_reserve_drift = {
	# [#425 ratchet-rule / other-writer audit] silver_reserve_size has THREE other engine writers that
	# also run for CHI — this drift is NOT the sole mover, but it composes safely with all of them:
	#   1. CURRENCY_all_governorships_send_to_reserves (oa_wealth_changes.txt:302, quarterly) adds each
	#      governorship's SILVER-GOOD production. Audited: NO Qing province produces the `silver` trade
	#      good (setup/provinces/*China/Yunnan/Sichuan/etc. = 0), so ~nil for CHI (only the small
	#      silver_reserve_accumulation_rate buy).
	#   2. CURRENCY_private_purchase_or_sell_reserves (se_CURRENCY.txt:1204, MONTHLY — 3×/quarter) adds/
	#      subtracts CURRENCY_reserve_accumulation_rate_from_inflation_or_deflation to the LARGEST reserve
	#      (always silver for CHI, gold=0): it SUBTRACTS under inflation, ADDS under deflation. This
	#      REINFORCES the 銀荒 narrative (silver leaking out = inflation = further specie drain), it does
	#      not fight the drift.
	#   3. INCOME_sell_reserves (se_INCOME.txt:561, quarterly) drains the largest reserve when CHI runs a
	#      deficit with auto-fund — a legitimate "sell the plate to pay the troops" mechanic.
	# COMPOSITION IS SAFE regardless of these: the positive-accumulation leg here is band-gated (< peak)
	# per the no-restoring-drift ratchet rule; the drain/crisis-bleed legs are ungated (so downward crisis
	# dynamics are never neutered); and BOTH ends are clamped [0, peak] every quarter, so no other writer
	# can push the reserve past its historic peak or below zero via this drift's window.
	if = {
		# guard on the engine reserve existing (always true for CHI — seeded per-tag in
		# CURRENCY_setup_all_reserves — but explicit so a mis-scope no-ops rather than erroring).
		limit = { has_variable = silver_reserve_size }
		# --- accumulation vs drain -------------------------------------------------
		if = {
			# High Qing era + a seated minister: strong accumulation toward the peak
			# (historically ~150萬兩/yr 1755-1777 ≈ ~35萬兩/quarter → ~350 千兩/quarter). [#372
			# review-fix] the base accumulation is gated ONLY on the office being FILLED, not on
			# finesse >= 6 — the era + a seated minister already implies the fisc is being run
			# competently, and the council autofill sorts by COMBINED stats (not finesse), so a
			# low-finesse general can hold the Revenue chair on some seeds; gating the strong branch
			# on finesse would then stall accumulation below the milestone and the peak event would
			# silently never fire. A high-finesse minister still earns an EXTRA +100 (千兩; the
			# competence reward) just below, so skill is rewarded without being a hard gate.
			limit = {
				has_variable = qing_high_qing_era
				has_variable = qing_office_revenue_holder
				var:qing_office_revenue_holder = { is_alive = yes  employer = ROOT }
				var:silver_reserve_size < qing_silver_reserve_peak_cmpsvalue
			}
			change_variable = { name = silver_reserve_size  add = 250 }
			# competence reward: a high-finesse Grand Minister accumulates faster still.
			if = {
				limit = { var:qing_office_revenue_holder = { finesse >= 6 } }
				change_variable = { name = silver_reserve_size  add = 100 }
			}
		}
		else = {
			# vacant office OR the post-era decline: the treasury drains.
			change_variable = { name = silver_reserve_size  subtract = 150 }
		}
		# --- the silver crisis (銀荒) bleeds the reserve on top of the above --------
		if = {
			limit = {
				has_variable = qing_currency_stress
				var:qing_currency_stress >= 40
			}
			change_variable = { name = silver_reserve_size  subtract = 100 }
		}
		# [#425 note] this drift writes silver_reserve_size with raw change_variable and deliberately
		# does NOT touch silver_reserve_actual_change (the engine's per-tick delta var that the VANILLA
		# economy panel's "silver accumulation rate" line reads). So the reserve SIZE moves each quarter
		# but that vanilla rate line won't attribute the drift's ±250-350 to it — cosmetic only, and on
		# the vanilla view, not the 戶部 panel (which reads the size directly). Left as-is: this drift is
		# an administrative/historical driver, not a market accumulation, so folding it into the market
		# accumulation-rate readout would misrepresent it.
		# --- clamp [0, peak] -------------------------------------------------------
		if = { limit = { var:silver_reserve_size > qing_silver_reserve_peak_cmpsvalue }  set_variable = { name = silver_reserve_size  value = var:qing_silver_reserve_peak } }
		if = { limit = { var:silver_reserve_size < 0 }  set_variable = { name = silver_reserve_size  value = 0 } }
		# --- [MO-ENH backfeed] a CRITICALLY DRAINED reserve worsens the silver crisis (bidirectional
		# coupling). Currency stress already bleeds the reserve (above); the reverse is equally historical
		# — an empty 戶部銀庫 (as in 1799, drained by the White Lotus war + the Heshen seizure) left the
		# throne no silver to stabilise the copper-cash exchange, so 銀貴錢賤 worsened. A near-empty
		# reserve (< 10000 千兩, ~1/8 of the zenith) adds a gentle quarterly push to currency stress; a
		# healthy reserve does nothing. Additive + clamped by QING_DECLINE_nudge; no runaway (stress
		# feeds reform pressure, not back into the reserve at the same magnitude).
		if = {
			limit = {
				var:silver_reserve_size < 10000
				has_variable = qing_currency_stress
			}
			QING_DECLINE_nudge = { var = qing_currency_stress  amount = 1 }
		}
		# --- peak milestone (盈庫之盛): fire once when the treasury first nears its high.
		# Gated on the shared court event slot (BT-28) + a one-shot flag, so it is a single
		# memorable beat, never spam. 80000 (~98% of 81820) counts as "reached the peak".
		if = {
			limit = {
				var:silver_reserve_size >= 80000
				NOT = { has_variable = qing_silver_reserve_peaked }
				NOT = { has_variable = qing_gc_event_slot_used }
			}
			set_variable = { name = qing_silver_reserve_peaked  value = 1 }
			set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [BT-28] claim court slot
			trigger_event = { id = qing_revenue.5  days = { 5 20 } }
			LOG_line = { sys = QING  msg = "revenue: silver reserve reached its historic peak for" }
		}
	}
}

# ======================================================================
# [#415] QING_revenue_salt_income — the STANDING salt-gabelle revenue (鹽課). Scope: country (CHI).
# Called each quarter from QING_revenue_pulse. Historically the salt tax was one of the largest
# single central revenues (~10-12%); this pays it as a routine quarterly treasury income, scaled by
# the gabelle's honesty + the minister's competence + realm size. Counter maths into a scratch var,
# clamped >= 0, added once. NOT era-gated (salt was collected throughout), but a vacant office /
# entrenched graft yields less.
# NOTE: this is a SECOND, distinct channel from the passive qing_salt_gabelle_reformed_mod /
# qing_salt_gabelle_graft country modifiers (which apply a global_tax_modifier % buff/malus). This
# effect is the flat quarterly 鹽課 line-item; the modifiers are the diffuse tax %. Tune both together.
#   base 6, then:
#     + reformed gabelle -> +4 (an honest, well-run monopoly); graft-modifier -> -2 (leaky)
#     + a seated finesse>=8 Grand Minister -> +3 (wrings more from the yards)
#     + realm scale: +1 per full 15 owned provinces (num_of_cities/15), the salt-yard base
#   floored at 1 so the monopoly always yields something.
# ======================================================================
QING_revenue_salt_income = {
	set_variable = { name = qing_salt_income_tmp  value = 6 }
	# honesty of the monopoly.
	if = {
		limit = { has_variable = qing_salt_gabelle_reformed  var:qing_salt_gabelle_reformed = 1 }
		change_variable = { name = qing_salt_income_tmp  add = 4 }
	}
	if = {
		limit = { has_country_modifier = qing_salt_gabelle_graft }
		change_variable = { name = qing_salt_income_tmp  subtract = 2 }
	}
	# a capable seated Grand Minister of Revenue wrings more from the yards.
	if = {
		limit = {
			has_variable = qing_office_revenue_holder
			var:qing_office_revenue_holder = { is_alive = yes  employer = ROOT  finesse >= 8 }
		}
		change_variable = { name = qing_salt_income_tmp  add = 3 }
	}
	# realm scale: the salt-yard base grows with the empire (+1 per 15 provinces), CAPPED at +14
	# so a vast conquest empire's salt line stays substantial-but-bounded (the Qing owns hundreds of
	# cities; without a cap the term would balloon). At the ~1763 extent this sits near the cap.
	set_variable = { name = qing_salt_scale_tmp  value = num_of_cities }
	change_variable = { name = qing_salt_scale_tmp  divide = 15 }
	if = { limit = { var:qing_salt_scale_tmp > 14 }  set_variable = { name = qing_salt_scale_tmp  value = 14 } }
	change_variable = { name = qing_salt_income_tmp  add = var:qing_salt_scale_tmp }
	remove_variable = qing_salt_scale_tmp
	# floor at 1 (the monopoly always yields something), then pay it.
	if = { limit = { var:qing_salt_income_tmp < 1 }  set_variable = { name = qing_salt_income_tmp  value = 1 } }
	add_treasury = var:qing_salt_income_tmp
	# publish for the Revenue panel read-out (the 鹽課 line item).
	set_variable = { name = qing_salt_income_last  value = var:qing_salt_income_tmp }
	remove_variable = qing_salt_income_tmp
	LOG_line = { sys = QING  msg = "revenue: the salt gabelle pays its quarterly take for" }
}

# ======================================================================
# QUARTERLY REVENUE PULSE — called from QING_GOV_pulse (quarterly cadence).
# Checks the Revenue holder's state and offers the domain events (salt gabelle,
# granary, silver crisis, treasury audit) on a throttled roll. A vacant office
# lets the domain drift (salt graft up, currency stress up, granaries neglected).
# ======================================================================
QING_revenue_pulse = {
	LOG_enter = { sys = QING  fn = "QING_revenue_pulse" }
	# [#372] drift the concrete silver reserve toward / away from its historic peak first,
	# so the panel read-out + any milestone reflect this quarter before the event roll below.
	QING_revenue_reserve_drift = yes

	# [#415] THE SALT GABELLE (鹽課) — a STANDING quarterly revenue, not just an event lever.
	# Historically the salt tax was ~10-12% of central revenue, a fiscal pillar (the Lianghuai
	# 兩淮 salt administration alone rivalled whole provinces). We pay a quarterly salt income to
	# the treasury, scaled by: base yield, whether the gabelle is REFORMED (honest, high yield) vs
	# GRAFT-ridden (leaky), the Revenue minister's finesse (a capable 尚書 wrings more from the
	# monopoly), and realm size (more provinces = more salt yards). All read into a scratch var,
	# clamped >= 0, added once. Independent of the event roll below (this is the routine take).
	QING_revenue_salt_income = yes

	# If the office is VACANT, let the domain drift negatively.
	if = {
		limit = { NOT = { has_variable = qing_office_revenue_holder } }
		# salt gabelle: without oversight, graft entrenches (+corruption)
		if = {
			limit = {
				has_variable = qing_salt_gabelle_reformed
				var:qing_salt_gabelle_reformed = 1
			}
			QING_DECLINE_nudge = { var = qing_corruption_level  amount = 1 }
			LOG_line = { sys = QING  msg = "revenue: office vacant -> salt gabelle oversight lapses (+corruption) for" }
		}
		# silver crisis: without a capable minister, currency stress worsens
		if = {
			limit = {
				has_variable = qing_currency_stress
				var:qing_currency_stress >= 20
			}
			QING_DECLINE_nudge = { var = qing_currency_stress  amount = 1 }
			LOG_line = { sys = QING  msg = "revenue: office vacant -> silver crisis unmanaged (+currency stress) for" }
		}
	}
	# Office filled: offer the domain events on a throttled roll.
	else_if = {
		limit = {
			has_variable = qing_office_revenue_holder
			var:qing_office_revenue_holder = { is_alive = yes  employer = ROOT }
			NOT = { has_variable = qing_revenue_event_cooldown }
			NOT = { has_variable = qing_gc_event_slot_used }  # [court-slot] share the quarterly court-event throttle
		}
		set_variable = { name = qing_revenue_event_cooldown  days = 270 }  # ~3 quarters
		random = {
			chance = 40
			random_list = {
				# salt gabelle reform (鹽政): revenue vs patronage trade-off
				20 = {
					trigger = {
						NOT = { has_variable = qing_salt_gabelle_reformed }
					}
					trigger_event = { id = qing_revenue.1  days = { 5 20 } }
					set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [court-slot] claim on the fire path
				}
				# granary stocking (常平倉): a good harvest lets you build reserves
				15 = {
					trigger = {
						# can only stock granaries if we have the capacity (Yellow River
						# provinces with cities and free building slots). [#1 follow-up] select by
						# REGION (Zhili/Shandong/Henan — the 常平倉 heartland), the same proven idiom
						# QING_DECLINE_granary_concrete uses. The old hardcoded province_id list was
						# WRONG (7229 = Peterwardein/Illyria, not 北京), so it never matched and the
						# roster stayed empty at a 1763 start.
						any_owned_province = {
							OR = {
								is_in_region = Zhili
								is_in_region = Shandong
								is_in_region = Henan
							}
							has_city_status = yes
						}
					}
					trigger_event = { id = qing_revenue.2  days = { 5 20 } }
					set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [court-slot] claim on the fire path
				}
				# silver crisis response (銀荒): can the minister act?
				20 = {
					trigger = {
						has_variable = qing_currency_stress
						var:qing_currency_stress >= 30
					}
					trigger_event = { id = qing_revenue.3  days = { 5 20 } }
					set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [court-slot] claim on the fire path
				}
				# treasury audit (uncovering embezzled land-tax)
				15 = {
					trigger = {
						has_variable = qing_corruption_level
						var:qing_corruption_level >= 40
					}
					trigger_event = { id = qing_revenue.4  days = { 5 20 } }
					set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [court-slot] claim on the fire path
				}
			}
		}
	}
	LOG_exit = { sys = QING  fn = "QING_revenue_pulse"  result = OK }
}

# ======================================================================
# HELPER: assess the Revenue holder's fitness for a given task.
# Writes a 0..100 score into qing_revenue_fitness based on the holder's:
#   - finesse (the core skill for fiscal administration)
#   - honesty/just traits vs corrupt trait
#   - numeric corruption stat
#   - loyalty (a disloyal minister won't act for the throne's interest)
# $task$ = "fiscal" (general finance), "oversight" (anti-corruption), or "crisis"
# (decisive action under pressure). Scope: country.
# ======================================================================
QING_revenue_assess_fitness = {
	LOG_enter = { sys = QING  fn = "QING_revenue_assess_fitness" }
	set_variable = { name = qing_revenue_fitness  value = 50 }

	if = {
		limit = {
			has_variable = qing_office_revenue_holder
			var:qing_office_revenue_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_revenue_holder = {
			# FINESSE is the core skill (administrator's gift)
			ROOT = {
				change_variable = { name = qing_revenue_fitness  add = prev.finesse }
				change_variable = { name = qing_revenue_fitness  subtract = 5 }
			}
			# HONESTY vs CORRUPTION
			if = {
				limit = { OR = { has_trait = honest  has_trait = just } }
				ROOT = { change_variable = { name = qing_revenue_fitness  add = 15 } }
			}
			if = {
				limit = { has_trait = corrupt }
				ROOT = { change_variable = { name = qing_revenue_fitness  subtract = 20 } }
			}
			if = {
				limit = { corruption >= 40 }
				ROOT = { change_variable = { name = qing_revenue_fitness  subtract = 15 } }
			}
			else_if = {
				limit = { corruption >= 20 }
				ROOT = { change_variable = { name = qing_revenue_fitness  subtract = 8 } }
			}
			# LOYALTY: a disloyal minister won't act for the throne
			if = {
				limit = { loyalty < 40 }
				ROOT = { change_variable = { name = qing_revenue_fitness  subtract = 20 } }
			}
			else_if = {
				limit = { loyalty >= 70 }
				ROOT = { change_variable = { name = qing_revenue_fitness  add = 10 } }
			}
			# TASK-SPECIFIC adjustments
			# "oversight" (anti-corruption) demands high honesty
			if = {
				limit = { ROOT.var:qing_revenue_task = flag:oversight }
				if = {
					limit = { OR = { has_trait = honest  has_trait = just } }
					ROOT = { change_variable = { name = qing_revenue_fitness  add = 10 } }
				}
				if = {
					limit = { has_trait = corrupt }
					ROOT = { change_variable = { name = qing_revenue_fitness  subtract = 15 } }
				}
			}
			# "crisis" (decisive action) demands high finesse + martial (nerve)
			else_if = {
				limit = { ROOT.var:qing_revenue_task = flag:crisis }
				ROOT = {
					change_variable = { name = qing_revenue_fitness  add = prev.martial }
					change_variable = { name = qing_revenue_fitness  subtract = 5 }
				}
			}
		}
	}
	# clamp 0..100
	if = { limit = { var:qing_revenue_fitness > 100 }  set_variable = { name = qing_revenue_fitness  value = 100 } }
	if = { limit = { var:qing_revenue_fitness < 0 }    set_variable = { name = qing_revenue_fitness  value = 0 } }

	LOG_line = { sys = QING  msg = "revenue fitness ($task$) = for" }
	LOG_exit = { sys = QING  fn = "QING_revenue_assess_fitness"  result = OK }
}

# ======================================================================
# CONCRETE: stock a granary in a Yellow River province (常平倉).
# Adds a qing_granary_building via add_building_level in a suitable province
# (city-status, Yellow River core). Scope: country.
# ======================================================================
QING_revenue_stock_granary = {
	LOG_enter = { sys = QING  fn = "QING_revenue_stock_granary" }
	# Pick a suitable Yellow River province that does NOT yet hold a granary and add the building.
	# [#355 review fix] the NOT-has_building guard stops the pick landing on an already-granaried
	# province (a redundant level for the same 60-gold cost the caller paid); ordered by population
	# so the largest uncovered city is stocked first, matching the works-build siblings' idiom.
	ordered_owned_province = {
		limit = {
			# [#1 follow-up] select by REGION (the 常平倉 heartland), matching
			# QING_DECLINE_granary_concrete. The old province_id list was wrong (7229 = Illyria).
			OR = {
				is_in_region = Zhili
				is_in_region = Shandong
				is_in_region = Henan
			}
			has_city_status = yes
			NOT = { has_building = qing_granary_building }
		}
		order_by = total_population
		max = 1
		add_building_level = qing_granary_building
		save_scope_as = granary_province
		LOG_line = { sys = QING  msg = "revenue: stocked granary (building +1) at for" }
	}
	LOG_exit = { sys = QING  fn = "QING_revenue_stock_granary"  result = OK }
}

# ======================================================================
# [playtest fix task #23] SEED the historical High-Qing ever-normal granary system (常平倉).
# Scope: country (CHI). Called ONCE from QING_DECLINE_init's High-Qing branch (start < 1772.1.1).
# The 常平倉 were at their zenith in the High Qing — the roster must NOT be empty at a 1763 start
# (the same rationale as the 62,000-tael silver-reserve seed). Places a qing_granary_building in
# each Yellow-River heartland city that lacks one (北京/天津/保定/開封/濟南 — the ids the Revenue
# stock/deplete paths already use), and pre-fills the pooled grain so the reserve reads stocked
# from turn one. Building placement drives qing_granary_count (Works sweep) and hence the pool cap;
# the qing_granary_food pre-fill is clamped to that cap on the first DECLINE pulse (harmless if the
# count lags one quarter — it simply tops out at the true capacity).
# ======================================================================
QING_revenue_seed_historical_granaries = {
	LOG_enter = { sys = QING  fn = "QING_revenue_seed_historical_granaries" }
	# [#1 follow-up] seed the 5 largest 常平倉-heartland cities (Zhili/Shandong/Henan — the same
	# region test QING_DECLINE_granary_concrete uses), NOT a hardcoded province_id list. The old
	# list was WRONG (7229 = Peterwardein in Illyria, 7230/7234/7235 likewise non-Chinese), so
	# every_owned_province never matched and no granary was ever built — the empty-roster bug.
	# ordered + max = 5 reproduces the intended 5 historical granaries without building one in
	# every heartland city.
	ordered_owned_province = {
		limit = {
			OR = {
				is_in_region = Zhili
				is_in_region = Shandong
				is_in_region = Henan
			}
			has_city_status = yes
			NOT = { has_building = qing_granary_building }
		}
		order_by = total_population
		max = 5
		add_building_level = qing_granary_building
		LOG_line = { sys = QING  msg = "revenue: SEEDED historical High-Qing granary (常平倉) at for" }
	}
	# pre-fill the pooled grain so the reserve begins stocked (not empty). Init if the DECLINE
	# pulse has not yet created the var; the first pulse clamps this to the real [0, count*200] cap.
	if = { limit = { NOT = { has_variable = qing_granary_food } }  set_variable = { name = qing_granary_food  value = 0 } }
	change_variable = { name = qing_granary_food  add = 600 }
	LOG_exit = { sys = QING  fn = "QING_revenue_seed_historical_granaries"  result = OK }
}

# ======================================================================
# CONCRETE: deplete a granary (famine draws down the reserves).
# Removes a qing_granary_building from a Yellow River province that has one.
# Scope: country.
# ======================================================================
QING_revenue_deplete_granary = {
	LOG_enter = { sys = QING  fn = "QING_revenue_deplete_granary" }
	random_owned_province = {
		limit = {
			# [#1 follow-up] select by REGION, matching the build paths above (old ids were wrong).
			OR = {
				is_in_region = Zhili
				is_in_region = Shandong
				is_in_region = Henan
			}
			num_of_qing_granary_building >= 1
		}
		remove_building_level = qing_granary_building
		save_scope_as = depleted_province
		LOG_line = { sys = QING  msg = "revenue: depleted granary (building -1) at for" }
	}
	LOG_exit = { sys = QING  fn = "QING_revenue_deplete_granary"  result = OK }
}

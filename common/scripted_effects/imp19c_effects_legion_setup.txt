# imp19c_effects_legion_setup.txt — historical starting-army setup
#
# create_unit / raise_legion follow the engine idiom verified against the
# feasibility oracles (Invictus, Terra-Indomita): every create_unit declares a
# base `sub_unit = <type>`, adds strength with `add_subunit = <type>`, and the
# commander is attached INSIDE the raise_legion block via
#   char:X = { add_to_legion = PREV } + random_legion_unit = { set_as_commander = char:X }
# (An earlier version used add_loyal_subunit / set_personal_loyalty / a pre-add
#  every_sub_unit loop / an external add_commander on a saved scope — none of
#  those are valid here; that is why these effects were disabled "due to errors".)
#
# Prefix = SE
########################################################################

########################################################################
# QING DUAL STANDING ARMY (#66) — the two permanent, separately-administered,
# ethnically-defined institutions of the 1815 Qing military, meeting only at the
# throne (see memory imp19c-qing-army-1815-research):
#
#   八旗 Eight Banners — the hereditary conquest-elite caste + strategic reserve.
#     ~half its soldiers (禁旅八旗) garrison BEIJING; the rest hold ~20 walled
#     "Manchu cities" 滿城 (mancheng) — Xi'an (largest interior), the Manchuria
#     homeland, the Xinjiang/NW frontier (Ili = seat of the 伊犁將軍), and interior
#     choke-points. Prestigious, but by 1815 militarily soft (banner-poverty
#     crisis 旗人生計, lost 國語騎射). We express that softness through the existing
#     qing_banner_decay bands (qing_banner_decay_mild / _severe country modifiers),
#     NOT by shrinking the paper rolls — hence garrisons are raised at full paper
#     size and the decay modifier carries the combat penalty ("big on paper,
#     brittle in field").
#
#   綠營 Green Standard — the Han professional TERRITORIAL/CONSTABULARY army,
#     deliberately fragmented (anti-warlordism) and dispersed across every
#     province for policing, escort, coast/canal patrol and first-line
#     suppression. Modelled as several mediocre provincial stacks under the great
#     regional commands (Zhili, Liangjiang, Huguang, Sichuan, Liangguang,
#     Shaan-Gan, Fujian incl. Taiwan, Yunnan-Guizhou).
#
# IMPLEMENTATION NOTE: every legion is raised from c:CHI.capital_scope.governorship
# (proven to exist) but LOCATED at its garrison province — exactly the pattern
# SE_occupation_of_france uses (all raised from the GBR capital governorship,
# located across France). This sidesteps any per-province state.governorship
# existence risk. Each raise is individually guarded on province existence +
# Qing ownership via the SE_qing_raise_garrison[_cmd] helpers, so a partitioned
# or shrunken Qing simply gets fewer garrisons (each miss logs LOG_fail) instead
# of a scope error. count = $size$ in the while-loop is the number of ADDITIONAL
# regular_infantry companies on top of the 1 base sub_unit (so total = size + 1).
########################################################################

# ----------------------------------------------------------------------
# Helper: raise a garrison legion of ($size$ + 1) regular_infantry companies at
# province $prov$, named $name$, with NO scripted commander (the player assigns
# one). Guards province existence + Qing ownership; raises from the Qing capital
# governorship and locates the unit at $prov$. Silent LOG_fail if $prov$ is not
# held. Assumes c:CHI exists (the caller SE_qing_armies guards that once).
# ----------------------------------------------------------------------
SE_qing_raise_garrison = {
	if = {
		# [#190-fix task #190] Accept BOTH CHI-owned land AND land held by a CHI
		# SUBJECT. The 1815 frontier banner garrisons (Ili 伊犁, Ürümqi 烏魯木齊,
		# Mukden 盛京, Heilongjiang 黑龍江, Kashgar 喀什噶爾) sit in provinces owned by
		# CHI's autonomous-governorship subjects (ILI/MKD/HLJ, XNG under ILI), not by
		# CHI directly — historically these WERE Qing frontier commands. The old guard
		# (owner = c:CHI only) silently dropped exactly these 5 garrisons via LOG_fail.
		# is_subject_of is a proven trigger (common/military_traditions/00_indian.txt;
		# Invictus parthia_interactions.txt). The legion is still RAISED from
		# c:CHI.capital_scope.governorship, so it belongs to Qing regardless.
		limit = { exists = $prov$  $prov$ = { OR = { owner = c:CHI  owner = { is_subject_of = c:CHI } } } }
		c:CHI.capital_scope.governorship = {
			raise_legion = {
				create_unit = {
					name = $name$
					location = $prov$
					sub_unit = regular_infantry
					save_scope_as = qing_garrison_legion
					while = { count = $size$  add_subunit = regular_infantry }
				}
			}
		}
		LOG_line = { sys = QING  msg = "Qing garrison raised: $name$ at $prov$ (+$size$ coy)" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "SE_qing_raise_garrison"  reason = "$prov$ not held by Qing; $name$ not raised" }
	}
}

# ----------------------------------------------------------------------
# Helper: as above, but attaches historical commander $cmd$ (a char scope) to the
# legion. Guards that $cmd$ exists too (a dead/absent character just falls through
# to LOG_fail rather than erroring). Commander attach follows the verified idiom:
# $cmd$ = { add_to_legion = PREV } then random_legion_unit = { set_as_commander }.
# ----------------------------------------------------------------------
SE_qing_raise_garrison_cmd = {
	if = {
		# [#190-fix task #190] Same subject-land acceptance as SE_qing_raise_garrison
		# above (Ili banner garrison under commander Changling char:244 is precisely
		# one of the frontier cases this restores). See that helper for the rationale.
		limit = { exists = $prov$  $prov$ = { OR = { owner = c:CHI  owner = { is_subject_of = c:CHI } } }  exists = $cmd$ }
		c:CHI.capital_scope.governorship = {
			raise_legion = {
				create_unit = {
					name = $name$
					location = $prov$
					sub_unit = regular_infantry
					save_scope_as = qing_garrison_legion
					while = { count = $size$  add_subunit = regular_infantry }
				}
				$cmd$ = { add_to_legion = PREV }
				random_legion_unit = { set_as_commander = $cmd$ }
			}
		}
		LOG_line = { sys = QING  msg = "Qing garrison raised: $name$ at $prov$ (+$size$ coy) under commander $cmd$" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "SE_qing_raise_garrison_cmd"  reason = "$prov$ unheld or $cmd$ absent; $name$ not raised" }
	}
}

# [#190-fix task #190] QING 1815 COASTAL WATER-FORCES (水師).
#
# THE BUG: the engine auto-generates every country's starting navy from
# STARTING_SHIPS_PER_PORT_POPULATION = 0.06 (common/defines/00_defines.txt:39).
# For China's huge coastal population that yields two ~85-ship modern fleets —
# grossly ahistorical: in 1815 the Qing had NO ocean-going battle-fleet, only
# provincial water-forces (水師) of shallow-draught war-junks for coastal defence,
# river/canal patrol and anti-piracy (the great pirate confederations had only
# just been suppressed, 1810). A blue-water steam/frigate navy did not arrive
# until the Self-Strengthening fleets (se_QING_SELFSTR.txt), which this mod
# already gates behind that mission tree.
#
# THE FIX: on game start, DISBAND every auto-generated CHI navy (destroy_unit,
# proven in Invictus 01_carthaginian_missions_08_naval.txt:152-157), then raise
# a modest historically-scaled junk squadron at the two principal maritime
# provincial commands — 廣東水師 (Guangdong, Canton estuary / anti-pirate) and
# 福建水師 (Fujian, Taiwan Strait). Modelled with `brig`, the only period-
# appropriate light warship key in this TC (screw_frigate/medium_steamer are the
# later modern hulls; the galley hulls are empty stubs — see qing_war_events.txt).
# Guarded on coastal ownership; each raise logs. Called once from
# oa_economy_setup.txt right after SE_qing_armies.
########################################################################
SE_qing_navy = {
	LOG_enter = { sys = QING  fn = "SE_qing_navy" }
	# Global sentinel: destroy_unit is DESTRUCTIVE, so — unlike the purely-additive
	# SE_qing_armies — this must never run twice (which would sink a navy the player
	# has since built). Guard so it fires exactly once at the true game start.
	if = {
		limit = {
			exists = c:CHI
			NOT = { has_global_variable = qing_navy_setup_done }
		}
		set_global_variable = qing_navy_setup_done
		c:CHI = {
			# 1) Disband the ahistorical auto-generated fleets. Every hull the engine
			#    seeded from the port-population define is removed; the water-forces
			#    below are then the ONLY navy the 1815 Qing fields.
			if = {
				limit = { any_navy = { always = yes } }
				every_navy = {
					destroy_unit = yes
				}
				LOG_line = { sys = QING  msg = "auto-generated Qing fleets disbanded (ahistorical 1815 blue-water navy removed)" }
			}

			# 2) 廣東水師 Guangdong Water-Force — Canton estuary, the empire's busiest
			#    maritime frontier (Canton System trade + Pearl River anti-piracy).
			#    p:9298 is Guangzhou (also the Liangguang garrison seat).
			if = {
				limit = { exists = p:9298  p:9298 = { owner = c:CHI  is_coastal = yes } }
				create_unit = {
					navy = yes
					name = "Guangdong Water-Force 廣東水師"
					location = p:9298
					sub_unit = brig
					while = { count = 5  add_subunit = brig }
				}
				LOG_line = { sys = QING  msg = "Guangdong Water-Force raised at Canton (p:9298), 6x war-junk" }
			}
			else = {
				LOG_fail = { sys = QING  fn = "SE_qing_navy"  reason = "p:9298 Canton not held/coastal; Guangdong Water-Force not raised" }
			}

			# 3) 福建水師 Fujian Water-Force — Taiwan Strait command (Fuzhou + Straits
			#    marine, the Green Standard marine squadrons). p:3651 is Fuzhou.
			if = {
				limit = { exists = p:3651  p:3651 = { owner = c:CHI  is_coastal = yes } }
				create_unit = {
					navy = yes
					name = "Fujian Water-Force 福建水師"
					location = p:3651
					sub_unit = brig
					while = { count = 3  add_subunit = brig }
				}
				LOG_line = { sys = QING  msg = "Fujian Water-Force raised at Fuzhou (p:3651), 4x war-junk" }
			}
			else = {
				# Fuzhou may be non-coastal in this map's province graph; fall back to
				# the largest coastal province owned, so Fujian still gets a squadron.
				if = {
					limit = { any_owned_province = { is_coastal = yes } }
					ordered_owned_province = {
						limit = { is_coastal = yes }
						order_by = total_population
						max = 1
						save_scope_as = qing_navy_fallback_port
					}
					create_unit = {
						navy = yes
						name = "Fujian Water-Force 福建水師"
						location = scope:qing_navy_fallback_port
						sub_unit = brig
						while = { count = 3  add_subunit = brig }
					}
					LOG_line = { sys = QING  msg = "Fujian Water-Force raised at fallback coastal port (Fuzhou not coastal), 4x war-junk" }
				}
				else = {
					LOG_fail = { sys = QING  fn = "SE_qing_navy"  reason = "no coastal province owned; Fujian Water-Force not raised" }
				}
			}

			# 4) [#191 task #191] 浙江水師 Zhejiang Water-Force — the Ningbo / Zhenhai
			#    (鎮海) command guarding the Zhejiang coast and the Yangzi approaches,
			#    the third principal provincial water-force (清史稿·兵志). Smallest of the
			#    three and, like them, a coastal-defence junk squadron outclassed by a
			#    single European steam frigate. p:2893 is Ningbo.
			if = {
				limit = { exists = p:2893  p:2893 = { owner = c:CHI  is_coastal = yes } }
				create_unit = {
					navy = yes
					name = "Zhejiang Water-Force 浙江水師"
					location = p:2893
					sub_unit = brig
					while = { count = 2  add_subunit = brig }
				}
				LOG_line = { sys = QING  msg = "Zhejiang Water-Force raised at Ningbo (p:2893), 3x war-junk" }
			}
			else = {
				LOG_fail = { sys = QING  fn = "SE_qing_navy"  reason = "p:2893 Ningbo not held/coastal; Zhejiang Water-Force not raised" }
			}
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "SE_qing_navy"  reason = "c:CHI absent or navy setup already done (sentinel set); no Qing water-force raised" }
	}
	LOG_exit = { sys = QING  fn = "SE_qing_navy"  result = OK }
}

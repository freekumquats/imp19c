# imp19c_effects_legion_setup.txt — historical starting-army setup
#
# create_unit / raise_legion follow the engine idiom verified against the
# feasibility oracles (Invictus, Terra-Indomita): every create_unit declares a
# base `sub_unit = <type>`, adds strength with `add_subunit = <type>`.
#
# TWO unit-creation idioms are used in this file, deliberately:
#   1. The QING GARRISON helpers (SE_qing_raise_garrison[_cmd]) use a BARE create_unit
#      (navy = no) issued directly in the province OWNER's country scope, with an
#      ABSOLUTE `location = <province>` token and an inline `commander = <char>` field.
#      [#352 F4] This is REQUIRED because raise_legion enforces ONE LEGION PER
#      GOVERNORSHIP: all ~26 garrisons raise from the same CHI capital governorship, so
#      only the first (Beijing) succeeds and the rest throw "Governorship already has
#      legion" (5080× in error.log — the boot test saw exactly ONE army). A bare
#      create_unit has NO such limit and honours the absolute location, so every garrison
#      spawns dispersed at its own province. PROVEN: Terra-Indomita galatian_invasion.txt
#      :1839/1455 (create_unit = { navy = no  location = p:X  commander = scope:Y }) and
#      Invictus invasions.txt:221/239 (two land units, one country scope, one tick).
#   2. The 1815 European coalition legions below (SE_occupation_of_france etc.) still
#      use raise_legion + `char:X = { add_to_legion = PREV }` + random_legion_unit =
#      { set_as_commander = char:X }, since each coalition power raises from its OWN
#      capital governorship (one legion each — within the per-governorship limit).
#
# Prefix = SE

# QING DUAL STANDING ARMY (#66) — the two permanent, separately-administered,
# ethnically-defined institutions of the 1815 Qing military, meeting only at the
# throne (see memory imp19c-qing-army-1815-research):
#
#   八旗 Eight Banners — the hereditary conquest-elite caste + strategic reserve.
#     ~half its soldiers (禁旅八旗) garrison BEIJING; the rest hold ~20 walled
#     "Manchu cities" 滿城 (mancheng) — Xi'an (largest interior), the Manchuria
#     homeland, the Xinjiang/NW frontier (Ili = seat of the 伊犁將軍), and interior
#     choke-points. Prestigious, but by 1815 militarily soft (banner-poverty
#     crisis 旗人生計, lost 國語騎射). We express that softness through the existing
#     qing_banner_decay bands (qing_banner_decay_mild / _severe country modifiers),
#     NOT by shrinking the paper rolls — hence garrisons are raised at full paper
#     size and the decay modifier carries the combat penalty ("big on paper,
#     brittle in field").
#
#   綠營 Green Standard — the Han professional TERRITORIAL/CONSTABULARY army,
#     deliberately fragmented (anti-warlordism) and dispersed across every
#     province for policing, escort, coast/canal patrol and first-line
#     suppression. Modelled as several mediocre provincial stacks under the great
#     regional commands (Zhili, Liangjiang, Huguang, Sichuan, Liangguang,
#     Shaan-Gan, Fujian incl. Taiwan, Yunnan-Guizhou).
#
# IMPLEMENTATION NOTE: [#352 F4] each garrison is created with a BARE create_unit
# (navy = no) in the province OWNER's country scope, LOCATED at its garrison province
# via an absolute `location = $prov$` token — NOT via raise_legion. raise_legion caps
# ONE legion per governorship, so raising all ~26 from the shared CHI capital
# governorship left only the first (Beijing) and errored the rest; a bare create_unit
# has no such cap and disperses correctly. Each raise is individually guarded on
# province existence + Qing (or CHI-subject) ownership via the SE_qing_raise_garrison
# [_cmd] helpers, so a partitioned or shrunken Qing simply gets fewer garrisons (each
# miss logs LOG_fail) instead of a scope error. count = $size$ in the while-loop is the
# number of ADDITIONAL regular_infantry companies on top of the 1 base sub_unit
# (so total = size + 1).
########################################################################

# ----------------------------------------------------------------------
# Helper: raise a garrison legion of ($size$ + 1) regular_infantry companies at
# province $prov$, named $name$, with NO scripted commander (the player assigns
# one). Guards province existence + Qing ownership; creates the unit (bare create_unit,
# no raise_legion) in the owner's scope located at $prov$. Silent LOG_fail if $prov$ is not
# held. Assumes c:CHI exists (the caller SE_qing_armies guards that once).
# ----------------------------------------------------------------------
SE_qing_raise_garrison = {
	if = {
		# [#190-fix task #190] Accept BOTH CHI-owned land AND land held by a CHI
		# SUBJECT. The 1815 frontier banner garrisons (Ili 伊犁, Ürümqi 烏魯木齊,
		# Mukden 盛京, Heilongjiang 黑龍江, Kashgar 喀什噶爾) sit in provinces owned by
		# CHI's autonomous-governorship subjects (ILI/MKD/HLJ, XNG under ILI), not by
		# CHI directly — historically these WERE Qing frontier commands. The old guard
		# (owner = c:CHI only) silently dropped exactly these 5 garrisons via LOG_fail.
		# is_subject_of is a proven trigger (common/military_traditions/00_indian.txt;
		# Invictus parthia_interactions.txt). The unit is created by $prov$'s OWNER, so
		# a subject-held frontier province raises under THAT subject (the intended
		# "dispersed across China AND its subjects" behaviour).
		limit = { exists = $prov$  $prov$ = { OR = { owner = c:CHI  owner = { is_subject_of = c:CHI } } } }
		# [#352 F4 — ROOT CAUSE, log-confirmed + the user's own data point] The R2-2/3/4 attempt #6
		# wrapped each garrison in `capital_scope.governorship = { raise_legion = { create_unit } }`.
		# raise_legion enforces ONE LEGION PER GOVERNORSHIP: all ~26 garrisons raise from the SAME CHI
		# capital governorship, so the FIRST (Beijing) succeeds and the other 25 throw
		# "Governorship already has legion" (5080× in error.log) — the boot test saw exactly ONE army.
		# The PRE-R2 code used bare create_unit (no legion limit → all 22 spawned) but stacked them at
		# Beijing because it entered the country scope THROUGH the province (`$prov$ = { owner = { } }`),
		# which re-resolved `location` to the owner's capital.
		# FIX (both bugs at once): the PROVEN land-army idiom is a BARE create_unit — NO raise_legion, so
		# NO governorship limit — issued directly in the OWNER's country scope, with an ABSOLUTE
		# `location = $prov$` token. Verified in Terra-Indomita galatian_invasion.txt:1839
		# (`create_unit = { navy = no  location = p:384  commander = scope:X  while = {...} }`, bare in the
		# country scope) and Invictus invasions.txt:221/239 (two land units, one country scope, one tick,
		# no raise_legion). $prov$ stays the absolute location token. Do NOT interpolate
		# $name$/$prov$/$size$ into a LOG string (breaks the arg compiler) — keep the log line static.
		# [BT-15 — supersedes the old owner-scope create] The earlier revision created in the province
		# OWNER's scope, so subject-owned frontier provinces spawned garrisons UNDER the subject's control
		# (the user's "subject-owned garrisons under subject control" report). The Son of Heaven garrisons
		# the marches directly, so we now ALWAYS create in c:CHI (below).
		# [BT-15] Issue the create_unit in the c:CHI scope ALWAYS (the Son of Heaven garrisons the
		# marches). The absolute location = $prov$ token resolves globally regardless of the creating
		# scope (proven TI galatian_invasion + se_QING_ILI raising at non-CHI frontier provinces), so a
		# subject-owned frontier province now yields a CHI-CONTROLLED garrison standing on that province
		# — fixing the "subject-owned garrisons under subject control" report.
		# [BT-15 review-fix] Issue create_unit BARE — SE_qing_armies runs in the c:CHI country_event
		# scope (qing_force_setup.1), so `this` is ALREADY c:CHI and a bare create_unit yields a
		# CHI-CONTROLLED garrison. The prior `c:CHI = { ... }` wrapper was a REDUNDANT re-entry — and
		# the navy fix below (~line 408) proved that re-entry RE-RESOLVES the absolute `location` to the
		# inland capital (Beijing), the recurring #224 all-garrisons-on-Beijing blocker. Bare-in-CHI
		# keeps CHI control without that re-resolution (matches header idiom #1 + the bare navy squadrons).
		create_unit = {
			navy = no
			name = $name$
			location = $prov$
			sub_unit = regular_infantry
			while = { count = $size$  add_subunit = regular_infantry }
		}
		LOG_line = { sys = QING  msg = "Qing garrison raised via bare create_unit at target province (CHI-controlled, no governorship limit)" }
	}
	else = {
		LOG_fail = { sys = QING  fn = "SE_qing_raise_garrison"  reason = "province not held by Qing; garrison not raised" }
	}
}

# ----------------------------------------------------------------------
# Helper: as above, but attaches historical commander $cmd$ (a char scope) to the
# legion. Guards that $cmd$ exists too (a dead/absent character just falls through
# to LOG_fail rather than erroring). Commander attach follows the verified idiom:
# $cmd$ = { add_to_legion = PREV } then random_legion_unit = { set_as_commander }.
# ----------------------------------------------------------------------
SE_qing_raise_garrison_cmd = {
	if = {
		# [#190-fix task #190] Same subject-land acceptance as SE_qing_raise_garrison
		# above (Ili banner garrison under commander Changling char:244 is precisely
		# one of the frontier cases this restores). See that helper for the rationale.
		limit = { exists = $prov$  $prov$ = { OR = { owner = c:CHI  owner = { is_subject_of = c:CHI } } }  exists = $cmd$ }
			# [#352 F4] Same bare-create_unit fix as SE_qing_raise_garrison (no raise_legion -> no
			# governorship limit -> every garrison spawns, dispersed at its own $prov$). Commander attach
			# uses the PROVEN TWO-STEP: create_unit { save_scope_as = U ... } then scope:U = { set_as_commander
			# = $cmd$ }. The boot test showed the INLINE `commander =` field rejected a stat-less setup officer
			# ('Failed to assign leader ( 596 ): EligibleForCommandOnly', error.log 4423873), whereas the
			# save-scope + set_as_commander form attached all 14 land commanders that same tick (debug.log).
			# PROVEN: Invictus startup_events.txt:133 (random_navy = { set_as_commander = char:253 }) and
			# egyptian_revolt_events.txt:176 (scope:thebes_unit = { set_as_commander = scope:thebes_general }).
			# Attach only when $cmd$ is a non-ruler EMPLOYED BY the province owner (a CHI commander cannot lead
			# a subject-owned frontier legion — e.g. Ili raises under the ILI subject); else muster commanderless.
			# [BT-15, review-corrected] Create a CHI-CONTROLLED garrison via a BARE create_unit in the
			# c:CHI scope this helper already runs in (NOT a c:CHI={} re-entry wrapper — see the review-fix
			# note in the branch below). Guard the commander on employer = c:CHI: a CHI-owned legion is led
			# by a CHI officer. PASSES for the 3 frontier _cmd garrisons (Ili/Ürümqi/Kashgar) whose CHI
			# commanders had failed the old employer = scope:garrison_owner guard. location = $prov$ keeps them placed.
			if = {
				limit = { $cmd$ = { is_alive = yes  is_ruler = no  employer = c:CHI } }
				# [BT-15 review-fix] BARE create_unit (already in c:CHI scope; see SE_qing_raise_garrison).
				create_unit = {
					navy = no
					name = $name$
					location = $prov$
					sub_unit = regular_infantry
					while = { count = $size$  add_subunit = regular_infantry }
					save_scope_as = qing_new_garrison
				}
				# proven two-step commander attach (see comment above): set the historical commander on the
				# just-created unit scope. Works for stat-less setup officers where inline `commander =` fails.
				scope:qing_new_garrison = { set_as_commander = $cmd$ }
				# [#331 user] mark this char as a serving Qing garrison/army commander so the
				# lifecycle event (QING_council_on_commander_death -> qing_office.32) can detect
				# that a member of the officer corps has died and prompt the War Minister for a
				# replacement recommendation. Stamped ONLY on the successfully-attached path.
				$cmd$ = { set_variable = { name = qing_officer_marker  value = 1 } }
				LOG_line = { sys = QING  msg = "Qing garrison raised under attached historical commander (save-scope set_as_commander)" }
			}
			else = {
				# [BT-15 review-fix] BARE create_unit (already in c:CHI scope; see SE_qing_raise_garrison).
				create_unit = {
					navy = no
					name = $name$
					location = $prov$
					sub_unit = regular_infantry
					while = { count = $size$  add_subunit = regular_infantry }
				}
				LOG_line = { sys = QING  msg = "Qing garrison raised commanderless (attached commander not eligible; player assigns)" }
			}
	}
	else = {
		LOG_fail = { sys = QING  fn = "SE_qing_raise_garrison_cmd"  reason = "province unheld or commander absent; garrison not raised" }
	}
}

# proven in Invictus 01_carthaginian_missions_08_naval.txt:152-157), then raise

# ----------------------------------------------------------------------
# [#223-fix task #223] NAVY SPAWN — WHY ONLY ONE FLEET APPEARED, AND THE FIX.
#
# THE BUG (user boot-test): only the Fujian squadron ever spawned. Root cause
# (oracle-confirmed): multiple `create_unit navy=yes` calls issued SEQUENTIALLY in
# a single scope on the SAME on_game_initialized tick collide — only one navy
# survives. Every PROVEN-WORKING multi-fleet setup in this mod raises each fleet on
# a DISTINCT tick: the four Self-Strengthening fleets (se_QING_SELFSTR.txt) each
# spawn from a different mission's on_completion, never in one batch.
#
# [BT-15 CORRECTION 2026-07-11] The prior note here ("is_coastal is NOT the culprit;
# p:9298/p:3651/p:2893 read coastal fine") was WRONG. error.log proved the surviving
# fleets were culled by a create_unit REJECTION: "Trying to create naval unit where it
# does not belong: PROVINCE,7190 Zhuhai / 2968 Xiangshan". is_coastal = yes is TRUE for
# river/lake-adjacent provinces and does NOT guarantee a province borders a real SEA
# ZONE, so the engine refused to berth a navy there. The ONLY proven sea-port predicate
# is is_port = yes (province trigger, common/unit_abilities/naval_raiding.txt:67). All
# the is_coastal berth checks below are now is_port so the fallback auto-relocates any
# river-port primary to a genuine sea port. Fuzhou p:3651 (the one that always survived)
# is_port; Zhuhai/Xiangshan/Canton are not.
#
# THE FIX (final, #314 B22): the timing changes (#223 stagger, #242 one-tick) did NOT
# work — the user still saw only the Fujian fleet. The real culprit was the PLACEMENT
# idiom: the country-scope `create_unit { navy=yes location=p:X }` form silently spawned
# only one of the three fleets. Oracle consult (2026-07-10) found the reliable multi-navy
# idiom is the SAME province->owner->create_unit form that fixed the land garrisons, but
# with an ABSOLUTE PROVINCE / SAVED-SCOPE location token (NOT the relative `prev.prev`).
# `prev.prev` at the owner scope resolves to ROOT's capital (Beijing) for ALL three fleets;
# the engine cannot berth three navies at one inland capital, so only one survived (Fujian).
# Each squadron below now locates at its OWN coastal home port (p:9298 / p:3651 / p:2893, or
# its fallback scope) — the Invictus galatian_invasion.txt:718 absolute-token idiom, which
# applies identically to navies. All three squadron effects use it (main + fallback). They
# still raise together on one tick (.11) after the .10 disband; same-tick multi-navy is
# proven safe (Terra-Indomita me_jomon.txt:1779 spawns 12 fleets on one tick).
# ----------------------------------------------------------------------

# Disband the engine's ahistorical auto-generated blue-water fleets (seeded from
# STARTING_SHIPS_PER_PORT_POPULATION). destroy_unit is DESTRUCTIVE, so the global
# sentinel guarantees this runs exactly once. Called first, on its own tick, BEFORE
# the three squadron raises — so the water-forces below are the only 1815 Qing navy.
SE_qing_navy_disband = {
	LOG_enter = { sys = QING  fn = "SE_qing_navy_disband" }
	if = {
		limit = {
			exists = c:CHI
			NOT = { has_global_variable = qing_navy_setup_done }
		}
		set_global_variable = qing_navy_setup_done
		c:CHI = {
			if = {
				limit = { any_navy = { always = yes } }
				every_navy = { destroy_unit = yes }
				LOG_line = { sys = QING  msg = "auto-generated Qing fleets disbanded (ahistorical 1815 blue-water navy removed)" }
			}
		}
	}
	else = {
		LOG_fail = { sys = QING  fn = "SE_qing_navy_disband"  reason = "c:CHI absent or navy setup already done (sentinel set)" }
	}
	LOG_exit = { sys = QING  fn = "SE_qing_navy_disband"  result = OK }
}

# 廣東水師 Guangdong Water-Force — Canton estuary, the empire's busiest maritime
# frontier (Canton System trade + Pearl River anti-piracy). p:9298 is Guangzhou.
# Raised on its OWN tick (see #223-fix note above). Raise/fallback logic verbatim
# from the pre-fix code. Called in country scope (ROOT = c:CHI).
SE_qing_navy_guangdong = {
	LOG_enter = { sys = QING  fn = "SE_qing_navy_guangdong" }
	if = {
		# [#352 F4-navy] Canton p:9298 is a RIVER port (its port sea-zone 10520 is in default.map's
		# river_provinces LIST, not sea_zones) -> create_unit rejected it 'not coastal' and the fallback
		# picked a non-naval province ('does not belong', error.log 4423870). Berth the Guangdong
		# water-force at Zhuhai p:7190 — the Pearl estuary MOUTH (sea zone 10323, a real sea), the
		# historical Humen/Bogue anti-pirate station. CHI-owned in setup/provinces/00_Guangdong.txt.
		limit = { exists = p:7190  p:7190 = { owner = c:CHI  is_port = yes } }
		# [R2-4 — B22 fix] Issue create_unit BARE in the CHI country scope (this effect is called from a
		# CHI country_event, so ROOT/this = c:CHI already). The old `c:CHI = { create_unit ... }` wrapper
		# RE-ENTERED the country scope, which re-resolved the absolute `location` token to the capital
		# (Beijing — inland), so all three squadrons berthed inland and the engine kept only one (Fujian,
		# via its coastal fallback) — the boot-test symptom. Oracle-proven: TI rel_flavor_kemetic.7 issues
		# navy create_unit BARE in the country scope (`create_unit = { navy = yes location = 2501 }`), and
		# me_tamil_king.txt uses an inline `commander =` on a navy create_unit — both work.
		create_unit = {
			navy = yes
			name = QING_UNIT_GUANGDONG_COASTAL_PATROL
			location = p:7190
			sub_unit = brig
			while = { count = 5  add_subunit = brig }
			save_scope_as = qing_new_gd_navy
		}
		# [#352 F4-navy] proven two-step commander attach (Invictus startup_events.txt:133
		# random_navy = { set_as_commander = char:253 }): set the historical 水師提督 on the new fleet
		# scope. The inline `commander =` field rejected this stat-less setup officer in the boot test.
		# [#352 F4-navy review] guard the attach: only seat the admiral if he exists, is a
		# living non-ruler in CHI service (the 1763 roster would be dead at an 1815 start —
		# death_date was stripped for the setup-attach, so the age system may kill them in the
		# 31-day gap before this fires). Else the fleet musters commanderless (player assigns).
		if = {
			limit = { exists = char:597  char:597 = { is_alive = yes  is_ruler = no  employer = c:CHI } }
			scope:qing_new_gd_navy = { set_as_commander = char:597 }  # Huang Shijian 黃仕簡
			char:597 = { set_variable = { name = qing_officer_marker  value = 1 } }
		}
		LOG_line = { sys = QING  msg = "Guangdong Coastal Patrol raised at Zhuhai/Humen (p:7190) under Huang Shijian, 6x war-junk" }
	}
	else = {
		# [#212-fix] fallback: raise at the largest owned coastal province, excluding
		# the other two water-forces' home ports (3651 Fuzhou / 2968 Xiangshan).
		if = {
			limit = { any_owned_province = { is_port = yes  NOT = { province_id = 3651 }  NOT = { province_id = 2968 } } }
			ordered_owned_province = {
				limit = { is_port = yes  NOT = { province_id = 3651 }  NOT = { province_id = 2968 } }
				order_by = total_population
				max = 1
				save_scope_as = qing_gd_fallback_port
			}
			# [#314 B22-fix] province->owner->create_unit { location = <scope> } form (see main branch).
			# [R2-4] create_unit issued BARE in the CHI country scope (ROOT=c:CHI here); the old c:CHI={ } wrapper re-entered the country scope and re-resolved location to the capital (Beijing, inland) so only Fujian survived. Oracle: TI kemetic.7 navy create_unit is bare in country scope.
			create_unit = {
				navy = yes
				name = QING_UNIT_GUANGDONG_COASTAL_PATROL
				location = scope:qing_gd_fallback_port
				sub_unit = brig
				while = { count = 5  add_subunit = brig }
			}
			LOG_line = { sys = QING  msg = "Guangdong Coastal Patrol raised at fallback sea port (Zhuhai p:7190 unavailable), 6x war-junk" }
		}
		else = {
			LOG_fail = { sys = QING  fn = "SE_qing_navy_guangdong"  reason = "p:7190 Zhuhai not held/coastal and no fallback sea port; Guangdong Coastal Patrol not raised" }
		}
	}
	LOG_exit = { sys = QING  fn = "SE_qing_navy_guangdong"  result = OK }
}

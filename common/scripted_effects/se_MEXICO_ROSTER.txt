########################################################################
# se_MEXICO_ROSTER.txt — THE HISTORICAL CAST OF THE MEXICAN REPUBLIC
# (task #96, concrete layer). The named men of Mexico's unstable half-century
# step onto the map as REAL on-map characters, keyed to the beat / band that is
# their moment: Iturbide crowns the first empire, Santa Anna loses the north,
# Juarez raises the Reforma, Miramon fights the Reform War, Zaragoza wins at
# Puebla, and Maximilian takes the throne the French plant. This is the
# CONCRETE-OVER-ABSTRACT layer (standing rule): the instability counters are the
# summary, but every crisis hangs a real, named caudillo, reformer or general off
# the map — the direct twin of se_USA_ROSTER.txt / se_QING_ROSTER.txt.
#
# ENGINE REALITY (identical to the USA roster): the mod's create_character takes
# no name argument and there is no set_first_name effect, so a conjured character
# gets a random, culture-appropriate GetName. We carry the historical identity
# two ways the engine DOES support:
#   (1) add_nickname = "<loc key>" pins the real name as his epithet, so he is
#       findable in the court list as e.g. Antonio "Santa Anna".
#   (2) the beat that spawns him already names him in its title/desc.
#
# Every spawn is guarded by a once-only ROOT flag (mex_spawned_*), so a figure
# summoned early by a band milestone is NOT summoned again by his dated beat (and
# vice versa) — the two drivers share one idempotent code path.
#
# create_character sets culture / religion / age explicitly (the oracle gotcha).
# Cultures verified in-repo: mexican (MEX primary), zapotec (Juarez, Oaxacan —
# historically correct, present in MEX poptype_rights + common/cultures/
# 00_mesoamerican.txt), austrian (Maximilian, common/cultures/00_german.txt).
# Trait keys verified against common/traits/ (orator, ambitious, aggressive,
# stubborn, proud, righteous, zealous, brave, tactician, disciplined, conqueror,
# honest, just, cruel, silver_tongued).
#
# AI-OWNED: called from the AI-autonomous Mexico arc (se_MEXICO / mex_instability
# events); no is_ai gate anywhere. All paths log sys = MEX. Prefix = MEX_spawn_.
#   ROOT / this = c:MEX throughout; the fresh character is saved as scope:mex_figure.
#
# Sources: Meyer/Sherman/Deeds, The Course of Mexican History; Krauze, Mexico:
#   Biography of Power (1997); Hamnett, Juarez (1994); Fowler, Santa Anna of
#   Mexico (2007); Ridley, Maximilian and Juarez (1992).
########################################################################

# ======================================================================
# FINALIZE — shared tail for every spawn. Runs in the new character's scope.
# Roots him in the Mexican court and pins his historical name. $nick$ = the
# NICKNAME_MEX_* loc key.
# ======================================================================
MEX_roster_finalize = {
	set_home_country = ROOT
	add_nickname = "$nick$"
	save_scope_as = mex_figure
	LOG_line = { sys = MEX  msg = "roster: ($nick$) entered the stage" }
}

# ---- AGUSTIN DE ITURBIDE — Emperor Agustin I (First Empire, 1822) ----
# The royalist officer who allied with the insurgents (Plan de Iguala) and crowned
# himself Emperor of a conservative, monarchist independence. The native precedent
# for a Mexican throne — the seed the French will echo forty years on.
MEX_spawn_iturbide = {
	if = {
		limit = { NOT = { has_variable = mex_spawned_iturbide } }
		set_variable = mex_spawned_iturbide
		create_character = {
			age = 38  culture = mexican  religion = catholic
			add_charisma = 7  add_finesse = 6  add_martial = 7  add_zeal = 5
			add_trait = ambitious  add_trait = proud  add_trait = brave
			MEX_roster_finalize = { nick = "NICKNAME_MEX_ITURBIDE" }
		}
	}
	else = { LOG_fail = { sys = MEX  fn = "MEX_spawn_iturbide"  reason = "Iturbide already on the stage for" } }
}

########################################################################
# se_LOG.txt — Diagnostic logging module
#
# Purpose: emit richly-tagged lines to logs/debug.log so a play session run
#          with the -debug_mode launch option produces a diagnostic trace that
#          can be handed back for analysis. The debug_log effect writes to
#          <userdir>/Paradox Interactive/Imperator/logs/debug.log (NOT game.log).
#          Engine errors land in logs/error.log automatically; logical failures
#          are surfaced here in debug.log with a loud tag.
#
# IMPORTANT: debug_log ONLY outputs when the game is launched in -debug_mode.
#            All lines below are silently dropped in a normal session, so this
#            module is zero-cost in ordinary play.
#
# All lines share the "[IMP19C]" prefix so the whole trace can be extracted:
#     grep "\[IMP19C\]" debug.log > imp19c_trace.log
# Per-system lines add a second tag, e.g. "[IMP19C][SUBJ]", so a single
# subsystem can be isolated:
#     grep "\[IMP19C\]\[SUBJ\]" debug.log > subj_trace.log
#
# Prefix = LOG
########################################################################

# ---- Core line writer -------------------------------------------------
# One tagged message. $sys$ = subsystem tag (SUBJ, TRADE, CURR, ...),
# $msg$ = free text. Pass numbers/vars pre-formatted into $msg$ by the caller.
LOG_line = {
	# Function: write a single tagged diagnostic line to logs/debug.log
	# Scope:    any
	debug_log = "[IMP19C][$sys$] $msg$"
}

# ---- Call tracing -----------------------------------------------------
# Mark entry into a code path. $fn$ = effect / event name being entered.
LOG_enter = {
	# Function: trace that a code path began executing
	# Scope:    any
	debug_log = "[IMP19C][$sys$] ENTER $fn$"
}

# Mark exit, recording the outcome. $result$ = OK / FAIL / SKIP / <detail>.
LOG_exit = {
	# Function: trace that a code path finished, with its result
	# Scope:    any
	debug_log = "[IMP19C][$sys$] EXIT  $fn$ result=$result$"
}

# ---- Game-state capture ----------------------------------------------
# Dump the full scope stack (ROOT, this, prev, and all saved/named scopes
# with their values) plus a header, so the log records exactly what state
# the code was operating on. Use this at any point a failure needs context.
LOG_state = {
	# Function: header line + full scope-stack dump for game-state capture
	# Scope:    any
	debug_log = "[IMP19C][$sys$] STATE $note$ (scope dump follows)"
	debug_log_scopes = yes
}

# ---- Failure surfacing -----------------------------------------------
# Loud, greppable failure line. $fn$ = where it failed, $reason$ = why.
# Pairs a FAIL tag with a scope dump so a silent no-op becomes a diagnosable
# event. Extract just failures with: grep "\[IMP19C\]\[FAIL\]" debug.log
LOG_fail = {
	# Function: surface a logical failure that would otherwise be a silent no-op
	# Scope:    any
	debug_log = "[IMP19C][FAIL][$sys$] $fn$ : $reason$"
	debug_log_scopes = yes
}

# ---- Guarded assertion -----------------------------------------------
# Evaluate $check$-style logic at the call site instead (script has no
# generic assert), then call LOG_ok or LOG_fail. This effect just logs a
# passed check for symmetry when tracing a validated precondition.
LOG_ok = {
	# Function: record that a precondition/branch passed as expected
	# Scope:    any
	debug_log = "[IMP19C][$sys$] OK    $fn$ : $detail$"
}

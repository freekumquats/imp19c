########################################################################
# se_QING_AMBAN.txt — the AMBAN / RESIDENT lifecycle (#113).
#
# The 駐紮大臣 (amban, resident commissioner) was the Qing throne's man on the
# ground in Inner Asian dependencies: Lhasa, Urga (Khüree, modern Ulaanbaatar),
# Kashgar, Ili, and the other frontier posts. The office of LIFAN YUAN (理藩院,
# Court of Colonial Affairs) governed relations with Mongolia, Tibet, Xinjiang
# and the Inner Asian march — the amban was its instrument, posted to a subject
# to oversee the dependency on behalf of the throne.
#
# The lifecycle: QING_amban_post creates a resident character at CHI, moves him
# to the subject country, grants him the qing_amban_resident modifier (duration
# -1), and stores the link on the OVERLORD (CHI) keyed by the subject. Every
# quarter QING_amban_evaluate scores each posted resident (a) against the Lifan
# Yuan holder via QING_pair_friction — a clash between resident and the Peking
# supervisor leads to drift (subject loyalty/liberty_desire) and a recall crisis;
# (b) against the throne via QING_char_affinity — a capable, loyal resident
# keeps the dependency firmly in hand. A VACANT Lifan Yuan office = the residents
# go unsupervised (drift). Teardown via QING_amban_recall.
#
# TARGETING: applies to Inner Asian dependencies — the subject types most apt
# for a resident are vassal_tribe (tribal vassals), sinosphere_tributary (the
# Ryukyu + Korea tier), protectorate, and feudatory. The engine allows gating
# on is_subject_type OR on being ANY subject at all (gated on is_subject = yes).
# We use the broader ANY-subject gate (safer, proven) with a culture/region
# filter: the resident is posted to dependencies where Manchu/Mongol/Tibetan
# cultures OR bodish/mongolic culture_groups are present (the Inner Asian march).
#
# DOES NOT double-count the Lifan Yuan accountability metric (se_QING_ACCOUNTABILITY
# line 74 already judges the office on subject loyalty aggregate). This layer adds
# CHARACTER-level interplay: a capable+loyal resident + capable Lifan holder =
# dependency held firm; a weak/clashing resident = drift.
#
# CHI-only, player-only. Called from QING_GOV_pulse quarterly. se_LOG throughout.
########################################################################

# ======================================================================
# POST AN AMBAN (駐紮大臣). $subject$ = the subject country scope to which we
# post the resident. Creates a character at CHI, moves him to the subject,
# applies the qing_amban_resident modifier (OUTSIDE the create_character block,
# per the #90 boot-crash gotcha), and stores the link on CHI keyed by the
# subject's tag. Country scope (ROOT = CHI).
# ======================================================================
QING_amban_post = {
	LOG_enter = { sys = QING  fn = "QING_amban_post" }
	# The resident is a minor character, adult, Manchu or Han (Inner Asian post
	# went to conquest-elite or sinicized officials), middling stats (not a
	# great statesman — the great offices kept the best men in Peking).
	create_character = {
		age = { 35 50 }
		culture = manchu          # [logfix] was ROOT.primary_culture (invalid create_character field); amban = Manchu conquest-elite
		religion = vajrayana      # [logfix] was ROOT.religion
		add_finesse = { 3 6 }
		add_martial = { 2 5 }
		add_charisma = { 3 6 }
		add_zeal = { 2 5 }
		save_scope_as = qing_amban_new
	}
	# [#392 refactor] the post-wiring (move to subject, role modifier, affinity, storage,
	# marker, tenure timer) is now the shared QING_amban_wire tail so the game-start seed
	# (QING_amban_seed_one) and this runtime poster stay behaviourally identical — no drift.
	QING_amban_wire = { subject = $subject$ }
	LOG_line = { sys = QING  msg = "amban: posted to for" }
	LOG_exit = { sys = QING  fn = "QING_amban_post"  result = OK }
}

# ======================================================================
# [#392] SHARED WIRING TAIL. Runs in ROOT (CHI) with scope:qing_amban_new already set
# to the freshly-created resident. $subject$ = the subject country scope. Moves the
# resident into the subject, applies the qing_amban_resident modifier (OUTSIDE
# create_character per the #90 boot-crash gotcha), scores his throne-affinity, stores
# the fixed-name link on the subject, and stamps the death-cleanup marker + 8-year
# tenure timer. Extracted verbatim from the old QING_amban_post body so both the
# runtime poster and the game-start seed share ONE definition.
# ======================================================================
QING_amban_wire = {
	# move him to the subject, apply the role modifier, store the link
	scope:qing_amban_new = {
		# [#157 fix] `minor_character = yes` inside create_character is an INVALID key
		# ("Unknown effect" — silently discarded), so residents were being spawned as
		# FULL major characters (office/succession-eligible, list clutter). The proven
		# way to mark a created character minor is the set_as_minor_character effect in
		# the character scope (events/annexation.txt:21). A resident is a minor courtier.
		set_as_minor_character = THIS
		move_country = $subject$
		add_character_modifier = { name = qing_amban_resident  duration = -1 }
		# score his compatibility with the throne (the Lifan holder will be
		# scored separately at evaluate time)
		ROOT = { save_scope_as = qing_overlord_for_affinity }
		scope:qing_overlord_for_affinity = {
			scope:qing_amban_new = { QING_char_affinity = yes }
		}
	}
	# [#113 integ-fix] STORAGE. The agent's original stored the link on CHI via a
	# variable NAME built from tag interpolation (name = "qing_amban_$subject$.GetTag")
	# — an UNPROVEN idiom (nothing else in the repo constructs a var name from a tag;
	# it can silently no-op). Replaced with the PROVEN shape: a FIXED-name variable
	# stored on the SUBJECT country itself (each dependency holds at most one resident),
	# so there is no dynamic name and no list-indexing. Guarded reads elsewhere use the
	# same fixed name in the subject's own scope.
	$subject$ = {
		set_variable = { name = qing_amban_here  value = scope:qing_amban_new }
	}
	# [#113 review-fix] stamp a marker on the resident himself so on_character_death
	# can recognise a dying amban and clear the subject's qing_amban_here (mirrors the
	# qing_office_held marker the office model keys death-cleanup off). While posted the
	# resident's employer IS the subject, so the death hook reaches the right country.
	# Without this, an amban who died of natural causes stranded qing_amban_here forever:
	# evaluate skipped the dead ref and post_sweep saw the var still present, so the
	# dependency was never re-staffed.
	scope:qing_amban_new = {
		set_variable = { name = qing_amban_marker  value = 1 }
		# [deferred-fix #2] TENURE TIMER. qing_amban.2 (gone native) originally gated on
		# `time_in_office >= 2920` — an UNPROVEN character token (absent from Terra-Indomita,
		# Invictus, and the rest of the repo; it silently failed the trigger so .2 could
		# never fire even once dispatched). Replaced with the PROVEN expiring-variable idiom
		# already used in this feature (qing_amban.3's qing_amban_unrest_suppressed_recently
		# value=1 days=730): stamp an 8-year (2920-day) timer at post time. While the var
		# EXISTS the resident is short-tenured; once the engine auto-expires it he has served
		# >= 8 years, so qing_amban.2 fires on `NOT = { has_variable = qing_amban_tenure_pending }`.
		set_variable = { name = qing_amban_tenure_pending  value = 1  days = 2920 }
	}
}

# ======================================================================
# RECALL AN AMBAN. $subject$ = the subject country scope, $reason$ = log text.
# Removes the stored var, strips the modifier, and either moves the character
# back to CHI or retires him (death). Country scope (ROOT = CHI).
# ======================================================================
QING_amban_recall = {
	LOG_enter = { sys = QING  fn = "QING_amban_recall" }
	# [#113 integ-fix] retrieve the stored resident from the SUBJECT's own fixed-name
	# var (qing_amban_here); guard on var-present + is_alive + employer match.
	if = {
		limit = {
			$subject$ = {
				has_variable = qing_amban_here
				var:qing_amban_here = { is_alive = yes  employer = $subject$ }
			}
		}
		$subject$.var:qing_amban_here = {
			remove_character_modifier = qing_amban_resident
			# [#113 review-fix] drop the amban marker: the recalled man rejoins the court
			# pool and is no longer a resident (so a later death does not run the amban
			# death-cleanup for a subject he no longer serves).
			if = { limit = { has_variable = qing_amban_marker }  remove_variable = qing_amban_marker }
			# [deferred-fix #2] drop the tenure timer too: a recalled man who is later
			# re-posted (or posted elsewhere) gets a fresh 8-year clock, not a stale one.
			if = { limit = { has_variable = qing_amban_tenure_pending }  remove_variable = qing_amban_tenure_pending }
			# recall: move back to CHI (he rejoins the court pool)
			move_country = ROOT
			save_scope_as = qing_amban_recalled
		}
		LOG_line = { sys = QING  msg = "amban: recalled from ($reason$) for" }
	}
	$subject$ = { remove_variable = qing_amban_here }
	LOG_exit = { sys = QING  fn = "QING_amban_recall"  result = OK }
}

# ======================================================================
# POST SWEEP — quarterly. Post a resident to any Inner Asian dependency (a
# subject whose ruler is of a mongolic/bodish culture group, i.e. Mongolia /
# Tibet / the Inner Asian march) that lacks one, PROVIDED the Lifan Yuan office
# is filled (a vacant 理藩院 has no man to post). This is what makes the coupling
# LIVE — the lifecycle events (clash / gone-native / reward / turnover) then run
# off the posted residents. Bounded: at most ONE posting per pulse (a random
# unattended dependency), so the court staffs the march gradually. Country scope
# (ROOT = CHI). [#113 integ-fix: the feature had NO posting entry point — only
# turnover re-posted — so it was inert; this sweep makes it live.]
# ======================================================================
QING_amban_post_sweep = {
	LOG_enter = { sys = QING  fn = "QING_amban_post_sweep" }
	# only post if the Lifan Yuan supervisor is in place
	if = {
		limit = {
			has_variable = qing_office_lifanyuan_holder
			var:qing_office_lifanyuan_holder = { is_alive = yes  employer = ROOT }
		}
		# find ONE Inner Asian dependency with no resident and post there
		random_subject = {
			limit = {
				NOT = { has_variable = qing_amban_here }
				# [#163] MANUAL OVERRIDE: skip any dependency the player is managing by hand
				# (qing_amban_manual flag, set from the amban panel in the subject's diplomatic
				# view). The auto-sweep never staffs, replaces, or fills a vacancy the player has
				# claimed — the manual layer owns those posts. Subjects the player never touches
				# keep the gradual auto-staffing exactly as before.
				NOT = { has_variable = qing_amban_manual }
				# [#392] SHARED warrant gate (was an inline culture-group predicate here) — now the
				# single QING_amban_warrants_resident_trigger the panel roster also reads, so a
				# posted resident is guaranteed to render (warrant == postable). This also brings the
				# autonomous_governorships ILI/ULS (伊犁/烏里雅蘇台將軍, Manchu-ruled) into scope, the
				# canonical resident-general posts the old culture-only test silently skipped.
				QING_amban_warrants_resident_trigger = yes
			}
			save_scope_as = qing_amban_post_target
			ROOT = { QING_amban_post = { subject = scope:qing_amban_post_target } }
			LOG_line = { sys = QING  msg = "amban: post-sweep staffed for" }
		}
	}
	LOG_exit = { sys = QING  fn = "QING_amban_post_sweep"  result = OK }
}

# ======================================================================
# EVALUATE ALL RESIDENTS — quarterly pulse. For each subject that has a posted
# amban, score (a) the resident vs the Lifan Yuan holder (QING_pair_friction —
# high friction = they clash, the dependency slips: subject loyalty/liberty_desire
# drifts), (b) the resident vs the throne (QING_char_affinity — a capable, loyal
# resident keeps it firm). A VACANT Lifan Yuan office = no supervision, residents
# drift. Country scope (ROOT = CHI).
# ======================================================================
QING_amban_evaluate = {
	LOG_enter = { sys = QING  fn = "QING_amban_evaluate" }
	# check if the Lifan Yuan office is filled (the supervisor)
	if = {
		limit = {
			has_variable = qing_office_lifanyuan_holder
			var:qing_office_lifanyuan_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_lifanyuan_holder = { save_scope_as = qing_lifan_holder }
	}
	# [#113 integ-fix] iterate every subject; for each with a posted amban (stored in
	# the subject's OWN fixed-name var qing_amban_here), evaluate. No tag interpolation.
	every_subject = {
		limit = {
			# guard: the amban must be alive and still in this subject's employ
			has_variable = qing_amban_here
			var:qing_amban_here = { is_alive = yes  employer = prev }
		}
		# save the subject scope + its resident, then do the scoring work in ROOT (CHI)
		save_scope_as = qing_amban_subject
		var:qing_amban_here = { save_scope_as = qing_amban_current }
		ROOT = {
			# (a) score resident vs Lifan holder (if holder exists)
			if = {
				limit = { exists = scope:qing_lifan_holder }
				QING_pair_friction = { a = scope:qing_lifan_holder  b = scope:qing_amban_current }
				if = {
					limit = { var:qing_pair_friction >= 50 }
					# CLASH: the resident and the Lifan holder disagree — the dependency
					# slips. Subject's opinion of the overlord worsens. Log it; a severe
					# clash may queue a recall event (not auto-fired here; the event
					# system handles it).
					scope:qing_amban_subject = {
						add_opinion = { modifier = qing_amban_clash_opinion  target = ROOT }
					}
					LOG_line = { sys = QING  msg = "amban: CLASH vs Lifan holder (friction) in , dependency slips for" }
					# queue a recall crisis at high friction
					if = {
						# [court-slot] qing_amban.1 is a MODAL Lifan Yuan (理藩院) court event fired inside every_subject —
						# several subjects clashing in one pulse would queue several modals AND dogpile the other court
						# beats. Share the quarterly court-event slot (qing_gc_event_slot_used, reset monthly): the FIRST
						# clash this pulse takes the slot, the rest hold until a later month. Claim only on the fire path.
						limit = {
							var:qing_pair_friction >= 65
							NOT = { has_variable = qing_gc_event_slot_used }  # [court-slot] share the court-event throttle
						}
						scope:qing_amban_subject = { save_scope_as = qing_amban_crisis_subject }
						scope:qing_amban_current = { save_scope_as = qing_amban_crisis_resident }
						scope:qing_lifan_holder = { save_scope_as = qing_amban_crisis_holder }
						trigger_event = { id = qing_amban.1  days = { 5 15 } }
						set_variable = { name = qing_gc_event_slot_used  value = 1 }  # [court-slot] claim on the fire path
					}
				}
				else = {
					# COOPERATION: resident and holder agree — dependency held firm.
					# Subject's opinion of the overlord improves.
					scope:qing_amban_subject = {
						add_opinion = { modifier = qing_amban_cooperation_opinion  target = ROOT }
					}
					LOG_line = { sys = QING  msg = "amban: cooperation + Lifan holder in , dependency firm for" }
				}
			}
			else = {
				# NO LIFAN HOLDER: the residents go unsupervised — drift.
				# Subject's opinion of the overlord worsens slightly.
				scope:qing_amban_subject = {
					add_opinion = { modifier = qing_amban_ineffective_opinion  target = ROOT }
				}
				LOG_line = { sys = QING  msg = "amban: in unsupervised (vacant Lifan Yuan), drift for" }
			}
			# (b) score resident vs throne (affinity — a loyal, capable resident is
			# a pillar; a disloyal one is a risk). Already scored at post time, but
			# re-score quarterly as the character's loyalty/stats shift.
			scope:qing_amban_current = { QING_char_affinity = yes }
			if = {
				limit = { scope:qing_amban_current = { var:qing_char_affinity < 40 } }
				# LOW AFFINITY: the resident is estranged from the throne — a risk.
				# Dependency slips. Subject's opinion of the overlord worsens.
				scope:qing_amban_subject = {
					add_opinion = { modifier = qing_amban_ineffective_opinion  target = ROOT }
				}
				LOG_line = { sys = QING  msg = "amban: in disloyal to throne (affinity), risk for" }
			}
			else_if = {
				limit = { scope:qing_amban_current = { var:qing_char_affinity >= 70 } }
				# HIGH AFFINITY: the resident is loyal and capable — a pillar.
				# Dependency held firmer still. Subject's opinion of the overlord improves.
				scope:qing_amban_subject = {
					add_opinion = { modifier = qing_amban_capable_opinion  target = ROOT }
				}
				LOG_line = { sys = QING  msg = "amban: in loyal+capable (affinity), dependency firm for" }
			}
			# [#163] MANUAL-OVERRIDE GATE: for a dependency the player is managing by hand
			# (qing_amban_manual on the subject, set from the amban panel in its diplomatic
			# view), suppress the AUTOMATIC lifecycle-event dispatch below (turnover / gone-
			# native / recall) — the player owns posting and recall for this post. The
			# friction/affinity SCORING and opinion effects ABOVE still run, so a poorly
			# chosen resident keeps its quarterly consequences; only auto-turnover is frozen.
			if = {
				limit = { NOT = { scope:qing_amban_subject = { has_variable = qing_amban_manual } } }
				# [deferred-fix #2] DISPATCH THE LIFECYCLE EVENTS. qing_amban.2 (gone native),
				# .3 (able resident), and .4 (turnover) were fully authored but had ZERO
				# trigger_event callers anywhere — dead content; only .1 (clash) ever fired.
				# Each carries its own defensive trigger + random_subject re-pick, so this
				# dispatch only needs to nudge them; we roll a low per-quarter chance gated on
				# the same conditions each event validates. Mutually exclusive priority
				# (turnover > gone-native > able) so at most one fires per resident per quarter.
				# .4 TURNOVER: the resident is old/unhealthy (a replacement crisis).
				if = {
					limit = {
						scope:qing_amban_current = {
							OR = { age >= 65  health < 0.4 }
						}
					}
					random = {
						chance = 15
						ROOT = {
							trigger_event = { id = qing_amban.4  days = { 10 40 } }
							LOG_line = { sys = QING  msg = "amban: turnover event queued (resident old/unwell) in for" }
						}
					}
				}
				# .2 GONE NATIVE: long-tenured (tenure timer expired) + estranged (affinity < 40).
				else_if = {
					limit = {
						scope:qing_amban_current = {
							NOT = { has_variable = qing_amban_tenure_pending }
							var:qing_char_affinity < 40
						}
					}
					random = {
						chance = 15
						ROOT = {
							trigger_event = { id = qing_amban.2  days = { 10 40 } }
							LOG_line = { sys = QING  msg = "amban: gone-native event queued (resident long-tenured + estranged) in for" }
						}
					}
				}
				# .3 ABLE RESIDENT: loyal+capable (affinity >= 70) and the dependency has unrest to quell.
				else_if = {
					limit = {
						scope:qing_amban_current = { var:qing_char_affinity >= 70 }
						scope:qing_amban_subject = {
							any_owned_province = { count >= 1  province_unrest >= 2 }
						}
					}
					random = {
						chance = 20
						ROOT = {
							trigger_event = { id = qing_amban.3  days = { 10 40 } }
							LOG_line = { sys = QING  msg = "amban: able-resident event queued (resident quells unrest) in for" }
						}
					}
				}
			}
		}
	}
	LOG_exit = { sys = QING  fn = "QING_amban_evaluate"  result = OK }
}

# ======================================================================
# [#392] GAME-START SEED — post the HISTORICAL 1763 resident-generals into the
# Inner-Asian frontier commands so the Lifan Yuan Directorate opens with its real
# roster (the amban system was live in 1763; an empty panel misrepresented that).
#
# QING_amban_seed_one — creates ONE named historical resident and wires him via the
# shared QING_amban_wire tail (identical to a runtime posting). Country scope (ROOT =
# CHI). Params:
#   $subject$ = the subject-country scope to post to (must be a live CHI subject).
#   $nick$    = the NICKNAME_QING_AMBAN_* loc key pinning his historical name.
#   $culture$ = his culture (manchu for the banner generals, mongolian for a Khalkha).
#   $fin$/$mar$/$cha$/$zea$ = fixed skill BONUSES (historical competence, not random,
#     so the roster is reproducible — the runtime poster keeps its random spread).
# religion = vajrayana matches the Inner-Asian (Tibetan-Buddhist) posting context, as
# the runtime poster uses. Guarded by the caller (only posts where none is present).
# ======================================================================
QING_amban_seed_one = {
	LOG_enter = { sys = QING  fn = "QING_amban_seed_one" }
	create_character = {
		age = { 40 55 }
		culture = $culture$
		religion = vajrayana
		add_finesse = $fin$
		add_martial = $mar$
		add_charisma = $cha$
		add_zeal = $zea$
		save_scope_as = qing_amban_new
	}
	# pin the historical name as an epithet (add_nickname takes the bare loc key —
	# the proven upstream form; NEVER interpolate $nick$ into a quoted LOG string).
	scope:qing_amban_new = { add_nickname = $nick$ }
	# shared wiring tail (move to subject, modifier, affinity, storage, marker, timer)
	QING_amban_wire = { subject = $subject$ }
	LOG_line = { sys = QING  msg = "amban: game-start historical resident seeded for" }
	LOG_exit = { sys = QING  fn = "QING_amban_seed_one"  result = OK }
}

# ======================================================================
# [#392] QING_amban_seed_gamestart — called ONCE from imp19c_qing_on_game_initialized
# (CHI only). Posts the three historical 1763 residents to the DIRECT CHI frontier
# subjects the Lifan Yuan panel can render (every_subject reaches direct subjects only;
# the nested XNG/MGA are unreachable, so seeding them would strand invisible chars):
#   • TIB 西藏  → 輔鼐 Funai        (Manchu; amban in Lhasa 1761-64; middling administrator)
#   • ILI 伊犁  → 明瑞 Mingrui      (Manchu; first 伊犁將軍 from 1762; distinguished general)
#   • ULS 烏里雅蘇台 → 成衮札布 Chenggünjab (Khalkha Mongol; 定邊左副將軍 1757-71; able, long-tenured)
# Each post is idempotent + guarded: only if the subject EXISTS, is a live CHI subject,
# has no resident yet (never clobbers a post the player/sweep already made), AND passes
# QING_amban_warrants_resident_trigger — the SAME gate the panel roster filters on, so we
# never seed a resident the panel could not render (defense-in-depth: keeps the seed honest
# even if a subject's generated ruler/subject-type ever shifts off the warrant set). Runs at
# init BEFORE QING_ministry_recompute_all_perf so the opening Lifan Yuan meter/roster
# already sees them. Country scope (ROOT = CHI).
# ======================================================================
QING_amban_seed_gamestart = {
	LOG_enter = { sys = QING  fn = "QING_amban_seed_gamestart" }
	# TIB — 輔鼐 Funai (Manchu amban, competent administrator)
	if = {
		limit = {
			exists = c:TIB
			c:TIB = { is_subject_of = ROOT  NOT = { has_variable = qing_amban_here }  QING_amban_warrants_resident_trigger = yes }
		}
		c:TIB = { save_scope_as = qing_amban_seed_target }
		QING_amban_seed_one = {
			subject = scope:qing_amban_seed_target
			nick = NICKNAME_QING_AMBAN_FUNAI
			culture = manchu
			fin = 4  mar = 2  cha = 4  zea = 3
		}
	}
	# ILI — 明瑞 Mingrui (Manchu banner general, distinguished commander)
	if = {
		limit = {
			exists = c:ILI
			c:ILI = { is_subject_of = ROOT  NOT = { has_variable = qing_amban_here }  QING_amban_warrants_resident_trigger = yes }
		}
		c:ILI = { save_scope_as = qing_amban_seed_target }
		QING_amban_seed_one = {
			subject = scope:qing_amban_seed_target
			nick = NICKNAME_QING_AMBAN_MINGRUI
			culture = manchu
			fin = 3  mar = 6  cha = 4  zea = 3
		}
	}

# se_QING_SUBPOSTS.txt — [#77] AUTO-STAFFING THE MINISTRY SUB-POSTS.
#
# Beneath the thirteen Grand Council offices (se_QING_COUNCIL.txt, filled by
# QING_council_autofill) sit the appointable CORPS the ministry panels manage:
#   • the Zongli Yamen's diplomats           (駐外使臣, marker qing_zongli_diplomat)
#   • the Censorate's investigating censors   (御史,   marker qing_is_censor_inspector)
#   • the Imperial Guard's captains           (侍衛,   marker qing_is_imperial_guardsman)
# Historically these posts were STAFFED BY THE STATE, not left for a sovereign to
# fill by hand — a censor bench sat, envoys were posted, guardsmen enrolled. Yet in
# the mod they started EMPTY and filled only when the player clicked each row. This
# file gives them the SAME game-start seed + quarterly vacancy-refill the ambans have
# (QING_amban_seed_gamestart / QING_amban_post_sweep): the throne staffs them without
# the player's intervention, and the player keeps the panel Recall/Discharge levers to
# replace an officeholder at will.
#
# 1:1 RULE (user): a character holds AT MOST ONE post. Every fill draws only a
# QING_subpost_eligible_candidate (qing_dynasty_triggers.txt) — an employed adult who is
# NOT the ruler/heir, NOT a governor/general/admiral, NOT on the Grand Council, and NOT
# already in ANY other sub-post/study/amban/eunuch role. So no man is ever double-booked.
#
# CANDIDATE SUPPLY (user): a fill prefers a FREE eligible courtier (the ablest, by
# combined_stats_council_svalue, exam laureates floating to the top). When the court has
# no free man to spare, it MINTS one through the examination process — QING_exam_mint_scholar
# — framed as an EXTRAORDINARY PROVINCIAL STUDENT who performed above expectations (marker
# qing_is_above_exp_student, counted in qing_above_exp_student_count for the Ministry of Rites
# panel). So the roster is always staffable; the exam is the source, exactly as for civil
# office (#34) and the ambans (#392).
#
# Scope: COUNTRY (ROOT = CHI) throughout. se_LOG sys = QING. Prefix = QING_subpost_.
# No modifiers granted inside create_character (#90 rule — the mint helper obeys it).
########################################################################

# ======================================================================
# QING_subpost_fill_one_minted — [BT-diplomat fix 2026-07-22] staff ONE seat by ALWAYS MINTING a
# BRAND-NEW character (never drawing an existing courtier). Country scope (ROOT = CHI).
# Params: $marker$, $count$ (as QING_subpost_fill_one).
# WHY THIS EXISTS: the "diplomats are also Grand Council ministers" double-booking bug was tried and
# FIXED REPEATEDLY (diplomat=commander strip + R2, qing_officer_marker BT-7 + R2, shared-picker
# eligibility guards, roster-rebuild strip, quarterly un-double-book sweep) and KEPT RETURNING every
# boot test — an existing courtier drawn into the corps could always later take (or already hold) an
# office through some ordering the strip missed. The user's decision: fill diplomat seats with
# BRAND-NEW characters, GUARANTEED to hold no other office. This effect skips branch (A) entirely and
# always runs the mint path — a fresh exam scholar who exists only to serve, so double-booking is
# structurally impossible. (Same mint mechanism QING_subpost_fill_one branch B uses; #90-safe — the
# corps marker is stamped in a FOLLOW-UP scope, never inside create_character.)
QING_subpost_fill_one_minted = {
	# [BT-appointment-redesign review-fix 2026-07-22] capture the COUNTRY from `this` (CHI in every current
	# caller: the day-32 seed event + the country-scope QING_GOV_pulse sweep) and use it for set_home_country
	# below, NOT ROOT — the GC-office twin (QING_council_autofill_office) had a real bug where ROOT was a
	# character on the runtime backfill path; this helper is currently only reached in country-ROOT contexts,
	# but saving `this` makes it path-agnostic and matches the council helper, so a future vacate-style caller
	# can't reintroduce the character-to-country-arg mismatch.
	save_scope_as = qing_subpost_country
	# [BT-officeholder 2026-07-22] CREATE a purpose-built official for the seat and assign him — like the
	# GC-office autofill (QING_council_autofill_office). Do NOT route through the exam POOL: the old
	# version called QING_exam_mint_scholar, which adds the new man to qing_scholar_pool as an eligible
	# FREE body — so the council autofill (or another corps) could then grab him, and he double-booked
	# (a minted diplomat seated as Minister of War — the reported regression). Creating him directly and
	# adding him to NO pool, then stamping only his corps marker, means
	# nothing can ever pick him for a second post. #90-safe: markers stamped in a FOLLOW-UP scope, never
	# inside create_character. Age/stats mirror the exam-scholar profile (a fresh juren-grade official).
	# $degree$ = the exam degree the post requires (juren/jinshi for civil corps, wu_jinshi for the
	# martial Guard). Granted via add_trait INSIDE create_character — the #90-safe placement (a trait
	# set during creation, not granted to a just-made char afterward), and this whole path is deferred
	# to a runtime event so it is off the construction crash path regardless.
	create_character = {
		age = 30
		culture = han
		religion = confucianism
		add_finesse = 7
		add_charisma = 5
		add_martial = 1
		add_zeal = 4
		add_trait = $degree$
		save_scope_as = qing_subpost_new
	}
	scope:qing_subpost_new = {
		set_home_country = scope:qing_subpost_country
		set_variable = { name = $marker$  value = 1 }
	}
	change_variable = { name = $count$  add = 1 }
	LOG_line = { sys = QING  msg = "subpost: created a BRAND-NEW official (holds no other office) for" }
}

# ======================================================================
# QING_subpost_staff_corps_minted — [BT-diplomat fix 2026-07-22] fill a corps up to $target$ using
# ONLY freshly-minted characters (QING_subpost_fill_one_minted). Country scope (ROOT = CHI).
# Params: $marker$, $count$, $target$. Recomputes the live count first (dropping any member who has
# since taken an office/command — same disqualifier set as QING_subpost_staff_corps), then mints new
# men for the remaining vacancies. Used for the DIPLOMAT corps so its members never double-book.
# ======================================================================
QING_subpost_staff_corps_minted = {
	# LIVE recount: rebuild $count$ from the marker, stripping anyone now disqualified (took an
	# office/command/governorship/harem/officer role since being posted). Mirror of
	# QING_subpost_staff_corps's recount pass.
	set_variable = { name = $count$  value = 0 }
	every_character = {
		limit = {
			employer = ROOT
			is_alive = yes
			has_variable = $marker$
		}
		if = {
			limit = { OR = { is_general = yes  is_admiral = yes  is_governor = yes  has_variable = qing_office_held  has_variable = qing_is_harem_consort  has_variable = qing_officer_marker } }
			remove_variable = $marker$
		}
		else = {
			ROOT = { change_variable = { name = $count$  add = 1 } }
		}
	}
	# [Design-2 chunk 2.5] resolve the LEGISLATED establishment target into a temp var, then mint up to
	# it. The Ministry Establishment law (qing_ministry_estab_law) sets qing_law_ministry_estab_target
	# to 3 (lean) / 4 (standard, the default + game-start seed) / 6 (expanded). We copy it into a local
	# temp so the comparison RHS can be a named script_value (a var-ref on a comparison RHS is illegal),
	# and DEFAULT to the caller's $target$ when the law var is not yet set — so this is byte-identical to
	# the old literal-4 behaviour if the law/seed hasn't run, and never reads an unset var on the RHS.
	if = {
		limit = { has_variable = qing_law_ministry_estab_target }
		set_variable = { name = qing_estab_target_tmp  value = var:qing_law_ministry_estab_target }
	}
	else = { set_variable = { name = qing_estab_target_tmp  value = $target$ } }
	# mint new men up to the legislated establishment — one per guarded rung. 6 rungs so the EXPANDED tier
	# (6) is reachable; lean/standard simply stop earlier. $degree$ threads through so each created official
	# gets the civil/martial degree its post requires.
	if = { limit = { var:$count$ < qing_estab_target_tmp_cmpsvalue }  QING_subpost_fill_one_minted = { marker = $marker$  count = $count$  degree = $degree$ } }
	if = { limit = { var:$count$ < qing_estab_target_tmp_cmpsvalue }  QING_subpost_fill_one_minted = { marker = $marker$  count = $count$  degree = $degree$ } }
	if = { limit = { var:$count$ < qing_estab_target_tmp_cmpsvalue }  QING_subpost_fill_one_minted = { marker = $marker$  count = $count$  degree = $degree$ } }
	if = { limit = { var:$count$ < qing_estab_target_tmp_cmpsvalue }  QING_subpost_fill_one_minted = { marker = $marker$  count = $count$  degree = $degree$ } }
	if = { limit = { var:$count$ < qing_estab_target_tmp_cmpsvalue }  QING_subpost_fill_one_minted = { marker = $marker$  count = $count$  degree = $degree$ } }
	if = { limit = { var:$count$ < qing_estab_target_tmp_cmpsvalue }  QING_subpost_fill_one_minted = { marker = $marker$  count = $count$  degree = $degree$ } }
}

# ======================================================================
# QING_subpost_strip_double_booked — the un-double-booking pass (STRIP ONLY, no fill).
# Params: $marker$, $count$. Rebuilds $count$ from the marker, removing it from any member
# who has since taken a role that breaks the 1:1 court-position rule (army/navy commander,
# provincial governor, a Grand Council great office, garrison field officer, or — legacy —
# a harem consort), and leaving the count reflecting the survivors. The quarterly sweep runs
# this UNGATED (a headless ministry must not keep a commander on its rolls), then does the
# minted fill-to-target separately, gated on a serving director. Country scope (ROOT = CHI).
# ======================================================================
QING_subpost_strip_double_booked = {
	# [diplomat=commander fix R2 2026-07-20] STRIP-ONLY reconcile: remove the corps marker from any
	# member who has taken a role that breaks the 1:1 court-position rule (army/navy commander,
	# provincial governor, a Grand Council great office, or — legacy — a harem consort), and leave the
	# count reflecting the survivors. This is the un-double-booking half of QING_subpost_staff_corps,
	# split out so the quarterly sweep can run it UNGATED (the fill half stays gated on a serving
	# director). Country scope (ROOT = CHI). Params: $marker$, $count$. No fill, so it never conjures
	# staff for a headless ministry — it only ever REMOVES a clash. Same merged single-pass shape as
	# QING_subpost_staff_corps's recount (PERF-AUDIT M1): recount from scratch, dropping the disqualified.
	set_variable = { name = $count$  value = 0 }
	every_character = {
		limit = {
			employer = ROOT
			is_alive = yes
			has_variable = $marker$
		}
		if = {
			# [BT-7 fix 2026-07-22] ADDED has_variable = qing_officer_marker — a garrison/army field officer
			# (the #331 set_as_commander marker) is not reliably caught by is_general (e.g. Tsedan 612, a
			# Green-Standard garrison officer under an amban, leaked into the Diplomat Corps). Matches the
			# shared gate QING_office_eligible_candidate + the zongli rebuild disqualifier set.
			limit = { OR = { is_general = yes  is_admiral = yes  is_governor = yes  has_variable = qing_office_held  has_variable = qing_is_harem_consort  has_variable = qing_officer_marker } }
			remove_variable = $marker$
		}
		else = {
			ROOT = { change_variable = { name = $count$  add = 1 } }
		}
	}
}

# ======================================================================
# QING_subpost_seed_gamestart — [#77] staff the three ministry corps at game start
# (called once from imp19c_qing_on_game_initialized, CHI only, BEFORE the ministry perf
# fold so the panels open with a manned roster). Also inits the above-expectations
# student counter. Country scope (ROOT = CHI). Targets: a working bench of 4 each (the
# panel caps are 6; 4 is the "healthy corps" baseline the perf meters score against).
# ======================================================================
QING_subpost_seed_gamestart = {
	LOG_enter = { sys = QING  fn = "QING_subpost_seed_gamestart" }
	# the exam bookkeeping (qing_scholar_pool_count etc.) must exist before a fill can mint
	# into the pool — this seed runs BEFORE QING_exam_init in the game-start order, so init it
	# here first. Idempotent (guards on qing_exam_initialized), so the later init call no-ops.
	QING_exam_init = yes
	if = { limit = { NOT = { has_variable = qing_above_exp_student_count } }  set_variable = { name = qing_above_exp_student_count  value = 0 } }
	# [Design-2 chunk 2.5] seed the ministry-establishment target to the standard 4 (the default law tier),
	# so QING_subpost_staff_corps_minted always reads a set value. The Ministry Establishment law overwrites
	# it on enactment (3 lean / 4 standard / 6 expanded). Idempotent: only seeds if not already set.
	if = { limit = { NOT = { has_variable = qing_law_ministry_estab_target } }  set_variable = { name = qing_law_ministry_estab_target  value = 4 } }
	# [BT-officeholder 2026-07-22] ALL THREE corps are staffed by CREATING brand-new officials (never
	# drawing existing courtiers) — the same model as the GC offices. Each created man is stamped its
	# corps marker and added to NO pool, so nothing can ever grab
	# him for a second post: double-booking (diplomats/censors as ministers/commanders) is structurally
	# impossible. Each corps gets the DEGREE its post requires: civil (juren) for the Zongli diplomats
	# and Censorate inspectors (civil-service posts), martial (wu_jinshi) for the Imperial Guard captains
	# (a martial corps). The degree is granted via add_trait inside QING_subpost_fill_one_minted — SAFE
	# because this whole seed is now DEFERRED to a day-32 runtime event (add_trait at construction is the
	# known boot-crash class), and runs AFTER any historical figures already hold their posts.
	QING_subpost_staff_corps_minted = { marker = qing_zongli_diplomat        count = qing_zongli_diplomat_count  target = 4  degree = juren }
	QING_subpost_staff_corps_minted = { marker = qing_is_censor_inspector    count = qing_censor_inspector_count  target = 4  degree = juren }
	QING_subpost_staff_corps_minted = { marker = qing_is_imperial_guardsman  count = qing_guard_corps_count       target = 4  degree = wu_jinshi }
	LOG_line = { sys = QING  msg = "subpost: game-start ministry corps staffed (all three corps minted fresh) for" }
	LOG_exit = { sys = QING  fn = "QING_subpost_seed_gamestart"  result = OK }
}

# ======================================================================
# QING_subpost_refill_sweep — [#77] quarterly vacancy refill (called from QING_GOV_pulse,
# beside QING_amban_post_sweep). Tops each corps back up to its establishment target so a
# member lost to death/departure/promotion is replaced without the player lifting a finger
# — the state staffs its own benches. The player's panel Recall/Discharge levers still let
# him drop a man; the next sweep simply refills the gap (so removing a specific officeholder
# to make room for a chosen replacement works: discharge, then appoint the man he wants, or
# let the sweep pick the ablest). Gated so it only runs while the relevant GC office is
# FILLED — a headless ministry does not conjure a staff (mirrors the amban sweep, which
# gates on the Lifan Yuan holder). Country scope (ROOT = CHI).
# ======================================================================
QING_subpost_refill_sweep = {
	if = {
		limit = { tag = CHI }
		# [diplomat=commander fix R2 2026-07-20] STRIP the double-booked FIRST, UNGATED — every quarter,
		# regardless of whether the office has a director. The reported "diplomats are also commanders"
		# clash RE-FORMS after game start whenever a sitting diplomat/censor/guardsman is later given an
		# army/navy command (by the player or the AI): the day-30/31 OOB reconcile cleans the initial
		# clash, but the ONLY ongoing reconcile was this sweep — and it was wholly gated on the office
		# being filled. At a 1763 start the Zongli Yamen (an 1861 institution) has NO director, so the
		# sweep never ran and the re-formed double-booking was never cleaned. The STRIP (un-double-book)
		# must be unconditional; only the FILL-to-target should wait for a director (a headless ministry
		# conjures no new staff, but it must not keep a commander on its rolls either).
		# QING_subpost_strip_double_booked = strip-only (the every_character marker-removal pass, no fill).
		QING_subpost_strip_double_booked = { marker = qing_zongli_diplomat        count = qing_zongli_diplomat_count }
		QING_subpost_strip_double_booked = { marker = qing_is_censor_inspector    count = qing_censor_inspector_count }
		QING_subpost_strip_double_booked = { marker = qing_is_imperial_guardsman  count = qing_guard_corps_count }
		# FILL to establishment — only while the relevant office has a serving director.
		# [BT-diplomat fix 2026-07-22] DIPLOMATS refill with BRAND-NEW MINTED characters
		# (QING_subpost_staff_corps_minted), never drawn from the existing court, so a refilled seat can
		# never be a sitting minister/commander (the recurring double-booking bug). Only while the Zongli
		# Yamen has a Director (a headless ministry conjures no new staff).
		if = {
			limit = {
				has_variable = qing_office_zongli_holder
				var:qing_office_zongli_holder = { is_alive = yes  employer = ROOT }
			}
			QING_subpost_staff_corps_minted = { marker = qing_zongli_diplomat  count = qing_zongli_diplomat_count  target = 4  degree = juren }
		}
		# investigating censors — only while the Censorate has a Grand Inspector.
		# [BT-officeholder 2026-07-22] refill with BRAND-NEW MINTED civil (juren) officials, never drawn from
		# the existing court/pool — the same rule as the diplomats and the GC offices: a vacated seat is filled
		# by a CREATED character who holds no other post, so a refilled censor can never be a sitting minister/
		# commander (the recurring double-booking bug) and nothing is ever drawn from the exam pool automatically.
		if = {
			limit = {
				has_variable = qing_office_censor_holder
				var:qing_office_censor_holder = { is_alive = yes  employer = ROOT }
			}
			QING_subpost_staff_corps_minted = { marker = qing_is_censor_inspector  count = qing_censor_inspector_count  target = 4  degree = juren }
		}
		# guard captains — only while the Guard has a Commandant.
		# [BT-officeholder 2026-07-22] refill with BRAND-NEW MINTED martial (wu_jinshi) officers, never drawn
		# from the existing court/pool — same rule as above; the Guard is a martial corps so its created men
		# carry the martial degree.
		if = {
			limit = {
				has_variable = qing_office_guard_commandant_holder
				var:qing_office_guard_commandant_holder = { is_alive = yes  employer = ROOT }
			}
			QING_subpost_staff_corps_minted = { marker = qing_is_imperial_guardsman  count = qing_guard_corps_count  target = 4  degree = wu_jinshi }
		}
	}
}

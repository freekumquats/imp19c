########################################################################
# se_QING_SECRETARIAT.txt — the Central Secretariat's working chancery (內閣 票擬),
# #361.
#
# The 內閣 (Central Secretariat) was the throne's edict-drafting chancery, headed by
# the Grand Secretary (大學士) — the SAME grand_secretariat office the Hanlin Academy
# (#358) feeds. The Hanlin is the TALENT RESERVOIR (the waiting jinshi corps); the 內閣
# is where those men actually WORK: turning the throne's will into drafted rescripts
# (票擬) and clearing the backlog that piles up when the chancery grinds (積壓).
#
# ONE OFFICE, ONE METER (house rule, mirrors the harem→household precedent #360): this
# is NOT a second Grand Council seat and gets NO perf meter of its own. The chancery's
# throughput folds into the EXISTING qing_min_perf_grand_secretariat (se_QING_MINISTRY.txt
# term (e) — no council edit needed, the grand_secretariat key is already enumerated in
# QING_council_fold_ministry_perf). A cleared chancery lifts the Grand Secretary's
# standing; a choked backlog drags it. The #273 qing_secretariat.* events (the Edict Mill,
# the Rescript Backlog, the Drafting Scandal) already model these beats as flavour; this
# gives the player a CONCRETE backlog to see + a lever to act on between them.
#
# STATE (CHI country scope):
#   qing_secretariat_backlog          int 0..100, rescripts awaiting drafting (票擬積壓)
#   qing_secretariat_initialized      flag (idempotent init guard, dedicated — not the
#                                     backlog var, which the perf recompute may touch)
#   qing_secretariat_convene_cooldown days: throttle the convene lever (~3 quarters)
#
# PREFIX: QING_secretariat_ / sys = QING. CHI-only. Init from on_game_initialized;
# drifts quarterly via QING_secretariat_pulse (se_QING_GOVERNANCE.txt).
########################################################################

# ======================================================================
# INIT — seed a moderate opening backlog. Idempotent. Called once at game start from
# on_game_initialized. Guard on a DEDICATED flag, not the backlog var: the perf recompute
# runs at game-start and reads the backlog (defaulting a bare var to 0), so guarding on
# the count would then wrongly skip seeding the opening backlog.
# ======================================================================
QING_secretariat_init = {
	LOG_enter = { sys = QING  fn = "QING_secretariat_init" }
	if = {
		limit = { NOT = { has_variable = qing_secretariat_initialized } }
		set_variable = { name = qing_secretariat_initialized  value = 1 }
		# a working chancery always carries some backlog — a moderate 30 (well within the
		# tolerable band, so a competent secretary keeps the meter neutral at start).
		set_variable = { name = qing_secretariat_backlog  value = 30 }
	}
	LOG_exit = { sys = QING  fn = "QING_secretariat_init"  result = OK }
}

# ======================================================================
# QING_secretariat_pulse — quarterly drift. Scope: COUNTRY (CHI). Rescripts ACCUMULATE
# each quarter (the throne's business never stops); a capable, present Grand Secretary
# works the backlog DOWN passively (his charisma = his throughput), while a vacant or
# weak chancery lets it climb. Net drift = fresh rescripts (+6/quarter) minus the
# secretary's throughput. Clamp 0..100. Called from the quarterly governance pulse AFTER
# QING_household_pulse / QING_harem_pulse.
# ======================================================================
QING_secretariat_pulse = {
	LOG_enter = { sys = QING  fn = "QING_secretariat_pulse" }
	QING_secretariat_init = yes
	# fresh rescripts always come up for drafting.
	change_variable = { name = qing_secretariat_backlog  add = 6 }
	if = {
		# a live, employed Grand Secretary works the backlog down by his throughput —
		# his charisma (the 內閣's governing skill), so an able hand (charisma 10) clears
		# ~10/quarter (net -4), a weak one (charisma 5) only ~5 (net +1, it creeps up).
		limit = {
			has_variable = qing_office_grand_secretariat_holder
			var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
		}
		var:qing_office_grand_secretariat_holder = { save_scope_as = qing_sec_holder }
		set_variable = { name = qing_sec_scratch  value = 0 }
		change_variable = { name = qing_sec_scratch  add = scope:qing_sec_holder.charisma }
		change_variable = { name = qing_secretariat_backlog  subtract = var:qing_sec_scratch }
		LOG_line = { sys = QING  msg = "secretariat: the chancery works the rescript backlog for" }
	}
	# a vacant chancery has no one to draft — the backlog climbs untended (already +6 above,
	# with no throughput subtracted). Log the drift.
	else = {
		LOG_line = { sys = QING  msg = "secretariat: an unmanned chancery lets the rescript backlog climb for" }
	}
	# clamp 0..100.
	if = { limit = { var:qing_secretariat_backlog < 0 }   set_variable = { name = qing_secretariat_backlog  value = 0 } }
	if = { limit = { var:qing_secretariat_backlog > 100 } set_variable = { name = qing_secretariat_backlog  value = 100 } }
	LOG_exit = { sys = QING  fn = "QING_secretariat_pulse"  result = OK }
}

# ======================================================================
# QING_secretariat_convene (票擬) — the lever: convene the chancery for a concentrated
# session of edict-drafting. Clears a large chunk of the backlog and, because a torrent
# of crisp rescripts marshals the realm's politics, generates political influence scaled
# by the Grand Secretary's charisma (echoing qing_secretariat.1, the Edict Mill). Scope:
# COUNTRY (CHI). Guarded on a present Grand Secretary + a backlog worth convening for +
# no cooldown; stamps a ~3-quarter cooldown (the engine auto-decrements day-typed vars).
# ======================================================================
QING_secretariat_convene = {
	LOG_enter = { sys = QING  fn = "QING_secretariat_convene" }
	if = {
		limit = {
			has_variable = qing_office_grand_secretariat_holder
			var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
			has_variable = qing_secretariat_backlog
			var:qing_secretariat_backlog >= 20
			NOT = { has_variable = qing_secretariat_convene_cooldown }
		}
		var:qing_office_grand_secretariat_holder = { save_scope_as = qing_sec_holder }
		# clear a solid chunk of the backlog (a concentrated drafting session).
		change_variable = { name = qing_secretariat_backlog  subtract = 30 }
		if = { limit = { var:qing_secretariat_backlog < 0 } set_variable = { name = qing_secretariat_backlog  value = 0 } }
		# the crisp rescripts marshal the realm's politics — influence scaled by the
		# secretary's charisma (an able hand gets more out of the session): charisma/2.
		set_variable = { name = qing_sec_scratch  value = 0 }
		change_variable = { name = qing_sec_scratch  add = scope:qing_sec_holder.charisma }
		change_variable = { name = qing_sec_scratch  divide = 2 }
		add_political_influence = var:qing_sec_scratch
		# a well-run drafting session steadies the chancery's own hand a touch.
		# [#361 review-fix] add_loyalty takes a NAMED loyalty-modifier key, not a raw scalar
		# (00_imp19c_loyalty.txt:35; a bare `= 2` silently no-ops). Use the existing commended key.
		scope:qing_sec_holder = { add_loyalty = loyalty_qing_commended }
		# throttle: ~3 quarters before the chancery can be convened in special session again.
		set_variable = { name = qing_secretariat_convene_cooldown  days = 270 }
		LOG_line = { sys = QING  msg = "secretariat: the chancery is convened to draft the backlog for" }
	}
	LOG_exit = { sys = QING  fn = "QING_secretariat_convene"  result = OK }
}

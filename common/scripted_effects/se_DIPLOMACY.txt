DIPLOMACY_cache_country_power_tradezone = {
	# Scope: Country
	# Function: Save the country's power in each tradezone, if above 100
	save_scope_as = country_scope
	# [#85 C3 perf] The base DIPLOMACY_power svalue (economy+military+tech+RECURSIVE
	# subjects+admin+stability) is tradezone-INDEPENDENT, yet this helper runs once
	# per (country, tradezone) and re-evaluated it all 22 times per country every
	# quarter. Read the once-per-country cache set by the caller
	# (DIPLOMACY_update_global_power_status pre-pass); fall back to a fresh compute
	# when the cache is absent (e.g. the DEBUG timetest harness calls this helper
	# without the pre-pass), so the value is identical for every caller. Only the
	# base term is hoisted — the TZ-specific multipliers below are unchanged.
	if = {
		limit = { has_variable = DIPLOMACY_power_base_cached }
		set_local_variable = { name = base_power  value = var:DIPLOMACY_power_base_cached }
	}
	else = {
		set_local_variable = { name = base_power  value = DIPLOMACY_power }
	}
	set_variable = {
		name = DIPLOMACY_power_$tradezone$
		value = {
			value = local_var:base_power
			multiply = var:TZ_penetration_$tradezone$
			# Bonus multiplier for home-turf, % of this country's GDP in the TZ
			# e.g. if this TZ contains 75% of the country's wealth, multiply the power rating by 1.75
			multiply = {
				value = 1
				add = {
					value = 0
					every_governorships = {
						#if = {
							limit = {
								capital_scope = {
									$tradezone$_tradezone = { PROVINCE = yes }
								}
							}
							add = {
								value = WEALTH_governorship_total
							}
						#}
					}
					min = 0.01
					divide = {
						value = var:WEALTH_total_private_moveable_wealth
						divide = 128
					}
				}

			}
		}
	}
}

DIPLOMACY_seed_feature_play_success = {
	# Scope: the INSTIGATOR country, called IMMEDIATELY after AI_begin_diplomatic_play
	# (which saves the freshly-created provobj as scope:diplomatic_play). Function:
	# give a MOD-FEATURE play a baseline success head-start so an aim that is already
	# justified at launch (a standing frontier claim #53, or years of demographic
	# pressure #57) is not near-certain to fizzle below the floor before its
	# success-moving events (agitator/settlers/...) have had a chance to fire.
	#
	# The seed is BELOW the fizzle floor, so the play must still develop to win — this
	# only removes the cold-start penalty an AI interest-play deservedly carries.
	# Guarded on the provobj existing (a launcher that aborted set no play), so a
	# no-op launch stays a no-op.
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = DIPLO  fn = "DIPLOMACY_seed_feature_play_success" }
	if = {
		limit = { exists = scope:diplomatic_play  scope:diplomatic_play = { has_variable = is_diplomatic_play } }
		scope:diplomatic_play = {
			set_variable = {
				name = diplomatic_play_success
				value = DIPLOMACY_play_success_seed_feature
			}
		}
		# THIS (not ROOT) is the instigator: this helper always runs in the instigator's
		# country scope, whereas at the migration call sites ROOT is still the pulsing
		# sender country. Log the actual instigator so the trace reads correctly.
		LOG_line = { sys = DIPLO  msg = "feature play seeded with baseline success for [This.GetTag]" }
	}
	else = {
		LOG_fail = { sys = DIPLO  fn = "DIPLOMACY_seed_feature_play_success"  reason = "no play provobj in scope (launch aborted); no baseline success seeded" }
	}
	LOG_exit = { sys = DIPLO  fn = "DIPLOMACY_seed_feature_play_success"  result = OK }
}

DIPLOMACY_complete_play = {
	# Scope: called with $diplomatic_play$ = the play provobj (a province tagged
	# is_diplomatic_play). Function: RESOLVE a play that has reached max progression
	# into a real gameplay outcome, then TEAR THE PLAY DOWN.
	#
	# CRITICAL: DIPLOMACY_diplomatic_play_update_status re-calls this EVERY monthly
	# tick while diplomatic_play_progression stays above the max (nothing lowers it),
	# so if we did not remove the play here the outcome would re-fire (re-transfer
	# territory) every month. The final AI_remove_diplomatic_play strips
	# is_diplomatic_play + all play_* vars, so the next update pass finds nothing.
	#
	# The outcome is scored on the play's accumulated diplomatic_play_success (0..100,
	# moved by the play events). Below the floor the play FIZZLES; at/above the
	# decisive line the goal is achieved in FULL; between, a PARTIAL outcome. This
	# means an AI interest-play that never developed through events (success stays
	# near 0) simply lapses — no territory changes hands on a play that went nowhere.
	$diplomatic_play$ = {
		save_scope_as = diplomatic_play

		# Guard: a well-formed play carries an instigator, a target area, a target
		# country and a goal. A malformed/partly-torn-down provobj is just removed.
		if = {
			limit = {
				has_variable = play_instigator
				has_variable = play_target_area
				has_variable = play_target_country
				has_variable = play_goal
			}
			var:play_instigator = { save_scope_as = play_instigator }
			var:play_target_country = { save_scope_as = play_target_country }
			var:play_target_area = { save_scope_as = play_target_area }

			LOG_line = { sys = DIPLO  msg = "play resolving: instigator [scope:play_instigator.GetTag] goal [scope:diplomatic_play.MakeScope.GetVariable('play_goal').GetValueAsFlagName] vs [scope:play_target_country.GetTag] over [scope:play_target_area.GetName], success [scope:diplomatic_play.MakeScope.GetVariable('diplomatic_play_success').GetValue|0]" }

			if = {
				# FIZZLE — the confrontation ran out of steam short of any result.
				# A play that went nowhere produces no popup (this is the common fate of
				# an AI interest-play that never developed through events); the trace line
				# records it. No state changes.
				limit = { var:diplomatic_play_success < DIPLOMACY_play_success_floor }
				LOG_line = { sys = DIPLO  msg = "play fizzled (success below floor): no outcome for [scope:play_instigator.GetTag]" }
			}
			else = {
				# a real outcome: branch on the goal.
				DIPLOMACY_trigger_diplomatic_play_finale_event = yes
			}
		}
		else = {
			LOG_fail = { sys = DIPLO  fn = "DIPLOMACY_complete_play"  reason = "play provobj missing one of instigator/target_area/target_country/goal at resolution; removed without an outcome" }
		}

		# TEAR DOWN — always, so the resolved play cannot re-fire next tick.
		AI_remove_diplomatic_play = yes
	}
}

DIPLOMACY_update_global_power_status = {
	# [#85 C3 perf] Base-power hoist. DIPLOMACY_cache_country_power_tradezone
	# multiplies the heavy, tradezone-INDEPENDENT DIPLOMACY_power svalue
	# (economy+military+tech+RECURSIVE subjects+admin+stability) by a per-TZ
	# penetration factor, once per (country, tradezone) — i.e. it re-evaluated the
	# whole base svalue 22× per country every quarter. Here we compute it ONCE per
	# country into DIPLOMACY_power_base_cached; the helper reads that cache and only
	# does the cheap per-TZ multiply. Set right before this country's TZ loop and
	# removed right after, so the variable never outlives the first pass and passes
	# 2-5 are byte-for-byte unchanged. Behaviour-identical: the helper falls back to
	# a fresh DIPLOMACY_power compute when the cache is absent (the DEBUG timetest
	# harness calls the helper without this pre-pass), so every caller sees the same
	# value; only the redundant recompute is removed.
	every_country = {
		set_variable = { name = DIPLOMACY_power_base_cached  value = DIPLOMACY_power }
		every_tradezone_complex = {
			APPLY = DIPLOMACY_cache_country_power_tradezone
		}
		remove_variable = DIPLOMACY_power_base_cached
	}
	every_country = {
		every_tradezone_complex = {
			APPLY = DIPLOMACY_add_cached_subject_power_tradezone
		}
	}
	every_country = {
		every_tradezone_complex = {
			APPLY = DIPLOMACY_finalise_cached_subject_power_tradezone
		}
	}
	every_tradezone_complex = {
		APPLY = DIPLOMACY_get_power_in_play_TZ
	}
	every_tradezone_complex = {
		APPLY = DIPLOMACY_get_top_players_tradezone
	}

	DIPLOMACY_get_top_global_powers = yes

}

DIPLOMACY_diplomatic_play_update_status = {
	# Scope: Province (diplomatic play)
	# Function: All observers check their allegiance in a diplomatic play, and the play's balance of power and attitude is recalculated
	# Called: On diplomatic play update on_action

	save_scope_as = diplomatic_play

	# Update the diplomatic play's age
	change_variable = {
		name = diplomatic_play_progression
		add = 1
	}
	# If the diplomatic play is progressed, push it to its conculsion instead
	if = {
		limit = {
			var:diplomatic_play_progression > DIPLOMACY_play_max_progression
		}
		DIPLOMACY_complete_play = {
			diplomatic_play = scope:diplomatic_play
		}
	}
	# Otherwise, update all values
	else = {
		if = {
			limit = {
				NOT = {
					has_variable = diplomatic_play_reevaluated_recently
				}
			}
			AI_diplomatic_play_do_support_change = yes
			AI_assess_adversary = { 
				diplomatic_play = scope:diplomatic_play
				AI_target = scope:diplomatic_play.var:play_target_country
			}
			AI_diplomatic_play_evaluate_attitude = {
				diplomatic_play = scope:diplomatic_play
			}
			AI_diplomatic_play_evaluate_war = {
				diplomatic_play = scope:diplomatic_play
			}
			set_variable = {
				name = diplomatic_play_reevaluated_recently
				days = 365
			}
		}
	}
	

}

DIPLOMACY_modify_play_success = {
	# Scope: Province (diplomatic play)
	# Function: Modify the success rating of a diplomatic play, to a maximum of 100
	if = {
		limit = {
			NOT = { $amt$ = 0 }
		}
		set_local_variable = {
			name = success_multiplier
			value = {
				# Get reciprocal of balance of power
				value = 1
				divide = {
					value = var:AI_play_power_balance_instigator_root
					min = 0.001
				}
				# Invert the reciprocal
				subtract = 1
				multiply = -1
				# Add to 1
				add = 1
				# audit-fix (#79, diplomacy): floor the multiplier at a positive value.
				# The formula is (2 - 1/balance); balance = instigator_power/target_power
				# is clamped high (max 90) in se_AI but NOT low, so a weaker instigator
				# (balance < 0.5) produced a ZERO or NEGATIVE multiplier — every positive
				# success event got nulled/reversed and the play could never succeed.
				# 0.25 keeps a weak instigator accruing success slowly (and bounds the
				# negative-event divide branch below to a x4 penalty rather than a blow-up).
				min = 0.25
			}
		}

		if = {
			limit = {
				$amt$ > 0
			}
			change_variable = {
				name = diplomatic_play_success
				add = {
					value = $amt$
					multiply = local_var:success_multiplier
				}
			}
			if = {
				limit = {
					var:diplomatic_play_success > 100
				}
				set_variable = {
					name = diplomatic_play_success
					value = 100
				}
			}
		}
		else = {
			change_variable = {
				name = diplomatic_play_success
				add = {
					value = $amt$
					divide = {
						value = local_var:success_multiplier
						min = 0.001
					}
				}
			}
			if = {
				limit = {
					var:diplomatic_play_success < 0
				}
				set_variable = {
					name = diplomatic_play_success
					value = 0
				}
			}
		}
	}
	
}

DIPLOMACY_trigger_diplomatic_play_war = {
	# Scope: Diplomatic play province
	# Function: Trigger a war involving all observers who have picked a side in a diplomatic play
	save_scope_as = diplomatic_play
	var:play_instigator = {
		FUNC_declare_war_with_wargoal_province = {
	        war_goal = conquer_wargoal
	        province = scope:diplomatic_play.var:play_target_area
	        target = scope:diplomatic_play.var:play_target_country
	    }
	    every_in_list = {
	    	variable = play_observers_support_instigator
	    	add_to_war = {
				target = scope:current_war_scope
				attacker = yes
				leader = no
			}
	    }
	    every_in_list = {
	    	variable = play_observers_support_target
	    	add_to_war = {
				target = scope:current_war_scope
				attacker = no
				leader = no
			}
	    }
	}
	# audit-fix (#79, diplomacy): tear the play down once it has been resolved by
	# war. Previously the play's state (is_diplomatic_play + play_* vars) was left
	# live, so DIPLOMACY_update_all_diplomatic_plays kept ticking it and, on
	# reaching max progression, DIPLOMACY_complete_play would ADDITIONALLY hand out
	# territory / subjugation on top of whatever the war produced. Reuse the same
	# canonical teardown the completion path (#58) uses. Scope is the play province.
	AI_remove_diplomatic_play = yes
}

DIPLOMACY_trigger_diplomatic_play_event = {
	# Scope: Province (diplomatic play)
	# Function: Trigger a diplomatic play event according to the status of the play
	
	# Events are triggered by a combination of success chance and progress
	# The war willingness and play attitude are also taken into account

	# EARLY EVENTS
	if = {
		limit = {
			var:diplomatic_play_progression < DIPLOMACY_play_progress_one_third
		}
		random_list = {
			1 = { 
				DIPLOMACY_kickoff_play_event = {
					id = minor_border_discussion.1
					pov = var:play_instigator
				} 
			}
			1 = { 
				DIPLOMACY_kickoff_play_event = {
					id = agitator_sponsorship.1
					pov = var:play_instigator
				} 
			}
			1 = { 
				DIPLOMACY_kickoff_play_event = {
					id = send_settlers.1
					pov = var:play_instigator
				} 
			}
			1 = { # Tangier crisis type (target supporter head of state visits target to promote target sovereignty)
				DIPLOMACY_kickoff_play_event = {
					id = head_of_state_visit_play_area.1
					pov = c:BAR # Stand-in for flag:random_supporter_target
				} 
			}
		}
	}
	# MIDDLE EVENTS
	else_if = {
		limit = {
			AND = {
				var:diplomatic_play_progression >= DIPLOMACY_play_progress_one_third
				var:diplomatic_play_progression <= DIPLOMACY_play_progress_two_thirds
			}
		}
		random_list = {
			1 = { # Tangier crisis type (target supporter head of state visits target to promote target sovereignty)
				DIPLOMACY_kickoff_play_event = {
					id = head_of_state_visit_play_area.1
					pov = c:BAR # Stand-in for flag:random_supporter_target
				} 
			}
			1 = { 
				DIPLOMACY_kickoff_play_event = {
					id = agitator_sponsorship.1
					pov = var:play_instigator
				} 
			}
			1 = { 
				DIPLOMACY_kickoff_play_event = {
					id = send_settlers.1
					pov = var:play_instigator
				} 
			}
		}

	}
	# LATE EVENTS
	else_if = {
		limit = {
			var:diplomatic_play_progression > DIPLOMACY_play_progress_two_thirds
		}
		random_list = {
			1 = { # Fashoda type
				DIPLOMACY_kickoff_play_event = {
					id = diplomatic_play.1
					pov = var:play_instigator
				} 
			}
			1 = { 
				DIPLOMACY_kickoff_play_event = {
					id = agitator_sponsorship.1
					pov = var:play_instigator
				} 
			}
			1 = { # Agadir crisis type event event (a supporter of the target sends a gunboat to circle the target area)
				DIPLOMACY_kickoff_play_event = {
					id = agadir_crisis_type.1
					pov = c:BAR # Stand-in flag:random_supporter_target
				} 
			}
		}
	}
}

DIPLOMACY_trigger_diplomatic_play_finale_event = {
	# Scope: Province (diplomatic play provobj). Expects the caller
	# (DIPLOMACY_complete_play) to have set scope:diplomatic_play, scope:play_instigator,
	# scope:play_target_country and scope:play_target_area, and to have already gated on
	# success >= floor. Function: deliver the play's GOAL as a real outcome, scaled by
	# whether success cleared the DECISIVE line (full) or only the floor (partial).
	#
	# Only two goals are ever passed by live callers today (flag:get_territory ×3 paths,
	# flag:influence ×1 — the migration nativist play); flag:subjugate_nation is handled
	# for completeness; anything else falls through to a territorial reading with a note.

	set_local_variable = {
		name = play_decisive
		value = 0
	}
	if = {
		limit = { var:diplomatic_play_success >= DIPLOMACY_play_success_decisive }
		set_local_variable = { name = play_decisive  value = 1 }
	}

	# --- GET TERRITORY: cede provinces of the target area to the instigator. ----
	# Uses LAND_transfer_province (grantee = instigator) — the SAME proven annexation
	# idiom diplomatic_play.2 uses. A DECISIVE win takes the whole target area; a
	# partial win takes only the single most valuable eligible province.
	if = {
		limit = { var:play_goal = flag:get_territory }
		DIPLOMACY_resolve_get_territory = yes
	}
	# --- INFLUENCE: plant foreign influence in the target area's state. ---------
	else_if = {
		limit = { var:play_goal = flag:influence }
		DIPLOMACY_resolve_influence = yes
	}
	# --- SUBJUGATE NATION: bind the target country as a subject of the instigator.
	else_if = {
		limit = { var:play_goal = flag:subjugate_nation }
		DIPLOMACY_resolve_subjugate = yes
	}
	# --- FALLBACK: an unhandled goal — treat as a territorial claim so the win is
	# not silently dropped, and surface it so the gap is visible in the trace.
	else = {
		LOG_fail = { sys = DIPLO  fn = "DIPLOMACY_trigger_diplomatic_play_finale_event"  reason = "unhandled play_goal for [scope:play_instigator.GetTag]; defaulting to a territorial claim on the target area" }
		scope:play_target_area = {
			if = {
				limit = {
					exists = owner
					NOT = { owner = scope:play_instigator }
				}
				add_claim = scope:play_instigator
			}
		}
	}
}

# ======================================================================
# RESOLVE GET_TERRITORY — cede target-area land to the instigator.
# Scope: the play provobj; expects scope:play_instigator / scope:play_target_area
# and local_var:play_decisive (1 = full, 0 = partial). Skips provinces the
# instigator already owns; only fires the annexation notice if land actually moved.
# ======================================================================
DIPLOMACY_resolve_get_territory = {
	# collect the eligible (foreign-owned, not-already-ours) provinces of the area
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = DIPLO  fn = "DIPLOMACY_resolve_get_territory" }
	scope:play_target_area.area = {
		every_area_province = {
			limit = {
				is_uninhabitable = no
				exists = owner
				NOT = { owner = scope:play_instigator }
			}
			add_to_list = play_cede_candidates
		}
	}

	if = {
		limit = { any_in_list = { list = play_cede_candidates  count >= 1 } }
		if = {
			# DECISIVE — the whole contested area changes hands. LAND_transfer_provinces
			# takes the entire list at once (it handles multiple source owners and the
			# governorship-wealth split internally), so it is called ONCE, not per-province.
			limit = { local_var:play_decisive = 1 }
			LAND_transfer_provinces = {
				target_provinces = play_cede_candidates
				grantee = scope:play_instigator
			}
			LOG_line = { sys = DIPLO  msg = "play get_territory DECISIVE: [scope:play_instigator.GetTag] annexes the [scope:play_target_area.GetName] area" }
		}
		else = {
			# PARTIAL — a single province is ceded. It must be the CONTESTED province
			# itself (the play_target_area — the very province the feature seeded a
			# claim on, #53/#57), so the award matches the claim rather than drifting
			# to some other, richer province that merely shares the area. Only if the
			# contested province is not itself cedable (already ours / uninhabitable)
			# do we fall back to the most valuable eligible province in the area.
			if = {
				limit = {
					scope:play_target_area = {
						is_uninhabitable = no
						exists = owner
						NOT = { owner = scope:play_instigator }
					}
				}
				scope:play_target_area = {
					save_scope_as = play_cede_prov
					LAND_transfer_province = {
						grantee = scope:play_instigator
					}
				}
				LOG_line = { sys = DIPLO  msg = "play get_territory PARTIAL: [scope:play_instigator.GetTag] takes the contested province [scope:play_cede_prov.GetName]" }
			}
			else = {
				ordered_in_list = {
					list = play_cede_candidates
					order_by = DIPLOMACY_province_wealth_value
					max = 1
					save_scope_as = play_cede_prov
					LAND_transfer_province = {
						grantee = scope:play_instigator
					}
				}
				LOG_line = { sys = DIPLO  msg = "play get_territory PARTIAL (contested province not cedable; most-valuable fallback): [scope:play_instigator.GetTag] takes [scope:play_cede_prov.GetName]" }
			}
		}
		# relations fallout (the loser resents the cession). Excludes the instigator
		# itself and the c:BAR placeholder that AI interest-plays normalise to.
		if = {
			limit = {
				exists = scope:play_target_country
				NOT = { scope:play_target_country = { this = scope:play_instigator } }
				NOT = { scope:play_target_country = { tag = BAR } }
			}
			scope:play_target_country = {
				add_opinion = { modifier = bitter_over_occupation  target = scope:play_instigator }
			}
		}
		# NOTE: the cession itself is surfaced to the player by the engine's own
		# land-transfer notification; we deliberately do NOT fire the .5/.6 EXPEDITION
		# events here — they require scopes (foreign_minister, play_target_area) the
		# resolution path does not set and that teardown strips, and their flavour
		# ("establishes an outpost"/"annexes a territory") is shaped for the naval
		# expedition flow, not a diplomatic-play cession.
	}
	else = {
		# the area held no foreign land to take (already ours / uninhabitable): the
		# win banks a claim instead so the aim is not wholly lost.
		LOG_line = { sys = DIPLO  msg = "play get_territory: no cedable province in [scope:play_target_area.GetName] for [scope:play_instigator.GetTag]; banking a claim" }
		scope:play_target_area = {
			if = {
				limit = { exists = owner  NOT = { owner = scope:play_instigator } }
				add_claim = scope:play_instigator
			}
		}
	}
	every_in_list = { list = play_cede_candidates  remove_from_list = play_cede_candidates }
	LOG_exit = { sys = DIPLO  fn = "DIPLOMACY_resolve_get_territory"  result = OK }
}

# ======================================================================
# RESOLVE INFLUENCE — the instigator plants foreign influence in the target
# area's state (the mod's DIPLOMACY_add_state_foreign_influence layer). No land
# changes hands. Scope: the play provobj.
# ======================================================================
DIPLOMACY_resolve_influence = {
	# The instigator projects influence onto the TARGET COUNTRY's home ground —
	# NOT the target area. The only live influence caller is the nativist migration
	# breaking-point play (se_MIGRATION.txt): there the instigator is the receiving
	# HOST state and play_target_country is the SENDER of the influx, while
	# play_target_area is a province the host ALREADY OWNS. The coherent win is the
	# host gaining sway over the sender (to stem the flow at its source), so we plant
	# influence in the target country's capital state, guarded so we never try to
	# influence ourselves or the c:BAR placeholder.
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = DIPLO  fn = "DIPLOMACY_resolve_influence" }
	if = {
		limit = {
			exists = scope:play_target_country
			NOT = { scope:play_target_country = { this = scope:play_instigator } }
			NOT = { scope:play_target_country = { tag = BAR } }
			scope:play_target_country = { exists = capital_scope.state }
		}
		scope:play_target_country.capital_scope.state = {
			# refresh any prior influence on the state, then add the instigator's
			DIPLOMACY_remove_state_foreign_influence = yes
			DIPLOMACY_add_state_foreign_influence = {
				influencer = scope:play_instigator
			}
		}
		LOG_line = { sys = DIPLO  msg = "play influence: [scope:play_instigator.GetTag] gains sway over [scope:play_target_country.GetTag]" }
	}
	else = {
		LOG_fail = { sys = DIPLO  fn = "DIPLOMACY_resolve_influence"  reason = "target country absent, is the instigator, or is c:BAR; no influence planted for [scope:play_instigator.GetTag]" }
	}
	LOG_exit = { sys = DIPLO  fn = "DIPLOMACY_resolve_influence"  result = OK }
}

# ======================================================================
# RESOLVE SUBJUGATE — bind the whole target country to the instigator as a
# subject. DECISIVE = a client_state; partial = a lighter tributary. Uses
# FUNC_make_subject (the mod's make_subject wrapper). Scope: the play provobj.
# ======================================================================
DIPLOMACY_resolve_subjugate = {
	# [#98-fix task #98] se_LOG enter/exit wiring (deep-audit)
	LOG_enter = { sys = DIPLO  fn = "DIPLOMACY_resolve_subjugate" }
	if = {
		limit = {
			exists = scope:play_target_country
			NOT = { scope:play_target_country = { this = scope:play_instigator } }
			NOT = { scope:play_target_country = { is_subject_of = scope:play_instigator } }
		}
		if = {
			limit = { local_var:play_decisive = 1 }
			FUNC_make_subject = {
				overlord = scope:play_instigator
				target = scope:play_target_country
				type = client_state
			}
			LOG_line = { sys = DIPLO  msg = "play subjugate DECISIVE: [scope:play_target_country.GetTag] becomes a client state of [scope:play_instigator.GetTag]" }
		}
		else = {
			FUNC_make_subject = {
				overlord = scope:play_instigator
				target = scope:play_target_country
				type = tributary
			}
			LOG_line = { sys = DIPLO  msg = "play subjugate PARTIAL: [scope:play_target_country.GetTag] becomes a tributary of [scope:play_instigator.GetTag]" }
		}
		# the new subject relationship is itself visible to the player; no .5/.6
		# expedition popup (wrong flavour + unset scopes — see the note in
		# DIPLOMACY_resolve_get_territory).
	}
	else = {
		LOG_fail = { sys = DIPLO  fn = "DIPLOMACY_resolve_subjugate"  reason = "target country absent, is the instigator, or already a subject; no subjugation for [scope:play_instigator.GetTag]" }
	}
	LOG_exit = { sys = DIPLO  fn = "DIPLOMACY_resolve_subjugate"  result = OK }
}

DIPLOMACY_add_state_foreign_influence = {
	# Scope: State
	# Function: Add influence from $influencer$ to the given state
	LOG_enter = { sys = DIPLO  fn = "DIPLOMACY_add_state_foreign_influence" }
	save_scope_as = target_state

	set_variable = {
		name = DIPLOMACY_influenced_by
		value = $influencer$
	}
	set_variable = {
		name = DIPLOMACY_state_influence_percent_of_country
		value = {
			every_state_province = {
				add = total_population
				add = {
					value = num_of_upper_strata
					multiply = 5
				}
			}
			divide = {
				value = owner.country_population
				owner = {
					add = {
						value = 0
						every_owned_province = {
							add = num_of_upper_strata
						}
						multiply = 5
					}
				}
				# audit-fix (#79, diplomacy): floor the denominator so a degenerate
				# owner (0 population, 0 upper strata) can't divide-by-zero → NaN.
				min = 1
			}
		}
	}

	$influencer$ = {
		add_to_variable_list = {
			name = DIPLOMACY_list_of_outgoing_state_influence

			target = scope:target_state
		}
		# Add target country to list of countries influenced by the influencer
		# Usage: Poll this list to get the total influence in each country
		if = {
			limit = {
				NOT = {
					is_target_in_variable_list = {
						name = DIPLOMACY_list_of_outgoing_influenced_countries
						target = scope:target_state.owner
					}
				}
			}
			add_to_variable_list = {
				name = DIPLOMACY_list_of_outgoing_influenced_countries

				target = scope:target_state.owner
			}
			add_to_variable_list = {
				name = DIPLOMACY_list_of_influenced_states

				target = scope:target_state
			}
		}
	}

	# Add foreign influencer country to owner country's list of foreign influencers
	# Usage: Poll this list to get the total amount of influence from each foreign influencer
	owner = {
		if = {
			limit = {
				NOT = {
					is_target_in_variable_list = {
						name = DIPLOMACY_list_of_incoming_influencers
						target = $influencer$
					}
				}
			}
			add_to_variable_list = {
				name = DIPLOMACY_list_of_incoming_influencers
				target = $influencer$
			}
		}
	}

	# TODO: Trigger event explaining influence has been added
	LOG_exit = { sys = DIPLO  fn = "DIPLOMACY_add_state_foreign_influence"  result = OK }
}

DIPLOMACY_remove_state_foreign_influence = {
	# Scope: State
	# Function: Remove a foreign influence from the state
	LOG_enter = { sys = DIPLO  fn = "DIPLOMACY_remove_state_foreign_influence" }
	save_scope_as = influenced_state
	if = {
		limit = {
			has_variable = DIPLOMACY_influenced_by
		}
		var:DIPLOMACY_influenced_by = {
			save_scope_as = influencer
		}
		# Remove the influencer from the country's list of foreign influencers if this is the only state it's influencing in the country
		# And vice versa
		owner = {
			save_scope_as = influenced_country
			set_local_variable = {
				name = count_influenced_by
				value = {
					value = 0
					every_in_list = {
						variable = DIPLOMACY_list_of_incoming_influencers
						add = 1
					}
				}
			}
			if = {
				limit = {
					local_var:count_influenced_by = 1
				}
				remove_list_variable = {
					name = DIPLOMACY_list_of_incoming_influencers
					target = scope:influencer
				}
				scope:influencer = {
					remove_list_variable = {
						name = DIPLOMACY_list_of_outgoing_influenced_countries
						target = scope:influenced_country
					}
					remove_list_variable = {
						name = DIPLOMACY_list_of_influenced_states
						target = scope:influenced_state
					}
				}
			}
		}

		remove_variable = DIPLOMACY_influenced_by
		remove_variable = DIPLOMACY_state_influence_percent_of_country
		LOG_exit = { sys = DIPLO  fn = "DIPLOMACY_remove_state_foreign_influence"  result = OK }
	}
	else = {
		LOG_fail = { sys = DIPLO  fn = "DIPLOMACY_remove_state_foreign_influence"  reason = "state has no DIPLOMACY_influenced_by to remove" }
	}
}

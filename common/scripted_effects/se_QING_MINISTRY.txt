########################################################################
# se_QING_MINISTRY.txt — [#349+] the MINISTRY-PERFORMANCE backbone.
#
# The Grand Council IS the set of filled great offices (se_QING_COUNCIL.txt). This
# file makes each office a working BUREAUCRACY: it computes a per-ministry 0..100
# performance meter qing_min_perf_<office> (50 = adequate) from that ministry's own
# concrete sub-state (the officer corps + garrisons for War, the Amban roster for the
# Lifan Yuan, the special buildings for Works, ...). Those meters are then FOLDED into
# the council's effectiveness target by QING_council_fold_ministry_perf (se_QING_COUNCIL.txt),
# so a well-run ministry raises its Minister's standing on the Grand Council and a
# poorly-run one drags it down — the leader answers for how his ministry is run.
#
# CADENCE: QING_ministry_recompute_all_perf is called at the TOP of QING_GOV_pulse
# (se_QING_GOVERNANCE.txt), before QING_council_recompute, so the fold reads fresh
# values. Each ministry adds ONE compute effect + its call here. All O(offices/roster)
# counter maths; no unbounded pop/province sweeps.
#
# Scope: COUNTRY (ROOT = CHI) throughout. se_LOG sys = QING. Prefix = QING_ministry_.
########################################################################

# ======================================================================
# DISPATCHER — recompute every ministry-performance meter. Each ministry-panel
# feature appends its own QING_ministry_recompute_perf_<office> call below. Gated
# to CHI (the meters only matter for the Qing player's Grand Council).
# ======================================================================
QING_ministry_recompute_all_perf = {
	if = {
		limit = { tag = CHI }
		QING_ministry_recompute_perf_war = yes
		QING_ministry_recompute_perf_lifanyuan = yes
		QING_ministry_recompute_perf_zongli = yes
		QING_ministry_recompute_perf_works = yes
		QING_ministry_recompute_perf_rites = yes
		QING_ministry_recompute_perf_revenue = yes
		QING_ministry_recompute_perf_grand_secretariat = yes
		QING_ministry_recompute_perf_chamberlain = yes
		# (Personnel #365, Justice #363, Censorate #362,
		#  Imperial Guard #364 each append their recompute call here as they ship.
		#  Central Secretariat #361 reuses the grand_secretariat compute already
		#  registered here for the Hanlin Academy #358.)
	}
}

# ======================================================================
# [#349] WAR MINISTRY (兵部) PERFORMANCE — qing_min_perf_war (0..100, 50 = adequate).
# The Board of War answered for the standing army: the officer corps and the
# garrisons that hold the realm. We build the meter from concrete state the mod
# already tracks:
#   • the Minister of War's own martial competence (the Board president 兵部尚書, x4),
#   • OFFICER-CORPS COVERAGE — the SIZE of the officer corps vs a healthy establishment
#     (a shrunken corps = posts unfilled = a weaker army). Deliberately NOT the corps'
#     average martial — that is already folded into council effectiveness directly
#     (se_QING_COUNCIL.txt ~404-408), so re-using it here would double-count.
# A VACANT War office floors the meter low (no one runs the Board). All reads guarded.
# ======================================================================
QING_ministry_recompute_perf_war = {
	# --- rebuild the OFFICER-CORPS ROSTER list (the concrete on-map commanders the War
	# panel #349 renders). Every serving unit-commander in CHI's employ, deduped by the
	# is_target_in_variable_list guard (a general can lead several sub-units). This is the
	# datamodel the panel's dynamicgridbox iterates. Bounded O(units).
	clear_variable_list = qing_war_officer_roster
	set_variable = { name = qing_war_roster_count  value = 0 }
	every_unit = {
		limit = { has_commander = yes  commander = { is_alive = yes  employer = ROOT } }
		commander = {
			ROOT = {
				if = {
					limit = { NOT = { is_target_in_variable_list = { name = qing_war_officer_roster  target = prev } } }
					add_to_variable_list = { name = qing_war_officer_roster  target = prev }
					change_variable = { name = qing_war_roster_count  add = 1 }
				}
			}
		}
	}
	# base: adequate.
	set_variable = { name = qing_min_perf_war  value = 50 }
	if = {
		# office FILLED by a live employed minister — score on his martial + the corps.
		limit = {
			has_variable = qing_office_war_holder
			var:qing_office_war_holder = { is_alive = yes  employer = ROOT }
		}
		# (a) the minister's martial, deviation from a competent baseline of 7, x4 (the
		# minister's own competence is the DOMINANT driver of how well the Board is run).
		set_variable = { name = qing_min_war_scratch  value = 0 }
		var:qing_office_war_holder = { save_scope_as = qing_war_minister }
		change_variable = { name = qing_min_war_scratch  add = scope:qing_war_minister.martial }
		change_variable = { name = qing_min_war_scratch  subtract = 7 }
		change_variable = { name = qing_min_war_scratch  multiply = 4 }
		change_variable = { name = qing_min_perf_war  add = var:qing_min_war_scratch }
		# (b) OFFICER-CORPS COVERAGE — is the standing army adequately officered? A count
		# term (deviation from a healthy establishment of ~10 serving commanders), x2. We
		# use COUNT, NOT average martial: the corps' average MARTIAL is ALREADY folded into
		# council effectiveness directly (se_QING_COUNCIL.txt ~404-408, the officer-corps
		# coupling), so re-scoring it here would DOUBLE-COUNT it. Coverage (are the posts
		# manned at all) is an orthogonal Board-of-War responsibility not folded elsewhere.
		# Reads qing_war_roster_count — the count WE JUST TALLIED above (fresh this pulse),
		# NOT the council's qing_officer_corps_count (which is recomputed LATER in the same
		# pulse, so reading it here would be one pulse stale).
		if = {
			limit = { var:qing_war_roster_count > 0 }
			set_variable = { name = qing_min_war_scratch  value = var:qing_war_roster_count }
			change_variable = { name = qing_min_war_scratch  subtract = 10 }   # deviation from a healthy establishment
			change_variable = { name = qing_min_war_scratch  multiply = 2 }
			change_variable = { name = qing_min_perf_war  add = var:qing_min_war_scratch }
		}
		else = {
			# no serving officer corps at all — a paper ministry; a real penalty.
			change_variable = { name = qing_min_perf_war  subtract = 10 }
		}
	}
	else = {
		# VACANT Board of War — the army drifts without a head. Floor it low.
		set_variable = { name = qing_min_perf_war  value = 25 }
	}
	# clamp 0..100.
	if = { limit = { var:qing_min_perf_war < 0 }   set_variable = { name = qing_min_perf_war  value = 0 } }
	if = { limit = { var:qing_min_perf_war > 100 } set_variable = { name = qing_min_perf_war  value = 100 } }
	LOG_line = { sys = QING  msg = "war-ministry performance recomputed" }
}

# ======================================================================
# [#350] LIFAN YUAN (理藩院) PERFORMANCE — qing_min_perf_lifanyuan (0..100, 50 = adequate).
# The Court of Colonial Affairs administered the Inner-Asian dependencies through its
# resident Ambans (駐防大臣). We build the meter from concrete state the amban system
# already tracks:
#   • the Grand Director's own charisma (the 理藩院尚書 — charisma is its governing skill),
#   • AMBAN COVERAGE — of the subjects that WARRANT a resident (the autonomous-governorship
#     Inner-Asian dependencies), how many actually HAVE a live amban posted. A dependency
#     left without a resident = the Court failing its remit.
# A VACANT Lifan Yuan office floors the meter low. Also (re)builds the roster list
# qing_lifanyuan_amban_subjects (the amban-holding subjects the panel #350 renders).
# All reads guarded; O(subjects).
# ======================================================================
QING_ministry_recompute_perf_lifanyuan = {
	# --- rebuild the amban-holding-subjects roster (panel datamodel) + tally coverage.
	clear_variable_list = qing_lifanyuan_amban_subjects
	set_variable = { name = qing_lifan_amban_count  value = 0 }   # subjects WITH a live amban
	set_variable = { name = qing_lifan_warrant_count  value = 0 } # subjects that WARRANT one
	every_subject = {
		limit = { is_subject_type = autonomous_governorship }
		ROOT = { change_variable = { name = qing_lifan_warrant_count  add = 1 } }
		if = {
			limit = {
				has_variable = qing_amban_here
				var:qing_amban_here = { is_alive = yes  employer = prev }
			}
			ROOT = {
				change_variable = { name = qing_lifan_amban_count  add = 1 }
				add_to_variable_list = { name = qing_lifanyuan_amban_subjects  target = prev }
			}
		}
	}
	# base: adequate.
	set_variable = { name = qing_min_perf_lifanyuan  value = 50 }
	if = {
		# office FILLED — score on the Director's charisma + amban coverage.
		limit = {
			has_variable = qing_office_lifanyuan_holder
			var:qing_office_lifanyuan_holder = { is_alive = yes  employer = ROOT }
		}
		# (a) the Director's charisma, deviation from a competent baseline of 7, x4.
		var:qing_office_lifanyuan_holder = { save_scope_as = qing_lifan_director }
		set_variable = { name = qing_min_lifan_scratch  value = 0 }
		change_variable = { name = qing_min_lifan_scratch  add = scope:qing_lifan_director.charisma }
		change_variable = { name = qing_min_lifan_scratch  subtract = 7 }
		change_variable = { name = qing_min_lifan_scratch  multiply = 4 }
		change_variable = { name = qing_min_perf_lifanyuan  add = var:qing_min_lifan_scratch }
		# (b) AMBAN COVERAGE — fraction of warrant subjects with a resident, scaled to +/-.
		# coverage% = 100 * amban / warrant; deviation from a target of 100% (full coverage),
		# divided down so an uncovered dependency is a real but bounded drag.
		if = {
			limit = { var:qing_lifan_warrant_count > 0 }
			set_variable = { name = qing_min_lifan_scratch  value = var:qing_lifan_amban_count }
			change_variable = { name = qing_min_lifan_scratch  multiply = 100 }
			change_variable = { name = qing_min_lifan_scratch  divide = var:qing_lifan_warrant_count }  # coverage %
			change_variable = { name = qing_min_lifan_scratch  subtract = 100 }   # deviation from full coverage (<=0)
			change_variable = { name = qing_min_lifan_scratch  divide = 4 }        # scale (0..-25)
			change_variable = { name = qing_min_perf_lifanyuan  add = var:qing_min_lifan_scratch }
		}
	}
	else = {
		# VACANT Court of Colonial Affairs — the dependencies drift. Floor it low.
		set_variable = { name = qing_min_perf_lifanyuan  value = 25 }
	}
	# clamp 0..100.
	if = { limit = { var:qing_min_perf_lifanyuan < 0 }   set_variable = { name = qing_min_perf_lifanyuan  value = 0 } }
	if = { limit = { var:qing_min_perf_lifanyuan > 100 } set_variable = { name = qing_min_perf_lifanyuan  value = 100 } }
	LOG_line = { sys = QING  msg = "lifan-yuan performance recomputed" }
}

# 50 = adequate). The Office for the General Management of Affairs administered the
# Qing's modern foreign relations. Its performance is built from:
#   • the Grand Director's own charisma (總理衙門大臣 — charisma is its governing skill, x4),
#   • the DIPLOMAT CORPS it fields — lower-ranked diplomats (駐外使臣) reporting to the

# [#351] MINISTRY OF WORKS (工部) PERFORMANCE — qing_min_perf_works (0..100, 50 = adequate).
# The Board of Works maintained the empire's hydraulic + monumental infrastructure: the
# Yellow River dikes (黃河河堤), the Grand Canal grain-tribute depots (漕運倉), the Great
# Wall (長城), and — in this mod — the state specialty manufactories (織造/御窯/茶廠/棉紡/鹽場,
# se_QING_WORKS.txt). We build the meter from concrete on-map state:
#   • the Grand Minister of Works' own FINESSE (工部尚書 — finesse is its governing skill,
#     the same stat se_QING_WORKS gates the build effects on, x4), and
#   • WORKS COVERAGE — how many public-works buildings actually stand across the realm vs a
#     healthy establishment of ~8 (a Board that has built little = neglected infrastructure),
#   • less a CORRUPTION drag — the 河工 graft that plagued river-works budgets: high realm
#     corruption saps the Board's real output no matter how it looks on paper.
# A VACANT Works office floors the meter low (no one maintains the works). Also (re)builds
# the roster list qing_works_provinces (every province holding a works building — the panel
# #351 datamodel) + tallies qing_works_building_count. All reads guarded; O(owned provinces).
# ======================================================================
QING_ministry_recompute_perf_works = {
	# --- rebuild the works-province roster (panel datamodel) + tally the count. A province
	# is on the roster if it holds ANY Board-of-Works building. Bounded O(owned provinces).
	clear_variable_list = qing_works_provinces
	set_variable = { name = qing_works_building_count  value = 0 }
	every_owned_province = {
		limit = {
			OR = {
				has_building = qing_dike_building
				has_building = qing_canal_depot_building
				has_building = qing_wall_section_building
				has_building = qing_silk_filature_building
				has_building = qing_porcelain_kiln_building
				has_building = qing_tea_workshop_building
				has_building = qing_cotton_workshop_building
				has_building = qing_salt_yard_building
			}
		}
		ROOT = {
			add_to_variable_list = { name = qing_works_provinces  target = prev }
			change_variable = { name = qing_works_building_count  add = 1 }
		}
	}
	# base: adequate.
	set_variable = { name = qing_min_perf_works  value = 50 }
	if = {
		# office FILLED by a live employed minister — score on his finesse + works coverage.
		limit = {
			has_variable = qing_office_works_holder
			var:qing_office_works_holder = { is_alive = yes  employer = ROOT }
		}
		# (a) the minister's finesse, deviation from a competent baseline of 7, x4 (finesse is
		# the Board's governing skill — the stat se_QING_WORKS gates its build routes on).
		var:qing_office_works_holder = { save_scope_as = qing_works_minister }
		set_variable = { name = qing_min_works_scratch  value = 0 }
		change_variable = { name = qing_min_works_scratch  add = scope:qing_works_minister.finesse }
		change_variable = { name = qing_min_works_scratch  subtract = 7 }
		change_variable = { name = qing_min_works_scratch  multiply = 4 }
		change_variable = { name = qing_min_perf_works  add = var:qing_min_works_scratch }
		# (b) WORKS COVERAGE — count of standing works vs a healthy establishment of ~8, x2.
		# A Board that has raised few dikes/depots/manufactories has neglected the realm's
		# infrastructure. Count, not any skill re-use, so nothing is double-folded.
		set_variable = { name = qing_min_works_scratch  value = var:qing_works_building_count }
		change_variable = { name = qing_min_works_scratch  subtract = 8 }   # deviation from a healthy establishment
		change_variable = { name = qing_min_works_scratch  multiply = 2 }
		change_variable = { name = qing_min_perf_works  add = var:qing_min_works_scratch }
		# (c) CORRUPTION DRAG — the 河工 graft. High realm corruption saps real output.
		# subtract corruption/10 (0..-10 at the corruption rails), guarded on the var existing.
		if = {
			limit = { has_variable = qing_corruption_level }
			set_variable = { name = qing_min_works_scratch  value = var:qing_corruption_level }
			change_variable = { name = qing_min_works_scratch  divide = 10 }
			change_variable = { name = qing_min_perf_works  subtract = var:qing_min_works_scratch }
		}
	}
	else = {
		# VACANT Board of Works — the dikes silt, the canal chokes. Floor it low.
		set_variable = { name = qing_min_perf_works  value = 25 }
	}
	# clamp 0..100.
	if = { limit = { var:qing_min_perf_works < 0 }   set_variable = { name = qing_min_perf_works  value = 0 } }
	if = { limit = { var:qing_min_perf_works > 100 } set_variable = { name = qing_min_perf_works  value = 100 } }
	LOG_line = { sys = QING  msg = "works-ministry performance recomputed" }
}

# ======================================================================
# [#353] MINISTRY OF RITES (禮部) PERFORMANCE — qing_min_perf_rites (0..100, 50 = adequate).
# The Board of Rites administered two great state functions this mod models concretely:
#   • the TRIBUTARY ORDER (朝貢體系, se_QING_TRIBUTE.txt) — the periodic embassies from the
#     Qing's tributary subjects (Korea/Ryukyu/Siam/...), the ceremony of empire; and
#   • the IMPERIAL EXAMINATIONS (科舉, se_QING_EXAM.txt) — the triennial 會試/殿試 that
#     recruited the scholar-officials, whose health is the qing_exam_pass_rate meter.
# We build the ministry meter from concrete state both systems already track:
#   • the Grand Minister of Rites' own ZEAL (禮部尚書 — zeal is its governing skill, x4),
#   • TRIBUTARY COVERAGE — the number of tributary subjects sending embassies (a full court
#     of vassals = the rites flourishing; an empty one = the ceremony hollow), and
#   • EXAM HEALTH — the qing_exam_pass_rate deviation from a healthy 50 (a well-run
#     examination ladder recruits talent; a corrupt/blocked one starves the bureaucracy).
# A VACANT Rites office floors the meter low. Also rebuilds qing_rites_tributaries (the
# tributary-subject roster the panel #353 renders). All reads guarded; O(subjects).
# ======================================================================
QING_ministry_recompute_perf_rites = {
	# --- rebuild the tributary roster (panel datamodel) + tally coverage. Uses the proven
	# tributary trigger (QING_tribute_is_tributary_trigger). Bounded O(subjects).
	clear_variable_list = qing_rites_tributaries
	set_variable = { name = qing_rites_tributary_count  value = 0 }
	every_subject = {
		limit = { QING_tribute_is_tributary_trigger = yes }
		ROOT = {
			add_to_variable_list = { name = qing_rites_tributaries  target = prev }
			change_variable = { name = qing_rites_tributary_count  add = 1 }
		}
	}
	# base: adequate.
	set_variable = { name = qing_min_perf_rites  value = 50 }
	if = {
		# office FILLED by a live employed minister — score on his zeal + tribute + exams.
		limit = {
			has_variable = qing_office_rites_holder
			var:qing_office_rites_holder = { is_alive = yes  employer = ROOT }
		}
		# (a) the minister's zeal, deviation from a competent baseline of 7, x4 (zeal is the
		# Board of Rites' governing skill — the ceremony + the examinations are its charge).
		var:qing_office_rites_holder = { save_scope_as = qing_rites_minister }
		set_variable = { name = qing_min_rites_scratch  value = 0 }
		change_variable = { name = qing_min_rites_scratch  add = scope:qing_rites_minister.zeal }
		change_variable = { name = qing_min_rites_scratch  subtract = 7 }
		change_variable = { name = qing_min_rites_scratch  multiply = 4 }
		change_variable = { name = qing_min_perf_rites  add = var:qing_min_rites_scratch }
		# (b) TRIBUTARY COVERAGE — count of tributary subjects vs a healthy ~4 court of vassals,
		# x2 (the tributary embassies are the rites' most visible expression).
		set_variable = { name = qing_min_rites_scratch  value = var:qing_rites_tributary_count }
		change_variable = { name = qing_min_rites_scratch  subtract = 4 }
		change_variable = { name = qing_min_rites_scratch  multiply = 2 }
		change_variable = { name = qing_min_perf_rites  add = var:qing_min_rites_scratch }
		# (c) EXAM HEALTH — the examination pass-rate meter's deviation from a healthy 50,
		# divided down (0..±10 band): a flourishing exam ladder lifts the Board, a blocked one
		# drags it. Guarded on the exam system being initialised.
		if = {
			limit = { has_variable = qing_exam_pass_rate }
			set_variable = { name = qing_min_rites_scratch  value = var:qing_exam_pass_rate }
			change_variable = { name = qing_min_rites_scratch  subtract = 50 }
			change_variable = { name = qing_min_rites_scratch  divide = 5 }

# ======================================================================
# [#355] MINISTRY OF REVENUE (戶部) PERFORMANCE — qing_min_perf_revenue (0..100, 50 = adequate).
# The Board of Revenue held the imperial fisc: the census, the land tax, the SALT GABELLE
# (鹽政 — a vast revenue riddled with patronage), the ever-normal GRANARIES (常平倉, grain
# reserves against famine), and the response to the 19thC SILVER CRISIS (銀荒/銀貴錢賤). This
# mod tracks all of these as concrete counters/buildings the decline engine already drives
# (se_QING_REVENUE.txt): qing_salt_gabelle_reformed, qing_currency_stress, qing_corruption_level,
# and the qing_granary_building in the Yellow-River provinces. We build the ministry meter
# from that concrete state:
#   • the Grand Minister of Revenue's own FINESSE (戶部尚書 — finesse is its governing skill,
#     the same stat QING_revenue_assess_fitness leans on, x4),
#   • GRANARY COVERAGE — the ever-normal reserves standing across the Yellow-River provinces
#     vs a healthy ~5 (empty granaries = famine unmanaged), x3,
#   • a SALT-GABELLE bonus (an honest, reformed 鹽政 lifts the fisc), and
#   • drags for the SILVER CRISIS (qing_currency_stress) and realm CORRUPTION (the graft that
#     ate the land-tax). Deliberately NOT the minister's honesty/loyalty traits — those feed
#     QING_revenue_assess_fitness for the domain EVENTS, an orthogonal path; the council fold
#     already scores the holder's own attributes via QING_council_score_office, so we score
#     ministry OUTPUT (reserves/salt/currency), not the man again.
# A VACANT Revenue office floors the meter low. Also rebuilds qing_revenue_granaries (the
# granary-province roster the panel #355 renders) + tallies qing_revenue_granary_count.
# All reads guarded; O(owned provinces).
# ======================================================================
QING_ministry_recompute_perf_revenue = {
	# --- rebuild the granary-province roster (panel datamodel) + tally the reserves. A
	# province is on the roster if it holds an ever-normal granary. Bounded O(owned provinces).
	clear_variable_list = qing_revenue_granaries
	set_variable = { name = qing_revenue_granary_count  value = 0 }
	every_owned_province = {
		limit = { has_building = qing_granary_building }
		ROOT = {
			add_to_variable_list = { name = qing_revenue_granaries  target = prev }
			change_variable = { name = qing_revenue_granary_count  add = 1 }
		}
	}
	# base: adequate.
	set_variable = { name = qing_min_perf_revenue  value = 50 }
	if = {
		# office FILLED by a live employed minister — score on his finesse + fiscal output.
		limit = {
			has_variable = qing_office_revenue_holder
			var:qing_office_revenue_holder = { is_alive = yes  employer = ROOT }
		}
		# (a) the minister's finesse, deviation from a competent baseline of 7, x4 (finesse is
		# the Board's governing skill — the administrator's gift for the fisc).
		var:qing_office_revenue_holder = { save_scope_as = qing_revenue_minister }
		set_variable = { name = qing_min_revenue_scratch  value = 0 }
		change_variable = { name = qing_min_revenue_scratch  add = scope:qing_revenue_minister.finesse }
		change_variable = { name = qing_min_revenue_scratch  subtract = 7 }
		change_variable = { name = qing_min_revenue_scratch  multiply = 4 }
		change_variable = { name = qing_min_perf_revenue  add = var:qing_min_revenue_scratch }
		# (b) GRANARY COVERAGE — ever-normal reserves standing vs a healthy establishment of ~5,
		# x3. Empty granaries = the realm one bad harvest from famine. Count, not skill re-use.
		set_variable = { name = qing_min_revenue_scratch  value = var:qing_revenue_granary_count }
		change_variable = { name = qing_min_revenue_scratch  subtract = 5 }   # deviation from a healthy reserve
		change_variable = { name = qing_min_revenue_scratch  multiply = 3 }
		change_variable = { name = qing_min_perf_revenue  add = var:qing_min_revenue_scratch }
		# (c) SALT GABELLE — a reformed, honest 鹽政 lifts the fisc (+10); an unreformed one is
		# the baseline (patronage tolerated, no bonus).
		if = {
			limit = {
				has_variable = qing_salt_gabelle_reformed
				var:qing_salt_gabelle_reformed = 1
			}
			change_variable = { name = qing_min_perf_revenue  add = 10 }
		}
		# (d) SILVER CRISIS DRAG — the 銀荒: the currency stress the Board is failing to arrest.
		# subtract currency_stress/6 (0..-~16 at the rails), guarded on the var existing.
		if = {
			limit = { has_variable = qing_currency_stress }
			set_variable = { name = qing_min_revenue_scratch  value = var:qing_currency_stress }
			change_variable = { name = qing_min_revenue_scratch  divide = 6 }
			change_variable = { name = qing_min_perf_revenue  subtract = var:qing_min_revenue_scratch }
		}
		# (e) CORRUPTION DRAG — the graft that ate the land-tax. subtract corruption/10
		# (0..-10 at the rails), guarded on the var existing.
		if = {
			limit = { has_variable = qing_corruption_level }
			set_variable = { name = qing_min_revenue_scratch  value = var:qing_corruption_level }
			change_variable = { name = qing_min_revenue_scratch  divide = 10 }
			change_variable = { name = qing_min_perf_revenue  subtract = var:qing_min_revenue_scratch }
		}
	}
	else = {
		# VACANT Board of Revenue — the salt graft entrenches, the granaries empty. Floor it low.
		set_variable = { name = qing_min_perf_revenue  value = 25 }
	}
	# clamp 0..100.
	if = { limit = { var:qing_min_perf_revenue < 0 }   set_variable = { name = qing_min_perf_revenue  value = 0 } }
	if = { limit = { var:qing_min_perf_revenue > 100 } set_variable = { name = qing_min_perf_revenue  value = 100 } }
	LOG_line = { sys = QING  msg = "revenue-ministry performance recomputed" }
}

# ======================================================================
# [#357] QING_zongli_play_boost — the Zongli Yamen's per-play success factor. Applied
# once per Qing pulse to EACH of the Qing's live diplomatic plays (from QING_gp_scan_plays,
# the (A) instigator=self branch). Scope: the PLAY PROVOBJ. A capable, well-staffed Zongli
# (qing_min_perf_zongli > 50) nudges the play's success UP; a poor/vacant one (< 50) drags
# it DOWN — the foreign service either advances the crisis or fumbles it. Counter-only +
# clamped (DIPLOMACY_modify_play_success clamps 0..100), so it is idempotent-safe to
# re-apply every pulse (like the GP-tension pressure it rides alongside). ROOT = CHI is
# passed as $home$ so the provobj can read the Qing's perf var.
# ======================================================================
QING_zongli_play_boost = {
	# convert (perf - 50) into a small per-pulse success delta: +/-50 perf -> +/-2 success.
	# read the Qing's perf var off the passed home country.
	set_variable = { name = qing_zongli_play_delta  value = $home$.var:qing_min_perf_zongli }
	change_variable = { name = qing_zongli_play_delta  subtract = 50 }
	change_variable = { name = qing_zongli_play_delta  divide = 25 }   # ~ +/-2 at the rails
	DIPLOMACY_modify_play_success = { amt = var:qing_zongli_play_delta }
	remove_variable = qing_zongli_play_delta
}

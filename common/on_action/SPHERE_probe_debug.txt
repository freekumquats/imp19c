# [develop #254] Register via the on_actions LIST form so this block actually
# fires. Bare inline-effect on_game_initialized blocks do NOT merge across files
# (only ONE file's inline effect runs); on_actions lists DO merge. See #254.
on_game_initialized = {
	on_actions = {
		imp19c_sphere_probe_on_game_initialized
	}
}

imp19c_sphere_probe_on_game_initialized = {
	effect = {
		# [#163 SPHERE PROBE DIAG] Unconditional beacon BEFORE any guard/scope switch, so the
		# next boot test distinguishes "on_game_initialized block never fired" (this line absent)
		# from "guard skipped it / c:CHI scope failed" (this line present, PROBE START absent).
		LOG_line = { sys = SPHERE  msg = "on_game_initialized effect ENTERED (pre-guard beacon)" }
		if = {
			limit = { exists = c:CHI }
			LOG_line = { sys = SPHERE  msg = "beacon: c:CHI EXISTS" }
		}
		else = {
			LOG_line = { sys = SPHERE  msg = "beacon: c:CHI DOES NOT EXIST (probe guard would skip)" }
		}
		if = {
			limit = { has_global_variable = sphere_probe_done }
			LOG_line = { sys = SPHERE  msg = "beacon: sphere_probe_done ALREADY SET (probe self-skips; stale flag from a prior run)" }
		}
		# [#163 SPHERE PROBE] Fire the throwaway PULL-vs-PUSH capability probe once,
		# only if CHI is in the game and we haven't already run it. Guarded by a
		# global flag so it fires exactly once per campaign. Delete this file and
		# se_SPHERE_probe.txt once the verdict is recorded.
		if = {
			limit = {
				NOT = { has_global_variable = sphere_probe_done }
				exists = c:CHI
			}
			set_global_variable = sphere_probe_done
			c:CHI = {
				SPHERE_probe_run = yes
			}
		}
	}
}

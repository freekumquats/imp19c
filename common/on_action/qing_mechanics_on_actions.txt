# seeds a tidy first-tick state for a fresh CHI game. [develop #254] Only the

# [1763 port] Dated beats = develop values + 19127 (rebased 1815.7.1 -> 1763.2.16);
# immediate days={1 ..} game-start events left unshifted.
# [develop #254] Register via the on_actions LIST form so this block actually
# fires. Bare inline-effect on_game_initialized blocks do NOT merge across files
# (only ONE file's inline effect runs); on_actions lists DO merge. See #254.
on_game_initialized = {
	on_actions = {
		imp19c_qing_on_game_initialized
	}
}

			# so a hardcoded `set_as_ruler=char:N` (and every father=/mother=/spouse=/commander=char:N)
			# binds to the WRONG character. That is exactly how the "wrong emperor" bug happened

						NOT = { current_ruler = { has_trait = chinese_emperor } }

# ======================================================================
# [#115/#116/#119] SUCCESSION HOOK. The engine's on_ruler_change (see
# 00_specific_from_code.txt:105) is merged across files, so this adds the Qing's
# own reaction to a change of sovereign:
#   • refresh the dynastic seats (the new emperor takes the 皇帝 seat, his heir
#     the 皇太子 seat, any prior regency is re-evaluated for the new reign);
#   • resolve the secret-succession tablet (qing_succession.2): a smooth accession
#     if 秘密立儲 was sealed, a disputed one if the princes were still jockeying.
# CHI-only, but fires for AI and player alike (a succession is a succession).
# ======================================================================
on_ruler_change = {
	effect = {
		if = {
			limit = { tag = CHI }
			QING_seat_refresh_all = yes
			# [#424] the privy purse (內帑) is an institution of the throne: carry the former
			# sovereign's private silver over to the new one (scope:former_ruler is set here).
			QING_household_carryover_purse = yes
			trigger_event = { id = qing_succession.2  days = { 1 3 } }
			# [#312/R2-8] PRAGMATIC-SANCTION contest. The heir line is (re)defined at an
			# accession, so this is where the "does a female heir now head an open female
			# line?" question is settled — NOT on a monthly poll (an heir does not change
			# monthly). The effect self-gates on the female-heir conditions and is idempotent
			# while already contested, so it is safe to fire on every accession. There is no
			# on_law_change on_action, so a law change made mid-reign takes effect at the next
			# accession (or the next heir shift, below).
			SUCCESSION_pragmatic_contest = yes
		}
	}
}

# [#312/R2-8] HEIR-SHIFT HOOK. A death mid-reign can change who the primary heir is
# (e.g. an elder son dies and a daughter becomes next in line), which is the other
# moment the Pragmatic-Sanction question can newly arise without a change of ruler.
# on_character_death scope is ROOT = the DYING CHARACTER (like the betrothal / office
# hooks in 00_specific_from_code.txt); the country is reached via `employer`. Fires
# only for a death within the CHI ruling family (from_ruler_family — proven, upstream
# oaths_of_allegiance_button.txt:40) — the only deaths that can reshuffle the heir line.
# The contest self-gates + is idempotent, so this is a cheap no-op the rest of the time.
on_character_death = {
	effect = {
		if = {
			limit = {
				from_ruler_family = yes
				exists = employer
				employer = { tag = CHI }
			}
			employer = { SUCCESSION_pragmatic_contest = yes }
		}
		# [#322 deferred] THE INVESTITURE (冊封). If the country whose ruler just changed is
		# a Qing TRIBUTARY, its new sovereign awaits the imperial patent of investiture that
		# renewed the tributary bond each reign. Notify the overlord (CHI) so the player gets
		# the investiture dilemma.
		# [T2 review-fix] on_character_death gives ROOT = the DYING CHARACTER, so is_subject_type
		# / overlord (COUNTRY triggers) were being evaluated against a CHARACTER scope — dead logic
		# that never fired (matching the FIRST branch, which correctly hops to `employer` to reach
		# the country). A ruler's employer IS his own country, so hop through employer here too:
		# the tributary country is `employer`, and it is what we save + pass to the overlord.
		else_if = {
			limit = {
				exists = employer
				employer = {
					is_subject_type = tributary
					exists = overlord
					overlord = { tag = CHI }
				}
			}
			employer = {
				save_scope_as = trib_investee
				overlord = {
					QING_tribute_invest_new_ruler = { vassal = scope:trib_investee }
				}
			}
		}
	}
}

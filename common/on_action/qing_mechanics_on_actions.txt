# [1763 port] Dated beats = develop values + 19127 (rebased 1815.7.1 -> 1763.2.16);
# immediate days={1 ..} game-start events left unshifted.
# [develop #254] Register via the on_actions LIST form so this block actually
# fires. Bare inline-effect on_game_initialized blocks do NOT merge across files
# (only ONE file's inline effect runs); on_actions lists DO merge. See #254.
on_game_initialized = {
	on_actions = {
		imp19c_qing_on_game_initialized
	}
}

imp19c_qing_on_game_initialized = {

		if = {
			limit = { tag = CHI }

			# distant-power web) become possible. Idempotent; logged.
			QING_open_global_diplomacy = yes
			# Default the inter-ethnic stance to the historical balanced dyarchy.
			if = {
				limit = { NOT = { has_variable = qing_ethnic_stance_active } }
				QING_set_ethnic_stance = { stance = dyarchy }

			# own internal fold is a clean no-op here (perf vars don't yet exist). All guarded.

			# ~6 points low until the first quarterly pulse self-heals it. Seeding here (harem_init

			QING_harem_init = yes
			QING_ministry_recompute_all_perf = yes
			QING_council_recompute = yes
			# [#115/#116] seed the permanent/dynastic Grand Council seats (Emperor
			# 皇帝, Crown Prince 皇太子) and evaluate a game-start regency (a child or
			# infirm sovereign at 1815). se_QING_SEATS.txt.
			QING_seat_refresh_all = yes
			# [#157] seed the dynastic-harmony meter (family dynamics of the
			# imperial house: emperor / crown prince / empress dowager) that

			QING_dynasty_init = yes

			# Foreign advisors (洋員, New#2): seed the advisor-slot counters so the

			QING_advisor_init = yes

			# hub reads a clean 0 before any minister is sent abroad.
			QING_legation_init = yes
			# Educational Mission (留學生, New#4): seed the students-abroad counters.
			QING_students_init = yes
			# Unequal treaties (條約口岸, New#5): seed the treaty/port/tariff counters.

# ======================================================================
# [#115/#116/#119] SUCCESSION HOOK. The engine's on_ruler_change (see
# 00_specific_from_code.txt:105) is merged across files, so this adds the Qing's
# own reaction to a change of sovereign:
#   • refresh the dynastic seats (the new emperor takes the 皇帝 seat, his heir
#     the 皇太子 seat, any prior regency is re-evaluated for the new reign);
#   • resolve the secret-succession tablet (qing_succession.2): a smooth accession
#     if 秘密立儲 was sealed, a disputed one if the princes were still jockeying.
# CHI-only, but fires for AI and player alike (a succession is a succession).
# ======================================================================
on_ruler_change = {
	effect = {
		if = {
			limit = { tag = CHI }
			QING_seat_refresh_all = yes
			trigger_event = { id = qing_succession.2  days = { 1 3 } }
		}
		# [#322 deferred] THE INVESTITURE (冊封). If the country whose ruler just changed is
		# a Qing TRIBUTARY, its new sovereign awaits the imperial patent of investiture that
		# renewed the tributary bond each reign. Notify the overlord (CHI) so the player gets
		# the investiture dilemma. `this` = the tributary; its overlord = CHI. Self-guards on
		# the tie via QING_tribute_is_tributary_trigger (evaluated in the overlord hop).
		else_if = {
			limit = {
				is_subject_type = tributary
				exists = overlord
				overlord = { tag = CHI }
			}
			save_scope_as = trib_investee
			overlord = {
				QING_tribute_invest_new_ruler = { vassal = scope:trib_investee }
			}
		}
	}
}

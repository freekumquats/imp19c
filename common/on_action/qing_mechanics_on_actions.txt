########################################################################
# qing_mechanics_on_actions.txt — startup hook for the Qing player-experience
# mechanics suite. The suite's monthly pulse (qing_mechanics_pulse_on_action in
# 00_monthly_country.txt) self-initialises via QING_DECLINE_init, so this only
# seeds a tidy first-tick state for a fresh CHI game. Engine merges multiple
# on_game_initialized blocks across files.
########################################################################

on_game_initialized = {
	effect = {
		if = {
			limit = { tag = CHI }
			QING_DECLINE_init = yes
			# Default the inter-ethnic stance to the historical balanced dyarchy.
			if = {
				limit = { NOT = { has_variable = qing_ethnic_stance_active } }
				QING_set_ethnic_stance = { stance = dyarchy }
			}
			# Governance layer: seed the bureaucracy/council meters and convene
			# an initial Grand Council from the most capable courtiers.
			QING_GOV_init = yes
			QING_council_init = yes
			QING_council_autofill = yes
			# Self-Strengthening Movement: seed the progress counter (advanced
			# later by the Self-Strengthening mission tree).
			QING_selfstr_init = yes
			# Great-power / foreign-reaction layer: seed the per-power tension
			# counters (raised by the expansion mission trees; se_QING_DIPLO.txt).
			QING_gp_init = yes
			# Reform end-states (#14): seed the Christian/Taiping-path meter so the
			# Heavenly-Kingdom road and its mission tree read a clean 0 at start.
			QING_taiping_init = yes
			# Foreign advisors (洋員, New#2): seed the advisor-slot counters so the
			# recruitment GUI reads a clean state before any expert is hired.
			QING_advisor_init = yes
			# Permanent legations (駐外使館, New#3): seed the legation count so the
			# hub reads a clean 0 before any minister is sent abroad.
			QING_legation_init = yes
			# Educational Mission (留學生, New#4): seed the students-abroad counters.
			QING_students_init = yes
			# Unequal treaties (條約口岸, New#5): seed the treaty/port/tariff counters.
			QING_treaty_init = yes
			# Tech-transfer bargains (洋務交易, New#6): seed the per-power favor-debt.
			QING_techtransfer_init = yes
			# Protect the Vassal (保藩, New#7): seed the suzerain-prestige meter.
			QING_vassal_init = yes
			# Missionary incidents (教案, New#8): seed the anti-Christian sentiment meter.
			QING_missionary_init = yes
			# The Ili Crisis (新疆/伊犁, New#9): seed the New-Dominion grip meter and stage marker.
			QING_ili_init = yes
			# The Emperor Emeritus (#65): seed the Napoleon reform-count counter.
			QING_napoleon_init = yes
			# The pre-Perry Japan arc (#81): seed the Nagasaki-intel / reform-faction /
			# shogun-grip counters so the pre-Perry mission tree reads a clean state.
			QING_jppre_init = yes
			# #80 (Amherst pinned to 1816): the Amherst embassy is the historical
			# kickoff of the Emperor Emeritus chain, but as one weight-10 branch of the
			# ~quarterly flavour roll it fired at an emergent, usually-too-late time
			# (median ~4 years, often past the <1821.5.5 Napoleon window). Pin it with a
			# dated one-shot: START_DATE is 1815.7.1, Amherst reached Peking in late
			# Aug 1816, so ~410-430 days lands it in its historical window and makes the
			# counterfactual dependably reachable. Player-only (is_ai=no), and the event
			# re-checks its own trigger (exists c:GBR + not yet resolved) at fire time;
			# the flavour-roll branch remains a harmless fallback (self-guarded on
			# qing_embassy_britain, so a no-op once this has fired).
			if = {
				limit = { is_ai = no }
				trigger_event = { id = qing_embassy.2  days = { 410 430 } }
			}
			# #81 (Perry hand-off): Commodore Perry reached Edo Bay in July 1853,
			# ~13,880 days after the 1815.7.1 START_DATE. Schedule the black-ships
			# one-shot with a wide window; the event re-checks its own date guard
			# (current_date >= 1853.1.1) and the not-yet-done guard at fire time, so
			# it lands in its historical window and fires exactly once. It closes the
			# pre-Perry arc and hands off to the post-Perry Japan tree. Player-only.
			if = {
				limit = { is_ai = no }
				trigger_event = { id = qing_japan_preperry.10  days = { 13820 13940 } }
			}
		}
	}
}

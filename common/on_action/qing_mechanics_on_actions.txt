# qing_mechanics_on_actions.txt — startup hook for the Qing player-experience
# mechanics suite. The suite's monthly pulse (qing_mechanics_pulse_on_action in
# 00_monthly_country.txt) self-initialises via QING_DECLINE_init, so this only
# seeds a tidy first-tick state for a fresh CHI game. [develop #254] Only the

# effect blocks, so this registers a NAMED action (see block below).

# [1763 port] Dated beats = develop values + 19127 (rebased 1815.7.1 -> 1763.2.16);
# immediate days={1 ..} game-start events left unshifted.
# [develop #254] Register via the on_actions LIST form so this block actually
# fires. Bare inline-effect on_game_initialized blocks do NOT merge across files
# (only ONE file's inline effect runs); on_actions lists DO merge. See #254.
on_game_initialized = {
	on_actions = {
		imp19c_qing_on_game_initialized
	}
}

imp19c_qing_on_game_initialized = {
	effect = {
		if = {
			limit = { tag = CHI }
			# [BT-9/BT-20 detector] CHARACTER-POSITION INTEGRITY SELF-CHECK. A gap in the global
			# setup character-id space silently COMPACTS runtime ids (runtime = written − missing_below),
			# so a hardcoded `set_as_ruler=char:N` (and every father=/mother=/spouse=/commander=char:N)
			# binds to the WRONG character. That is exactly how the "wrong emperor" bug happened
			# (Qianlong char:214 → runtime 205; the throne got a toddler's kin, Yongrong). This is a
			# PURE DIAGNOSTIC (no ruler mutation — the real fix is keeping the id space contiguous,
			# see memory imp19c-setup-char-id-rule): if CHI's seated ruler is not an adult holder of
			# the chinese_emperor trait, scream a loud, greppable FAIL line + scope dump so the id-gap
			# regression surfaces on the very next boot instead of only in a playtest.
			#   grep "IMP19C FAIL QING : QING_boot_ruler_integrity" debug.log
			if = {
				limit = {
					OR = {
						NOT = { current_ruler = { has_trait = chinese_emperor } }
						current_ruler = { is_adult = no }
					}
				}
				LOG_fail = {
					sys = QING
					fn = "QING_boot_ruler_integrity"
					reason = "CHI seated ruler is NOT an adult chinese_emperor — likely a setup character-id GAP compacting runtime ids and mis-binding set_as_ruler=char:214 (see memory imp19c-setup-char-id-rule; check global id contiguity)"
				}
			}
			else = {
				LOG_line = { sys = QING  msg = "boot ruler integrity OK: CHI seated an adult chinese_emperor" }
			}
			QING_DECLINE_init = yes
			# [#92] Global diplomatic reach: the engine gates real diplomacy on
			# in_diplomatic_range, leaving the trans-Pacific USA (and other far
			# powers) permanently unreachable at the base range. Grant the Qing a
			# permanent global-reach modifier at game start so US relations (and the
			# distant-power web) become possible. Idempotent; logged.
			QING_open_global_diplomacy = yes
			# Default the inter-ethnic stance to the historical balanced dyarchy.
			if = {
				limit = { NOT = { has_variable = qing_ethnic_stance_active } }
				QING_set_ethnic_stance = { stance = dyarchy }
			}
			# Governance layer: seed the bureaucracy/council meters and convene
			# an initial Grand Council from the most capable courtiers.
			QING_GOV_init = yes
			QING_council_init = yes
			QING_council_autofill = yes
			# [#115/#116] seed the permanent/dynastic Grand Council seats (Emperor
			# 皇帝, Crown Prince 皇太子) and evaluate a game-start regency (a child or
			# infirm sovereign at 1815). se_QING_SEATS.txt.
			QING_seat_refresh_all = yes
			# [#157] seed the dynastic-harmony meter (family dynamics of the
			# imperial house: emperor / crown prince / empress dowager) that
			# feeds Grand Council effectiveness. se_QING_DYNASTY.txt.
			QING_dynasty_init = yes
			# [#158] seed the conservative-vs-reformist council-faction meters
			# (reform lean + polarization) that overlay the Grand Council. Derived
			# each quarter in QING_council_recompute; feeds qing_reform_faction_balance
			# and docks a deadlock penalty on council effectiveness. se_QING_FACTION.txt.
			QING_faction_init = yes
			# Self-Strengthening Movement: seed the progress counter (advanced
			# later by the Self-Strengthening mission tree).
			QING_selfstr_init = yes
			# Great-power / foreign-reaction layer: seed the per-power tension
			# counters (raised by the expansion mission trees; se_QING_DIPLO.txt).
			QING_gp_init = yes
			# Reform end-states (#14): seed the Christian/Taiping-path meter so the
			# Heavenly-Kingdom road and its mission tree read a clean 0 at start.
			QING_taiping_init = yes
			# Foreign advisors (洋員, New#2): seed the advisor-slot counters so the
			# recruitment GUI reads a clean state before any expert is hired.
			QING_advisor_init = yes
			# Permanent legations (駐外使館, New#3): seed the legation count so the
			# hub reads a clean 0 before any minister is sent abroad.
			QING_legation_init = yes
			# Educational Mission (留學生, New#4): seed the students-abroad counters.
			QING_students_init = yes
			# Unequal treaties (條約口岸, New#5): seed the treaty/port/tariff counters.
			QING_treaty_init = yes
			# Tech-transfer bargains (洋務交易, New#6): seed the per-power favor-debt.
			QING_techtransfer_init = yes
			# Protect the Vassal (保藩, New#7): seed the suzerain-prestige meter.
			QING_vassal_init = yes
			# [#322] The tribute-embassy rhythm (朝貢體系): seed the mission-tempo counter
			# so the quarterly scheduler can dispatch tributary embassies. (se_QING_TRIBUTE.txt)
			QING_tribute_init = yes
			# [#323] The grain-tribute logistics (漕運): seed the canal-condition + capital
			# grain-reserve counters so the quarterly canal tick can run. (se_QING_CANAL.txt)
			QING_canal_init = yes
			# [#352 F11] Seed the Canton trade-regime marker. The 一口通商 single-port
			# Canton System was decreed 1757, so the 1763 start is OPEN. Without this the
			# var is unset at game-start, so BOTH the Open and Close buttons show at once
			# (each gates on NOT var:qing_canton_regime = flag:<other>, so an unset var
			# satisfies both). Idempotent: only seed if not already set.
			if = {
				limit = { NOT = { has_variable = qing_canton_regime } }
				set_variable = { name = qing_canton_regime  value = flag:open }
			}
			# Missionary incidents (教案, New#8): seed the anti-Christian sentiment meter.
			QING_missionary_init = yes
			# [#321] The examination hiring pool (科舉): seed the pool bookkeeping so the
			# keju cycle can mint graduate characters into it and office-appointment can
			# draw on them first. (se_QING_EXAM.txt)
			QING_exam_init = yes
			# The Ili Crisis (新疆/伊犁, New#9): seed the New-Dominion grip meter and stage marker.
			QING_ili_init = yes
			# The Emperor Emeritus (#65): seed the Napoleon reform-count counter.
			QING_napoleon_init = yes
			# The pre-Perry Japan arc (#81): seed the Nagasaki-intel / reform-faction /
			# shogun-grip counters so the pre-Perry mission tree reads a clean state.
			QING_jppre_init = yes
			# [OVERNIGHT WAVE] #126 Household 內務府: seed the private imperial treasury
			# 內帑 (qing_privy_purse = 50) — the one net-new counter of the office wave
			# (a private purse distinct from the state 戶部 treasury; no existing counter
			# fit). Drifts quarterly via QING_household_pulse. (se_QING_HOUSEHOLD.txt)
			QING_household_init = yes
			# [#165] Four-power sphere-of-influence: build the contested frontier
			# ring and seed each march's opening China/Britain/France/Russia balance
			# from its theatre. Evolved yearly by QING_sphere_pulse. (se_QING_SPHERE.txt)
			QING_sphere_init = yes
			# [OVERNIGHT WAVE] #107 province-level ethnic tension: seed the Inner Asian
			# march (Turkestan 20 / Tibet 15 / Mongolia 10 / Qinghai+Kham 10) so the
			# bottom-up tension engine starts from a historically-restive frontier.
			# (se_QING_ETHNIC_TENSION.txt)
			QING_ethnic_tension_init = yes
			# #80 (Amherst pinned to 1816): the Amherst embassy is the historical
			# kickoff of the Emperor Emeritus chain, but as one weight-10 branch of the
			# ~quarterly flavour roll it fired at an emergent, usually-too-late time
			# (median ~4 years, often past the <1821.5.5 Napoleon window). Pin it with a
			# dated one-shot: START_DATE is 1815.7.1, Amherst reached Peking in late
			# Aug 1816, so ~410-430 days lands it in its historical window and makes the
			# counterfactual dependably reachable. Player-only (is_ai=no), and the event
			# re-checks its own trigger (exists c:GBR + not yet resolved) at fire time;
			# the flavour-roll branch remains a harmless fallback (self-guarded on
			# qing_embassy_britain, so a no-op once this has fired).
			if = {
				limit = { is_ai = no }
				trigger_event = { id = qing_embassy.2  days = { 19537 19557 } }
			}
			# #81/#94 COUPLING INVERSION: Perry is now OWNED and scheduled by the
			# Japan bakumatsu arc. japan_bakumatsu.0 (tag = TKG, AI-autonomous) fires
			# this Qing hand-off (qing_japan_preperry.10) as a CONSUMER the moment Perry
			# reaches Edo Bay — the Qing's pre-Perry narrative now reacts to Japan's
			# event instead of the Qing driving Japan's clock, and Japan owns 100% of
			# Japan's simulation state (on TKG/domain tags, not on CHI). This CHI-side
			# dated schedule is RETAINED as an idempotent backstop only: it closes the
			# Qing's OWN pre-Perry bookkeeping at the historical date should the Japan
			# arc never fire (e.g. TKG annexed before 1853). Whichever fires first sets
			# qing_jppre_perry_done; the other no-ops (preperry.10 self-guards on it).
			# Player-only.
			if = {
				limit = { is_ai = no }
				trigger_event = { id = qing_japan_preperry.10  days = { 32947 33067 } }
			}
		}
	}
}

# ======================================================================
# [#115/#116/#119] SUCCESSION HOOK. The engine's on_ruler_change (see
# 00_specific_from_code.txt:105) is merged across files, so this adds the Qing's
# own reaction to a change of sovereign:
#   • refresh the dynastic seats (the new emperor takes the 皇帝 seat, his heir
#     the 皇太子 seat, any prior regency is re-evaluated for the new reign);
#   • resolve the secret-succession tablet (qing_succession.2): a smooth accession
#     if 秘密立儲 was sealed, a disputed one if the princes were still jockeying.
# CHI-only, but fires for AI and player alike (a succession is a succession).
# ======================================================================
on_ruler_change = {
	effect = {
		if = {
			limit = { tag = CHI }
			QING_seat_refresh_all = yes
			trigger_event = { id = qing_succession.2  days = { 1 3 } }
			# [#312/R2-8] PRAGMATIC-SANCTION contest. The heir line is (re)defined at an
			# accession, so this is where the "does a female heir now head an open female
			# line?" question is settled — NOT on a monthly poll (an heir does not change
			# monthly). The effect self-gates on the female-heir conditions and is idempotent
			# while already contested, so it is safe to fire on every accession. There is no
			# on_law_change on_action, so a law change made mid-reign takes effect at the next
			# accession (or the next heir shift, below).
			SUCCESSION_pragmatic_contest = yes
		}
	}
}

# [#312/R2-8] HEIR-SHIFT HOOK. A death mid-reign can change who the primary heir is
# (e.g. an elder son dies and a daughter becomes next in line), which is the other
# moment the Pragmatic-Sanction question can newly arise without a change of ruler.
# on_character_death scope is ROOT = the DYING CHARACTER (like the betrothal / office
# hooks in 00_specific_from_code.txt); the country is reached via `employer`. Fires
# only for a death within the CHI ruling family (from_ruler_family — proven, upstream
# oaths_of_allegiance_button.txt:40) — the only deaths that can reshuffle the heir line.
# The contest self-gates + is idempotent, so this is a cheap no-op the rest of the time.
on_character_death = {
	effect = {
		if = {
			limit = {
				from_ruler_family = yes
				exists = employer
				employer = { tag = CHI }
			}
			employer = { SUCCESSION_pragmatic_contest = yes }
		}
	}
}

########################################################################
# qing_startruler_on_actions.txt — [#31] mark every country's GAME-START ruler so the
# vanilla "seek a spouse" ambition (scheme_seek_a_spouse, common/ambitions/01_schemes.txt)
# cannot re-marry a historical head-of-state to an AI-conjured spouse.
#
# WHY: the #329 setup fix (removing death_date from the historical start rulers so the
# setup snapshot stops reading them as already-dead) correctly seats live rulers like
# Catherine II — but a LIVE, WIDOWED (Pyotr III d.1762), unmarried, AI-controlled ruler is
# exactly the profile scheme_seek_a_spouse.can_be_picked targets (is_ruler + is_adult +
# is_married=no). The AI then marries her to a fresh create_character ("Nikita Badayev" in
# the boot test). Before #329 these rulers were dead-placeholder/auto-picked, so the scheme
# never saw them — that is why it only started happening now.
#
# FIX: stamp the game-start ruler of EVERY country with qing_historical_start_ruler (a
# permanent character var), and gate scheme_seek_a_spouse.can_be_picked on
# NOT = { has_variable = qing_historical_start_ruler }. This blocks re-marriage ONLY for the
# specific historical figures seated at 1763.2.16; every mid-game ruler (a successor who
# takes the throne after load, and thus never carries the marker) can still seek a spouse.
#
# [develop #254] Registered via the on_actions LIST form (bare inline on_game_initialized
# blocks do NOT merge across files — only one wins). Runs once, at game init.
########################################################################

on_game_initialized = {
	on_actions = {
		imp19c_startruler_mark_on_game_initialized
	}
}

imp19c_startruler_mark_on_game_initialized = {
	effect = {
		every_country = {
			limit = { exists = current_ruler }
			current_ruler = {
				set_variable = { name = qing_historical_start_ruler  value = 1 }
			}
		}
	}
}

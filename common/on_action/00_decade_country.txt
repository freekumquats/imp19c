decade_country_pulse = {
	on_actions = {
		# Bottom-up migration: each country lets a grievance-weighted pop trickle
		# from its worst provinces to the best adjacent one. Spread across the
		# decade window (1..3650 days) so the whole world does not migrate on the
		# same tick. See common/scripted_effects/se_MIGRATION.txt.
		delay = { days = { 1 3650 } }
		MIGRATION_decade_pulse
		# Layer 4 separatism: alien-ruled, disaffected homelands (seeded by the
		# Layer 1-3 de jure system) may break away as a new country and be
		# fought over. Rare and self-throttled; piggybacks this existing pulse so
		# it adds NO new polling hook. See common/scripted_effects/se_SEPARATISM.txt.
		SEPARATISM_decade_pulse
	}
}

# Child on_action: the country-scope effect the pulse fires. Split out so the
# pulse block stays a pure schedule (mirrors the vanilla / reference-mod idiom).
MIGRATION_decade_pulse = {
	effect = {
		MIGRATION_country_pulse = yes
	}
}

SEPARATISM_decade_pulse = {
	effect = {
		SEPARATISM_country_pulse = yes
	}
}

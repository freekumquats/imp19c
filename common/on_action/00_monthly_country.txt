# Fires reaction events (unrest / scandal / riot / harvest / festival) for any
# subject the Qing player is actively integrating. Gated to CHI and throttled to
# roughly twice a year via a cooldown variable so pushes stay eventful, not spammy.
qing_integration_pulse_on_action = {
	trigger = {
		tag = CHI
		is_ai = no
		NOT = { has_variable = qing_integration_pulse_cooldown }
		any_subject = {
			has_variable = SUBJ_integration_active
			NOT = { has_variable = SUBJ_integration_suspended }
			is_subject_type = autonomous_governorship
		}
	}
	effect = {
		LOG_line = { sys = SUBJ  msg = "monthly integration pulse firing for overlord=" }
		set_variable = { name = qing_integration_pulse_cooldown  days = 180 }
		SUBJ_QING_integration_pulse = yes
	}
}

# Qing player-experience mechanics suite: the shared decline-engine pulse.
# Reads O(1) counters and re-applies modifier bands; fires weighted reaction
# events. Gated to the single human Qing player and throttled to ~quarterly via
# a cooldown variable, so it is negligible for the AI (rejected on tag = CHI).
qing_mechanics_pulse_on_action = {
	trigger = {
		tag = CHI
		is_ai = no
		NOT = { has_variable = qing_mechanics_pulse_cooldown }
	}
	effect = {
		set_variable = { name = qing_mechanics_pulse_cooldown  days = 90 }
		# [BT-28] TWO-SLOT EVENT BUDGET, cleared at the TOP of the pulse (before ANY roller runs, incl. the
		# ministry pulses inside QING_GOV_pulse). At most ONE court event (dynasty / faction / justice-via-
		# frontier / industry-via-works / foreign-spouse / officer) and ONE external event (tribute) fire per
		# quarterly pulse. Each roller CLAIMS its slot only on the branch that actually schedules an event
		# (BT-10 pattern); a no-op roll leaves the slot free. Stops the game-start + steady-state pile-up.
		remove_variable = qing_gc_event_slot_used
		remove_variable = qing_ext_event_slot_used
		QING_DECLINE_pulse = yes
		QING_ethnic_stance_drift = yes
		# Great-Game diplomatic-play pressure (#52): scan the Qing's live plays and
		# mount tension on any great power whose sphere a play reaches into (以夷制夷
		# rivals eased in turn). Runs BEFORE the governance pulse so the fresh
		# tension is converted to rivalry bands by QING_gp_apply_bands this pulse.
		QING_gp_scan_plays = yes
		# [#165] Four-power sphere-of-influence: rebuild the contested frontier ring
		# and evolve each march's China/Britain/France/Russia balance. Self-throttled
		# to ~yearly (own 365-day cooldown), so it is cheap on the quarterly pulse.
		# Runs right after the play scan so a flip that raises GP tension is banded by
		# QING_gp_apply_bands in the same governance pulse below. (se_QING_SPHERE.txt)
		QING_sphere_pulse = yes
		# Governance layer: recompute council effectiveness + the abstracted
		# bureaucracy meters and re-apply their bands. Runs AFTER the decline
		# pulse so it reads fresh corruption/currency counters.
		QING_GOV_pulse = yes
		# [#346] frontier-garrison overlay: a Beijing-commanded garrison on a subject's
		# Inner-Asian soil tightens integration, chafes the dependency, and (if a resident
		# Amban shares the ground) triggers a War-Ministry vs Lifan-Yuan turf war.
		# (se_QING_FRONTIER.txt)
		QING_fgar_scan = yes
		# [#385] Sinicization rate-of-change: refresh the live Han-dominant province
		# count (via the same governorship descent the report list uses, so the figure
		# matches the listed rows) and roll the year-old baseline annually, so the
		# Sinicization report can show a year-over-year trend, not just a static list.
		# Cheap count every quarter; baseline rolls once a year. (se_QING_SINICIZATION.txt)
		QING_sinic_pulse = yes
		# [BT-28] frontier/justice flavour reactions now share the court slot (was ungated -> justice.1
		# fired alongside dynasty/tribute/industry). Gated on the slot being free; the roll claims it when it fires.
		if = {
			limit = { NOT = { has_variable = qing_gc_event_slot_used } }
			QING_frontier_flavour_roll = yes
		}
		# [#157] Family dynamics of the imperial house (emperor / crown prince /
		# empress dowager). Low chance so the court's private life is an occasional
		# beat, not spam; each branch re-checks its own trigger. Runs AFTER the
		# governance pulse so the harmony meter it moves is read fresh next quarter.
		# [BT-10] gated on the shared quarterly GC-event slot; the roll claims it when it fires.
		if = {
			limit = { NOT = { has_variable = qing_gc_event_slot_used } }
			random = {
				chance = 25
				QING_dynasty_flavour_roll = yes
			}
		}
		# [#158] Conservative-vs-reformist council politics (the faction layer that
		# overlays the Grand Council + the three dynastic figureheads). Same low
		# chance as the family beat; each branch re-checks its own trigger. Runs
		# after the governance pulse so the stance/lean/polarization meters it reads
		# were refreshed this quarter by QING_council_recompute. (se_QING_FACTION.txt)
		# [BT-10] gated on the shared quarterly GC-event slot; the roll claims it when it fires.
		if = {
			limit = { NOT = { has_variable = qing_gc_event_slot_used } }
			random = {
				chance = 25
				QING_faction_flavour_roll = yes
			}
		}
		# #65 Emperor Emeritus: re-score Napoleon's conditional loyalty (HIGH while
		# his revenge project is served — reforms advancing AND GBR/RUS tension held
		# high; DROPS on appeasement / stalled reform). Self-guards if he is absent.
		QING_napoleon_loyalty_pulse = yes
		# Japan/Meiji sub-layer (#22): Japan modernizes on its own clock and
		# the accord/rivalry bands are re-applied. Runs after the flavour roll.
		QING_japan_pulse = yes
		# §11 L4 separatism (Qing): tension-scaled breakaway risk, layered ON TOP
		# of the generic decade-cadence engine. Only bites once ethnic tension has
		# boiled past its floor; self-throttled by its own ~3yr cooldown. Runs
		# after the decline pulse so it reads a fresh qing_ethnic_tension.
		SEPARATISM_qing_pulse = yes
		# [#314 B17] Collect the quarterly tribute from every subject carrying a demand
		# tier (set via the diplomatic-view small/medium/large selector). Concrete
		# treasury transfer scaled by tier + subject income; se_SUBJECT_QING.txt.
		QING_subject_collect_tribute = yes
		# [#330 user] Foreign-spouse council lobby: if a foreign-born consort wed into the
		# imperial house now holds a Grand Council office, her birth court's interests
		# occasionally reach the council table (qing_office.30). Low chance so it is an
		# occasional court intrigue, not spam; the check self-guards on an eligible member
		# existing, so it is a clean no-op when no foreign spouse sits. (se_QING_COUNCIL.txt)
		# [BT-10] gated on the shared quarterly GC-event slot; the check claims it when it fires.
		if = {
			limit = { NOT = { has_variable = qing_gc_event_slot_used } }
			random = {
				chance = 20
				QING_council_foreign_spouse_check = yes
			}
		}
		# [#331 user] War Minister's Report: the Grand Minister of War (兵部尚書) — or the
		# Grand Commandant of the Imperial Guard (領侍衛內大臣) for the metropolitan garrison —
		# periodically reports on a serving garrison commander (qing_office.31). Low chance so
		# it is an occasional court dilemma; self-guards on a seated War minister and a living
		# legion commander, a clean no-op otherwise. (se_QING_COUNCIL.txt)
		# [BT-10] gated on the shared quarterly GC-event slot; the check claims it when it fires.
		if = {
			limit = { NOT = { has_variable = qing_gc_event_slot_used } }
			random = {
				chance = 12
				QING_council_officer_report_check = yes
			}
		}
		# [#321] Maintain the examination hiring pool: retire the aged/dead un-placed
		# graduates and keep the count var honest. Cheap, self-guards on the list.
		QING_exam_pool_tick = yes
		# [#321 deferred] The TRIENNIAL examination clock (鄉試/會試 on the fixed 3-year
		# cadence). Owns convening qing_keju.1 now (replacing the old stochastic weight in
		# QING_frontier_flavour_roll); self-throttled by its own 1095-day cooldown, so on
		# most pulses this is a clean no-op.
		QING_exam_schedule_triennial = yes
		# [#322] The tribute-embassy rhythm: dispatch at most one due tributary mission
		# this quarter (off the real subject graph), then fold the recent-mission tempo
		# into the suzerain-prestige drift. Self-guards on having tributary subjects.
		QING_tribute_schedule_missions = yes
		QING_tribute_prestige_from_missions = yes
		# [#322 deferred] the tributary order's stress tests, both self-throttled + self-
		# guarded (clean no-ops at the High Qing zenith): a restless vassal tests the bond
		# when suzerain prestige has fallen low, and a great power meddles in a tributary once
		# its China tension has built. Both inert until the 19th-c. pressures actually arrive.
		QING_tribute_breakaway_check = yes
		QING_tribute_gp_interference_check = yes
	}
}

		#Add income for reparations...

				value = subtract_10_percent_monthly_income
			}
			var:paying_5_year_reparations = {
				add_treasury = {
					value = add_10_percent_monthly_income
				}

		}
	}

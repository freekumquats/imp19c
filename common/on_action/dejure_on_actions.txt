# engine (§11 Layer 3, se_DEJURE.txt).
#
# One job: freeze every settled province's 1815 plurality culture into its
# dejure_culture variable, ONCE, at game start. DEJURE_freeze_setup self-guards
# on a global variable, so it is safe no matter how the engine fires this block
# (once globally, or once per country). every_province is a global iterator, so
# no particular root scope is required. [develop #254] Only the on_actions LIST
# form of on_game_initialized merges across files (not bare inline effect blocks),
# so this registers a NAMED action and coexists with the Qing suite's own block
# in qing_mechanics_on_actions.txt.
########################################################################

# [develop #254] Register via the on_actions LIST form so this block actually
# fires. Bare inline-effect on_game_initialized blocks do NOT merge across files
# (only ONE file's inline effect runs); on_actions lists DO merge. See #254.
on_game_initialized = {
	on_actions = {
		imp19c_dejure_on_game_initialized
	}
}

imp19c_dejure_on_game_initialized = {
	effect = {
		DEJURE_freeze_setup = yes
	}
}

currency_law = {
	potential = {
		is_tribal = no
	}
	# Hard specie: coins carry their metal value. Trustworthy and hard to debase
	# (low corruption, steadier stability), but the rigid money supply drags commerce.
	precious_metal_content_coinage = {
		modifier = {
			monthly_corruption = -0.05
			stability_monthly_change = 0.02
			global_commerce_modifier = -0.05
			global_upper_strata_happyness = 0.05
		}
	}
	# A silver standard (the China-relevant regime): a modest commercial base and
	# light merchant confidence, no strong drag either way.
	silver_standard_currency = {
		modifier = {
			global_commerce_modifier = 0.05
			global_middle_strata_output = 0.05
			global_tax_modifier = 0.05
		}
	}
	# Promissory notes / early bank paper: credit lubricates trade at the cost of a
	# graft/counterfeiting risk.
	promissory_notes = {
		allow = {
			civic_tech >= 6
		}
		modifier = {
			global_commerce_modifier = 0.1
			global_capital_trade_routes = 1
			global_middle_strata_happyness = 0.05
			monthly_corruption = 0.05
		}
	}
	# Fiat currency: the most flexible money supply (strong commerce and revenue),
	# but losing the metal anchor is destabilising and leans on state credibility.
	fiat_currency = {
		allow = {
			civic_tech >= 6
		}
		modifier = {
			global_commerce_modifier = 0.15
			global_tax_modifier = 0.1
			global_capital_trade_routes = 1
			stability_monthly_change = -0.03
			diplomatic_reputation = -0.5
		}
	}
}

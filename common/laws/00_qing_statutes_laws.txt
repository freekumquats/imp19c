# 00_qing_statutes_laws.txt — Qing 會典 (Collected Statutes): the player-
# legislated policy of the Qing ministries, expressed as NATIVE ENGINE LAW
# GROUPS (see overnight_designs.md Design/Impl 2). Each group is CHI-only
# (potential = { tag = CHI }); the FIRST option is the current-behavior
# default so a fresh game holds it and nothing changes until the player
# legislates. Enacted via the stock Laws tab (change_law) at the global
# enact_law price (35 PI / 15 stability, common/prices/00_hardcoded.txt).
#
# EACH new group MUST also be registered in gui/government_view.gui's
# hand-enumerated Laws tab (a laws_widget_area entry) or it will be
# change_law-able but INVISIBLE — see the documented succession_law case.
#
# THIS FILE (chunk 1) proves the pipeline on the two cheapest, pure
# modifier-swap groups (no on_enact, no var/effect plumbing): the Penal
# Code (刑部/大清律例) and Ritual Orthodoxy (禮部). Both reuse the effect
# shapes of existing event modifiers (qing_justice_* / qing_rites_*), applied
# here as a STANDING law modifier rather than a temporary event modifier.
#
# UTF-8 BOM at file head (matches every other common/laws/ file).
########################################################################

# ======================================================================
# PENAL CODE REGIME (大清律例) — 刑部. How the dynasty applies its law:
# a merciful, assize-tempered code; the full deterrent rigor; or a revised,
# clarified modern code. Mirrors the qing_justice_* event-modifier effects
# as a standing stance.
# ======================================================================
qing_penal_code_law = {
	potential = { tag = CHI }

	# DEFAULT (current behavior): balanced application — no standing modifier.
	qing_penal_balanced = {
		modifier = { }
	}

	# [review-fix] Merciful, assize-tempered justice (秋審 mercy): the LEGITIMACY option, NOT the order
	# option — mercy wins the throne benevolent legitimacy and eases tyranny, but calms the streets only
	# modestly (−0.5 vs harsh's −1). This makes merciful↔harsh a real legitimacy-vs-order axis rather than
	# a free lunch. Legitimacy trimmed 0.03→0.02 (a STANDING law is milder than the temporary event mod
	# qing_justice_merciful_assize it echoes; matches the ritual-tier de-escalation).
	qing_penal_merciful = {
		modifier = {
			global_unrest = -0.5
			monthly_legitimacy = 0.02
			monthly_tyranny = -0.02
		}
	}

	# The full rigor of the code — harsh deterrence: the ORDER option — the strongest calm (−1 unrest) +
	# stability, bought with legitimacy. (Cf. qing_justice_harsh_code.)
	qing_penal_harsh = {
		modifier = {
			global_unrest = -1
			monthly_legitimacy = -0.02
			stability_monthly_change = 0.01
		}
	}

	# A revised, clarified code (大清律例 reform) — just law binds the realm: broad calm + legitimacy +
	# state loyalty. [review-fix] Gated on the mid-game reform track (qing_reform_track_unlocked, set when
	# reform pressure ≥50 in se_QING_DECLINE.txt) rather than civic_tech — a real progression gate, not the
	# no-op civic_tech>=4 (this mod starts tech low, so the tech gate opened turn one). Legitimacy trimmed
	# 0.02 as a standing modifier. (Cf. qing_justice_code_revised.)
	qing_penal_revised = {
		allow = { has_variable = qing_reform_track_unlocked }
		modifier = {
			global_unrest = -0.5
			monthly_legitimacy = 0.02
			global_monthly_state_loyalty = 0.02
		}
	}
}

# ======================================================================
# RITUAL ORTHODOXY (禮部) — the court's stance on the ancestral rites and
# the Mandate: strict orthodoxy vs pragmatic adaptation. Mirrors the
# qing_rites_* event-modifier effects as a standing stance.
# ======================================================================
qing_ritual_orthodoxy_law = {
	potential = { tag = CHI }

	# DEFAULT (current behavior): conventional observance — no standing modifier.
	qing_ritual_conventional = {
		modifier = { }
	}

	# Strict ancestral orthodoxy (慎終追遠): the rites upheld to the letter — legitimacy and stability from
	# ritual propriety, but [review-fix] rigid conservatism CHOKES commercial dynamism (−0.03 commerce).
	# This makes orthodox↔pragmatic a genuine tradition-vs-modernity axis rather than orthodox being a
	# no-cost dominant pick. (Cf. qing_rites_ancestral_piety, halved from its temporary event magnitude.)
	qing_ritual_orthodox = {
		modifier = {
			monthly_legitimacy = 0.03
			stability_monthly_change = 0.01
			global_commerce_modifier = -0.03
		}
	}

	# Pragmatic adaptation of the rites to the age: a freer hand for commerce, at a cost to ritual
	# legitimacy. [review-fix] Gated on the mid-game reform track (qing_reform_track_unlocked) not the
	# no-op civic_tech>=4 — adapting the ancestral rites is a reform-era move, not a turn-one option.
	# (Cf. qing_rites_pragmatic_adaptation.)
	qing_ritual_pragmatic = {
		allow = { has_variable = qing_reform_track_unlocked }
		modifier = {
			global_commerce_modifier = 0.05
			monthly_legitimacy = -0.01
		}
	}
}

# ======================================================================
# CHUNK 2.2 — var-driven fiscal/frontier stance laws. These groups drive an
# EXISTING backing var via on_enact = { set_variable ... }; the var is already
# read by the domain pulse (se_QING_OPIUM / se_QING_CARAVAN / se_QING_REVENUE),
# so the law simply becomes the player's control surface for it. The FIRST
# option matches the var's init value (the pulse seeds it), so holding the
# default is byte-identical to today; on_enact only runs on an active change.
# set_variable is a trivial, safe on_enact (no heavy chain / iterator / recompute).
# ======================================================================

# OPIUM POLICY (鴉片) — 戶部/刑部. Drives qing_opium_posture (0 prohibit / 1
# tolerate-for-revenue / 2 treaty-legalized), read by se_QING_OPIUM.txt.
qing_opium_policy_law = {
	potential = { tag = CHI }

	# DEFAULT: prohibition (the 1729/1796/1839 bans) — matches init qing_opium_posture=0.
	qing_opium_prohibit = {
		on_enact = { set_variable = { name = qing_opium_posture  value = 0 } }
	}
	# Tolerate for the revenue (a licensed opium excise) — trades order/health for coin.
	qing_opium_tolerate = {
		on_enact = { set_variable = { name = qing_opium_posture  value = 1 } }
	}
	# Treaty legalization (post-1858 Tianjin tariff on opium) — full revenue, worst epidemic.
	# Gated on the treaty system standing (the legalization was imposed by treaty, not chosen freely).
	qing_opium_legalize = {
		allow = { has_variable = qing_treaty_track }
		on_enact = { set_variable = { name = qing_opium_posture  value = 2 } }
	}
}

qing_caravan_customs_law = {

}

# INTER-ETHNIC GOVERNANCE DOCTRINE (旗漢之辨) — the constitutional balance between
# the Manchu banner establishment and the Han majority. Drives the existing
# QING_set_ethnic_stance effect (se_QING_MECHANICS.txt), which swaps the three
# qing_ethnic_stance_* country modifiers + sets qing_ethnic_stance_active. This is
# the SAME control the scripted-GUI stance buttons use — the law is a parallel
# surface. on_enact calls a proven effect that only add/removes country modifiers
# (no heavy chain / iterator), so it is trampoline-safe as-is.
qing_ethnic_governance_law = {
	potential = { tag = CHI }

	# DEFAULT: dyarchy (滿漢一體) — the balanced High-Qing settlement (stability,
	# state loyalty). Matches the gamestart stance seeded by se_QING_MECHANICS.
	qing_ethnic_doctrine_dyarchy = {
		on_enact = { QING_set_ethnic_stance = { stance = dyarchy } }
	}
	# Banner supremacy (首崇滿洲): martial vigor + manpower, at a fiscal + unrest cost.
	qing_ethnic_doctrine_banner = {
		on_enact = { QING_set_ethnic_stance = { stance = banner } }
	}
	# Han conciliation (重用漢人): tax + promotion + assimilation, at a manpower +
	# subject-loyalty cost.
	qing_ethnic_doctrine_han = {
		on_enact = { QING_set_ethnic_stance = { stance = han } }
	}
}

# OFFICE-SELLING POLICY (捐納) — 吏部. Whether posts may be purchased. Toggles the
# qing_office_purchased_ranks country modifier, whose presence the bureau-integrity
# pulse (se_QING_GOVERNANCE.txt) reads (−20 to the integrity target). add/remove of
# a country modifier is a trivial, safe on_enact.
qing_office_selling_law = {
	potential = { tag = CHI }

	# DEFAULT: examination-only appointment (科舉正途) — no purchased ranks. Matches
	# the gamestart (no qing_office_purchased_ranks modifier present).
	qing_office_exam_only = {
		on_enact = { remove_country_modifier = qing_office_purchased_ranks }
	}
	# Open sale of office (捐納廣開): a fast fiscal expedient — quick revenue via the
	# purchase price, but the bureaucracy's integrity rots (the pulse docks it).
	qing_office_open_sale = {
		on_enact = { add_country_modifier = { name = qing_office_purchased_ranks  duration = -1 } }
	}
}

# CANTON TRADE REGIME (廣州體制) — the maritime-trade stance at the southern port.
# Drives qing_canton_regime (flag:open | flag:closed), read by se_QING_CANTON.txt.
# (Tariff autonomy and censorate vigor are pulse-COMPUTED progress meters, not policy
# knobs, so they are NOT exposed as laws — a law must not overwrite a computed value.)
qing_canton_regime_law = {
	potential = { tag = CHI }

	# DEFAULT: the open Canton System (一口通商) — matches the flag:open seeded at 1763.
	qing_canton_open = {
		on_enact = { set_variable = { name = qing_canton_regime  value = flag:open } }
	}
	# Hard closure (閉關): shut the port to foreign trade — control and insulation up,
	# commerce down (the se_QING_CANTON pulse swaps the qing_canton_* modifiers).
	qing_canton_closed = {
		on_enact = { set_variable = { name = qing_canton_regime  value = flag:closed } }
	}
}

# ======================================================================
# CHUNK 2.4 — ministry-cadence law (exam frequency). This group drives an
# existing pulse literal (the exam cooldown) by setting a small selector var
# the scheduler branches on. on_enact = set_variable (trivial/safe); the
# BRANCHING lives in se_QING_EXAM.txt (proven if/else_if literal-days pattern,
# cf. se_QING_TRIBUTE.txt:257-266), NOT a var-driven duration (unproven).
# ======================================================================

# EXAMINATION CADENCE (科舉) — 禮部. How often the metropolitan examination is held.
# Sets qing_law_exam_cadence (0 triennial / 1 biennial / 2 sexennial); the exam
# scheduler (QING_exam_schedule_triennial) reads it to pick the cooldown length.
qing_exam_cadence_law = {
	potential = { tag = CHI }

	# DEFAULT: the triennial cadence (三年大比) — the historical norm. Selector 0 =
	# the 1095-day cooldown the scheduler already used, so this is a no-op default.
	qing_exam_triennial = {
		on_enact = { set_variable = { name = qing_law_exam_cadence  value = 0 } }
	}
	# Frequent exams (增科取士): more sittings — a wider, faster-renewed literati pool
	# and a legitimacy dividend, at a higher administrative burden. ~biennial.
	qing_exam_frequent = {
		on_enact = { set_variable = { name = qing_law_exam_cadence  value = 1 } }
	}
	# Sparse exams (減科): fewer sittings ease the burden but starve the ladder of
	# advancement — a thinner degree-holding elite. ~sexennial.
	qing_exam_sparse = {
		on_enact = { set_variable = { name = qing_law_exam_cadence  value = 2 } }
	}
}

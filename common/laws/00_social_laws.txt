		modifier = {
			happiness_for_same_religion_modifier=-0.05
			global_population_happiness=0.10
			global_pop_conversion_speed_modifier=-0.50
		}
		allow = {
			NOT = { religion = secular }
		}
	}
	state_secularism = {
		modifier = {
			global_population_happiness=0.20
			global_pop_conversion_speed_modifier=-0.90
		}
		on_enact = {
			set_country_religion = secular
		}
		allow = {
			religion = secular
			# [#61 2026-07-23] Design 3 reconcile: never while the country holds an IDEOLOGY (a coexisting
			# religion entry) — enacting this would set_country_religion=secular and silently WIPE the
			# ideology. The ideology channel and the secularism laws both own the country religion slot; this
			# gate keeps them mutually exclusive. (Already gated religion=secular, so this is belt-and-braces.)
			NOT = { is_ideology_religion = yes }
		}
	}
	state_atheism = {
		modifier = {
			happiness_for_same_religion_modifier=0.30
			global_population_happiness=-0.10
		}
		on_enact = {
			set_country_religion = atheism
		}
		allow = {
			religion = secular
			# [#61 2026-07-23] Design 3 reconcile: never while holding an IDEOLOGY (would wipe it via
			# set_country_religion=atheism). Keeps the ideology channel + secularism laws mutually exclusive.
			NOT = { is_ideology_religion = yes }
		}
	}

### General Deity Checks ###

deity_war1 = { # WAR
	trigger = {
		# [B7] the Qing state religion (confucianism) ships its OWN themed pantheon
		# (common/deities/03_confucian_pantheon.txt); these generic region1 deities have no
		# localized names in this mod, so leaving them enabled for confucianism showed raw
		# keys (omen_deity_war1 ...) in the Pantheon panel. Exclude confucianism; every other
		# religion keeps the full generic set unchanged.
		NOT = { religion = confucianism }
	}
	icon = deity_war
	passive_modifier = { discipline = deity_discipline_svalue }
	omen = { global_freemen_happyness = omen_global_freemen_happiness_svalue } #Bonus from Omen
	on_activate = {
		military_apotheosis_capital_freemen_effect = yes
	}
	religion = region1
	deity_category = war #This Deity uses a War slot.

	deification_trigger = {
		can_deify_trigger = no
	}
}

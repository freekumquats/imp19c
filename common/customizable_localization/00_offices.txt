# ======================================================================
# [char-window title row] QING_court_position_name — the COURT POSITION a character
# holds, for the "Governor of X / Huangdi"-style subtitle row under the name in the
# character window. Character scope. Names, in priority order:
#   1. a GRAND COUNCIL great office (the 13 offices) — via the same qing_office_held
#      flag map QING_office_held_name uses, so it stays in lockstep with that;
#   2. a MINISTRY SUB-POST (Zongli diplomat / Censorate inspector / Guard captain);
#   3. an inner-court STUDY attendant (Southern / Upper Study);
#   4. an AMBAN / resident commissioner (駐紮大臣);
#   5. a palace EUNUCH (太監).
# The 1:1 rule (a man holds at most one post — QING_office_appoint strips sub-post
# markers on seating, and QING_subpost_eligible_candidate blocks double-booking) means
# at most one branch matches; priority order is belt-and-suspenders. Falls through to a
# BLANK string when the character holds no court position (so the GUI row, gated on the
# same has-a-position test, simply renders nothing / is hidden). All keys pre-exist
# except the sub-post/study/amban/eunuch ones added in qing_governance_l_english.yml.
# ======================================================================
QING_court_position_name = {
	type = character
	# --- 1. the thirteen Grand Council great offices (reuse the office-name keys) ---
	text = { localization_key = QING_OFFICE_NAME_CHANCELLOR         trigger = { has_variable = qing_office_held  var:qing_office_held = flag:chancellor } }

	text = { localization_key = QING_GC_OFFICE_LIFANYUAN           trigger = { has_variable = qing_office_held  var:qing_office_held = flag:lifanyuan } }
	text = { localization_key = QING_GC_OFFICE_CHAMBERLAIN         trigger = { has_variable = qing_office_held  var:qing_office_held = flag:chamberlain } }
	text = { localization_key = QING_GC_OFFICE_ZONGLI             trigger = { has_variable = qing_office_held  var:qing_office_held = flag:zongli } }

	# --- 5. palace eunuch ---
	text = { localization_key = QING_SUBPOST_NAME_EUNUCH  trigger = { has_variable = qing_is_palace_eunuch } }
	# --- fallthrough: the GUI row is hidden for anyone holding NO position, so this is reached
	# in-window only in the theoretical case of a qing_office_held flag outside the 15 enumerated
	# above; name it "a great office" (matching QING_office_held_name's fallthrough) rather than a
	# blank line, so the visible row never renders empty. QING_COURT_POSITION_NONE ("") is retained
	# for any caller that wants the empty form. ---
	text = { localization_key = QING_OFFICE_NAME_GENERIC  trigger = { always = yes } }

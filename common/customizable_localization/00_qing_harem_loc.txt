# [#438] Imperial harem rank names (後宮位號). Character-scope custom loc mapping a consort's
# qing_consort_rank variable (1..4) to her graded title, below the Empress (皇后). first_valid = yes
# picks the first matching branch top-down, so the highest rank is tested first. Referenced in
# gui/qing_harem.gui via [Character.Custom('qing_consort_rank_name')]. Idiom mirrors the mod's
# era_name_loc (000_GOVERNMENT_custom_loc.txt): text{ localization_key + trigger }.
qing_consort_rank_name = {
	type = character
	first_valid = yes

	text = {
		localization_key = qing_consort_rank_4
		trigger = { var:qing_consort_rank = 4 }
	}
	text = {
		localization_key = qing_consort_rank_3
		trigger = { var:qing_consort_rank = 3 }
	}
	text = {
		localization_key = qing_consort_rank_2
		trigger = { var:qing_consort_rank = 2 }
	}
	# entry rank (嬪/貴人): the default for a rank-1 consort or a legacy consort without the var.
	text = {
		localization_key = qing_consort_rank_1
		trigger = { always = yes }
	}
}

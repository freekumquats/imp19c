# [#34] KEJU DEGREE PRESTIGE — the soft-preference weight that makes the examination
# the SOURCE of the empire's civil servants. Added to every candidate-ranking value
# below, so a keju graduate outscores an equally-skilled commoner wherever the council,
# the ambanate, or a legation picks a man — WITHOUT a hard lock (a degreeless character
# can still be seated when the graduate pool is empty, so the realm never soft-locks for
# want of a laureate). The ladder mirrors the degree hierarchy (translate/academy floor
# → jinshi/hanlin apex), and is deliberately large relative to the 0..~9 skill axis so a
# degree is the dominant sort key: even a modest graduate heads the bench over a brilliant
# commoner, exactly as the historical bureaucracy privileged the metropolitan laureate.
# Evaluated in a CHARACTER scope. Conditional add-by-trait is the proven script-value
# idiom (cf. religionator_svalue, 00_governor_policies.txt). Degrees are mutually exclusive
# (opposites), so at most one branch adds.
qing_degree_prestige_svalue = {
	value = 0
	if = { limit = { has_trait = hanlin }       add = 30 }   # 翰林 — the Academy apex
	else_if = { limit = { has_trait = jinshi }       add = 25 }   # 進士 — the metropolitan laureate
	else_if = { limit = { has_trait = fanyi_jinshi } add = 25 }   # 繙譯進士 — the banner-track laureate (amban stock)
	else_if = { limit = { has_trait = gongshi }      add = 20 }   # 貢士 — passed the metropolitan, awaiting the palace
	else_if = { limit = { has_trait = juren }        add = 15 }   # 舉人 — the provincial laureate
	else_if = { limit = { has_trait = shengyuan }    add = 10 }   # 生員/秀才 — the licentiate
	else_if = { limit = { has_trait = jiansheng }    add = 5  }   # 監生 — the purchased/academy floor
}

# [#48] WUJU (武舉) MARTIAL DEGREE PRESTIGE — the martial parallel to qing_degree_prestige_svalue
# above. Added ONLY into council_sort_martial (the war-minister / guard-commandant bench), so a
# 武舉 laureate heads the MILITARY appointment list the way a civil 進士 heads the civil offices —
# and, deliberately, a civil degree no longer inflates the martial bench (a bookish 進士 should not
# out-rank a 武進士 for a field command). Weights mirror the civil ladder (the exam ranked below the
# civil one socially, but WITHIN the martial track the apex laureate should still dominate; the
# lower social prestige is expressed by the wuju feeding only ONE bench, not all four). Degrees are
# mutually exclusive with the civil ones and with each other (opposites), so at most one branch adds.
# Evaluated in a CHARACTER scope; conditional add-by-trait is the proven script-value idiom.
qing_wu_degree_prestige_svalue = {
	value = 0
	if = { limit = { has_trait = wu_zhuangyuan } add = 30 }   # 武狀元 — the martial optimus (一等侍衛)
	else_if = { limit = { has_trait = wu_jinshi }     add = 25 }   # 武進士 — the metropolitan/palace laureate
	else_if = { limit = { has_trait = wu_juren }      add = 15 }   # 武舉人 — the provincial graduate
	else_if = { limit = { has_trait = wu_shengyuan }  add = 10 }   # 武生員/武秀才 — the military licentiate
}

# [#34] KEJU PASS CHANCE — the 0..100 probability a coming-of-age court character passes
# the examination (used as `chance = qing_keju_pass_chance_svalue` in QING_exam_sit_candidate).
# Evaluated in the CANDIDATE'S character scope. Blends the realm's metropolitan pass-rate
# (employer.var:qing_exam_pass_rate, itself ladder-health net of corruption/捐納 drag) with the
# candidate's own scholarly aptitude (finesse + charisma), so an able child of a vigorous realm
# passes readily while a dull one in a clogged ladder rarely does. Bounded 5..90 (the exam is
# never a certainty nor wholly closed). Reading a country var via the employer link is done
# through a guarded `if = { limit = { employer = { var:X >= N } } }` band (the proven form —
# a bare `add = employer.var:X` accessor is not attested), so the pass-rate enters as a coarse
# floor rather than a continuous term; the finesse/charisma terms then separate candidates.
qing_keju_pass_chance_svalue = {
	value = 20
	# scholarly aptitude: finesse is the examiner's chief measure, charisma the lesser.
	add = finesse
	add = finesse
	add = charisma
	# the realm's pass-rate enters as a coarse band (guarded employer read).
	if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 75 } }  add = 40 }
	else_if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 60 } }  add = 30 }
	else_if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 45 } }  add = 20 }
	else_if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 30 } }  add = 10 }
	# bound: never a certainty, never wholly closed.
	min = 5
	max = 90
}

# [#48] WUJU PASS CHANCE — the martial-exam counterpart of qing_keju_pass_chance_svalue.
# The 武舉 was a test of 弓馬技勇 (bow, horse, strength), so MARTIAL is the chief measure
# here where finesse was the civil exam's. Blends the candidate's martial with the realm's
# pass-rate band (same guarded employer read; the wuju rode the same triennial calendar and
# ladder-health as the civil exam, so it reuses qing_exam_pass_rate rather than a second meter).
# Evaluated in the CANDIDATE'S character scope. Bounded 5..90.
qing_wuju_pass_chance_svalue = {
	value = 20
	# martial prowess: the examiner's chief measure on the outer field.
	add = martial
	add = martial
	add = charisma
	# the realm's pass-rate enters as a coarse band (guarded employer read).
	if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 75 } }  add = 40 }
	else_if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 60 } }  add = 30 }
	else_if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 45 } }  add = 20 }
	else_if = { limit = { exists = employer  employer = { has_variable = qing_exam_pass_rate  var:qing_exam_pass_rate >= 30 } }  add = 10 }
	# bound: never a certainty, never wholly closed.
	min = 5
	max = 90
}

# Combined 4-skill worth of a character, used to rank council/office candidates
# (order_by) and — historically — to weight the council's effectiveness. Mirrors
# combined_stats_olympics_svalue in 00_event_values.txt. Evaluated in a
# character scope. [#34] The keju degree prestige is folded in so graduates head
# the generalist (chancellor / regent / first-vacant) benches.
combined_stats_council_svalue = {
	value = 0
	add = martial
	add = finesse
	add = charisma
	add = zeal
	add = qing_degree_prestige_svalue
}

# [#35 review-fix] A Deliberative-Council banner lord's OPENING court backing (0..100):
# his combined-stats worth + a loyal-ish base of 30. Character scope. Wrapped in a SINGLE
# script value so QING_delib_seed_backing can seed it with ONE set_variable and NO
# change_variable read-back — a lord conjured by create_character cannot have his own script
# vars read back the same tick (the documented keju #26 antipattern: the immediate
# change_variable read-back fails and floods the log). combined_stats_council_svalue reads only
# engine stats + traits (immediately readable on a fresh character), so this is flood-free.
qing_delib_backing_seed_svalue = {
	value = 0
	add = combined_stats_council_svalue
	add = 30
	min = 0
	max = 100
}

# [#221] Single-skill ranking values for the per-office appoint pickers. Each
# office's picker sorts its candidate bench DESCENDING by that office's ONE
# governing skill (a war minister ranked on martial, a rites minister on zeal,
# etc.), so the ablest man FOR THAT POST heads the list. The Grand Chancellor /
# Grand Regent pickers instead use combined_stats_council_svalue above (sum of
# all four), since those summit posts are judged as generalists. Bare skill
# tokens are not demonstrated as order_by exports in this codebase, so each is
# wrapped in an explicit script value (proven order_by pattern). Character scope.
# [#34] Each folds in the keju degree prestige too, so a graduate heads even a
# single-skill office bench over an equally-skilled commoner (the soft preference).
council_sort_martial  = { value = 0  add = martial   add = qing_wu_degree_prestige_svalue }   # [#48] martial bench ranks on the WUJU (武舉) ladder, not the civil one
council_sort_finesse  = { value = 0  add = finesse   add = qing_degree_prestige_svalue }
council_sort_charisma = { value = 0  add = charisma  add = qing_degree_prestige_svalue }
council_sort_zeal     = { value = 0  add = zeal      add = qing_degree_prestige_svalue }

# [#384] New World crop adoption probability for an eligible marginal target province.
# Evaluated in PROVINCE scope inside QING_COLON_nwcrop_diffuse, where qing_nwcrop_adj has
# just been tallied = the number of adjacent frozen crop-source provinces. Base 15% per crop
# neighbour, capped at 60%. Wrapping the variable-driven chance in a script value is the
# proven `random = { chance = <svalue> }` idiom (Invictus 00_mission_cyrene.txt:33
# chance = current_ruler.cyr_martial_prob_svalue); a bare `chance = var:X` is NOT attested.
qing_nwcrop_flip_chance_svalue = {
	value = 0
	add = var:qing_nwcrop_adj
	multiply = 15
	max = 60
}

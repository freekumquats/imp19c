DEMAND_luxury_elasticity = {
	# DEFUNCT
	if = {
		limit = {
			has_variable = governorship_population
		}
		value = var:governorship_population
	}
	else = {
		value = 0
		every_governorship_state = {
			every_state_province = {
				add = total_population
			}
		}
	}
}

# [#281] rifles DEMAND is MILITARY, not a population luxury. Re-based off the same
# army/arsenal/fortress driver as munitions (mirrors DEMAND_early_munitions_base in
# DEMAND_svalues.txt) and tech-gated on tech_firearms, so demand scales with the
# military that also produces the supply (arsenals/fortresses) rather than with pop
# wealth. A country with no army and no arsenals demands ~0 rifles -> no phantom
# shortage. Rifles are infantry small-arms, so the army term is the dominant driver.
# (Formerly a universal DEMAND_set_demand_from_luxury good, which gave every populated
# governorship rifle demand regardless of whether it fielded troops.)
DEMAND_rifles_base = {
	value = 0
	if = {
		# [#281/#290] rifles are consumed by the CONCRETE elite `riflemen` unit (army_riflemen.txt),
		# not by all cohorts. In the 18thC (incl. the 1763 start) line infantry carried smoothbore
		# MUSKETS — rifles were rare, slow-loading arms for skirmishers/jaeger only, and did not
		# become general infantry issue until rifling was industrialised (Minie ball / percussion,
		# ~1850s). Gate demand on ACTUALLY FIELDING riflemen: the riflemen unit itself requires
		# invention=tech_rifles + a rifles surplus to raise (see its allow block), so a country
		# with no riflemen demands ~0 rifles, and at the 1763 musket-era start (no tech_rifles,
		# no riflemen) global rifle demand is ~0 -> no early phantom shortage.
		#
		# NOTE: num_of_unit_type is PROVEN only as a TRIGGER (value >= N; se_LOGISTICS steamer
		# check), NOT as a value-read (the artillery DEMAND TODO flags the value form as absent).
		# So we gate on "fields >=1 riflemen" and scale by army size as the demand proxy, rather
		# than reading the exact riflemen count. When a value-read of num_of_unit_type is proven,
		# swap the DEMAND_military_from_armies term for the literal riflemen count (see #290).
		# [logfix] this svalue evaluates in GOVERNORSHIP scope, but num_of_unit_type is a
		# COUNTRY trigger (error.log: "num_of_unit_type trigger [ Wrong scope: governorship,
		# expected country ]" — 4071 lines/session). Hop to country via owner alongside the
		# invention gate (which was already correctly hopped).
		limit = {
			owner = {
				invention = tech_rifles
				num_of_unit_type = { type = riflemen  value >= 1 }
			}
		}
		add = DEMAND_military_from_armies
		every_governorship_state = {
			every_state_province = {
				add = num_of_arsenal_building
			}
		}
		add = {
			value = JOBS_military_governorship
			min = 1
		}
		multiply = DEMAND_army_ratio_rifles
	}
	multiply = owner.MODIFIER_military_supply_consumption
}

DEMAND_rifles = { # Total
	value = DEMAND_rifles_base
	multiply = DEMAND_military_supplies_modifier
	multiply = DEMAND_elasticity_impact
	min = 0
}

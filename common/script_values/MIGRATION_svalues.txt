########################################################################
# MIGRATION_svalues.txt — push / pull scoring for bottom-up migration.
#
# DESIGN (firm): migration is BOTTOM-UP. Pops leave a province because life
# there is bad for them (push) and arrive somewhere because life there is
# better (pull). The government / player is only ONE factor — it nudges the
# scores with modifiers, it never orders a move. se_MIGRATION.txt reads these
# two province scores each decade pulse, moves a grievance-weighted pop from a
# high-push province to the best-pull neighbour, and (if the settlers flip the
# destination's plurality) seeds a claim — the organic route to the settler
# colonialism the Qing missions do by fiat.
#
# REUSE (firm): these scores RECOMPUTE NOTHING. Every term is an already-cached
# economy field read straight from its scope:
#   JOBS_unemployed_pops_province / JOBS_available_slots  (JOBS_svalues.txt)
#   province_unrest, civilization_value, total_population  (engine, per-tick)
#   governorship.ECON_governorship_food_shortage           (ECON_svalues.txt,
#       itself a sum of the cached var:shortage_<food> the consume step writes)
#   governorship.WEALTH_governorship_proletariat_per_capita vs the national
#       average (owner.var:WEALTH_total_private_moveable_wealth / country_population,
#       both cached by WEALTH_cache_national_wealth) — the wage/wealth adequacy
#       signal (see MIGRATION_wealth_relative_province below).
# So a pulse is O(provinces) cheap reads, not a resimulation.
#
# Prefix = MIGRATION_
########################################################################

# ======================================================================
# WEALTH ADEQUACY — the missing wage->welfare->behaviour link. The wage engine
# (se_ECON_wealth.txt / JOBS_svalues.txt) already PAYS wages into the strata
# wealth pools every quarter and nets cost-of-living against them, but nothing
# read that wealth back into how pops BEHAVE. This helper turns a province's
# per-capita wealth into a scale-free migration signal: how it compares to the
# NATIONAL average.
#   Scope: province. Returns a fraction clamped to [-1, 1]:
#     0  = local per-capita is exactly at the national per-capita average
#    <0  = poorer than average  (impoverishment -> emigration pressure)
#    >0  = richer than average   (prosperity   -> immigration draw)
#
# LIKE-FOR-LIKE BASIS (critical, review-fix): BOTH sides are ALL-STRATA per-capita.
# The numerator is governorship.WEALTH_governorship_per_capita (= governorship
# total private wealth / governorship population, WEALTH_svalues.txt:447, floored
# at 0.001 so it is never 0); the denominator is the national total private
# wealth / national population. An earlier version compared PROLETARIAT-only
# per-capita against the ALL-STRATA national mean — a units mismatch that made
# the ratio < 1 almost everywhere (the proletariat is a poorer-than-average
# stratum), so the deviation was negative nearly everywhere: the push leg fired
# empire-wide and the pull leg was dead, and a governorship with no proletariat
# read as maximally destitute (per-capita 0 -> deviation -1 -> full push). Using
# the all-strata per-capita on BOTH sides makes the signal genuinely symmetric
# around 0 (a governorship richer than the national average draws people; poorer
# sheds them) and removes the zero-stratum edge (the all-strata per-capita has a
# 0.001 floor and is populated for every governorship).
#
# All operands are already-cached reads. Guarded so it is a clean 0 before the
# wealth system has initialised (no wealth_setup_done) or on a country without the
# moveable-wealth cache — costing nothing when unset.
# ======================================================================
MIGRATION_wealth_relative_province = {
	value = 0
	if = {
		limit = {
			has_global_variable = wealth_setup_done
			owner = { has_variable = WEALTH_total_private_moveable_wealth }
		}
		value = governorship.WEALTH_governorship_per_capita
		divide = {
			# national per capita = cached national private wealth / population
			# (all strata, matching the all-strata numerator above)
			value = owner.var:WEALTH_total_private_moveable_wealth
			divide = owner.country_population
			min = 0.0001 # floor the denominator, never div/0
		}
		subtract = 1 # ratio -> deviation from the average
		min = -1
		max = 1
	}
}

# ======================================================================
# PUSH — how strongly a province makes its pops want to LEAVE. Sum of the
# hardships a pop can flee: joblessness/overpopulation, unrest, hunger, and
# the disruption of war/occupation. Per-pop grievances that vary WITHIN a
# province (being a cultural/religious minority, being poor) are NOT here —
# they belong in se_MIGRATION's per-pop weight, so only the *who leaves*
# depends on them, not *whether this province sheds anyone*.
#   Scope: province
# ======================================================================
MIGRATION_push_province = {
	value = 0

	# Overpopulation / joblessness: pops with no slot to work are the readiest
	# to leave. Already cached (total_population*0.5 - used slots).
	add = {
		value = JOBS_unemployed_pops_province
		min = 0
	}

	# Unrest — a restless province bleeds people. province_unrest is ~0..10.
	add = {
		value = province_unrest
		multiply = 2
	}

	# Hunger — the governorship's average food shortage (sum over food goods,
	# each ~0..1). A famine is one of history's great migration engines.
	add = {
		value = governorship.ECON_governorship_food_shortage
		multiply = 8
	}

	# War & occupation — people flee the front. Both read off the province's
	# own owner / controller, no recompute.
	if = {
		limit = { owner = { war = yes } }
		add = 4
	}
	if = {
		limit = {
			exists = controller
			NOT = { controller = owner }
		}
		add = 10
	}

	# Impoverishment — a province whose proletariat is poorer than the national
	# average sheds people (the wage/wealth adequacy link: low wages net of cost
	# of living -> a thin wealth pool -> emigration pressure). Only the poor-half
	# pushes (max 0 on the deviation), so prosperity does not create negative push;
	# it draws people via the pull score instead. Scaled to 6 at the -1 floor
	# (destitute vs the average), comparable to the war term — real but not
	# swamping unrest/famine. Reads MIGRATION_wealth_relative_province (guarded,
	# all-cached; 0 before the wealth system inits).
	add = {
		value = MIGRATION_wealth_relative_province
		max = 0    # only the below-average leg pushes
		multiply = -6
	}

	# GOVERNMENT LEVER (the "one factor among many"): a government may tag a
	# province with migr_gov_push to encourage people to leave (e.g. clearing a
	# crowded heartland). Optional — read behind a has_variable guard so it costs
	# nothing when unset. This is how the state INCENTIVISES, never orders.
	if = {
		limit = { has_variable = migr_gov_push }
		add = var:migr_gov_push
	}

	min = 0
}

# ======================================================================
# PULL — how strongly a province ATTRACTS incomers. The mirror of push:
# spare jobs, calm, food, and development draw people in; a province that is
# full, restless, hungry or fought-over repels them (the score can go
# negative, which se_MIGRATION treats as "do not settle here").
#   Scope: province
# ======================================================================
MIGRATION_pull_province = {
	value = 0

	# Opportunity — unfilled job slots are the classic magnet. Cached, and
	# already negative when a province is overfull (so it self-repels).
	add = {
		value = JOBS_available_slots
	}

	# Development / prosperity — civilised, built-up land draws settlers.
	add = {
		value = civilization_value
		divide = 20
	}

	# Calm attracts; unrest deters.
	subtract = {
		value = province_unrest
		multiply = 2
	}

	# A hungry destination is no refuge.
	subtract = {
		value = governorship.ECON_governorship_food_shortage
		multiply = 8
	}

	# No one migrates toward a war zone.
	if = {
		limit = { owner = { war = yes } }
		subtract = 6
	}
	if = {
		limit = {
			exists = controller
			NOT = { controller = owner }
		}
		subtract = 20
	}

	# Prosperity — a province whose proletariat is richer than the national
	# average draws incomers (the mirror of the push impoverishment leg: fat wage
	# pools = opportunity). Only the above-average leg attracts (min 0 on the
	# deviation), so a poor destination does not gain spurious pull here — it is
	# already repelling via its own push and its thin opportunity. Scaled to 6 at
	# the +1 ceiling, matching the push weight so the signal is symmetric.
	add = {
		value = MIGRATION_wealth_relative_province
		min = 0    # only the above-average leg attracts
		multiply = 6
	}

	# GOVERNMENT LEVER (the "one factor among many"): a government may tag a
	# province with migr_gov_pull to make it more attractive to settlers (e.g.
	# a frontier the state wants peopled — 屯田 / 闖關東). Optional, guarded, and
	# it only tilts the odds: the pop still moves of its own accord, and only if
	# the destination still out-scores where it lives. The state incentivises,
	# it does not order.
	if = {
		limit = { has_variable = migr_gov_pull }
		add = var:migr_gov_pull
	}
}

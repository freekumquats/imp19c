########################################################################
# MIGRATION_svalues.txt — push / pull scoring for bottom-up migration.
#
# DESIGN (firm): migration is BOTTOM-UP. Pops leave a province because life
# there is bad for them (push) and arrive somewhere because life there is
# better (pull). The government / player is only ONE factor — it nudges the
# scores with modifiers, it never orders a move. se_MIGRATION.txt reads these
# two province scores each decade pulse, moves a grievance-weighted pop from a
# high-push province to the best-pull neighbour, and (if the settlers flip the
# destination's plurality) seeds a claim — the organic route to the settler
# colonialism the Qing missions do by fiat.
#
# REUSE (firm): these scores RECOMPUTE NOTHING. Every term is an already-cached
# economy field read straight from its scope:
#   JOBS_unemployed_pops_province / JOBS_available_slots  (JOBS_svalues.txt)
#   province_unrest, civilization_value, total_population  (engine, per-tick)
#   governorship.ECON_governorship_food_shortage           (ECON_svalues.txt,
#       itself a sum of the cached var:shortage_<food> the consume step writes)
# So a pulse is O(provinces) cheap reads, not a resimulation.
#
# Prefix = MIGRATION_
########################################################################

# ======================================================================
# [migration-push fix] POP OVER-CAPACITY OVERFLOW (province scope). The prior push term guessed
# `has_province_modifier = overpopulation` — but that hardcoded engine auto-modifier is NOT queryable
# by name from script, so the term was inert (boot-test: over-capacity drove no migration). Compute
# the overflow DIRECTLY instead: total_population minus the province's readable pop ceiling
# (modifier:local_population_capacity — the proven numeric read, AI_svalues.txt:379 / MODIFIER_svalues.txt).
# Positive only when the province is genuinely over its ceiling; 0 otherwise (min = 0). Pure arithmetic —
# no trigger, no var-on-RHS comparison (safe per the RHS-operator rule).
MIGRATION_pop_overcapacity_overflow = {
	# Scope: province
	value = total_population
	subtract = modifier:local_population_capacity
	min = 0
}

# [migration-push fix] LOW STATE FOOD (province scope, reads its state). A state whose stored food is
# low against its capacity is a hungry state whose pops want out. has_state_food / has_state_food_capacity
# are the proven bare-readable state-food numerics (governor_policies:140 / se_QING_DECLINE granary), read
# here via the proven `state = { add = <primitive> }` scope-wrapper (JOBS_svalues idiom) rather than a
# dotted `state.has_state_food` accessor (unproven for engine primitives). Score = the shortfall of stored
# food below HALF the capacity, normalised by capacity and ×20: a full granary scores 0, an empty one ~10.
# Guarded so a 0-capacity state (deserts, sea-tiles) contributes nothing and never divides by zero.
MIGRATION_state_food_deficit = {
	# Scope: province
	value = 0
	if = {
		limit = { state = { has_state_food_capacity > 0 } }
		# shortfall below the comfort line (half capacity), via the state scope-wrapper
		add = {
			value = 0
			state = { add = has_state_food_capacity }
			multiply = 0.5
		}
		subtract = {
			value = 0
			state = { add = has_state_food }
		}
		# normalise by capacity and scale
		divide = {
			value = 0
			state = { add = has_state_food_capacity }
		}
		multiply = 20
		min = 0
	}
}

# ======================================================================
# PUSH — how strongly a province makes its pops want to LEAVE. Sum of the
# hardships a pop can flee: joblessness/overpopulation, unrest, hunger, and
# the disruption of war/occupation. Per-pop grievances that vary WITHIN a
# province (being a cultural/religious minority, being poor) are NOT here —
# they belong in se_MIGRATION's per-pop weight, so only the *who leaves*
# depends on them, not *whether this province sheds anyone*.
#   Scope: province
# ======================================================================
MIGRATION_push_province = {
	value = 0

	# Overpopulation / joblessness: pops with no slot to work are the readiest
	# to leave. Already cached (total_population*0.5 - used slots).
	add = {
		value = JOBS_unemployed_pops_province
		min = 0
	}

	# [migration-push fix / user boot-test] POPULATION OVER CAPACITY — the historical 人地矛盾 push
	# (Hong Liangji): people leave the crowded core for the frontier. The prior draft gated this on
	# `has_province_modifier = overpopulation` (the engine's hardcoded auto-modifier) — but the boot
	# test confirmed that term does NOTHING, because that auto-modifier is not queryable by name from
	# script. REPLACED with a DIRECT arithmetic overflow: how many pops the province holds ABOVE its
	# readable ceiling (MIGRATION_pop_overcapacity_overflow = total_population - modifier:local_population_capacity,
	# min 0). Weighted heavy (per user: over-capacity should drive a lot of migration). A province at or
	# under its ceiling contributes 0; one 10 pops over adds ~15.
	add = {
		value = MIGRATION_pop_overcapacity_overflow
		multiply = 1.5
	}
	# [migration-push fix / user boot-test] LOW STATE FOOD CAPACITY — a state whose granary is running
	# empty against its capacity is a hungry state whose pops seek fuller lands. DIRECT read of the state
	# food store vs capacity (MIGRATION_state_food_deficit), distinct from the governorship shortage term
	# below (which is a per-good production shortfall, not the stored-food level). Empty granary adds ~10.
	add = {
		value = MIGRATION_state_food_deficit
	}

	# Unrest — a restless province bleeds people. province_unrest is ~0..10.
	add = {
		value = province_unrest
		multiply = 2
	}

	# Hunger — the governorship's average food shortage (sum over food goods,
	# each ~0..1). A famine is one of history's great migration engines.
	add = {
		value = governorship.ECON_governorship_food_shortage
		multiply = 8
	}

	# War & occupation — people flee the front. Both read off the province's
	# own owner / controller, no recompute.
	if = {
		limit = { owner = { war = yes } }
		add = 4
	}
	if = {
		limit = {
			exists = controller
			NOT = { controller = owner }
		}
		add = 10
	}

	# GOVERNMENT LEVER (the "one factor among many"): a government may tag a
	# province with migr_gov_push to encourage people to leave (e.g. clearing a
	# crowded heartland). Optional — read behind a has_variable guard so it costs
	# nothing when unset. This is how the state INCENTIVISES, never orders.
	if = {
		limit = { has_variable = migr_gov_push }
		add = var:migr_gov_push
	}

	min = 0
}

# ======================================================================
# PULL — how strongly a province ATTRACTS incomers. The mirror of push:
# spare jobs, calm, food, and development draw people in; a province that is
# full, restless, hungry or fought-over repels them (the score can go
# negative, which se_MIGRATION treats as "do not settle here").
#   Scope: province
# ======================================================================
MIGRATION_pull_province = {
	value = 0

	# Opportunity — unfilled job slots are the classic magnet. Cached, and
	# already negative when a province is overfull (so it self-repels).
	add = {
		value = JOBS_available_slots
	}

	# Development / prosperity — civilised, built-up land draws settlers.
	add = {
		value = civilization_value
		divide = 20
	}

	# Calm attracts; unrest deters.
	subtract = {
		value = province_unrest
		multiply = 2
	}

	# A hungry destination is no refuge.
	subtract = {
		value = governorship.ECON_governorship_food_shortage
		multiply = 8
	}

	# [#49] OVER CAPACITY REPELS — mirror of the push term: a province already past its
	# pop ceiling (engine `overpopulation` auto-modifier) or starving (`state_starving`)
	# is no place to move TO. Partly self-repels via the negative JOBS_available_slots
	# above, but weight it explicitly so migrants steer AWAY from the crowded core toward
	# the frontier, not merely shuffle between two overfull heartland provinces.
	if = {
		limit = { has_province_modifier = overpopulation }
		subtract = 12
	}
	# [#49 review-fix] province-scope `starving_city` (see the push-side note) — not the
	# unproven state_starving chain.
	if = {
		limit = { has_province_modifier = starving_city }
		subtract = 10
	}

	# No one migrates toward a war zone.
	if = {
		limit = { owner = { war = yes } }
		subtract = 6
	}
	if = {
		limit = {
			exists = controller
			NOT = { controller = owner }
		}
		subtract = 20
	}

	# GOVERNMENT LEVER (the "one factor among many"): a government may tag a
	# province with migr_gov_pull to make it more attractive to settlers (e.g.
	# a frontier the state wants peopled — 屯田 / 闖關東). Optional, guarded, and
	# it only tilts the odds: the pop still moves of its own accord, and only if
	# the destination still out-scores where it lives. The state incentivises,
	# it does not order.
	if = {
		limit = { has_variable = migr_gov_pull }
		add = var:migr_gov_pull
	}
}

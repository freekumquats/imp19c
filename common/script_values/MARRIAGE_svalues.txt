# MARRIAGE_svalues.txt — [#311] script_values for the dynastic-marriage POWER-PARITY gate.
#
# [user] "not just rank ... Great Power has an economic + military metric" — equal standing
# for a match is measured on the mod's OWN great-power SCORE, DIPLOMACY_power (economy +
# military + tech + RECURSIVE subjects + admin + stability; see DIPLOMACY_svalues.txt), NOT
# the formal rank enum. The parity trigger (00_marriage_triggers.txt) needs a RATIO band, but
# a value-block RHS inside a trigger is unattested here — so the scaling is hoisted into this
# svalue and compared via the PROVEN `scope:target.<svalue>` cross-scope read (the
# `scope:target.power_base` / `scope:target.martial` idiom). One svalue suffices: own power ×3.
#
# BOM + LF (common/script_values convention). Prefix = MARRIAGE_.
########################################################################

# MARRIAGE_power_x3_own — COUNTRY. This country's great-power SCORE multiplied by three: the
# ceiling a partner's score may reach before the gulf is too wide to wed freely. Read on BOTH
# houses (self and scope:target) so the parity test is symmetric: a pair is of comparable
# standing when NEITHER house's power exceeds THREE TIMES the other's — a near-peer band that
# lets two great powers, or two minor houses, marry freely, but forces the great-power-to-
# petty-princeling match through the explicit dispensation/dowry BRIDGE (as the history had it).
MARRIAGE_power_x3_own = {
	# Scope: Country
	value = DIPLOMACY_power
	multiply = 3
}


# [#313] the ORDERING key for the candidate list. Scope = a candidate country (evaluated in the
# ordered_country pass of marriage_open_candidates). Returns the cached display var selected by
# ROOT's marriage_sort_mode: power (DIPLOMACY_power, default) | age (best eligible child's age) |
# skill (best child's top attribute). ordered_country builds the list highest-value-first, so a
# higher return = earlier in the list.
MARRIAGE_sort_svalue = {
	# Scope: Country (the candidate)
	if = {
		limit = { ROOT = { has_variable = marriage_sort_mode  var:marriage_sort_mode = flag:age } }
		value = 0
		if = { limit = { has_variable = marriage_display_best_age }  add = var:marriage_display_best_age }
	}
	else_if = {
		limit = { ROOT = { has_variable = marriage_sort_mode  var:marriage_sort_mode = flag:skill } }
		value = 0
		if = { limit = { has_variable = marriage_display_best_skill }  add = var:marriage_display_best_skill }
	}
	else = {
		# default: power (the same economy+military score the parity gate uses)
		value = DIPLOMACY_power
	}
}

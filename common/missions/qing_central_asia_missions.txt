########################################################################
# qing_central_asia_missions.txt — THE WESTERN REGIONS (經略西域), #448.
#
# A dedicated expansion tree carrying the Qing PAST the Xinjiang consolidation
# (qing_xinjiang_missions.txt, #422, which pacifies Dzungaria + the Tarim) and out
# into Central Asia proper: the Kazakh steppe (the three Juz — GKH/ORT/KSH), and the
# khanates of the Ferghana and Transoxiana (Kokand KOK, Bukhara BUK, Khiva KHV).
#
# Historically the Qianlong conquest of Xinjiang (1755-59) brought the empire to the
# doorstep of the Kazakh hordes and the Central Asian khanates, who sent tribute to
# Beijing; this tree lets the player push that dominance further — the steppe hordes
# brought into the tributary system, the caravan cities of the Silk Road overawed, and
# (at the far reach) the Ferghana valley itself held.
#
# As with the Burma (#421) tree, the PLAYER does the real work: the task goals gate on
# the Qing genuinely bringing the steppe tags under the wing (is_subject_of) or
# CONTROLLING the ground (provinces in the Central Asian regions). What is scripted
# (se_QING_CENTRAL_ASIA.txt) is only the texture — the tributary embrace of the hordes
# and khanates (FUNC_make_subject) and the lasting dominion of the Western Regions.
#
# CAPSTONE — 西域大定 THE WESTERN REGIONS PACIFIED: the steppe and the caravan cities
# under the Son of Heaven; the lasting qing_western_regions dominion modifier sealed.
#
# Gating: CHI-only, player-only (ai_chance factor 0). Requires the Xinjiang tree's
# consolidation as the launch-point is unnecessary as a hard gate (the player may drive
# west by any road), but the potential requires at least one Central Asian neighbour to
# exist. Tasks cost treasury / political influence. Icons reuse stock test1/2/3.
#
# Sources: Peter Perdue, China Marches West: The Qing Conquest of Central Eurasia (2005);
#   James Millward, Beyond the Pass: Economy, Ethnicity and Empire in Qing Xinjiang (1998)
#   & Eurasian Crossroads: A History of Xinjiang (2007); Joseph Fletcher, "Ch'ing Inner
#   Asia" (Cambridge History of China v.10).
########################################################################

qing_central_asia_mission = {
	icon = test3
	header = mission_image_test

	repeatable = no

	chance = { }
	ai_chance = { factor = 0 }   # player-only

	potential = {
		tag = CHI
		is_ai = no
		# there must be a Central Asian power to reach for — a Kazakh horde or a khanate.
		OR = {
			exists = c:GKH
			exists = c:ORT
			exists = c:KSH
			exists = c:KOK
			exists = c:BUK
			exists = c:KHV
		}
	}

	abort = {
		NOT = { tag = CHI }
	}

	on_start = {
		LOG_line = { sys = QING  msg = "MISSION the Western Regions (經略西域) started for" }
	}

	on_completion = {
		LOG_line = { sys = QING  msg = "MISSION the Western Regions COMPLETED for" }
	}

	# ==================================================================
	# ROOT — THE ROAD BEYOND THE PASS (出關). Commit the frontier armies and the caravan
	# revenues to the westward push past Jiayuguan and the Ili valley — the point from
	# which the empire reaches into the steppe and the khanates.
	# ==================================================================
	qing_ca_beyond_pass = {
		icon = test1
		allow = {
			treasury >= 100
			political_influence >= 25
		}
		on_start = {
			add_treasury = -100
			add_political_influence = -25
		}
		on_completion = {
			custom_tooltip = qing_ca_beyond_pass_tt
			current_ruler = { add_popularity = 10 }
			LOG_line = { sys = QING  msg = "MISSION task Central Asia road beyond the pass for" }
		}
	}

	# ==================================================================
	# THE KAZAKH STEPPE (撫哈薩克). Bring the three Juz — the Great, Middle and Little
	# Hordes — into the tributary system, as they historically sent tribute to Qianlong
	# after the fall of the Dzungars. At least two of the three hordes under the wing.
	# ==================================================================
	qing_ca_kazakh = {
		icon = test2
		requires = { qing_ca_beyond_pass }
		allow = {
			political_influence >= 30
			custom_tooltip = {
				text = qing_ca_needs_kazakh_tt
				# at least one of the three Kazakh hordes must be on the map to be reached.
				OR = { exists = c:GKH  exists = c:ORT  exists = c:KSH }
			}
		}
		on_start = {
			add_political_influence = -30
		}
		on_completion = {
			custom_tooltip = qing_ca_kazakh_tt
			# bring every extant Kazakh horde into the tributary fold (the make-subject-by-fiat
			# idiom the southern-frontier tree uses; each guarded on existence + not-already-bound).
			QING_ca_embrace_kazakh = yes
			current_ruler = { add_popularity = 15 }
			LOG_line = { sys = QING  msg = "MISSION task Central Asia Kazakh hordes embraced for" }
		}
	}

	# ==================================================================
	# THE CARAVAN CITIES (懾服浩罕). Overawe the khanates of the Silk Road — Kokand,
	# Bukhara, Khiva — into the tributary system. At least one khanate under the wing.
	# ==================================================================
	qing_ca_khanates = {
		icon = test2
		requires = { qing_ca_beyond_pass }
		allow = {
			political_influence >= 30
			custom_tooltip = {
				text = qing_ca_needs_khanates_tt
				# at least one of the Silk Road khanates must be on the map to be reached.
				OR = { exists = c:KOK  exists = c:BUK  exists = c:KHV }
			}
		}
		on_start = {
			add_political_influence = -30
		}
		on_completion = {
			custom_tooltip = qing_ca_khanates_tt
			# overawe every extant Silk Road khanate into the tributary fold (guarded per tag).
			QING_ca_embrace_khanates = yes
			current_ruler = { add_popularity = 15 }
			LOG_line = { sys = QING  msg = "MISSION task Central Asia khanates overawed for" }
		}
	}

	# ==================================================================
	# THE FERGHANA VALLEY (據費爾干納). The deepest reach: hold the ground of the Ferghana
	# — the rich valley of Kokand and Tashkent, the heart of Transoxiana's Silk Road.
	# The player must actually take and hold the ground.
	# ==================================================================
	qing_ca_ferghana = {
		icon = test3
		requires = { qing_ca_khanates }
		allow = {
			treasury >= 80
			custom_tooltip = {
				text = qing_ca_needs_ferghana_tt
				# the Qing control ground in the Ferghana region (Kokand / Tashkent).
				any_owned_province = { count >= 1  is_in_region = Fergana }
			}
		}
		on_start = {
			add_treasury = -80
		}
		on_completion = {
			custom_tooltip = qing_ca_ferghana_tt
			current_ruler = { add_popularity = 20 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION task Central Asia Ferghana held for" }
		}
	}

	# ==================================================================
	# CAPSTONE — 西域大定 THE WESTERN REGIONS PACIFIED. The steppe hordes and the caravan
	# cities under the Son of Heaven, and Qing ground in the heart of Transoxiana. The
	# lasting qing_western_regions dominion modifier is sealed — the Silk Road revived
	# under the imperial peace, tribute and trade flowing east.
	# ==================================================================
	qing_ca_capstone = {
		icon = test1
		requires = { qing_ca_kazakh  qing_ca_ferghana }
		allow = {
			custom_tooltip = {
				text = qing_ca_needs_pacified_tt
				# the steppe embraced AND ground held in the Ferghana heart.
				any_subject = {
					count >= 2
					OR = { tag = GKH  tag = ORT  tag = KSH  tag = KOK  tag = BUK  tag = KHV }
				}
				any_owned_province = { count >= 1  is_in_region = Fergana }
			}
		}
		on_completion = {
			custom_tooltip = qing_ca_capstone_tt
			add_country_modifier = { name = qing_western_regions  duration = -1 }
			current_ruler = { add_popularity = 40 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION the Western Regions CAPSTONE (西域大定) for" }
		}
	}
}

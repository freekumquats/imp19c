########################################################################
# qing_settle_frontier_missions.txt — SETTLE THE NOMADS (定牧墾邊), #451.
#
# The Qing consolidation of the great northern frontier — the Mongol steppe and the
# Manchurian homeland — from a belt of autonomous military governorships and banner
# pastures into settled, directly-held imperial ground. 定牧 "fix the pastures" (the
# banner/league system that pinned the once-mobile Mongols to fixed grazing) + 墾邊
# "reclaim the frontier" (the 屯田 military-agricultural colonies and the slow tide of
# Han settlement that broke the sod of Inner Mongolia and the Northeast).
#
# The 1763 board (setup/main/00_default.txt): the north is held for the Qing (CHI) not
# directly but through a tier of subjects — the Mongol military governorships (ULS
# Uliastai / KBD Kobdo, with MGA the Urga monastic state nested under ULS) and the
# Manchurian ones (MNC Manchuria / MKD Mukden / HLJ Heilongjiang), plus TNN (Tannu Tuva /
# the Uriankhai) as a tributary. This tree is the arc that INTEGRATES that tier — the
# player brings the governorships' ground under direct rule and settles it.
#
# As with the other Qing enterprise trees, the PLAYER does the real work: the task goals
# gate on the Qing genuinely OWNING ground in the frontier regions (Mongolia; Liaoning +
# the Far East), which the player reaches by integrating / annexing the autonomous
# governorships that hold it — and on the Tannu tie. What is scripted
# (se_QING_SETTLE_FRONTIER.txt) is only the texture: the reclamation colonies laid on the
# settled ground, and (a safety net) the tributary embrace of Tannu if it has broken away.
#
# It FORKS after the shared opening (the frontier policy set):
#   ARC N — 定漠北 THE MONGOL LANDS: integrate the Mongol governorships and settle the
#     steppe — hold ground across Mongolia, the banners pinned and the colonies planted.
#   ARC E — 實東北 THE MANCHURIAN HOMELAND: integrate the Northeast governorships and
#     settle the taiga — hold ground across Liaoning and the Far East, the willow palisade
#     opened to the plough.
#
# CAPSTONE — 定牧墾邊 THE FRONTIER SETTLED: the steppe and the taiga under direct rule and
# the plough, the northern march made imperial ground; the lasting qing_frontier_settled
# dominion modifier sealed.
#
# Gating: CHI-only, player-only (ai_chance factor 0). Tasks cost treasury / political
# influence and gate on the advancing OWNERSHIP of the frontier regions. Icons reuse stock
# test1/2/3 (the whole Qing suite's placeholder convention).
#
# Sources: Peter Perdue, China Marches West (2005); Owen Lattimore, Inner Asian Frontiers
#   of China (1940); Mark Elliott, The Manchu Way: The Eight Banners and Ethnic Identity in
#   Late Imperial China (2001); C. Patterson Giersch on Qing frontier settlement.
########################################################################

qing_settle_frontier_mission = {
	icon = test3
	header = mission_image_test

	repeatable = no

	chance = { }
	ai_chance = { factor = 0 }   # player-only

	potential = {
		tag = CHI
		is_ai = no
	}

	abort = {
		NOT = { tag = CHI }
	}

	on_start = {
		LOG_line = { sys = QING  msg = "MISSION settle the nomads (定牧墾邊) started for" }
	}

	on_completion = {
		LOG_line = { sys = QING  msg = "MISSION settle the nomads COMPLETED for" }
	}

	# ==================================================================
	# ROOT — THE FRONTIER POLICY (定邊策). Set the policy of the settled frontier: fix the
	# banners to their pastures (定牧), open the willow palisade, and commit the treasury to
	# the military-agricultural colonies. The point of commitment for the northern march.
	# ==================================================================
	qing_settle_policy = {
		icon = test1
		allow = {
			treasury >= 120
			political_influence >= 25
		}
		on_start = {
			add_treasury = -120
			add_political_influence = -25
		}
		on_completion = {
			custom_tooltip = qing_settle_policy_tt
			# a safety net: if Tannu Tuva has slipped the leash, bring it back under the wing.
			QING_settle_embrace_tannu = yes
			current_ruler = { add_popularity = 10 }
			LOG_line = { sys = QING  msg = "MISSION task settle-frontier policy set for" }
		}
	}

	# ==================================================================
	# ARC N — 定漠北 THE MONGOL LANDS. Integrate the Mongol military governorships (ULS/KBD,
	# and MGA at Urga) and settle the steppe: hold the ground of Mongolia directly, the
	# banners pinned to their leagues and the colonies planted among them.
	# ==================================================================
	qing_settle_mongolia = {
		icon = test2
		requires = { qing_settle_policy }
		allow = {
			treasury >= 80
			custom_tooltip = {
				text = qing_settle_needs_mongolia_tt
				# the Qing hold ground directly across the Mongol lands (reached by integrating
				# the autonomous governorships that hold them as CHI subjects in 1763).
				any_owned_province = { count >= 8  is_in_region = Mongolia }
			}
		}
		on_start = {
			add_treasury = -80
		}
		on_completion = {
			custom_tooltip = qing_settle_mongolia_tt
			# lay the reclamation colonies on the settled Mongol ground.
			QING_settle_reclaim_mongolia = yes
			current_ruler = { add_popularity = 15 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION task settle-frontier Mongol lands settled for" }
		}
	}

	# ==================================================================
	# ARC E — 實東北 THE MANCHURIAN HOMELAND. Integrate the Northeast governorships
	# (MNC/MKD/HLJ) and settle the taiga: hold the ground of Liaoning and the Far East
	# directly, the willow palisade opened to the plough and the homeland peopled.
	# ==================================================================
	qing_settle_manchuria = {
		icon = test2
		requires = { qing_settle_policy }
		allow = {
			treasury >= 80
			custom_tooltip = {
				text = qing_settle_needs_manchuria_tt
				# the Qing hold ground directly across the Manchurian homeland (Liaoning + the
				# Far East), reached by integrating the autonomous governorships that hold them.
				any_owned_province = {
					count >= 8
					OR = { is_in_region = Liaoning  is_in_region = Far_East }
				}
			}
		}
		on_start = {
			add_treasury = -80
		}
		on_completion = {
			custom_tooltip = qing_settle_manchuria_tt
			# lay the reclamation colonies on the settled Manchurian ground.
			QING_settle_reclaim_manchuria = yes
			current_ruler = { add_popularity = 15 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION task settle-frontier Manchurian homeland settled for" }
		}
	}

	# ==================================================================
	# CAPSTONE — 定牧墾邊 THE FRONTIER SETTLED. The steppe and the taiga alike under direct
	# rule and the plough — the Mongol lands and the Manchurian homeland made imperial ground,
	# the autonomous governorships consolidated and the frontier peopled. The lasting
	# qing_frontier_settled dominion modifier is sealed: the northern march secured for good.
	# ==================================================================
	qing_settle_capstone = {
		icon = test1
		requires = { qing_settle_mongolia  qing_settle_manchuria }
		allow = {
			custom_tooltip = {
				text = qing_settle_needs_settled_tt
				# ground held directly across BOTH the Mongol lands and the Manchurian homeland.
				any_owned_province = { count >= 12  is_in_region = Mongolia }
				any_owned_province = {
					count >= 12
					OR = { is_in_region = Liaoning  is_in_region = Far_East }
				}
			}
		}
		on_completion = {
			custom_tooltip = qing_settle_capstone_tt
			add_country_modifier = { name = qing_frontier_settled  duration = -1 }
			current_ruler = { add_popularity = 40 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION settle the nomads CAPSTONE (定牧墾邊) for" }
		}
	}
}

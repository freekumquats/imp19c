########################################################################
# qing_nanyang_missions.txt — THE SOUTH SEAS OPENING (南洋乘機), scripted-revolutions.
#
# When the French Revolutionary and Napoleonic Wars cut the European colonies of the
# East Indies off from their metropoles (the Dutch East Indies severed by the Batavian
# Republic in 1795; Spanish Manila isolated by the Peninsular War from 1808), a stronger
# Qing could have pressed into the archipelago — protecting the Chinese communities of
# the Nanyang (南洋), taking the Lanfang Kongsi (蘭芳公司) firmly under its wing, and
# contesting the soft, cut-off colonies of a distracted Europe.
#
# HISTORICAL ANCHOR: the Qing already holds Lanfang as a NOMINAL VASSAL at the 1763
# start (setup: CHI -> LAF nominal_vassal). Lanfang was the Hakka gold-mining republic
# of West Borneo founded by Luo Fangbo (羅芳伯) in 1777, which made tributary overtures
# to Peking and long resisted the Dutch (absorbed only in 1884). The 華僑 (overseas
# Chinese) of Batavia, Borneo, Manila and the Malay states were a vast, un-championed
# diaspora the Qing court traditionally disdained. This tree is the road not taken:
# the Son of Heaven becomes the protector of the South Seas Chinese.
#
# GATING: CHI-only, player-only (ai_chance 0). The tree UNLOCKS on the Napoleonic
# opening — the asia_napoleonic ripple events set the global flags asia_dei_cutoff /
# asia_phi_cutoff and stamp the asia_colony_cut_off modifier on the soft colonies. As
# with the India (#449) and Burma (#421) trees, the PLAYER does the real work: the task
# goals gate on genuine control of the ground or the tributary embrace (is_subject_of).
# Texture (make-subject, claims, the great-power reaction) is via se_QING_NANYANG.txt.
#
# It forks after the shared opening (championing the Nanyang Chinese):
#   ARC W — 固蘭芳 CONSOLIDATE LANFANG: raise the Lanfang nominal vassal to a firm
#     tributary and secure West Borneo (Pontianak) — the Chinese republic under the wing.
#   ARC E — 撫南洋 THE SEVERED INDIES: exploit the cut-off Dutch East Indies (and the
#     isolated Spanish Philippines) — take Sulu under the wing, and press Batavia/Manila.
#
# CAPSTONE — 南洋乘機 MASTER OF THE SOUTH SEAS: Lanfang secured, the Nanyang Chinese
# championed, and a cut-off colony broken or embraced; the lasting qing_nanyang_hegemony
# dominion modifier sealed.
#
# Sources: Yuan Bingling (袁冰凌), Chinese Democracies: A Study of the Kongsis of West
#   Borneo 1776-1884 (2000); Leonard Blussé, Strange Company (1986); Wang Gungwu, The
#   Chinese Overseas (2000); 莊國土, 華僑華人與中國的關係 (2001); M.C. Ricklefs, A History
#   of Modern Indonesia (2001).
########################################################################

qing_nanyang_mission = {
	icon = test2
	header = mission_image_test

	repeatable = no

	chance = { }
	ai_chance = { factor = 0 }   # player-only

	potential = {
		tag = CHI
		is_ai = no
		# unlocks on the Napoleonic South Seas opening (a cut-off European colony) OR
		# the Lanfang vassal existing to build on. Either gives the tree its footing.
		OR = {
			has_global_variable = asia_dei_cutoff
			has_global_variable = asia_phi_cutoff
			exists = c:LAF
		}
	}

	abort = {
		NOT = { tag = CHI }
	}

	on_start = {
		LOG_line = { sys = QING  msg = "MISSION the South Seas opening (南洋乘機) started for" }
	}
	on_completion = {
		LOG_line = { sys = QING  msg = "MISSION the South Seas opening COMPLETED for" }
	}

	# ==================================================================
	# ROOT — CHAMPION THE NANYANG CHINESE (護僑南洋). Break with the court's traditional
	# disdain for the men who sailed south, and proclaim the Son of Heaven the protector
	# of the 華僑 — the Chinese of Batavia, Borneo, Manila and the Malay coasts. Commit the
	# treasury and the political capital to a maritime forward policy.
	# ==================================================================
	qing_nanyang_champion = {
		icon = test1
		allow = {
			treasury >= 120
			political_influence >= 30
		}
		on_start = {
			add_treasury = -120
			add_political_influence = -30
		}
		on_completion = {
			custom_tooltip = qing_nanyang_champion_tt
			QING_nanyang_proclaim_protector = yes
			current_ruler = { add_popularity = 10 }
			LOG_line = { sys = QING  msg = "MISSION task Nanyang protector proclaimed for" }
		}
	}

	# ==================================================================
	# ARC W — 固蘭芳 CONSOLIDATE LANFANG. The Lanfang Kongsi is already a nominal vassal;
	# raise it to a firm sinosphere tributary, formally under the wing, so the Chinese
	# republic of West Borneo is a real Qing dependency rather than a distant curiosity.
	# ==================================================================
	qing_nanyang_lanfang = {
		icon = test2
		requires = { qing_nanyang_champion }
		allow = {
			political_influence >= 25
			custom_tooltip = {
				text = qing_nanyang_needs_lanfang_tt
				exists = c:LAF
			}
		}
		on_start = {
			add_political_influence = -25
		}
		on_completion = {
			custom_tooltip = qing_nanyang_lanfang_tt
			# raise Lanfang from nominal vassal to a firm tributary under the wing.
			QING_nanyang_embrace_lanfang = yes
			current_ruler = { add_popularity = 12 }
			LOG_line = { sys = QING  msg = "MISSION task Lanfang consolidated for" }
		}
	}

	# ==================================================================
	# ARC W (cont.) — 據坤甸 SECURE WEST BORNEO (Pontianak). Hold the gold coast of West
	# Borneo — Pontianak (2604, the Lanfang seat) under Qing control or held by the
	# Lanfang tributary — so the Chinese republic rests on secured ground.
	# ==================================================================
	qing_nanyang_borneo = {
		icon = test3
		requires = { qing_nanyang_lanfang }
		allow = {
			custom_tooltip = {
				text = qing_nanyang_needs_borneo_tt
				# the Lanfang seat held — by the Qing directly, or by the Lanfang tributary.
				OR = {
					AND = { exists = p:2604  p:2604 = { controller = ROOT } }
					AND = {
						exists = c:LAF
						c:LAF = { is_subject_of = ROOT }
						exists = p:2604
						p:2604 = { owner = c:LAF }
					}
				}
			}
		}
		on_completion = {
			custom_tooltip = qing_nanyang_borneo_tt
			if = { limit = { exists = p:2604  NOT = { owns = 2604 } }  add_claim = p:2604 }
			current_ruler = { add_popularity = 15 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION task West Borneo secured for" }
		}
	}

	# ==================================================================
	# ARC E — 撫蘇祿 THE SULU EMBRACE. Sulu (蘇祿) had renewed its tribute to the Qing in
	# 1726-1733 (a relationship revived after a ~300-year lapse since the Ming), the gifts
	# and trade permits worth more to the sultanate than the tribute it rendered. With Manila
	# cut off from Spain, take the Sulu sultanate firmly under the wing: the pivot of the
	# southern archipelago trade. [Research (QingCJK): the 1726-33 tributary tie is the real
	# anchor; an earlier draft's "1753 submission offer" was unattested and has been removed.]
	# ==================================================================
	qing_nanyang_sulu = {
		icon = test2
		requires = { qing_nanyang_champion }
		allow = {
			political_influence >= 25
			custom_tooltip = {
				text = qing_nanyang_needs_sulu_tt
				exists = c:SLU
			}
		}
		on_start = {
			add_political_influence = -25
		}
		on_completion = {
			custom_tooltip = qing_nanyang_sulu_tt
			# take Sulu under the wing (the 1753 overture finally accepted).
			QING_nanyang_embrace_sulu = yes
			current_ruler = { add_popularity = 12 }
			LOG_line = { sys = QING  msg = "MISSION task Sulu embraced for" }
		}
	}

	# ==================================================================
	# ARC E (cont.) — 乘虛擊夷 STRIKE THE SEVERED COLONY. Press the cut-off European
	# colony while it is soft: take and hold a stronghold of the Dutch East Indies
	# (Batavia, 6270) OR the Spanish Philippines (Manila, 2004) — the mastery of the
	# archipelago's key port wrested from a metropole that cannot answer.
	# ==================================================================
	qing_nanyang_severed = {
		icon = test3
		requires = { qing_nanyang_sulu }
		allow = {
			treasury >= 100
			custom_tooltip = {
				text = qing_nanyang_needs_severed_tt
				# a cut-off colonial capital under Qing control (Batavia or Manila).
				OR = {
					AND = { exists = p:6270  p:6270 = { controller = ROOT } }
					AND = { exists = p:2004  p:2004 = { controller = ROOT } }
				}
			}
		}
		on_start = {
			add_treasury = -100
		}
		on_completion = {
			custom_tooltip = qing_nanyang_severed_tt
			if = { limit = { exists = p:6270  p:6270 = { controller = ROOT }  NOT = { owns = 6270 } }  add_claim = p:6270 }
			if = { limit = { exists = p:2004  p:2004 = { controller = ROOT }  NOT = { owns = 2004 } }  add_claim = p:2004 }
			current_ruler = { add_popularity = 20 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION task severed colony struck for" }
		}
	}

	# ==================================================================
	# CAPSTONE — 南洋乘機 MASTER OF THE SOUTH SEAS. Lanfang secured under the wing, the
	# Nanyang Chinese championed, and a severed colony broken or a southern sultanate
	# embraced. The lasting qing_nanyang_hegemony dominion modifier is sealed — the Son
	# of Heaven the paramount power of the South Seas, seized in Europe's hour of distraction.
	# ==================================================================
	qing_nanyang_capstone = {
		icon = test1
		requires = { qing_nanyang_borneo  qing_nanyang_severed }
		allow = {
			custom_tooltip = {
				text = qing_nanyang_needs_hegemony_tt
				# Lanfang under the wing (or absorbed) AND a key archipelago port held.
				OR = { NOT = { exists = c:LAF }  c:LAF = { is_subject_of = ROOT } }
				OR = {
					AND = { exists = p:6270  p:6270 = { controller = ROOT } }
					AND = { exists = p:2004  p:2004 = { controller = ROOT } }
					AND = { exists = p:2604  p:2604 = { controller = ROOT } }
				}
			}
		}
		on_completion = {
			custom_tooltip = qing_nanyang_capstone_tt
			add_country_modifier = { name = qing_nanyang_hegemony  duration = -1 }
			QING_nanyang_close_campaign = yes
			current_ruler = { add_popularity = 40 }
			add_stability = 2
			LOG_line = { sys = QING  msg = "MISSION the South Seas opening CAPSTONE (南洋乘機) for" }
		}
	}
}

########################################################################
# qing_summer_palace_missions.txt — THE GARDEN OF GARDENS (圓明園／頤和園, task #74).
# CHI-only, player-driven.
#
# The imperial garden-palaces of the Qing. The 圓明園 (Yuanmingyuan, the "Garden
# of Perfect Brightness") was begun under Kangxi and brought to its zenith by
# Qianlong — a vast pleasure-ground north-west of Beijing, studded with the
# Jesuit-designed 西洋樓 (European Pavilions, Giuseppe Castiglione's baroque
# fountains) and housing one of the four copies of the 四庫全書. It was the true
# seat of the court, grander than the Forbidden City itself.
#
# This tree lets a self-strengthening (or merely ambitious) Qing raise and adorn
# the garden-palace as a monument to the dynasty's prestige — and then confronts
# it with the palace's two real historical fates:
#
#   • THE SACK OF 1860 (reactive, qing_summer_palace.1). In October 1860, at the
#     close of the Second Opium War, Lord Elgin's Anglo-French expedition looted
#     and burned the Yuanmingyuan to the ground — the single most infamous act of
#     the century of humiliation. If the garden stands and the Qing is at war
#     with Britain or France after 1856, the sack can fall: the garden modifiers
#     are stripped and a lasting scar of humiliation is left — a shock that BOTH
#     galvanises reform pressure AND forges civic (national) feeling.
#
#   • THE NAVY-FUNDS DILEMMA OF 1888 (the capstone, qing_summer_palace.2). The
#     historical Cixi diverted the Beiyang naval budget to rebuild the garden as
#     the 頤和園 (Yiheyuan, the "Garden of Nurtured Harmony") for her sixtieth
#     birthday — a splendour bought at the price of the fleet that would be sunk
#     at the Yalu in 1894. The capstone poses that exact choice: divert the navy
#     funds (court prestige now, a crippled fleet later) or pay honestly from the
#     treasury (the reformed Qing that can afford both).
#
# Machinery: se_QING_SUMMER_PALACE.txt (QING_sp_* verbs), qing_summer_palace_
# modifiers.txt, qing_summer_palace_events.txt. The sack is offered through
# QING_frontier_flavour_roll (se_QING_DECLINE.txt). All paths logged sys = QING.
#
# Flags the tree owns (country scope, ROOT = CHI):
#   qing_sp_built  — set when the garden root completes; gates the 1860 sack and
#                    the rebuild capstone.
#   qing_sp_burnt  — set when the sack fires; prevents re-fire, flips the capstone
#                    from "adorn" to "raise from the ashes".
#
# NOTE (cosmetic): icons reuse stock test1/2/3 + mission_image_test, as the
# reform / self-strengthening trees do; swap for bespoke art later.
#
# Sources: Young-tsu Wong, A Paradise Lost: The Imperial Garden Yuanming Yuan
#   (2001); Greg Thomas, "The Looting of Yuanming and the Translation of Chinese
#   Art in Europe" (2008); Hevia, English Lessons (2003, the 1860 sack); on the
#   navy-funds diversion, Benjamin Elman & the Beiyang historiography, and
#   Cambridge History of China v.11.
########################################################################

qing_summer_palace_mission = {
	icon = test1
	header = mission_image_test

	repeatable = no

	chance = { }
	ai_chance = { factor = 0 }   # player-only, like the rest of the Qing suite

	potential = {
		tag = CHI
		is_ai = no
	}

	abort = {
		NOT = { tag = CHI }
	}

	on_start = {
		LOG_line = { sys = QING  msg = "MISSION summer palace started for [ROOT.GetTag]" }
	}

	on_completion = {
		LOG_line = { sys = QING  msg = "MISSION summer palace COMPLETED for [ROOT.GetTag]" }
	}

	# ==================================================================
	# ROOT — 圓明園 restore & expand the Garden of Perfect Brightness.
	# Qianlong's garden of gardens, north-west of Beijing (Haidian). Raising it
	# sets qing_sp_built, which arms both the 1860 sack and the rebuild capstone.
	# ==================================================================
	qing_sp_yuanmingyuan = {
		icon = test1
		allow = {
			treasury >= 120
			custom_tooltip = {
				text = qing_sp_needs_capital_tt
				owns_or_subject_owns = 8363
			}
		}
		on_start = {
			add_treasury = -120
			LOG_line = { sys = QING  msg = "MISSION task sp_yuanmingyuan STARTED for [ROOT.GetTag]" }
		}
		on_completion = {
			custom_tooltip = qing_sp_yuanmingyuan_tt
			set_variable = qing_sp_built
			add_country_modifier = { name = qing_sp_garden_of_brightness  duration = -1 }
			current_ruler = { add_prestige = 25 }
			LOG_line = { sys = QING  msg = "MISSION task sp_yuanmingyuan done (garden raised) for [ROOT.GetTag]" }
		}
	}

	# ==================================================================
	# 西洋樓 the European Pavilions — Castiglione's Jesuit baroque, the marble
	# fountains and the Hall of Calm Seas. A wonder of cultural exchange — and, in
	# 1860, the looters' first and favourite target.
	# ==================================================================
	qing_sp_european_pavilions = {
		icon = test2
		requires = { qing_sp_yuanmingyuan }
		allow = {
			treasury >= 100
		}
		on_start = {
			add_treasury = -100
		}
		on_completion = {
			custom_tooltip = qing_sp_european_pavilions_tt
			# If the court painting cultural modifier is active (Castiglione 郎世寧 designed
			# the 西洋樓), grant a synergy bonus (extra prestige from the cultural fusion).
			if = {
				limit = { has_country_modifier = qing_culture_court_painting }
				custom_tooltip = qing_sp_pavilions_court_painting_synergy_tt
				current_ruler = { add_prestige = 10 }
			}
			add_country_modifier = { name = qing_sp_european_marvels  duration = -1 }
			current_ruler = { add_prestige = 15 }
			LOG_line = { sys = QING  msg = "MISSION task sp_european_pavilions done for [ROOT.GetTag]" }
		}
	}

	# ==================================================================
	# 四庫全書 the Imperial Library — house the Wenyuan Pavilion copy of the Complete
	# Library of the Four Treasuries, Qianlong's monument of learning.
	# ==================================================================
	qing_sp_siku_library = {
		icon = test3
		requires = { qing_sp_yuanmingyuan }
		allow = {
			political_influence >= 20
		}
		on_start = {
			add_political_influence = -20
		}
		on_completion = {
			custom_tooltip = qing_sp_siku_library_tt
			# If the player has already commissioned the Siku Quanshu cultural project,
			# this library task represents the Yuanmingyuan COPY (one of seven) — synergy.
			if = {
				limit = { has_country_modifier = qing_culture_siku_quanshu }
				custom_tooltip = qing_sp_siku_cultural_synergy_tt
				add_stability = 1
			}
			add_country_modifier = { name = qing_sp_imperial_library  duration = -1 }
			QING_DECLINE_nudge = { var = qing_civic_identity  amount = 3 }
			LOG_line = { sys = QING  msg = "MISSION task sp_siku_library done for [ROOT.GetTag]" }
		}
	}

	# ==================================================================
	# CAPSTONE — 頤和園 the Garden of Nurtured Harmony (the navy-funds dilemma).
	# Whether adorning the standing garden or raising it from the 1860 ashes, the
	# court must find the money. The capstone fires qing_summer_palace.2, which
	# poses the historical choice: divert the Beiyang naval budget (court splendour
	# now, a hollow fleet later) or fund it honestly from a full treasury.
	# ==================================================================
	# ==================================================================
	# BRANCH B — 清漪園 QINGYI YUAN (the Garden of Clear Ripples, predecessor).
	# A SEPARATE site from the Yuanmingyuan, built by Qianlong on Kunming Lake
	# (昆明湖) and Longevity Hill (萬壽山). This garden was ALSO burned in 1860,
	# and it is on THESE grounds that Cixi will raise the Yiheyuan in 1888.
	# Branch B is INDEPENDENT of Branch A — it does not require the Yuanmingyuan.
	# ==================================================================
	qing_sp_qingyi = {
		icon = test2
		allow = {
			treasury >= 80
			custom_tooltip = {
				text = qing_sp_needs_capital_tt
				owns_or_subject_owns = 8363
			}
		}
		on_start = {
			add_treasury = -80
			LOG_line = { sys = QING  msg = "MISSION task sp_qingyi STARTED for [ROOT.GetTag]" }
		}
		on_completion = {
			custom_tooltip = qing_sp_qingyi_tt
			set_variable = qing_sp_qingyi_built
			add_country_modifier = { name = qing_sp_qingyi_garden  duration = -1 }
			current_ruler = { add_prestige = 20 }
			LOG_line = { sys = QING  msg = "MISSION task sp_qingyi done (Qingyi Yuan raised) for [ROOT.GetTag]" }
		}
	}

	# ==================================================================
	# OPTIONAL — 同治中興 THE TONGZHI RESTORATION (1873 failed rebuild attempt).
	# The historical attempt to rebuild the Yuanmingyuan ruins collapsed after
	# 11 months for lack of funds and official opposition. Requires the Yuanmingyuan
	# to have been burnt. This is an OPTIONAL player path — it costs treasury and
	# political capital, likely fails, and gives only a small consolation.
	# ==================================================================
	qing_sp_tongzhi_restoration = {
		icon = test1
		requires = { qing_sp_yuanmingyuan }
		allow = {
			treasury >= 150
			political_influence >= 30
			custom_tooltip = {
				text = qing_sp_needs_burnt_tt
				has_variable = qing_sp_burnt
			}
			custom_tooltip = {
				text = qing_sp_tongzhi_after_1873_tt
				current_date >= 1873.1.1
			}
		}
		on_start = {
			add_treasury = -150
			add_political_influence = -30
			LOG_line = { sys = QING  msg = "MISSION task sp_tongzhi_restoration STARTED for [ROOT.GetTag]" }
		}
		on_completion = {
			custom_tooltip = qing_sp_tongzhi_tt
			# Fire the restoration attempt event (qing_summer_palace.4) which will likely fail
			trigger_event = { id = qing_summer_palace.4  days = { 3 10 } }
			LOG_line = { sys = QING  msg = "MISSION task sp_tongzhi_restoration done (attempt posed) for [ROOT.GetTag]" }
		}
	}


	# ==================================================================
	# CAPSTONE — 頤和園 YIHEYUAN (the Garden of Nurtured Harmony, 1888).
	# Cixi's rebuilt garden on the Qingyi Yuan site, for her 1894 sixtieth birthday.
	# This is Branch B's capstone — it requires the Qingyi Yuan (NOT the Yuanmingyuan).
	# The navy-funds dilemma is posed here. If the player has raised a Great Work
	# elsewhere (the engine's wonder system), they get a prestige bonus.
	# ==================================================================
	qing_sp_yiheyuan = {
		icon = test1
		requires = { qing_sp_qingyi }
		allow = {
			political_influence >= 25
			custom_tooltip = {
				text = qing_sp_yiheyuan_after_1884_tt
				current_date >= 1884.1.1
			}
		}
		on_start = {
			add_political_influence = -25
			LOG_line = { sys = QING  msg = "MISSION task sp_yiheyuan STARTED (funding dilemma) for [ROOT.GetTag]" }
		}
		on_completion = {
			custom_tooltip = qing_sp_yiheyuan_tt
			# If the Qing has also raised a Great Work (the engine's wonder system), the
			# narrative gardens and the real wonder synergize — a burst of prestige.
			if = {
				limit = { has_great_work = yes }
				custom_tooltip = qing_sp_yiheyuan_wonder_synergy_tt
				current_ruler = { add_prestige = 15 }
			}
			# The Yiheyuan rebuild was FOR Cixi's 60th birthday (萬壽) — if the jubilee
			# cultural modifier is already active, extend it; otherwise grant a short burst.
			if = {
				limit = { has_country_modifier = qing_culture_jubilee }
				custom_tooltip = qing_sp_yiheyuan_jubilee_extend_tt
				remove_country_modifier = qing_culture_jubilee
				add_country_modifier = { name = qing_culture_jubilee  duration = 1095 }
			}
			else = {
				custom_tooltip = qing_sp_yiheyuan_jubilee_grant_tt
				add_country_modifier = { name = qing_culture_jubilee  duration = 730 }
			}
			# The funding choice is a genuine dilemma — pose it as an event.
			trigger_event = { id = qing_summer_palace.2  days = { 3 10 } }
			LOG_line = { sys = QING  msg = "MISSION task sp_yiheyuan done; funding dilemma posed for [ROOT.GetTag]" }
		}
	}

}

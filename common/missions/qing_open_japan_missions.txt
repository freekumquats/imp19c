########################################################################
# qing_open_japan_missions.txt — OPEN THE CLOSED COUNTRY (叩關破鎖國), #447.
#
# An anachronistic NAVAL opening of Sakoku-era Japan roughly a CENTURY before
# Commodore Perry's black ships (1853-54): the Qing, having rebuilt a blue-water
# fleet, sail into the closed isles and force the Tokugawa Shogunate (TKG) to open.
#
# GATING ON THE TREASURE FLEET (#439): this tree is the sea-power's payoff. It is
# potential ONLY once the New Treasure Fleet enterprise is complete — the permanent
# qing_treasure_fleet_glory country modifier its 下西洋 capstone grants. Without a
# fleet to dwarf the war-junks, the closed country cannot be forced.
#
# As with the Burma (#421) and Treasure-Fleet (#439) trees, the PLAYER does the real
# work: the task goals gate on the Qing genuinely CONTROLLING the ground (provinces
# in Kyushu, then the Kanto/Edo heartland), so the fighting/coercion is his to do on
# the map. What is scripted (se_QING_OPEN_JAPAN.txt) is only the texture — the coastal
# pressure the great fleet lays on the shogunate, the claims that let control convert
# to a settlement, and the northern diplomacy with the Ainu.
#
# It FORKS after the shared opening (the fleet arrives off the coast):
#   ARC S — 南路 THE SOUTHERN GATE: force the shogunate's window on the world.
#     Take and hold Kyushu (the Nagasaki approach), then drive on the Kanto heartland
#     and Edo (3684) itself — the shogunate opened at the cannon's mouth.
#   ARC N — 北路 THE AINU ROAD (蝦夷): the northern lever. Back the Ainu (AIN) of Ezo
#     (Hokkaido) against the Matsumae clan (MAE) pushing north — take them under the
#     Qing wing (a sinosphere tributary) and hold Ezo, hemming Japan from the north.
#
# CAPSTONE — 開國 THE COUNTRY OPENED: the shogunate's isolation is broken and the
# Qing flag flies over its northern and southern gates; the lasting qing_japan_opened
# dominion modifier is sealed.
#
# Gating: CHI-only, player-only (ai_chance factor 0). Tasks cost treasury / political
# influence and gate on the advancing control of the isles + the Ainu tie, so the tree
# cannot resolve in Japan's (or the Ainu's) absence. Icons reuse stock test1/2/3 (the
# whole Qing suite's placeholder convention).
#
# Sources: W.G. Beasley, The Meiji Restoration (1972) & Japan Encounters the Barbarian
#   (1995); Marius Jansen, The Making of Modern Japan (2000); Brett Walker, The Conquest
#   of Ainu Lands: Ecology and Culture in Japanese Expansion 1590-1800 (2001); Ronald
#   Toby, State and Diplomacy in Early Modern Japan (1984, on Sakoku and the four gates).
########################################################################

qing_open_japan_mission = {
	icon = test3
	header = mission_image_test

	repeatable = no

	chance = { }
	ai_chance = { factor = 0 }   # player-only

	potential = {
		tag = CHI
		is_ai = no
		# the sea-power's payoff: only once the New Treasure Fleet (#439) enterprise is
		# complete — the permanent glory its 下西洋 capstone grants — can the closed country
		# be forced open by sea.
		has_country_modifier = qing_treasure_fleet_glory
		# and there must be a Tokugawa realm to open.
		exists = c:TKG
	}

	abort = {
		NOT = { tag = CHI }
	}

	on_start = {
		LOG_line = { sys = QING  msg = "MISSION open the closed country (叩關破鎖國) started for" }
	}

	on_completion = {
		LOG_line = { sys = QING  msg = "MISSION open the closed country COMPLETED for" }
	}

	# ==================================================================
	# ROOT — THE FLEET ARRIVES (寶船抵日). The treasure fleet stands off the closed coast
	# and the court presents its demand that the isles open. The shogunate stiffens; the
	# great hulls lay their pressure on the seaward domains. The point of commitment.
	# ==================================================================
	qing_openjapan_arrive = {
		icon = test1
		allow = {
			treasury >= 120
			political_influence >= 30
		}
		on_start = {
			add_treasury = -120
			add_political_influence = -30
		}
		on_completion = {
			custom_tooltip = qing_openjapan_arrive_tt
			# the fleet's presence bears down on the seaward shogunate (the coastal-pressure
			# texture) and the court's demand is pressed home.
			QING_openjapan_open_demand = yes
			current_ruler = { add_popularity = 10 }
			LOG_line = { sys = QING  msg = "MISSION task open-Japan fleet arrived for" }
		}
	}

	# ==================================================================
	# ARC S — 南路 THE SOUTHERN GATE (南路叩關). Force the shogunate's one window on the
	# world: take and hold Kyushu (the Nagasaki approach through which the whole Sakoku
	# trade was funnelled). A claim on the ground is granted so a settlement can convert
	# control to cession.
	# ==================================================================
	qing_openjapan_kyushu = {
		icon = test2
		requires = { qing_openjapan_arrive }
		allow = {
			custom_tooltip = {
				text = qing_openjapan_needs_kyushu_tt
				# the southern gate actually held by the Qing.
				any_owned_province = { count >= 1  is_in_region = Kyushu }
			}
		}
		on_completion = {
			custom_tooltip = qing_openjapan_kyushu_tt
			# grant a claim on Edo so the drive on the heartland has a war-goal.
			QING_openjapan_claim_heartland = yes
			current_ruler = { add_popularity = 15 }
			LOG_line = { sys = QING  msg = "MISSION task open-Japan southern gate held for" }
		}
	}

	# ==================================================================
	# ARC S (cont.) — THE MARCH ON EDO (進逼江戶). Drive into the Kanto heartland and take
	# the shogun's seat (Edo, 3684) itself. With the capital held, the Bakufu's will to
	# keep the country closed is broken.
	# ==================================================================
	qing_openjapan_edo = {
		icon = test3
		requires = { qing_openjapan_kyushu }
		allow = {
			treasury >= 80
			custom_tooltip = {
				text = qing_openjapan_needs_edo_tt
				# the shogun's seat under Qing control.
				exists = p:3684
				p:3684 = { controller = ROOT }
			}
		}
		on_start = {
			add_treasury = -80
		}
		on_completion = {
			custom_tooltip = qing_openjapan_edo_tt
			if = { limit = { exists = p:3684  NOT = { owns = 3684 } }  add_claim = p:3684 }
			current_ruler = { add_popularity = 20 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION task open-Japan Edo taken for" }
		}
	}

	# ==================================================================
	# ARC N — 北路 THE AINU ROAD (撫蝦夷). The northern lever: reach past the shogunate to
	# the Ainu (AIN) of Ezo, whom the Matsumae clan (MAE) has been grinding north for two
	# centuries. Take them under the Qing wing as a tributary — a foothold at Japan's back.
	# ==================================================================
	qing_openjapan_ainu = {
		icon = test2
		requires = { qing_openjapan_arrive }
		allow = {
			political_influence >= 25
			custom_tooltip = {
				text = qing_openjapan_needs_ainu_tt
				# the Ainu must be on the map to be reached.
				exists = c:AIN
			}
		}
		on_start = {
			add_political_influence = -25
		}
		on_completion = {
			custom_tooltip = qing_openjapan_ainu_tt
			# take the Ainu under the Qing wing as a tributary (the make-subject-by-fiat idiom
			# the southern-frontier tree uses, se_QING_HIMALAYA/himalaya_seasia; guarded so a
			# repeat or an already-bound Ainu is a no-op).
			QING_openjapan_embrace_ainu = yes
			current_ruler = { add_popularity = 15 }
			LOG_line = { sys = QING  msg = "MISSION task open-Japan Ainu taken under the wing for" }
		}
	}

	# ==================================================================
	# ARC N (cont.) — HOLD EZO (據蝦夷地). Secure Hokkaido: take and hold the ground of Ezo
	# so the Matsumae window closes and Japan is hemmed from the north.
	# ==================================================================
	qing_openjapan_ezo = {
		icon = test3
		requires = { qing_openjapan_ainu }
		allow = {
			custom_tooltip = {
				text = qing_openjapan_needs_ezo_tt
				# Ezo (Hokkaido) held — either by the Qing directly or by the Ainu tributary.
				OR = {
					any_owned_province = { count >= 1  is_in_region = Ezo }
					AND = {
						exists = c:AIN
						c:AIN = { is_subject_of = ROOT }
						c:AIN = { any_owned_province = { is_in_region = Ezo } }
					}
				}
			}
		}
		on_completion = {
			custom_tooltip = qing_openjapan_ezo_tt
			current_ruler = { add_popularity = 15 }
			LOG_line = { sys = QING  msg = "MISSION task open-Japan Ezo held for" }
		}
	}

	# ==================================================================
	# CAPSTONE — 開國 THE COUNTRY OPENED. The shogunate's isolation is broken: the southern
	# gate forced (Edo held) OR the northern one shut on it (the Ainu under the wing, Ezo
	# held). The lasting qing_japan_opened dominion modifier is sealed and the fleet's
	# pressure on the coast lifts — the Sakoku is ended a century early.
	# ==================================================================
	qing_openjapan_capstone = {
		icon = test1
		requires = { qing_openjapan_edo  qing_openjapan_ezo }
		allow = {
			custom_tooltip = {
				text = qing_openjapan_needs_opened_tt
				# both gates carried: the heartland (Edo) held AND the north (Ezo) secured.
				exists = p:3684
				p:3684 = { controller = ROOT }
				OR = {
					any_owned_province = { count >= 1  is_in_region = Ezo }
					AND = { exists = c:AIN  c:AIN = { is_subject_of = ROOT } }
				}
			}
		}
		on_completion = {
			custom_tooltip = qing_openjapan_capstone_tt
			add_country_modifier = { name = qing_japan_opened  duration = -1 }
			# the country is opened — the fleet's coastal pressure lifts.
			QING_openjapan_close_campaign = yes
			current_ruler = { add_popularity = 40 }
			add_stability = 1
			LOG_line = { sys = QING  msg = "MISSION open the closed country CAPSTONE (開國) for" }
		}
	}
}

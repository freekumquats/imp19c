########################################################################
# qing_customs_modifiers.txt — modifiers for the Imperial Maritime Customs
# Service (海關 / 中國海關總稅務司署). See se_QING_CUSTOMS.txt.
#
# The service is a 0..100 efficiency meter + a 0..100 foreign-control meter on
# the Qing country. Bands are (re)applied by the governance pulse. The named
# institution grants a lasting modifier; the Inspector-General bears a character
# modifier. Revenue itself is granted quarterly via CURRENCY_grant_country_wealth
# in the pulse, not via a modifier.
#
# All keys verified present in common/modifiers/00_*.txt. Country scope unless
# noted. Prefix: qing_customs_
#
# Sources: van de Ven, Breaking with the Past (2014); Bickers, The Scramble for
#   China (2011); Spence, The Search for Modern China.
########################################################################

# The service exists — a modern, treaty-based tariff administration. Reliable
# revenue and a modern statistical/postal arm; the base institutional bonus.
qing_customs_service_active = {
	global_tax_modifier = 0.08
	global_commerce_modifier = 0.05
	global_export_commerce_modifier = 0.05
}

# A strong (foreign) Inspector-General in post — the Hart effect: honest, able
# central management. Lifts revenue quality and checks graft.
qing_customs_strong_ig = {
	global_tax_modifier = 0.05
	monthly_corruption = -0.04
	diplomatic_reputation = 1
}

# EFFICIENCY BANDS -----------------------------------------------------
# An efficiently-run service — the funds flow and the books balance.
qing_customs_efficient = {
	global_tax_modifier = 0.05
	global_commerce_modifier = 0.05
}

# A leaky, ill-run service — revenue slips away through the old squeeze.
qing_customs_leaky = {
	global_tax_modifier = -0.05
	monthly_corruption = 0.02
}

# FOREIGN-CONTROL OVERLAY ----------------------------------------------
# A foreign-dominated service (control >= 66) — the standing reproach to Qing
# sovereignty. The revenue is real, but so is the humiliation: the reform and
# revolutionary factions point to it, and legitimacy suffers.
qing_customs_foreign_dominated = {
	monthly_legitimacy = -0.03
	global_monthly_state_loyalty = -0.02
}

# ---- Character-scope: borne by the sitting Inspector-General (foreigner) -----
qing_customs_inspector_general = { # character scope
	monthly_character_prominence = 0.05
	monthly_character_popularity = -0.02
	finesse = 1
}

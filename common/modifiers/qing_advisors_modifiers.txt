########################################################################
# qing_advisors_modifiers.txt — modifiers for the foreign advisors / technical
# experts (洋員 / 顧問). See se_QING_ADVISORS.txt.
#
# One "field active" country modifier per advisor field (naval/army/fiscal/
# technical), granted while an expert of that field serves, plus a character
# modifier borne by the serving expert. The Self-Strengthening advance the
# advisor gives is applied as a counter nudge in QING_advisor_recruit, not here.
#
# All keys verified present in common/modifiers/00_*.txt. Country scope unless
# noted. Prefix: qing_advisor_
#
# Sources: Pong, Modernization and Politics in China (1994); Richard Smith,
#   Mercenaries and Mandarins (1978); van de Ven, Breaking with the Past (2014).
########################################################################

# 海軍顧問 — naval expert (Giquel at Fuzhou, William Lang with the Beiyang Fleet):
# dockyard method and fleet training.
qing_advisor_naval_active = {
	ship_cost = -0.10
	naval_morale_modifier = 0.08
	navy_maintenance_cost = -0.05
}

# 陸軍教習 — army drill-master (Ward, Gordon): Western drill and modern tactics.
qing_advisor_army_active = {
	discipline = 0.05
	land_morale_modifier = 0.08
	heavy_infantry_offensive = 0.05
}

# 財政顧問 — fiscal expert (Hart and the Customs staff): honest accounts, loans,
# and revenue method.
qing_advisor_fiscal_active = {
	global_tax_modifier = 0.06
	monthly_corruption = -0.03
	global_commerce_modifier = 0.05
}

# 技術顧問 — technical expert (arsenal, mine, telegraph and railway engineers):
# the transfer of industrial method.
qing_advisor_technical_active = {
	research_points_modifier = 0.08
	build_cost = -0.05
	global_monthly_civilization = 0.05
}

# ---- Character-scope: borne by the serving foreign advisor -------------------
qing_foreign_advisor = { # character scope

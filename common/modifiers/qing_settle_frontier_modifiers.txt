# qing_settle_frontier_modifiers.txt — modifiers for SETTLE THE NOMADS
# (定牧墾邊, the Qing consolidation of the Mongol and Manchurian frontier), #451.
#
# The mission tree (common/missions/qing_settle_frontier_missions.txt) makes the PLAYER
# do the real work: task goals gate on the Qing OWNING ground directly in the frontier
# regions (Mongolia; Liaoning + the Far East). These modifiers SCRIPT the two things the
# player cannot: the reclamation colonies that take root on the settled ground, and the
# lasting dividend of the pacified, peopled northern march.
#
# Keys are the proven province/country vocabulary the mod's other event modifiers use
# (local_monthly_food_modifier / local_tax_modifier province rewards; the qing_western_regions
# / qing_japan_opened dominion rewards). The reclamation modifier carries a finite duration so
# it LAPSES on its own as the colonists take root (self-expiring, matching the burma-miasma /
# open-Japan pattern).
#
# NO BOM (common/modifiers/*.txt convention here).

# ---- THE RECLAMATION (墾荒) — province scope -----------------------------------
# Stamped on the Qing-held frontier ground (Mongolia; Liaoning + the Far East) as a
# settlement task completes: the 屯田 military-agricultural colonies, fixed pasture and
# broken sod. A food + tax texture. Stamped by QING_settle_reclaim_mongolia /
# QING_settle_reclaim_manchuria; finite duration (10 years) so it lapses as the colonists
# take root and the ground's own development carries it.
qing_frontier_reclamation = { # province scope
	local_monthly_food_modifier = 0.15
	local_tax_modifier = 0.10
	local_population_growth = 0.05
}

# ---- THE FRONTIER SETTLED (定牧墾邊) — country scope, permanent -----------------
# The capstone reward: the steppe and the taiga under direct rule and the plough, the
# northern march made imperial ground. A lasting dividend of manpower (the banner levies
# and the settled frontier), revenue, and legitimacy. Added permanently (duration = -1) by
# the capstone task.
qing_frontier_settled = { # country scope
	global_manpower_modifier = 0.10
	global_tax_modifier = 0.05
	monthly_legitimacy = 0.05
	land_morale_modifier = 0.05
}

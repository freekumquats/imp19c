# [#429] EMPTY GRANARY (糧倉告罄) — the ever-normal granary BUILDING still stands (an empty
# warehouse does not collapse), but its stores are exhausted in famine: it can no longer sell
# grain to steady the price (its stabilisation teeth go silent) and offers no buffer against the
# next bad harvest. Negates the food + stabilisation bonuses qing_granary_building grants while
# the reserve is drained; lifted when the granary is restocked (stock recovers). Replaces the old
# "raze the granary in famine" model (which wrongly destroyed the physical institution).
qing_granary_empty = {
	local_monthly_food_modifier = -0.05
	local_population_happiness = -0.05
	local_monthly_state_loyalty = -0.02
}

# [#429] GRAIN RELIEF (常平倉賑糶) — the national ever-normal reserve (qing_granary_stock) has
# SHIPPED grain into this province to relieve a REAL, LOCALIZED food shortage. This is the thing
# the vanilla per-governorship food sim cannot do on its own: pool surplus grain nationally and
# move it to a deficit region. Grants real food (local_monthly_food) for the duration, easing the
# shortage that would otherwise starve pops via shortage.1. Only sustainable while the reserve holds
# and the famine is LOCAL — a national dearth overwhelms the pool and the relief cannot be granted.
qing_grain_relief = {
	local_monthly_food = 2                    # the shipped grain itself
	local_population_happiness = 0.10          # a fed populace is a content one
	local_monthly_state_loyalty = 0.05         # the province's loyalty to the crown that fed it
	local_unrest = -2                          # relief quells the famine-riot that dearth would breed
}

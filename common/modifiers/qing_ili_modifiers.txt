# ---- Zuo's army in the field: the mobilisation is a real maintenance drain, but a long
# hard campaign forges an experienced, high-morale host (緩進急戰). Temporary.
qing_ili_campaign_mobilised = {
	army_maintenance_cost = 0.15
	land_morale = 0.05
	monthly_military_experience = 0.02
	movement_speed = 0.05
}

# ---- the coastal-defence dividend: Xinjiang abandoned, the treasury turned to the fleet.
# A lasting maritime edge — but the shame of the abandonment saps legitimacy. Permanent.
qing_ili_coastal_priority = {
	naval_morale = 0.10
	global_ship_recruit_speed = 0.15
	monthly_legitimacy = -0.02
}

# ---- the empire on a war footing against Russia over Ili: a defensive, mobilised posture
# that is costly to hold and stiffens the frontier, but strains the fisc. Temporary.
qing_ili_war_footing = {
	global_defensive = 0.10
	global_manpower_modifier = 0.10
	army_maintenance_cost = 0.10
	global_tax_modifier = -0.05
}

# ---- the frontier secured by Zeng Jize's treaty: the one clean diplomatic win — standing
# abroad, a settled and confident far west. Permanent.
qing_ili_frontier_secured = {
	diplomatic_reputation = 2
	monthly_legitimacy = 0.03
	global_manpower_modifier = 0.05
}

# ---- band: a firm grip on the New Dominion — the reconquered west yields men and prestige.
qing_xinjiang_grip_firm = {
	global_manpower_modifier = 0.05
	diplomatic_reputation = 1
	land_morale = 0.03
}

# ---- band: the far march bleeding — a lost or contested frontier drains the treasury and
# unsettles the borderlands.
qing_xinjiang_grip_lost = {
	global_tax_modifier = -0.05
	global_population_happiness = -0.02
	fort_maintenance_cost = 0.10
}

# absolute_cognatic_succession — the "Absolute-Cognatic" tier. The engine has no
# fourth succession type and only three global female-penalty constants (oracle
# finding E), so true gender-equal succession is NOT per-country scriptable. This
# modifier rides ON TOP of the cognatic_succession_law: mechanically the realm is
# still cognatic (males slightly preferred), but the modifier + tooltip present it
# as equal succession and grant a modest legitimacy/reputation nod for a modernised,
# uncontested line. Granted/stripped by the succession-reform actions.
#
# pragmatic_sanction_contested — applied to a realm whose succession-law change was
# NOT recognised by the other powers (the 1713 Habsburg precedent): a diplomatic-
# reputation + stability drag until the contest lapses.
########################################################################

	monthly_legitimacy = 0.03
	diplomatic_reputation = 1
}

pragmatic_sanction_contested = {
	diplomatic_reputation = -2
	stability_monthly_change = -0.02
}

# qing_burma_war_modifiers.txt — modifiers for the Qing invasion of Burma
# (清緬戰爭, the Sino-Burmese War of 1765-69), #421.
#
# The historical campaign was Qianlong's "disguised defeat" (Dai Yingcong): four
# invasions across the Yunnan frontier, each wrecked less by the Konbaung armies
# than by the jungle and malaria that annihilated the banner and Green-Standard
# columns in the Irrawaddy lowlands. This tree makes the PLAYER wage the war on the
# map (goals gate on real province CONTROL), while these modifiers SCRIPT the two
# things the player cannot — the pestilential country that bleeds an army marching
# into it, and the strain the campaign puts on the treasury and the war-weary realm.
#
# Keys are the proven province/country vocabulary the mod's other event modifiers use
# (00_from_events_province.txt local_hostile_attrition/local_defensive/
# local_monthly_food_modifier; the qing_ili_campaign_mobilised country strain band).
# All province modifiers are stamped with a finite duration so they LAPSE on their own
# (the self-expiring idiom — no remove_province_modifier needed, matching the #91
# qing_unit_*_rot pattern) as the campaign season turns.
#
# NO BOM (common/modifiers/*.txt convention here).

# ---- THE PESTILENTIAL LOWLANDS (瘴疫) — province scope -------------------------
# Stamped on the Burmese heartland provinces the campaign marches through. The
# Irrawaddy jungle in the wet season was the real killer of the Qing columns: heavy
# hostile attrition on any army standing here, and a countryside that cannot feed an
# invader. Defence bonus reflects the Konbaung stockades (寨) and the fever that
# broke every siege. Stamped by QING_burma_stamp_miasma; finite duration, so a
# secured, garrisoned province recovers once the fighting moves on.
qing_burma_miasma = { # province scope
	local_hostile_attrition = 0.25
	local_monthly_food_modifier = -0.30
	local_defensive = 0.15
	local_unrest = 2
}

# ---- THE CAMPAIGN STRAIN (南征) — country scope --------------------------------
# The running cost of the southern war on the empire: the treasury drain of
# provisioning an army beyond the passes, and the war-weariness of a hard, sickly
# campaign. A standing country modifier lifted only when the war is broken off or
# won (the mission tasks remove it). Mirrors qing_ili_campaign_mobilised (the
# Xinjiang reconquest strain band).
qing_burma_campaign_strain = { # country scope
	land_morale_modifier = -0.05
	war_exhaustion = 0.02
	global_tax_modifier = -0.04
	army_maintenance_cost = 0.10
}

# ---- THE CONQUEST OF BURMA (平緬) — country scope, permanent -------------------
# The capstone reward: the Konbaung realm subdued and the Irrawaddy brought under
# the Son of Heaven. A lasting prestige + revenue dividend from the new southern
# dominion. Added permanently (duration = -1) by the capstone task.
qing_burma_conquered = { # country scope
	global_tax_modifier = 0.06
	diplomatic_reputation = 2
	monthly_legitimacy = 0.05
	global_manpower_modifier = 0.05
}

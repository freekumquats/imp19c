# qing_open_japan_modifiers.txt — modifiers for OPEN THE CLOSED COUNTRY
# (叩關破鎖國, the anachronistic Qing naval opening of Sakoku Japan), #447.
#
# The mission tree (common/missions/qing_open_japan_missions.txt) makes the PLAYER
# force the isles open on the map (goals gate on real province CONTROL + the Ainu tie).
# These modifiers SCRIPT the two things the player cannot: the pressure the great
# treasure fleet lays on the shogunate's closed coast, and the lasting dividend of an
# opened Japan.
#
# Keys are the proven province/country vocabulary the mod's other event modifiers use
# (qing_burma_war_modifiers local_defensive / local_unrest; 00_from_events_country
# blockade_efficiency; the qing_treasure_fleet_glory dominion reward). The coastal-
# pressure modifier carries a finite duration so it LAPSES on its own (self-expiring,
# matching the burma-miasma / #91 qing_unit_*_rot pattern); the capstone also strips it.
#
# NO BOM (common/modifiers/*.txt convention here).

# ---- THE COASTAL PRESSURE (海壓) — province scope ------------------------------
# Stamped on the shogunate's seaward heartland (Edo + the Kyushu/Honshu ports) when the
# treasure fleet arrives and presses its demand. The great hulls standing off the coast
# throttle the port and sap the defenders' will — a blockade texture. Stamped by
# QING_openjapan_open_demand; finite duration, so a port recovers once the fleet moves
# on, and the capstone lifts it outright.
qing_japan_coastal_pressure = { # province scope
	local_defensive = -0.15
	local_unrest = 2
	local_monthly_food_modifier = -0.15
}

# ---- THE COUNTRY OPENED (開國) — country scope, permanent ----------------------
# The capstone reward: the Sakoku ended a century early, the isles opened to the Qing
# by the northern and southern gates alike. A lasting prestige + maritime-commerce
# dividend from the opened trade. Added permanently (duration = -1) by the capstone task.
qing_japan_opened = { # country scope
	diplomatic_reputation = 2
	global_commerce_modifier = 0.06
	monthly_legitimacy = 0.05
	naval_morale_modifier = 0.05
}

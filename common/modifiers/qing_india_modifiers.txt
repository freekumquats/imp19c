# qing_india_modifiers.txt — modifiers for CONTEST THE COMPANY
# (逐夷復印, the Qing bid to contest the British East India Company for Hindustan), #449.
#
# The mission tree (common/missions/qing_india_missions.txt) makes the PLAYER break the
# Company on the map (goals gate on real province CONTROL of the Company seat + the Nawab
# and Mughal tributary ties). These modifiers SCRIPT the two things the player cannot: the
# pressure the Qing descent lays on the Company's Bengal factories, and the lasting dividend
# of the paramountcy of Hindustan.
#
# Keys are the proven province/country vocabulary the mod's other event modifiers use
# (qing_japan_coastal_pressure local_defensive / local_unrest / local_monthly_food_modifier;
# the qing_japan_opened / qing_western_regions dominion rewards). The company-pressure
# modifier carries a finite duration so it LAPSES on its own (self-expiring, matching the
# burma-miasma / open-Japan pattern); the capstone also strips it.
#
# NO BOM (common/modifiers/*.txt convention here).

# ---- THE COMPANY PRESSURE (逼夷) — province scope ------------------------------
# Stamped on the Company's Bengal/Bihar factory ports (Calcutta + the EIC-held coast) when
# the Qing descent opens. The imperial descent throttles the factory's trade and saps the
# Company garrison's will — a blockade texture. Stamped by QING_india_open_demand; finite
# duration, so a factory recovers once the campaign moves on, and the capstone lifts it.
qing_india_company_pressure = { # province scope
	local_defensive = -0.15
	local_unrest = 2
	local_monthly_food_modifier = -0.15
	blockade_efficiency = 0.15
}

# ---- THE PARAMOUNTCY OF HINDUSTAN (印度霸權) — country scope, permanent ----------
# The capstone reward: the Company broken at Calcutta and the two thrones of Hindustan (the
# Nawab of Bengal and the Timurid emperor) under the wing of the Son of Heaven. A lasting
# dividend of diplomatic weight, the reopened wealth of the Indian trade, and prestige.
# Added permanently (duration = -1) by the capstone task.
qing_india_paramountcy = { # country scope
	diplomatic_reputation = 2
	global_commerce_modifier = 0.08
	monthly_legitimacy = 0.05
	global_tax_modifier = 0.05
}

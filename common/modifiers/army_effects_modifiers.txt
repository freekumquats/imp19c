########################################################################
# army_effects_modifiers.txt — [#347] STANDING-ARMY PROVINCE EFFECTS.
#
# The modifiers a stationed field army stamps on the province it garrisons, and
# the friction it breeds with the provincial governor. Applied/refreshed by the
# monthly pulse in army_effects_on_actions.txt (ARMY_pulse_province), removed when
# the garrison leaves. Kept deliberately small + generic so ANY country's stationed
# army pacifies its ground; the Qing-frontier overlay (#346) layers on top.
#
# All are PROVINCE-scope modifiers (local_unrest is the proven province key,
# common/modifiers/00_from_events_province.txt:8). Timed short (refreshed each
# pulse) so a departed army's pacification lapses on its own within the window.
########################################################################

# A stationed garrison pacifies its province — soldiers on the ground deter unrest
# (billeting, patrols, the visible arm of the state). The light band: a modest
# standing garrison.
army_garrison_pacified = {
	local_unrest = -2
}

# A large stationed garrison (a field host, not a token watch) presses harder on
# local order — but see army_garrison_billeting_strain below for the cost side.
army_garrison_pacified_heavy = {
	local_unrest = -4
}

# BILLETING STRAIN — a large army quartered on a province eats its surplus and
# frets the population (foraging, requisition, the soldiery's disorder). The
# happiness cost that balances the heavy-pacification benefit, so a huge host
# parked on a small province is a mixed blessing, not a free unrest sink.
army_garrison_billeting_strain = {
	local_population_happiness = -0.05
}

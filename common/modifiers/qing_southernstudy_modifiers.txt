########################################################################
# qing_southernstudy_modifiers.txt — modifiers for the Southern Study (南書房), #336.
#
# The Study confers no POLICY power (that is the Grand Council's) — its benefit is the
# LUSTER of a throne served by the realm's finest letters: court prestige + dynastic
# legitimacy, scaled to the corps' literary depth. Applied/refreshed each pulse by
# QING_southernstudy_apply_luster (se_QING_SOUTHERNSTUDY.txt), which picks the tier from
# qing_southernstudy_literary (summed finesse+charisma of the serving attendants) and
# removes-then-reapplies so a shrinking cabinet steps the buff back down. Country scope.
########################################################################

# [#336 review-fix] monthly_prestige is a CK3 token, NOT an Imperator one — a grep of the
# reference repos returns only CK3 minor_titles hits (Tianxia), and vanilla Imperator has no
# such country modifier, so it silently NO-OPS (the luster's prestige benefit never accrued).
# The proven Imperator country-scope token for the throne's standing is ruler_popularity_gain
# (80 uses across Invictus/TI modifiers). monthly_legitimacy IS confirmed valid and is kept.

# ---- 南書房 LUSTER (minor) — a modest literary cabinet (literary depth >= 20) --------
# A Study staffed with able men lends the reign a burnish of letters.
qing_southernstudy_luster_minor = {
	ruler_popularity_gain = 0.05
	monthly_legitimacy = 0.03
}

# ---- 南書房 LUSTER (major) — a brilliant literary cabinet (literary depth >= 45) -----
# The realm's finest scholars at the emperor's own side — the height of the High-Qing
# literary court, a throne whose letters command the whole gentry's respect.
qing_southernstudy_luster_major = {
	ruler_popularity_gain = 0.10
	monthly_legitimacy = 0.06
	global_pop_assimilation_speed_modifier = 0.05
}

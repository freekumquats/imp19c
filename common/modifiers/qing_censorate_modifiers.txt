########################################################################
# qing_censorate_modifiers.txt — modifiers for Censorate (都察院) events.
#
# Companion to qing_governance_modifiers.txt (which owns qing_censorate_vigilant,
# already used in qing_office.2). The NEW modifiers here are event-applied
# debuffs (cowed, captured, dissolved) that mark a failing or weaponized
# censorate. Country scope unless noted.
########################################################################

# A cowed censorate: the throne punished a censor for remonstrance, chilling
# oversight. Corruption drifts upward, unrest rises (officials no longer fear
# scrutiny), and political influence falls (the court is paralysed by fear).
qing_censorate_cowed = {
	monthly_corruption = 0.05
	global_unrest = 1
	monthly_political_influence_modifier = -0.05
}

# A captured censorate: weaponized for factional purges dressed as anti-
# corruption. Tyranny rises (the censorate is a purge tool, not a check), and
# loyalty among officials falls (they see the danger).
qing_censorate_captured = {
	monthly_tyranny = 0.05
	global_monthly_state_loyalty = -0.03
	monthly_political_influence_modifier = -0.05
}

# The censorate dissolved entirely: an extreme act of autocratic overreach.
# Corruption runs unchecked, unrest spikes (officials know the last internal
# check is gone), and legitimacy bleeds.
qing_censorate_dissolved = {
	monthly_corruption = 0.10
	global_unrest = 2
	monthly_legitimacy = -0.05
	monthly_political_influence_modifier = -0.10
}

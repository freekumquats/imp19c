# qing_embassy_modifiers.txt — modifier for the Western-embassy engine
# (西使覲見). See se_QING_EMBASSY.txt.
#
#   qing_embassy_trade   a received embassy has opened a channel of foreign trade
#                        with the West — a lasting commerce/export lift (10 years).
#
# The trade grant is deliberately modest and time-boxed (duration 3650 in the
# effect): a single received embassy is an opening, not the treaty-port economy
# (that is the separate qing_treaty_ports_* engine, which a received embassy also
# feeds by one port where the treaty order already stands). Applied once and not
# re-stacked while active (guarded in QING_embassy_grant_trade).
#
# All keys verified present in common/modifiers/00_*.txt. Country scope.
# Prefix: qing_embassy_
#
# Sources: se_QING_EMBASSY.txt header (Macartney, Amherst, Titsingh, Golovkin,
#   Cushing/Wanghia).
########################################################################

qing_embassy_trade = {
	global_commerce_modifier = 0.05
	global_export_commerce_modifier = 0.05
	diplomatic_reputation = 1
}

########################################################################
# THE EARLY-INDUSTRIALIZATION DIVERGENCE (洋務先聲). See se_QING_EARLYINDUS.txt.
# Launched from the Macartney EMBRACE option — an alt-history industrial turn ~67
# years before the historical Self-Strengthening Movement. Reuses the SS engine
# (shared qing_selfstr_progress counter + bands); these are the DIVERGENCE-SPECIFIC
# modifiers: the harbinger buff, the four era-appropriate 1790s institutions, and
# the conservative-court reaction band that drags until the program consolidates.
# All keys verified present in common/modifiers/00_*.txt. Country scope.
########################################################################

# The imperial commitment to learning the barbarians' arts — a research + reform lift
# that stands for the life of the program (mirrors the harbinger of a modern state).
qing_earlyindus_harbinger = {
	research_points_modifier = 0.10
	global_monthly_civilization = 0.05
	diplomatic_reputation = 1
}

# 火器局 Imperial Cannon Foundry — the English gift of cannon/muskets made domestic.
qing_earlyindus_foundry = {
	global_manpower_modifier = 0.05
	artillery_cost = -0.10
	artillery_offensive = 0.05
}

# 譯館 Translation & Science Bureau — Western science carried into Chinese.
qing_earlyindus_translation = {
	research_points_modifier = 0.08
	global_monthly_civilization = 0.03
}

# 機器船廠 Steam-Experiment Yard — the beginnings of a modern navy.
qing_earlyindus_yard = {
	ship_cost = -0.10
	naval_morale_modifier = 0.05
	navy_maintenance_cost = -0.05
}

# 機器製造局 Imperial Machine-Works — the machines that make machines.
qing_earlyindus_manufactory = {
	global_tax_modifier = 0.05
	build_cost = -0.05
	research_points_modifier = 0.05
}

# The conservative-court reaction — the drag a lifetime-early modernization pays until
# it proves itself (lifted by QING_earlyindus_consolidate once all four stand + progress>=40).
qing_earlyindus_conservative_reaction = {
	monthly_political_influence_modifier = -0.05
	happiness_for_same_culture_modifier = -0.03
	global_unrest = 1
}

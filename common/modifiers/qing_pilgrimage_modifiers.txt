# --- The throne, patron of the faith — pilgrimage well-sponsored -------
# Applied when the throne fully sponsors and recognizes a pilgrimage. The
# emperor's role as patron of Tibetan Buddhism (Qianlong's 1780 Panchen
# Lama reception; the Golden Urn lottery) cements Inner Asian loyalty.
# [deferred-fix #5] COUNTRY modifier (subject_loyalty / monthly_legitimacy are country
# keys). monthly_character_prestige was INVALID here — per the #74 note it is not a
# valid country-modifier key; use monthly_prestige (proven in qing_household_modifiers.txt:9).
qing_pilgrim_patron_of_faith = {
	monthly_legitimacy = 0.05
	subject_loyalty = 5
	monthly_prestige = 0.10
}

# --- Devotion turns to a rival — a neighbouring power encroaches --------
# Applied when a rival power (Russia over the Buryat) successfully woos a
# Mongol pilgrim away from Peking's orbit. The faith-as-cement bond weakens.
qing_pilgrim_devotion_to_rival = {
	subject_loyalty = -10
	monthly_legitimacy = -0.03
}

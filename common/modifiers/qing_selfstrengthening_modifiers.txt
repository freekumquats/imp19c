########################################################################
# qing_selfstrengthening_modifiers.txt — modifiers for the Self-Strengthening
# Movement (自強運動 / 洋務運動, 1861–1895) golden-age track.
#
# The movement is modelled as a 0..100 progress meter (qing_selfstr_progress,
# see se_QING_SELFSTR.txt) advanced by completing the Self-Strengthening mission
# tree (common/missions/qing_selfstrengthening_missions.txt). Progress bands
# below are (re)applied by the governance pulse exactly like the decline engine's
# bands. The named institutions each grant a lasting country modifier when the
# corresponding mission task is completed.
#
# ti-yong framing (中學為體，西學為用): the bonuses are industrial/military/fiscal,
# NOT political — political modernization is the separate reform track.
# Cross-wires to the economy: arsenals/yards raise research + build speed + army,
# the fleet bonuses are naval, and mission task effects grant real buildings
# (add_building_level) + treasury (CURRENCY_grant_country_wealth) + industry.
#
# All keys verified present in common/modifiers/00_*.txt + qing_*_modifiers.txt.
# Country scope. Prefix: qing_selfstr_
#
# Sources: Pong, "Modernization and Politics in China: Shen Pao-chen and the
#   Self-Strengthening Movement" (1994); Kwang-Ching Liu & Richard Smith in
#   Cambridge History of China vol. 11; Benjamin Elman, "Naval Warfare and the
#   Refraction of China's Self-Strengthening Reforms" (2004).
########################################################################

# ======================================================================
# PROGRESS BANDS — the momentum of the movement as a whole.
# nascent (>=25) → advancing (>=50) → a genuine industrial GOLDEN AGE (>=85).
# ======================================================================
qing_selfstr_nascent = {          # progress >= 25
	research_points_modifier = 0.05
	global_commerce_modifier = 0.05
	build_cost = -0.05
}

qing_selfstr_advancing = {        # progress >= 50
	research_points_modifier = 0.10
	global_commerce_modifier = 0.10
	build_cost = -0.10
	global_tax_modifier = 0.05
	land_morale_modifier = 0.05
}

qing_selfstr_golden_age = {       # progress >= 85 — the alternate-history apex
	research_points_modifier = 0.20
	global_commerce_modifier = 0.15
	global_export_commerce_modifier = 0.10
	build_cost = -0.15
	global_tax_modifier = 0.10
	land_morale_modifier = 0.10
	naval_morale_modifier = 0.10
	global_manpower_modifier = 0.10
	global_population_happiness = 1
	monthly_legitimacy = 0.05
}

# ======================================================================
# NAMED-INSTITUTION MODIFIERS — one per Self-Strengthening mission task.
# Each is granted (duration = -1) when the task completes; they stack with the
# progress bands. The buildings themselves are added via add_building_level in
# the mission effects; these represent the institution's national impact.
# ======================================================================

# 江南製造總局 (Jiangnan Arsenal, 1865) — the largest arsenal in East Asia + its
# translation bureau (翻譯館). Guns, ammunition, and the transfer of Western
# science into Chinese.
qing_selfstr_jiangnan_arsenal = {
	research_points_modifier = 0.08
	global_manpower_modifier = 0.08
	heavy_infantry_discipline = 0.05
	build_cost = -0.05
}

# 福州船政局 (Fuzhou Navy Yard, 1866) + its naval college (船政學堂). The cradle of
# a modern officer corps. The YARD itself is delivered as these naval modifiers +
# a port_building (it builds capacity, not hulls); the fleets it enabled are the
# separate 北洋/南洋/福建/廣東 tasks, which now raise REAL squadrons on completion
# (see QING_selfstr_raise_fleet in se_QING_SELFSTR.txt).
qing_selfstr_fuzhou_yard = {
	ship_cost = -0.15
	naval_morale_modifier = 0.10
	navy_maintenance_cost = -0.10
}

# 北洋水師 (Beiyang Fleet) — the great northern fleet, Li Hongzhang's pride
# (Dingyuan 定遠 / Zhenyuan 鎮遠, bases Lüshun 旅順 / Weihaiwei 威海衛). This modifier
# is the LASTING fleet-quality + cheap-construction bonus that RIDES ALONGSIDE the
# real squadron the beiyang task now spawns (QING_selfstr_found_beiyang raises the
# hulls AND adds this modifier). Stacks with the Fuzhou yard cut.
qing_selfstr_beiyang_fleet = {
	ship_cost = -0.20
	naval_morale_modifier = 0.15
	navy_movement_speed = 0.10
}

# 南洋水師 (Nanyang Fleet) — the southern fleet defending the Yangzi approaches
# and the south coast.
qing_selfstr_nanyang_fleet = {
	naval_morale_modifier = 0.10
	navy_supplies_modifier = 0.10
	navy_maintenance_cost = -0.05
}

# 福建水師 (Fujian Fleet) — the Fuzhou-yard squadron. Built beside the 福州船政局,
# so ships come cheaper; historically the yard's own defenders.
qing_selfstr_fujian_fleet = {
	ship_cost = -0.10
	naval_morale_modifier = 0.05
	navy_supplies_modifier = 0.05
}

# 廣東水師 (Guangdong Fleet) — the Canton squadron of the Pearl River delta; a
# nimble southern coastal-defence force.
qing_selfstr_guangdong_fleet = {
	navy_movement_speed = 0.10
	naval_morale_modifier = 0.05
	navy_maintenance_cost = -0.05
}

# 常勝軍 (Ever-Victorious Army) — the Western-drilled, modern-armed land force
# first raised at Shanghai 1860 under Frederick Townsend Ward (華爾) and famously
# commanded by Charles "Chinese" Gordon (戈登) from 1863; the seed of Li
# Hongzhang's Huai Army (淮軍) and of all later Qing military modernization. Per
# the design lock this is the ABILITY TO BUILD MODERN LAND UNITS CHEAPLY (not a
# spawn): a cost cut on the modern infantry/cavalry line + a discipline/morale
# edge, so the player fields a Western-drilled army by construction. The mod's
# active Victorian units (regular_infantry/conscripts/artillery) read the
# classical heavy/light-infantry category modifiers.
qing_selfstr_ever_victorious = {
	heavy_infantry_cost = -0.15
	light_infantry_cost = -0.15
	heavy_cavalry_cost = -0.15
	army_maintenance_cost = -0.10
	heavy_infantry_discipline = 0.10
	heavy_infantry_offensive = 0.05
	land_morale_modifier = 0.05
}

# 同文館 (Tongwen Guan, 1862) → naval/technical colleges → 京師大學堂. Interpreters,
# engineers, and the men who will staff the new bureaucracy.
qing_selfstr_tongwen_college = {
	research_points_modifier = 0.10
	global_pop_promotion_speed_modifier = 0.10
	monthly_political_influence_modifier = 0.05
}

# 輪船招商局 (China Merchants Steam Navigation Co., 1872) + telegraph (電報總局) —
# the 官督商辦 "official-supervision, merchant-management" commercial arm.
qing_selfstr_merchant_steam = {
	global_commerce_modifier = 0.10
	global_export_commerce_modifier = 0.10
	ship_cost = -0.05
}

# Railways (唐胥鐵路 1881 → 京張鐵路) + 開平礦務局 (Kaiping Mines) + 漢陽鐵廠
# (Hanyang Ironworks) — coal, iron, and rail: the sinews of industry.
qing_selfstr_rail_and_iron = {
	build_cost = -0.10
	global_tax_modifier = 0.05
	global_commerce_modifier = 0.05
	army_movement_speed = 0.05
}

# 電報總局 (Imperial Telegraph Administration, 1881) — the national wire, split
# out from rail as its own institution. Its distinctive value is administrative
# REACH: the centre can command the provinces in hours. Delivered as commerce +
# a research edge; the bureau-capacity lift is applied as a counter nudge in the
# founding effect (QING_selfstr_found_telegraph), not modelled here.
qing_selfstr_telegraph_network = {
	global_commerce_modifier = 0.08
	research_points_modifier = 0.05
	global_monthly_state_loyalty = 0.10
}

# 官督商辦 ("official supervision, merchant management") TENSION — the cost that
# rides with the merchant/telegraph conglomerates. Bureaucratic control fused to
# private profit bred graft and managerial drag (the Sheng Xuanhuai 盛宣懷
# critique). A lasting drag on tax efficiency + honest administration; the
# corruption cost itself is a counter nudge in the founding effect.
qing_selfstr_guandu_tension = {
	global_tax_modifier = -0.05
	monthly_political_influence_modifier = -0.05
}

# HOLLOW MODERNIZATION — the "refraction" malus (Benjamin Elman). Applied by
# QING_selfstr_refraction_test whenever qing_selfstr_progress runs far ahead of
# the bureau/council/integrity machinery that must carry it (gap >= 25). The
# fleets and arsenals exist on paper but are brittle: poorly coordinated, badly
# funded, and thinly cadred — the Yalu (1894) failure mode.
qing_selfstr_hollow = {
	land_morale_modifier = -0.10
	naval_morale_modifier = -0.10
	discipline = -0.05
	global_tax_modifier = -0.05
}

# P1 — land munitions/artillery shortage, minor tier (worst-case shortage >= 0.15).
# A supply pinch: units bleed a little and recover morale more slowly.
country_munitions_shortage_minor = {
	land_unit_attrition = 0.10
	land_morale_recovery = -0.10
}

# P1 — land munitions/artillery shortage, severe tier (worst-case shortage >= 0.50).
# A supply collapse: heavier attrition, badly stalled morale recovery, and columns
# slow to a crawl as they forage / cannibalise for parts.
country_munitions_shortage_severe = {
	land_unit_attrition = 0.30
	land_morale_recovery = -0.25
	army_movement_speed = -0.15
}

# P3 — coal / naval-supplies shortage for a steam-fleet owner (worst-case >= 0.15).
# Coal-starved boilers and missing stores drive up losses at sea. Only stamped on
# countries actually fielding steam hulls (see LOGISTICS_apply_coal_shortage_penalty).
country_coal_shortage = {
	naval_unit_attrition = 0.20
}

########################################################################
# qing_integration_modifiers.txt — modifiers for the Qing subject
# integration event chain (events/imp19c_mod_events/qing_subject_integration.txt).
#
# These back the positive/negative "reaction" events fired while the player
# integrates an autonomous_governorship. Historical anchors are noted so the
# balance intent stays legible; the event localisation carries the full cites.
#
# Prefix = qing_integ_
########################################################################

# --- NEGATIVE: a botched push breeds resentment across the subject network ---
# Echoes the way the Three Feudatories' revolt (1673) and the White Lotus rising
# (1794-1804) shook confidence in Qing control far beyond the province in revolt.
qing_integ_frontier_resentment = {
	subject_loyalty = -15
	stability_monthly_change = -0.05
}

# --- NEGATIVE: a corruption scandal (the Heshen 和珅 archetype) drains the ---
# treasury's credibility and slows the machinery of integration.
qing_integ_corruption_scandal = {
	integrate_speed = -0.25
	global_tax_modifier = -0.10
}

# --- NEGATIVE: open ethnic strife (Miao / Panthay / Dungan risings) ---
# on the frontier: settlers and natives at each other's throats. (Country
# scope: localized unrest is applied as a province modifier by the event.)
qing_integ_ethnic_strife = {
	subject_loyalty = -10
	stability_monthly_change = -0.03
}

# --- POSITIVE: a well-run investiture / Southern-Tour style progress ---
# buys goodwill and speeds the paperwork of absorption (rou yuan 柔遠).
qing_integ_imperial_favor = {
	subject_loyalty = 15
	integrate_speed = 0.20
}

# --- POSITIVE: tuntian 屯田 colonies bring in a bumper harvest, ---
# feeding the garrisons and settlers that hold the frontier down.
qing_integ_bountiful_frontier = {
	global_population_happiness = 0.05
	global_tax_modifier = 0.10
}

# --- CHARACTER: an official bribed to smooth the integration through ---
# (the "give alms / grease the palm" option). Historical: yanglian yin 養廉銀,
# the "nourish honesty" allowances Yongzheng paid to blunt bribe-taking.
qing_integ_bribed_official = {
	character_loyalty = 8
}

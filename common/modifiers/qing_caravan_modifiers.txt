########################################################################
# qing_caravan_modifiers.txt — [#370] the qing_caravan_prosperity band modifiers (商道繁榮 / 回疆貿易).
#
# Two tiers swapped by QING_DECLINE_apply_caravan_band (se_QING_DECLINE.txt) over the standing
# qing_caravan_prosperity meter (se_QING_CARAVAN.txt) — the health of the Kashgar–Yarkand caravan
# trade between Qing Altishahr and the Central Asian khanates:
#   qing_caravan_boom        (prosperity >= 70) — the routes thrive: the oasis bazaars full, the
#       customs flowing, a firmer, better-fed frontier (a prosperous oasis rebels less).
#   qing_caravan_depression  (prosperity <= 25) — the route is cut/starved: the customs dry up,
#       smuggling and banditry rise, western unrest climbs (the soil the khoja pretenders work).
#
# Country scope. Keys mirror the existing qing_customs_* / qing_xj_* bands (all proven valid
# country-scope modifier keys: global_commerce_modifier, global_export_commerce_modifier,
# global_tax_modifier, global_unrest, monthly_corruption). The quarterly customs treasury trickle
# itself is granted in the pulse (add_treasury), not here — these are the standing frontier effects.
#
# Sources: Millward, Beyond the Pass (1998) & Eurasian Crossroads (2007); Newby, The Empire and the
#   Khanate (2005) on the Kokand trade and the 1832 settlement; Fletcher in the Cambridge History of
#   China vol. 10 on the Altishahr caravan economy and the 協餉 frontier deficit.
########################################################################

# --- CARAVAN BOOM (商道繁榮) — the trade node thrives, the frontier prospers. ---
qing_caravan_boom = {
	global_commerce_modifier = 0.05
	global_export_commerce_modifier = 0.05
	global_tax_modifier = 0.03
	global_unrest = -0.5
}

# --- CARAVAN DEPRESSION (商道斷絕) — the route is cut, smuggling and banditry rise. ---
qing_caravan_depression = {
	global_commerce_modifier = -0.05
	global_export_commerce_modifier = -0.05
	global_unrest = 1
	monthly_corruption = 0.02
}

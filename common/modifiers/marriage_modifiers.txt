# [#311 hook #10] a LOVELESS / incompatible dynastic marriage — court discord and a shadow
# over the succession (Joseph II's barren second marriage; Frederick the Great's estrangement).
# A standing drag on legitimacy + court happiness until the reign turns over. Applied by
# MARRIAGE_apply_dysfunction; a lasting (duration -1) modifier the union/succession events hang.
royal_marriage_dysfunction = {
	monthly_legitimacy = -0.03
	global_population_happiness = -0.02
}

# [#betrothal-rework / GP-alliance workaround] a DYNASTIC ENTENTE — the diplomatic warmth + mutual-
# defence understanding a royal marriage creates when a FORMAL alliance is impossible. In Imperator
# vanilla a GREAT POWER cannot hold ordinary alliances (engine-hardcoded rank restriction), so a
# marriage involving a great power cannot seal the Coburg-style alliance a lesser pair gets. Rather
# than rely on add_alliance silently failing, such a match instead lays down this standing modifier
# on BOTH courts: an extra diplomatic slot (the dynastic tie IS a relationship), a defensive bonus
# (the two houses stand together), and a reputation nudge (a prestigious match). Applied by
# MARRIAGE_apply_marriage_bond when either party is a great power; a lasting (duration -1) bond the
# decay pass strips when the generation that made it passes (mirroring the alliance lapse). This is
# the "marriages work around the GP no-alliance rule" path the user asked for.
royal_marriage_entente = {
	diplomatic_relations = 1
	global_defensive = 0.05
	diplomatic_reputation = 1
}

# [#330 user] a DYNASTIC TIE TO THE IMPERIAL HOUSE — when the marriage is between an OVERLORD and its
# own SUBJECT (a Qing tributary monarchy wed into the Aisin Gioro, the historically-common binding of a
# vassal court to the throne), it does NOT create an alliance or entente (a subject cannot ally its
# overlord). Instead it TIGHTENS the bond. This modifier is applied to the OVERLORD (subject_loyalty is
# an OVERLORD-SIDE key — it raises the loyalty ALL that country's subjects feel toward it; cf.
# qing_office_lifanyuan_active applied to CHI). MARRIAGE_apply_marriage_bond ALSO bumps the specific
# subject's opinion toward the overlord (the per-subject loyalty lever), so THIS tie tightens even as the
# modifier's fleet-wide subject_loyalty buff rewards the overlord broadly. A lasting (duration -1)
# modifier the decay pass strips when the bond lapses (mirroring the alliance/entente lapse). A little
# monthly_legitimacy reflects the prestige of binding a vassal house into the dynasty. [#330 review fix:
# was wrongly applied to the SUBJECT, where subject_loyalty is inert for a tributary with no sub-vassals.]
royal_marriage_subject_tie = {
	subject_loyalty = 15
	monthly_legitimacy = 0.02
}

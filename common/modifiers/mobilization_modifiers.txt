########################################################################
# mobilization_modifiers.txt — UNIT-SCOPE readiness debuffs for the 19th-century
# mobilization / break-in system  (#192, overnight_decisions2.md §14b).  sys = MOBIL
#
# A newly-raised unit is NOT ready to fight: it must drill, gel into a cohesive
# body, and grow the logistical tail that feeds it (van Creveld: the tail is BUILT,
# not conjured). We model that as a MULTI-AXIS, FRONT-LOADED decay using only the
# proven engine idiom — self-expiring add_unit_modifier layers (oracle: legion =>
# every_legion_unit => add_unit_modifier { duration }; imp19c se_QING_MECHANICS.txt
# already stamps qing_unit_* this way). Each axis gets a heavy RAW layer (short
# duration, expires first) plus a lighter SETTLING layer (longer duration): so the
# penalty steps down (raw+settling) -> (settling) -> (none), front-loaded like the
# historical exp-decay curve, with NO per-tick variable bookkeeping.
#
# Three axes mature at different rates (Showalter, van Creveld, Lynn):
#   drill/discipline  — FAST   (raw layer short)   -> `discipline`
#   cohesion/morale   — MEDIUM (raw layer medium)  -> `land_morale_modifier`
#   supply/logistics  — SLOW   (raw layer long)    -> `land_unit_attrition`,`siege_ability`
#
# All keys below are verified valid at UNIT scope (add_unit_modifier): they are the
# same keys imp19c's qing_unit_banner_rot / qing_unit_greenstandard_rot already use.
# Positive land_unit_attrition = MORE losses; negative discipline/morale/siege = penalty.
#
# NO BOM (common/modifiers/*.txt convention here).
########################################################################

# ---- DRILL / DISCIPLINE axis (fast) --------------------------------------
# Raw recruits can't hold formation or volley on command. Clears fastest.
mobil_drill_raw = {          # heavy, short — the parade-ground shambles
	discipline = -0.20
}
mobil_drill_settling = {     # light, longer — still rough at the edges
	discipline = -0.08
}

# ---- COHESION / MORALE axis (medium) -------------------------------------
# Strangers in a stack break under pressure until they trust their file-mates.
mobil_cohesion_raw = {       # heavy, medium
	land_morale_modifier = -0.25
}
mobil_cohesion_settling = {  # light, longer
	land_morale_modifier = -0.10
}

# ---- SUPPLY / LOGISTICS axis (slow; the hardest to build) ----------------
# No baggage, no depots forward, no habit of resupply: the unit bleeds on the
# march and can't press a siege. Slowest to clear; for irregulars a residual
# floor persists (see the levy variant applied by MOBIL_stamp_unit).
mobil_supply_raw = {         # heavy, long
	land_unit_attrition = 0.30
	siege_ability = -0.20
}
mobil_supply_settling = {    # light, longest — the tail is still forming
	land_unit_attrition = 0.12
	siege_ability = -0.08
}

# ---- IRREGULAR SUPPLY FLOOR (levies/militia only) ------------------------
# Hastily-raised levies never fully acquire a professional logistical tail
# within a campaign season. A small, long-lived residual that a nearby supply
# depot / railway REMOVES at raise (the concrete van-Creveld depot hook).
mobil_supply_floor_irregular = {
	land_unit_attrition = 0.08
}

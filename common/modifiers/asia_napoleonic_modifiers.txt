# A European colony severed from its metropole by the wars in Europe (Dutch East
# Indies 1795; Spanish Philippines 1808). No reinforcement, no orders, no fleet — the
# governors hold on alone. A lasting weakness while the wars last (removed 1816 when
# the Europeans return, asia_napoleonic.4). This is the "opening" the Qing Nanyang tree
# exploits: a cut-off colony is soft.
asia_colony_cut_off = { # country scope
	global_manpower_modifier = -0.30
	manpower_recovery_speed = -0.30
	land_morale_modifier = -0.15
	global_defensive = -0.20
	global_supply_limit_modifier = -0.20
	global_monthly_state_loyalty = -0.10
}

# The East India Company stretched to its limit: Bonaparte threatening the road to
# India from Egypt, and Tipu Sultan's Mysore in arms and courting France (1798-99). A
# temporary strain (duration ~15 yrs) that opens a window the #449 India tree reads.
asia_company_overstretched = { # country scope
	global_manpower_modifier = -0.20
	land_morale_modifier = -0.10
	global_defensive = -0.15
}

# Russia pinned in the west by the 1812 invasion and the wars of the Sixth Coalition:
# the Far East (already thin) is stripped of men and attention. A temporary opening
# (~3 yrs) the Qing colonization tree's Amur branch can exploit. Country scope.
asia_russia_pinned_west = { # country scope
	global_manpower_modifier = -0.25
	manpower_recovery_speed = -0.25
	land_morale_modifier = -0.10
	global_defensive = -0.15
}

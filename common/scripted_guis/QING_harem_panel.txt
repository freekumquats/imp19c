########################################################################
# QING_harem_panel.txt — scripted GUIs for the Imperial Harem (後宮) — #360.
#
# The inner court (後宮) is part of the Imperial Household Department (內務府) run by the
# chamberlain (總管內務府大臣); its health folds into HIS Grand Council standing via
# qing_min_perf_chamberlain (se_QING_MINISTRY.txt term (e) — no council edit needed, the
# chamberlain office key is already enumerated in the perf fold). This is NOT a separate
# council seat — the harem is a facet of the chamberlain's charge.
#
# A Grand-Council-clone L4 window surfacing the LIVING CONSORTS (qing_harem_consorts) as a
# central CHARACTER roster + the inner-court depth (qing_harem_consort_count) as a read-out,
# with two ACTION LEVERS reusing the proven se_QING_HAREM effects — no new harem code:
#   • HOLD A DRAFT (選秀): QING_harem_hold_draft — mint new consorts (gated below a court of 8),
#   • FAVOUR A CONSORT (臨幸): QING_harem_favour_consort — beget a recognised imperial child
#     (gated on a male reigning emperor + a non-empty harem, mirroring the effect's own guard).
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_harem_open_panel_button (a button in gui/qing_household.gui — the harem sits
# under the Household). The window itself is gui/qing_harem.gui. Prefix = qing_harem_.
########################################################################

# ---- OPEN THE HAREM PANEL (button in the Household panel) ------------------
qing_harem_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# [#443 boot-crash fix] rebuild the roster + refresh the candidate list via the hidden
		# qing_harem.3 event (a RUNTIME trigger_event, never compile-inlined) instead of calling
		# QING_harem_refresh_candidates directly here — its body is an `ordered_character {
		# order_by = charisma ... }` SORTING iterator, and a scripted_gui compile-inlines its
		# button's effect chain at PARSE time, which access-violates the loader on a sorting
		# char-iterator (same class as the censorate/justice levers). Behaviour is identical.
		trigger_event = { id = qing_harem.3 }
		LOG_line = { sys = QING  msg = "EVENT qing_harem_open_panel_button fired; opening the Imperial Harem panel for" }
	}
}

# ---- HAREM-POPULATED indicator (the GUI reveals the roster when non-empty) --
qing_harem_populated = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_harem_consort_count
		var:qing_harem_consort_count > 0
	}
}

# ---- HOLD A DRAFT (選秀) — mint new consorts into the inner court. Gated below a court of 8
# (the effect self-caps too, but disable the button once full so it is not a no-op affordance).
qing_harem_hold_draft = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		# the throne must be occupied to keep a harem, and the court not yet crowded.
		exists = current_ruler
		has_variable = qing_harem_consort_count
		var:qing_harem_consort_count < 8
	}
	effect = {
		QING_harem_hold_draft = yes
		LOG_line = { sys = QING  msg = "harem panel: a consort draft is held for" }
	}
}

# ---- FAVOUR A CONSORT (臨幸) — open the FAVOUR PICKER. [#9 boot] The panel button no longer
# favours a random consort; it refreshes the favourable-consort list (the qing_harem.4 trampoline —
# a SORTING iterator, so trigger_event-fired, never inlined) and opens qing_harem_favour_window, where
# a row-click favours the CHOSEN woman (qing_harem_favour_selected). is_valid mirrors the effect's own
# guard so the button greys on an empty/ineligible court (a male reigning emperor + a favourable consort).
qing_harem_favour_consort = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		exists = current_ruler
		current_ruler = { is_male = yes }
		has_variable_list = qing_harem_consorts
		# a consort who is alive AND not already pregnant — mirrors the effect's own guard so
		# the button is not a no-op affordance when every consort already carries a child.
		any_in_list = { variable = qing_harem_consorts  is_alive = yes  is_pregnant = no }
	}
	effect = {
		trigger_event = { id = qing_harem.4 }
		LOG_line = { sys = QING  msg = "harem panel: opening the favour-a-consort picker for" }
	}
}

# ---- FAVOUR THE CHOSEN CONSORT (臨幸) — row action on the favour picker. The clicked consort comes
# via the picker's character datacontext (ROOT = the woman); scope:player = CHI. Re-roots to CHI
# through qing_harem.5 (the effect is written for ROOT = CHI reading scope:qing_favour_target), exactly
# the take-consort re-root pattern. is_valid re-guards the emperor + the woman's eligibility.
qing_harem_favour_selected = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
	}
	is_valid = {
		is_alive = yes
		is_female = yes
		is_pregnant = no
		has_variable = qing_is_harem_consort
		scope:player = {
			exists = current_ruler
			current_ruler = { is_male = yes }
		}
	}
	effect = {
		save_scope_as = qing_favour_target
		scope:player = {
			trigger_event = { id = qing_harem.5 }
			LOG_line = { sys = QING  msg = "harem panel: the emperor favours a chosen consort for" }
		}
	}
}

# ---- [#438] PROMOTE A CONSORT (晉封) — open the PROMOTE PICKER. [#9 boot] The panel button no longer
# promotes the most-favoured consort automatically; it refreshes the promotable-consort list (the
# qing_harem.6 trampoline — a SORTING iterator, so trigger_event-fired) and opens
# qing_harem_promote_window, where a row-click elevates the CHOSEN woman (qing_harem_promote_selected).
# is_valid mirrors the effect's own guard: a living consort below the top rank whose next tier under its
# cap (皇貴妃 1 / 貴妃 2 / 妃 4) still has an open place — so the button greys when the ladder is full.
qing_harem_promote_consort = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		exists = current_ruler
		has_variable_list = qing_harem_consorts
		any_in_list = {
			variable = qing_harem_consorts
			is_alive = yes
			var:qing_consort_rank < 4
			OR = {
				AND = { var:qing_consort_rank = 3  ROOT.var:qing_harem_rank4_count < 1 }
				AND = { var:qing_consort_rank = 2  ROOT.var:qing_harem_rank3_count < 2 }
				AND = { var:qing_consort_rank = 1  ROOT.var:qing_harem_rank2_count < 4 }
			}
		}
	}
	effect = {
		trigger_event = { id = qing_harem.6 }
		LOG_line = { sys = QING  msg = "harem panel: opening the promote-a-consort picker for" }
	}
}

# ---- PROMOTE THE CHOSEN CONSORT (晉封) — row action on the promote picker. The clicked consort comes
# via the picker's character datacontext (ROOT = the woman); scope:player = CHI. Re-roots to CHI through
# qing_harem.7 (the effect is written for ROOT = CHI reading scope:qing_promote_target). is_valid re-guards
# her rank against the effect's caps — but the caps read here are ROOT.var on the CHARACTER root, which
# is wrong, so the per-tier cap check is deferred to the effect (which recomputes on CHI); the row only
# guards the woman is a living consort below the top rank.
qing_harem_promote_selected = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
	}
	is_valid = {
		is_alive = yes
		has_variable = qing_is_harem_consort
		var:qing_consort_rank < 4
		scope:player = { exists = current_ruler }
	}
	effect = {
		save_scope_as = qing_promote_target
		scope:player = {
			trigger_event = { id = qing_harem.7 }
			LOG_line = { sys = QING  msg = "harem panel: a chosen consort is elevated in rank for" }
		}
	}
}

# ---- [#418] TAKE AS CONSORT (納為妃嬪) — the emperor takes an existing court woman into the inner
# court. Row action on the court-candidate picker (the char comes via the picker datacontext; ROOT =
# the woman, scope:player = the country), mirroring qing_guard_enrol_guardsman. Unlike the triennial
# DRAFT (which mints a NEW banner woman), this elevates a woman ALREADY on the map — an unmarried adult
# female of the emperor's household who is not already a consort. Self-caps at a court of 8. The effect
# runs at COUNTRY scope reading scope:qing_take_target, so save the picked char + call on the player.
qing_harem_take_consort = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
	}
	is_valid = {
		# an eligible, unmarried adult court woman of the player, not already a consort; court not full.
		is_alive = yes
		is_female = yes
		is_adult = yes
		is_married = no
		employer = scope:player
		NOT = { has_variable = qing_is_harem_consort }
		scope:player = {
			exists = current_ruler
			OR = {
				NOT = { has_variable = qing_harem_consort_count }
				var:qing_harem_consort_count < 8
			}
		}
	}
	effect = {
		# [review-fix / BT-13 pattern] This scripted_gui runs at CHARACTER root (ROOT = the clicked woman),
		# but QING_harem_take_consort + QING_harem_recompute_roster + QING_harem_refresh_candidates are all
		# written for ROOT = CHI (they use `employer = ROOT` / `set_home_country = ROOT` / `ROOT = { add_to_list }`).
		# `scope:player = { ... }` rebases `this` to CHI but leaves ROOT on the character — so firing the
		# ROOT-based machinery here would wipe the roster and spam errors. Save the target and re-root to CHI
		# through the hidden qing_harem.2 country_event (exactly the qing_office.40 fix); saved scopes carry
		# into trigger_event (proven: Invictus bribe_mercenary_button -> warfare.3).
		save_scope_as = qing_take_target
		scope:player = {
			trigger_event = { id = qing_harem.2 }
			LOG_line = { sys = QING  msg = "harem panel: a court woman is taken as a consort for" }
		}
	}
}

# ---- CLOSE (button in the panel window) ------------------------------------
qing_harem_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = { }
}

# QING_works_ministry_panel.txt — scripted GUIs for the Ministry of Works panel (#351).
#
# The Board of Works (工部) bureaucracy: a Grand-Council-clone L4 window headed by the
# Grand Minister of Works (工部尚書), surfacing the public-works buildings he answers for
# (dikes 河堤, canal depots 漕運倉, Great Wall sections 長城, state manufactories) as a
# central roster, the ministry-performance meter qing_min_perf_works (which folds into his
# Grand Council standing — se_QING_MINISTRY.txt / QING_council_fold_ministry_perf), and
# — unlike the read-only War/Lifan panels — COMMISSION BUTTONS that let the Works minister
# raise a dike / canal depot / Great Wall section directly (the capable route of the proven
# se_QING_WORKS.txt build effects, finesse-gated + costed there).
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_works_ministry_open_panel_button (a button in government_view.gui).
# The window itself is gui/qing_works_ministry.gui. Prefix = qing_works_ministry_.
########################################################################

# ---- OPEN THE WORKS MINISTRY PANEL (button in the government view) ----------
qing_works_ministry_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# The window reads the works roster + qing_min_perf_works directly from ROOT.
		LOG_line = { sys = QING  msg = "EVENT qing_works_ministry_open_panel_button fired; opening the Board of Works panel for" }
	}
}

# ---- WORKS MINISTER HOLDER INDICATOR (the GUI reveals his card when filled) -
qing_works_ministry_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_works_holder
		var:qing_office_works_holder = { is_alive = yes  employer = ROOT }
	}
}

# ---- VACANT indicator (the GUI shows a "post vacant" note) ------------------
qing_works_ministry_vacant = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = {
				has_variable = qing_office_works_holder
				var:qing_office_works_holder = { is_alive = yes  employer = ROOT }
			}
		}
	}
}

# ---- COMMISSION A DIKE (河堤) — the capable route (cheap = no). Requires a filled
# Works office with adequate finesse, treasury for the build, and a province lacking a dike.
qing_works_ministry_build_dike = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		has_variable = qing_office_works_holder
		var:qing_office_works_holder = { is_alive = yes  employer = ROOT  finesse >= 7 }
		treasury >= 90
		any_owned_province = { NOT = { has_building = qing_dike_building } }
	}
	effect = {
		QING_works_build_dike = { cheap = no }
	}
}

# ---- CLOSE (button in the panel window) ------------------------------------
qing_works_ministry_close_panel = {
	scope = country

# [#434] THE GREAT MONUMENTS — raise the Great Wall (長城) / Grand Canal (大運河) proper. Great
# labour projects: levy treasury AND manpower, raise-once. is_valid mirrors the effect's guards.
qing_works_ministry_build_great_wall = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI  NOT = { any_owned_province = { has_building = qing_great_wall_building } } }
	is_valid = {
		has_variable = qing_office_works_holder
		var:qing_office_works_holder = { is_alive = yes  employer = ROOT  finesse >= 7 }
		treasury >= 240
		manpower >= 10
		exists = capital_scope
	}
	effect = { QING_works_build_great_wall = yes }
}

# [#440] COUNTRYWIDE RAILWAY NETWORK — extend the national rail grid one province at a
# time. REPEATABLE (unlike the raise-once monuments): shown once railways are a real
# thing (tech_steam_locomotive, unlocked by the Self-Strengthening rail task) and while a
# core province still lacks rail. is_valid mirrors QING_rail_network_extend's guards.
qing_works_ministry_extend_railway = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		tag = CHI
		invention = tech_steam_locomotive
		any_owned_province = { NOT = { has_building = INF_railway_upgrade } }
	}
	is_valid = {
		has_variable = qing_office_works_holder
		var:qing_office_works_holder = { is_alive = yes  employer = ROOT  finesse >= 7 }
		treasury >= 60
		manpower >= 5
		any_owned_province = { NOT = { has_building = INF_railway_upgrade } }
	}
	effect = { QING_rail_network_extend = yes }
}

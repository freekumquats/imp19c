########################################################################
# QING_religion_panel.txt — scripted GUIs for the Qing Religion panel (#268).
#
# Read-only surfacing of the religious-tension state the mod already tracks on the
# Qing (CHI) player: the anti-Christian sentiment meter (仇教情緒), sectarian pressure
# (sect_pressure), the concrete missionary reach (mission-station count), reform
# pressure, the active sentiment band, the anti-Christian agitator (if he has taken
# the stage), and the Taiping / Heavenly-Kingdom progress. This is the player's
# window onto the missionary / sectarian / Taiping systems that #275 built but that
# had no dedicated surface for the player to read.
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# The panel is opened by a button in the religion view (see the splice in
# gui/religion_view.gui). The GUI window itself is defined in gui/qing_religion.gui.
#
# Mirrors the proven QING_greatgame_panel.txt idiom exactly (read-only, stateless,
# console-command open/close). Prefix = qing_religion_
########################################################################

# ---- OPEN THE RELIGION PANEL (button in the religion view) ------------------
qing_religion_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	# No state to set — the window reads the religion counters directly from ROOT
	# (the Qing player). The GUI is stateless + read-only.
	effect = {
	}
}

# ---- BAND INDICATORS (the GUI shows one line matching the live band) --------
# Fever band: sentiment >= 66 (扶清滅洋). Mirrors QING_missionary_apply_band.
qing_religion_band_fever = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_antichristian_sentiment
		var:qing_antichristian_sentiment >= 66
	}
}

# Simmering band: 40..65. Mirrors QING_missionary_apply_band.
qing_religion_band_simmering = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_antichristian_sentiment
		var:qing_antichristian_sentiment >= 40
		NOT = { var:qing_antichristian_sentiment >= 66 }
	}
}

# Calm band: < 40 (or unset). Shown when neither of the above holds.
qing_religion_band_calm = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = { has_variable = qing_antichristian_sentiment  var:qing_antichristian_sentiment >= 40 }
		}
	}
}

# ---- ANTI-CHRISTIAN AGITATOR (仇教 firebrand) — shown if he has taken the stage
# QING_missionary_spawn_agitator sets qing_spawned_antichr_agitator and saves the
# character; the GUI reveals his portrait only once he exists.
qing_religion_agitator_present = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_spawned_antichr_agitator
		# [#268 review-fix] re-validate the character is still alive AND still ours before the GUI
		# reads Var('qing_antichr_agitator').GetCharacter — the spawn flag is never cleared on death
		# (no on_character_death cleanup for the agitator), so gating on the flag alone would leave a
		# stale dead-man / blank portrait in the panel. Mirrors the proven qing_greatgame_zongli_filled
		# guard (var:...holder = { is_alive = yes }).
		var:qing_antichr_agitator = { is_alive = yes }
	}
}

# ---- TAIPING / HEAVENLY-KINGDOM progress — shown once the overture has opened
# the tree (qing_taiping_overture). The bar reads qing_taiping_progress (0..100).
qing_religion_taiping_active = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_taiping_overture
	}
}

# ---- CLOSE THE RELIGION PANEL (button in the panel window) ------------------
qing_religion_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = {
		# The window closes via GUI.ClearWidgets on the button; no state to clear
		# (the panel is read-only).
	}
}

########################################################################
# QING_princes_panel.txt — scripted GUIs for Court Intrigue + Succession (宮闈鬥爭 / 立儲之爭) — #368.
#
# The succession contest is the EMPEROR'S charge, not a Grand Council office — so, like the Harem
# (#360) and Upper Study (#337), it folds into the council transitively through the shared
# qing_dynastic_harmony meter (QING_council_recompute additively converts harmony into
# qing_council_eff_target). A house riven by a succession war drags the council; a sealed, secure
# succession lets the great offices work. No council-perf edit — the harmony coupling is the fold.
#
# A Grand-Council-clone L4 window surfacing the reigning emperor's ADULT PRINCES (qing_princes) as
# a central CHARACTER roster, each with his court backing (qing_prince_backing), plus the four
# proven se_QING_PRINCES levers — no new engine code here:
#   • FAVOUR A PRINCE (培植)            — QING_princes_favour
#   • SECRETLY DESIGNATE THE HEIR (秘密立儲) — QING_princes_seal_succession   (the teeth)
#   • INVESTIGATE AN OVER-MIGHTY PRINCE (查辦) — QING_princes_investigate     (gated on an over-mighty prince)
#   • REOPEN THE SUCCESSION (廢儲)      — QING_princes_reopen                 (gated on a sealed succession)
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_princes_open_panel_button (a button in gui/government_view.gui). The window itself
# is gui/qing_princes.gui. Prefix = qing_princes_. Mirrors QING_harem_panel.txt.
########################################################################

# ---- OPEN THE PRINCES PANEL (button in the government view) ----------------
qing_princes_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# rebuild the roster so the window opens on a fresh prince list.
		QING_princes_recompute_roster = yes
		LOG_line = { sys = QING  msg = "EVENT qing_princes_open_panel_button fired; opening the Court Intrigue panel for" }
	}
}

# ---- SUCCESSION-SEALED indicator (the GUI reveals the sealed banner when set) --
qing_princes_is_sealed = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { has_variable = qing_secret_succession_sealed }
}

# ---- FAVOUR A PRINCE (培植) — groom the ablest prince (+prestige, +backing). Needs a prince to
# groom; while unsealed it inflames the losers (the effect applies the harmony hit).
qing_princes_favour = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		exists = current_ruler
		has_variable = qing_prince_count
		var:qing_prince_count >= 1
	}
	effect = {
		QING_princes_favour = yes
		LOG_line = { sys = QING  msg = "princes panel: the throne grooms a favoured prince for" }
	}
}

# ---- SECRETLY DESIGNATE THE HEIR (秘密立儲) — the teeth. Seal the ablest prince as the hidden
# heir. Needs a prince to name and an as-yet UNSEALED succession (mirror the effect's own state).
qing_princes_seal_succession = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		exists = current_ruler
		has_variable = qing_prince_count
		var:qing_prince_count >= 1
		NOT = { has_variable = qing_secret_succession_sealed }
	}
	effect = {
		QING_princes_seal_succession = yes
		LOG_line = { sys = QING  msg = "princes panel: 秘密立儲 sealed on the front-runner for" }
	}
}

# ---- INVESTIGATE AN OVER-MIGHTY PRINCE (查辦) — strip a dangerously over-backed prince. MIRROR
# the effect's own guard (a prince at backing >= 70), so the button is not a live no-op.
qing_princes_investigate = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		has_variable_list = qing_princes
		any_in_list = { variable = qing_princes  var:qing_prince_backing >= 70 }
	}
	effect = {
		QING_princes_investigate = yes
		LOG_line = { sys = QING  msg = "princes panel: an over-mighty prince is investigated for" }
	}
}

# ---- REOPEN THE SUCCESSION (廢儲) — depose the sealed heir. Only when a succession is sealed
# (mirror the effect's own guard).
qing_princes_reopen = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		has_variable = qing_secret_succession_sealed
	}
	effect = {
		QING_princes_reopen = yes
		LOG_line = { sys = QING  msg = "princes panel: 廢儲 — the sealed succession is reopened for" }
	}
}

# ---- CLOSE (button in the panel window) ------------------------------------
qing_princes_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = { }
}

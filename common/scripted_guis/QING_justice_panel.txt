########################################################################
# QING_justice_panel.txt — scripted GUIs for the Board of Punishments panel (#363).
#
# The Board of Punishments (刑部) bureaucracy: a Grand-Council-clone L4 window headed by the
# Grand Minister of Justice (刑部尚書), surfacing the ministry-performance meter
# qing_min_perf_justice (which folds into his Grand Council standing — a fold that had NEVER
# been computed before #363, the same latent bug the Censorate #362 fixed), the realm's
# corruption level, and the concrete DOCKET (待審) of accused officials the Board holds and
# tries. Three action levers: ACCUSE AN OFFICIAL (#6: opens a corruption-sorted PICKER window
# so the player chooses whom to charge onto the docket), CONVICT (定罪 — disgrace + penal
# servitude) and ACQUIT (開釋) each docketed man.
#
# Distinct from the Censorate (都察院, #362) which impeaches in one blow: the 刑部 is the high
# COURT — it CHARGES first, then TRIES. Scoped to the Qing player (ROOT = CHI); AI never
# evaluates these (ai_is_valid = no). Opened by qing_justice_open_panel_button (a button in
# government_view.gui). Window: gui/qing_justice.gui. Prefix = qing_justice_. se_LOG sys = QING.
########################################################################

# ---- OPEN THE BOARD OF PUNISHMENTS PANEL (button in the government view) ----
qing_justice_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# recompute so the meter + docket roster are fresh the moment the panel opens.
		QING_ministry_recompute_perf_justice = yes
		LOG_line = { sys = QING  msg = "EVENT qing_justice_open_panel_button fired; opening the Board of Punishments panel for" }
	}
}

# ---- MINISTER-OF-JUSTICE HOLDER INDICATOR ----------------------------------
qing_justice_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_justice_holder
		var:qing_office_justice_holder = { is_alive = yes  employer = ROOT }
	}
}

qing_justice_vacant = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = {
				has_variable = qing_office_justice_holder
				var:qing_office_justice_holder = { is_alive = yes  employer = ROOT }
			}
		}
	}
}

# ---- ACCUSE AN OFFICIAL — the panel button's ENABLE-GATE (#6).
# [#6] This verb no longer CHARGES anyone directly: the button now fires
# qing_justice_refresh_accusable_btn + opens the qing_justice_accuse_window picker, and a
# row-click there charges the chosen man (qing_justice_accuse_selected). This verb survives
# only as the button's is_valid gate — it stays dead (button disabled) unless a Minister of
# Justice is seated AND at least one accusable official exists, so the picker is never opened
# empty. Its is_valid limit MUST stay byte-identical to QING_justice_refresh_accusable's.
# (The old auto-pick effect — trigger_event qing_justice.5 → QING_justice_find_accusable —
# is retained in events for reference but no longer invoked from here.) Country scope.
# ----------------------------------------------------------------------
qing_justice_accuse_official = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		# a present Minister of Justice must be seated to lay a charge...
		has_variable = qing_office_justice_holder
		var:qing_office_justice_holder = { is_alive = yes  employer = ROOT }
		# [#13 boot] ...the throne must be able to AFFORD the political capital a formal indictment
		# spends (mirror qing_justice_accuse_selected + the qing_justice.8 charge cost). Scope = CHI.
		political_influence >= 20
		# ...and there must BE an accusable official not already on the docket (mirror
		# find_accusable's limit, so no dead-affordance click).
		any_character = {
			employer = ROOT
			is_ruler = no
			is_adult = yes
			is_alive = yes
			NOT = { has_variable = qing_rebel_foil }
			NOT = { has_variable = qing_justice_accused }
			# [#14 boot] STRIP-THEN-ACCUSE: office-holders/commanders/governors are chargeable now (they
			# are relieved of the shielding post on charge), so the old NOT={has_variable=qing_office_held}
			# + is_general/is_admiral/is_governor=no exclusions are dropped. Keep only imprison's un-strippable
			# bars. MUST stay byte-identical to QING_justice_find_accusable / _refresh_accusable / _accuse_selected.
			# [#14 review-fix] but the sitting Justice minister cannot try himself (self-charge would vacate
			# his seat and strand his docket entry un-triable).
			NOT = { AND = { has_variable = qing_office_held  var:qing_office_held = flag:justice } }
			is_co_ruler = no
			is_primary_heir = no
			is_head_of_family = no
			is_clan_chief = no
			OR = {
				corruption >= 20
				has_trait = corrupt
				has_trait = cruel
				has_trait = arbitrary
				loyalty < 50
			}
		}
	}
	effect = {
		# [#6] NO-OP. This verb is now only the button's enable-gate (see block header); the
		# panel button's onclick fires qing_justice_refresh_accusable_btn + opens the picker,
		# and does NOT Execute this verb — so this effect never runs. Left empty (rather than
		# the old auto-pick trigger_event qing_justice.5) so that if anything ever does invoke
		# it, it cannot silently auto-charge a man behind the player's back.
	}
}

# ---- CONVICT AN ACCUSED (定罪) — the trial goes against a docketed man. Row action
# on the docket roster; the accused comes via datacontext (GuiScope char + player).
# Disgrace + strip office + imprison (下獄) via QING_justice_convict_accused. Mirrors
# qing_censorate_recall_inspector's char+player scope shape.
# ----------------------------------------------------------------------
qing_justice_convict = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
		has_variable = qing_justice_accused
	}
	is_valid = {
		has_variable = qing_justice_accused
		employer = scope:player
		# only a seated Minister of Justice may bring a case to judgement.
		scope:player = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = { is_alive = yes  employer = scope:player }
		}
	}
	effect = {
		# [#443 boot-crash fix] route through the hidden CHARACTER event qing_justice.6 (a RUNTIME
		# trigger_event, never compile-inlined) instead of calling QING_justice_convict_accused
		# directly here — it reaches the vanilla `imprison` engine effect, and a scripted_gui
		# compile-inlines its button's effect chain at PARSE time, which access-violates the loader
		# (same class as the censorate/harem/justice-accuse/princes/xinjiang iterators). ROOT is the
		# convict (as convict_accused expects); the saved player scope carries into trigger_event.
		trigger_event = { id = qing_justice.6 }
		LOG_line = { sys = QING  msg = "justice: an accused convicted from the panel for" }
	}
}

# ---- ACQUIT AN ACCUSED (開釋) — the trial clears a docketed man. Row action. ----
qing_justice_acquit = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
		has_variable = qing_justice_accused
	}
	is_valid = {
		has_variable = qing_justice_accused
		employer = scope:player
		# [#363-R2 review-fix] acquittal (開釋) is a trial outcome, so it too requires a seated
		# Minister of Justice — mirrors the convict gate. Without this, a docket populated while a
		# minister was seated could be cleared by clemency during a later vacancy (a headless Board
		# dispensing pardons + free loyalty_qing_commended), an asymmetry with the convict lever.
		scope:player = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = { is_alive = yes  employer = scope:player }
		}
	}
	effect = {
		QING_justice_acquit_accused = yes
		LOG_line = { sys = QING  msg = "justice: an accused acquitted from the panel for" }
		scope:player = { QING_ministry_recompute_perf_justice = yes }
	}
}

# ---- [#6] REFRESH THE ACCUSABLE-OFFICIAL PICKER ROSTER ----------------------
# Country-scope verb the ACCUSE button fires before opening the picker window. Rebuilds
# qing_justice_accusable (most-corrupt-first) via the scripted-effect (safe sorting inline).
qing_justice_refresh_accusable_btn = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# [#6 crash-safety] route the SORTING roster build through the hidden qing_justice.7
		# RUNTIME event (never compile-inlined) instead of calling QING_justice_refresh_accusable
		# directly here — its body is an `ordered_character order_by=corruption` sorting iterator,
		# and a scripted_gui compile-inlines its button effect chain at PARSE time (the same class
		# that boot-crashed THIS panel's accuse lever in #443). Mirrors the .5/.6 trampolines.
		trigger_event = { id = qing_justice.7 }
		LOG_line = { sys = QING  msg = "justice: accuse-picker roster refreshed from the panel for" }
	}
}

# ---- [#6] ACCUSE A SELECTED OFFICIAL (待審) — row-click in the picker window. --------
# The chosen courtier comes via datacontext (GuiScope char + player), exactly like the
# convict/acquit row levers. Marks him qing_justice_accused so the docket rebuild picks
# him up, then recomputes. No trampoline needed: no sorting iterator, no imprison — this
# is a bare set_variable + recompute (same shape as qing_justice_acquit's effect).
qing_justice_accuse_selected = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
		employer = scope:player
		NOT = { has_variable = qing_justice_accused }
	}
	is_valid = {
		# a present Minister of Justice must be seated to lay a charge (mirror the panel gate).
		scope:player = {
			has_variable = qing_office_justice_holder
			var:qing_office_justice_holder = { is_alive = yes  employer = scope:player }
			# [#13 boot] the throne must afford the political capital an indictment spends.
			political_influence >= 20
		}
		# the selected man must still be chargeable — mirror QING_justice_refresh_accusable's
		# limit so a row that went stale (e.g. gained a bar since the refresh) is dead.
		employer = scope:player
		is_ruler = no
		is_adult = yes
		is_alive = yes
		NOT = { has_variable = qing_rebel_foil }
		NOT = { has_variable = qing_justice_accused }
		# [#14 boot] office-holders/commanders/governors ARE chargeable (stripped on charge); keep
		# only imprison's un-strippable bars. Byte-identical to the shared accusable limit.
		# [#14 review-fix] the sitting Justice minister cannot try himself.
		NOT = { AND = { has_variable = qing_office_held  var:qing_office_held = flag:justice } }
		is_co_ruler = no
		is_primary_heir = no
		is_head_of_family = no
		is_clan_chief = no
		OR = {
			corruption >= 20
			has_trait = corrupt
			has_trait = cruel
			has_trait = arbitrary
			loyalty < 50
		}
	}
	effect = {
		# [#13/#14 boot] route the charge through the hidden CHARACTER event qing_justice.8 (a RUNTIME
		# trigger_event, never compile-inlined): it STRIPS the shielding position (QING_justice_strip_for_trial
		# reaches remove_command / remove_as_governor / QING_office_vacate_dispatch — vacate itself fires a
		# nested trigger_event, which must not be compile-inlined into this button), spends the political
		# influence, adds the small tyranny of a politically-charged indictment, then sets the accused mark.
		# ROOT = the charged man (as strip_for_trial expects); the saved player scope carries into trigger_event.
		trigger_event = { id = qing_justice.8 }
		LOG_line = { sys = QING  msg = "justice: an official charged onto the docket via the picker for" }
	}
}


# ---- CLOSE ------------------------------------------------------------------
qing_justice_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = { }
}

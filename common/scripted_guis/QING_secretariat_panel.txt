########################################################################
# QING_secretariat_panel.txt — scripted GUIs for the Central Secretariat panel (#361).
#
# The 內閣 (Central Secretariat) is the throne's edict-drafting chancery, headed by the
# Grand Secretary (大學士) — the SAME grand_secretariat office the Hanlin Academy (#358)
# feeds. The Hanlin is the talent RESERVOIR; the 內閣 is where those men WORK, turning the
# throne's will into drafted rescripts (票擬) and clearing the backlog (積壓) that piles up
# when the chancery grinds.
#
# ONE OFFICE, ONE METER (mirrors the harem→household precedent #360): this is NOT a separate
# council seat and reuses the EXISTING qing_min_perf_grand_secretariat meter — the chancery's
# throughput folds in via se_QING_MINISTRY.txt term (e), no council edit. The window surfaces
# the concrete rescript BACKLOG (qing_secretariat_backlog) as a read-out + a single ACTION
# LEVER (CONVENE THE CHANCERY 票擬) reusing the proven QING_secretariat_convene effect.
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_secretariat_open_panel_button (a button in gui/qing_hanlin.gui — the 內閣 sits
# alongside the Hanlin under the same Grand Secretary). The window is gui/qing_secretariat.gui.
# Prefix = qing_secretariat_.
########################################################################

# ---- OPEN THE CENTRAL SECRETARIAT PANEL (button in the Hanlin panel) -------
qing_secretariat_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# the window reads the backlog + qing_min_perf_grand_secretariat directly from ROOT;
		# make sure a fresh backlog value exists before it opens.
		QING_secretariat_init = yes
		LOG_line = { sys = QING  msg = "EVENT qing_secretariat_open_panel_button fired; opening the Central Secretariat panel for" }
	}
}

# ---- GRAND SECRETARY HOLDER INDICATOR (the GUI reveals his card when filled) -
qing_secretariat_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_grand_secretariat_holder
		var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
	}
}

# ---- VACANT indicator (the GUI shows a "post vacant" note) ------------------
qing_secretariat_vacant = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = {
				has_variable = qing_office_grand_secretariat_holder
				var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
			}
		}
	}
}

# ---- CONVENE THE CHANCERY (票擬) — a concentrated drafting session that clears a chunk of
# the backlog + marshals political influence. MIRROR the effect's own guard (a present Grand
# Secretary + a backlog worth convening for + no cooldown) so the button and effect agree —
# without this it would be a live-but-no-op affordance on an empty backlog, a vacant office,
# or during the cooldown.
qing_secretariat_convene = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		has_variable = qing_office_grand_secretariat_holder
		var:qing_office_grand_secretariat_holder = { is_alive = yes  employer = ROOT }
		has_variable = qing_secretariat_backlog
		var:qing_secretariat_backlog >= 20
		NOT = { has_variable = qing_secretariat_convene_cooldown }
	}
	effect = {
		QING_secretariat_convene = yes
		LOG_line = { sys = QING  msg = "secretariat panel: the chancery is convened for a drafting session for" }
	}
}

# ---- CLOSE (button in the panel window) ------------------------------------
qing_secretariat_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = { }
}

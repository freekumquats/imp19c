########################################################################
# QING_court_position.txt — [B19] gate for the character-window court-position title row.
#
# The f867bcb69 title row used a hand-rolled 8-way Or() of .MakeScope.Var(...).IsSet checks in
# the GUI `visible`, which never rendered for anyone (static-clean but broken). Replace that gate
# with this proven ScriptedGui.IsShown pattern (the same idiom the character-window scheme row +
# technology_view office rows use: ScriptedGui.IsShown( GuiScope.SetRoot( <char>.MakeScope ).End )).
#
# is_shown = the character holds SOME Qing court position: a Grand Council great office
# (qing_office_held), a ministry sub-post (Zongli diplomat / investigating censor / guard captain),
# an inner-court study attendant (Southern / Upper Study), an amban, or a palace eunuch. Character
# scope. Names come from the QING_court_position_name custom_loc, keyed off the same markers.
########################################################################

qing_has_court_position = {
	scope = character
	ai_is_valid = { always = no }
	is_shown = {
		OR = {
			has_variable = qing_office_held
			has_variable = qing_zongli_diplomat
			has_variable = qing_is_censor_inspector
			has_variable = qing_is_imperial_guardsman
			has_variable = qing_is_southernstudy
			has_variable = qing_is_upperstudy
			has_variable = qing_amban_marker
			has_variable = qing_is_palace_eunuch
		}
	}
	is_valid = { always = yes }
	effect = { }
}

########################################################################
# SUB_QING_subject_interactions.txt — scripted GUIs for per-subject actions
#
# These are the gated entry points the diplomatic-view buttons call. Each is
# scoped to the player (ROOT = overlord) and reads scope:target as the subject
# currently shown in the diplomacy view, passed from the GUI via:
#     .AddScope('target', DiplomaticView.GetTargetCountry.MakeScope )
#
# Gating lives in is_valid (influence cost + valid ladder position + cooldown),
# surfaced to the player with custom_tooltip red-reason lines. The actual work
# is done by the SUBJ_QING_* effects in se_SUBJECT_QING.txt.
#
# AI is disabled on every interaction (ai_is_valid = { always = no }) — this is
# a deliberately player-only feature set.
#
# Prefix = qing_subject_
########################################################################

# ---- TYPE INDICATORS (read-only, for the diplomatic-view "Type:" label) ---
# There is no engine datafunction that returns a subject's type name, so the
# UI reads the type off these boolean scripted GUIs via ScriptedGui.IsShown
# nested in SelectLocalization. Each is true iff scope:target is that type.
# autonomous_governorship needs no indicator: it is the final SelectLocalization
# fallback once the other three are false. AI never evaluates these.
qing_subject_type_is_tributary = {
	scope = country
	saved_scopes = { target }
	ai_is_valid = { always = no }
	is_shown = {
		exists = scope:target
		# [1763-hist-fix 2026-07-17] The bottom rung of the Qing ladder is now the sinosphere_tributary
		# (朝貢國) — the historically-apt ritual tributary (protected, event-based tribute, non-cancellable).
		# Match BOTH so any legacy plain-tributary subject still labels + promotes correctly.
		scope:target = { OR = { is_subject_type = sinosphere_tributary  is_subject_type = tributary } }
	}
}

qing_subject_type_is_nominal_vassal = {
	scope = country
	saved_scopes = { target }
	ai_is_valid = { always = no }
	is_shown = {
		exists = scope:target
		scope:target = { is_subject_type = nominal_vassal }
	}
}

qing_subject_type_is_feudatory = {

# ---- PROMOTE: move the subject one rung tighter -----------------------
# tributary -> nominal_vassal -> feudatory -> autonomous_governorship
qing_subject_promote_button = {
	scope = country
	saved_scopes = { target }

	is_shown = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
		# Only meaningful for the four Qing ladder types, and not at the top rung.
		scope:target = {
			OR = {
				is_subject_type = sinosphere_tributary   # [1763-hist-fix] bottom rung of the ladder
				is_subject_type = tributary               # legacy plain-tributary still promotable
				is_subject_type = nominal_vassal
				is_subject_type = feudatory
			}
		}
	}

	ai_is_valid = { always = no }

	is_valid = {
		custom_tooltip = {
			text = qing_subject_needs_influence_75_tt
			political_influence >= 75
		}
		# Cooldown is tracked per-subject, so acting on one subject does not
		# lock out interactions with the others.
		custom_tooltip = {
			text = qing_subject_promote_cooldown_tt
			scope:target = { NOT = { has_variable = qing_recently_changed_type } }
		}
	}

	effect = {
		add_political_influence = -75
		scope:target = { set_variable = { name = qing_recently_changed_type  days = 730  value = 1 } }

		# Step one rung tighter based on the subject's current type.
		scope:target = { SUBJ_QING_derive_rung = yes }
		if = {
			limit = { scope:target = { OR = { is_subject_type = sinosphere_tributary  is_subject_type = tributary } } }
			SUBJ_QING_change_type = { direction = promote  new_type = nominal_vassal }
		}
		else_if = {
			limit = { scope:target = { is_subject_type = nominal_vassal } }
			SUBJ_QING_change_type = { direction = promote  new_type = feudatory }
		}
		else_if = {
			limit = { scope:target = { is_subject_type = feudatory } }
			SUBJ_QING_change_type = { direction = promote  new_type = autonomous_governorship }
		}
		else = {
			LOG_fail = { sys = SUBJ  fn = "qing_subject_promote_button"  reason = "target not on a promotable rung" }
		}
	}
}

# ---- INTEGRATE: push integration progress, absorb at threshold --------
qing_subject_integrate_button = {
	scope = country
	saved_scopes = { target }

	is_shown = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
		# Only the tightest rung can be integrated toward full absorption.
		scope:target = { is_subject_type = autonomous_governorship }
		# While suspended the player must RESUME first (the resume button shows
		# instead); pushing integration would otherwise bypass the suspension.
		scope:target = { NOT = { has_variable = SUBJ_integration_suspended } }
	}

	ai_is_valid = { always = no }

	is_valid = {
		custom_tooltip = {
			text = qing_subject_needs_influence_50_tt
			political_influence >= 50
		}
		custom_tooltip = {
			text = qing_subject_integrate_cooldown_tt
			scope:target = { NOT = { has_variable = qing_recently_integrated } }
		}
	}

	effect = {
		add_political_influence = -50
		scope:target = { set_variable = { name = qing_recently_integrated  days = 365  value = 1 } }
		# A deliberate push advances the process and then rolls for a reaction
		# (the deliberate roll has no "quiet" weight, so a push is always eventful).
		SUBJ_QING_advance_integration = { steps = 1 }
		if = {
			limit = { scope:target = { is_subject_of = ROOT } }
			SUBJ_QING_roll_reaction = { ambient = no }
		}
	}
}

# ---- EXPAND SINOSPHERE: convert a plain tributary / tribal vassal into a 朝貢國 -----
# [1763-hist-fix 2026-07-17] "Expand Sinosphere" — draw a vanilla-type subject (a common
# tributary or a tribal vassal) INTO the Confucian tributary system (sinosphere_tributary):
# it gains the throne's protection + event-based 朝貢 tribute, becomes non-cancellable, and
# takes its place on the bottom rung of the Qing ladder (whence it can be promoted). CHI-only,
# player-only. Shown when the target is our subject and currently a plain tributary or tribal
# vassal (not already sinosphere). Costs influence; per-subject cooldown shared with type changes.
qing_subject_expand_sinosphere_button = {
	scope = country
	saved_scopes = { target }

	is_shown = {
		tag = CHI
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
		scope:target = {
			OR = {
				is_subject_type = tributary
				is_subject_type = vassal_tribe
			}
		}
	}

	ai_is_valid = { always = no }

	is_valid = {
		custom_tooltip = {
			text = qing_subject_needs_influence_50_tt
			political_influence >= 50
		}
		custom_tooltip = {
			text = qing_subject_promote_cooldown_tt
			scope:target = { NOT = { has_variable = qing_recently_changed_type } }
		}
	}

	effect = {
		add_political_influence = -50
		scope:target = { set_variable = { name = qing_recently_changed_type  days = 730  value = 1 } }
		# convert to the sinosphere tributary (the mod's make_subject wrapper preserves the overlord bond).
		FUNC_make_subject = { overlord = ROOT  target = scope:target  type = sinosphere_tributary }
		scope:target = { SUBJ_QING_derive_rung = yes }
		LOG_line = { sys = SUBJ  msg = "expand sinosphere: subject drawn into the 朝貢 tributary system by" }
	}
}

# ---- RESUME: lift a suspension so the process (and pulse) run again ----
# Only appears when the current subject's integration is suspended (a reaction
# event's "suspend" option set the flag). Free and un-cooldowned: the player
# already paid to reach this progress, so restarting should not be punished.
qing_subject_resume_integration_button = {
	scope = country
	saved_scopes = { target }

	is_shown = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
		scope:target = { is_subject_type = autonomous_governorship }
		scope:target = { has_variable = SUBJ_integration_suspended }
	}

	ai_is_valid = { always = no }

	is_valid = {
		always = yes
	}

	effect = {
		SUBJ_QING_resume_integration = yes
	}
}

# ---- INCORPORATE: convert an off-ladder protectorate onto the ladder ---
# A protectorate (Tibet under the amban settlement) sits off the control ladder
# and cannot be promoted or integrated. This one-way, expensive act — modelled on
# Zhao Erfeng's forced incorporation of Kham (改土歸流) — converts it straight to
# an autonomous_governorship (rung 4), after which the ordinary integrate verb
# applies. Priced in prestige + influence, NOT date-restricted: available from
# the 1815 start at the cost the historical court would have borne.
qing_subject_incorporate_button = {
	scope = country
	saved_scopes = { target }

	is_shown = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
		scope:target = { is_subject_type = protectorate }
	}

	ai_is_valid = { always = no }

	is_valid = {
		custom_tooltip = {
			text = qing_subject_needs_influence_100_tt
			political_influence >= 100
		}
		custom_tooltip = {
			text = qing_subject_needs_legitimacy_50_tt
			legitimacy >= 50
		}
		custom_tooltip = {
			text = qing_subject_incorporate_cooldown_tt
			scope:target = { NOT = { has_variable = qing_recently_changed_type } }
		}
	}

	effect = {
		add_political_influence = -100
		# [#314 B23 fix] cost LEGITIMACY, not prestige — incorporating a protectorate is a
		# country-level sovereign act; prestige is a CHARACTER resource and was the wrong scope.
		add_legitimacy = -50
		scope:target = { set_variable = { name = qing_recently_changed_type  days = 730  value = 1 } }
		SUBJ_QING_incorporate_protectorate = yes
	}
}

# ---- [#314 B17] TRIBUTE-SIZE selector (small / medium / large) ---------
# Repurposes the three dead "Overseer" policy toggles. Each button sets the tribute tier
# on the shown subject via QING_subject_set_tribute; the "shown" state is any subject of
# ROOT, "valid" is a cheap influence gate, and the pressed/active tier is read off the
# subject's qing_tribute_tier var so the GUI can highlight the current selection.
qing_subject_tribute_small_button = {
	scope = country
	saved_scopes = { target }
	ai_is_valid = { always = no }
	is_shown = {
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
	}
	is_valid = {
		custom_tooltip = {
			text = qing_subject_needs_influence_25_tt
			political_influence >= 25
		}
		# already on this tier -> no-op guard
		custom_tooltip = {
			text = qing_tribute_already_small_tt
			scope:target = { OR = { NOT = { has_variable = qing_tribute_tier }  NOT = { var:qing_tribute_tier = flag:small } } }
		}
	}
	effect = {
		add_political_influence = -25
		QING_subject_set_tribute = { tier = small }
	}
}

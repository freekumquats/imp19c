########################################################################
# QING_rites_ministry_panel.txt — scripted GUIs for the Ministry of Rites panel (#353).
#
# The Board of Rites (禮部) bureaucracy: a Grand-Council-clone L4 window headed by the
# Grand Minister of Rites (禮部尚書), surfacing the two great state functions he answers
# for — the TRIBUTARY ORDER (朝貢體系, the vassal embassies) as a central roster, and the
# IMPERIAL EXAMINATIONS (科舉, the qing_exam_pass_rate ladder) as a read-out — plus the
# ministry-performance meter qing_min_perf_rites (which folds into his Grand Council
# standing — se_QING_MINISTRY.txt / QING_council_fold_ministry_perf). Read-only dashboard:
# the tribute embassies + exams run on their own pulses (se_QING_TRIBUTE / se_QING_EXAM);
# this panel is the central read-across.
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_rites_ministry_open_panel_button (a button in government_view.gui).
# The window itself is gui/qing_rites_ministry.gui. Prefix = qing_rites_ministry_.
########################################################################

# ---- OPEN THE RITES MINISTRY PANEL (button in the government view) ----------
qing_rites_ministry_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		LOG_line = { sys = QING  msg = "EVENT qing_rites_ministry_open_panel_button fired; opening the Board of Rites panel for" }
	}
}

# ---- RITES MINISTER HOLDER INDICATOR (the GUI reveals his card when filled) -
qing_rites_ministry_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_rites_holder
		var:qing_office_rites_holder = { is_alive = yes  employer = ROOT }
	}
}

# ---- VACANT indicator (the GUI shows a "post vacant" note) ------------------
qing_rites_ministry_vacant = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = {
				has_variable = qing_office_rites_holder
				var:qing_office_rites_holder = { is_alive = yes  employer = ROOT }
			}
		}
	}
}

# ---- CLOSE (button in the panel window) ------------------------------------
qing_rites_ministry_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = { }
}

# MARRIAGE_PLAY_actions.txt — [#356] the marriage-proposal picker's scripted_guis.
#
# [2026-07-21] The target-realm picker (screen 1) was DELETED as dead code — nothing opened it.
# With it went its scripted_guis marriage_play_open + marriage_play_pick_realm and the
# MARRIAGE_PLAY_build_realms effect (se_MARRIAGE_PLAY.txt). The ONLY entry point is the Diplomatic
# View's "Royal Marriage" button -> marriage_play_diplo_open, which knows the target (the viewed
# realm) and opens screen 2 directly. Remaining flow:
#   marriage_play_diplo_open -> pick OWN char (marriage_play_pick_own) -> pick THEIR char
#   (marriage_play_pick_their) -> LAUNCH (marriage_play_launch -> MARRIAGE_PLAY_begin).
#
# Scope: COUNTRY (ROOT = CHI, the player). se_LOG sys = MARRIAGE. Prefix = marriage_play_.
########################################################################

# ---- OPEN FROM THE DIPLOMATIC VIEW ("Royal Marriage" action) ---------------
# [B1] The realm-picker window (screen 1) is bypassed: the diplomatic view already knows the
# TARGET realm (the viewed country), so this entry point takes it as saved scope:target and does
# the work of marriage_play_open + marriage_play_pick_realm in one shot — clears stale picks, sets
# marriage_play_target = the viewed realm, and builds BOTH our-own and their candidate lists — so
# the button can open screen 2 (marriage_play_own_window) directly. Scope: COUNTRY (ROOT = CHI).
# scope:target = the viewed country (passed from DiplomaticView.GetTargetCountry.MakeScope).
marriage_play_diplo_open = {
	scope = country
	saved_scopes = { target }
	ai_is_valid = { always = no }
	# CHI only, a real foreign realm with a ruler to marry into (not ourselves, not a barbarian
	# with no court). Mirrors the marriage_play_pick_realm gate + the CHI gate marriage_play_open had.
	is_shown = { tag = CHI }
	is_valid = {
		tag = CHI
		exists = scope:target
		NOT = { scope:target = { this = ROOT } }
		scope:target = { exists = current_ruler }
	}
	effect = {
		LOG_line = { sys = MARRIAGE  msg = "marriage picker: opened from the diplomatic view for" }
		# clear any stale selection from a prior open (as marriage_play_open does).
		if = { limit = { has_variable = marriage_play_target }      remove_variable = marriage_play_target }
		if = { limit = { has_variable = marriage_play_our_pick }    remove_variable = marriage_play_our_pick }
		# [#356 row-highlight] clear the per-char selection marker off the prior their-pick before dropping the var.
		if = { limit = { has_variable = marriage_play_their_pick }  var:marriage_play_their_pick = { remove_variable = qing_marriage_their_selected }  remove_variable = marriage_play_their_pick }
		# fix the target to the viewed realm (what marriage_play_pick_realm would have done).
		set_variable = { name = marriage_play_target  value = scope:target }
		# build our own unmarried candidates (screen 2) + the target's opposite-... (all sexes until
		# our pick fixes the filter) candidates (screen 3). Same builders marriage_play_open uses.
		MARRIAGE_PLAY_build_own = yes
		MARRIAGE_PLAY_build_their = yes
	}
}

# ---- STEP 2: pick OUR character --------------------------------------------
marriage_play_pick_own = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = { scope:player = { tag = CHI } }
	is_valid = { is_alive = yes  is_married = no  employer = scope:player }
	effect = {
		scope:player = {
			set_variable = { name = marriage_play_our_pick  value = prev }
			# our pick's sex fixes the opposite-gender filter for the target list; rebuild it.
			if = { limit = { has_variable = marriage_play_target }  MARRIAGE_PLAY_build_their = yes }
			# [#356 review-fix, MAJOR] a re-picked own character may now be the SAME sex as an
			# already-chosen their_pick (which is no longer on the rebuilt opposite-sex list). Drop
			# a now-invalid their_pick so the player must re-choose from the correctly-sexed column.
			if = {
				limit = {
					has_variable = marriage_play_their_pick
					OR = {
						AND = { var:marriage_play_our_pick = { is_female = yes }  var:marriage_play_their_pick = { is_female = yes } }
						AND = { var:marriage_play_our_pick = { is_male = yes }    var:marriage_play_their_pick = { is_male = yes } }
					}
				}
				# [#356 row-highlight] clear the per-char marker off the dropped pick so no stale row stays lit.
				var:marriage_play_their_pick = { remove_variable = qing_marriage_their_selected }
				remove_variable = marriage_play_their_pick
			}
		}
		LOG_line = { sys = MARRIAGE  msg = "marriage picker: our character chosen" }
		# [BT-G DIAGNOSTIC 2026-07-19 — REMOVE after screen-3 dead-end is localised] confirm this effect
		# completed and report the state screen 3 will read: this is the LAST script that runs before the
		# button's second onclick fires `gui.createwidget ... marriage_play_their_window`. If this line is
		# in the log but screen 3 never appears, the failure is in the WINDOW instantiation (GUI-layer),
		# not the script; if target is unset here, build_their left the their-lists empty.
		scope:player = {
			LOG_line = { sys = MARRIAGE  msg = "BT-G pick_own DONE — about to createwidget marriage_play_their_window" }
			if = { limit = { has_variable = marriage_play_target }        LOG_line = { sys = MARRIAGE  msg = "BT-G pick_own: target IS set" } }
			if = { limit = { NOT = { has_variable = marriage_play_target } }  LOG_line = { sys = MARRIAGE  msg = "BT-G pick_own: target UNSET" } }
			if = { limit = { has_variable = marriage_play_our_pick }      LOG_line = { sys = MARRIAGE  msg = "BT-G pick_own: our_pick IS set" } }
		}
	}
}

# ---- STEP 3: pick THEIR character ------------------------------------------
marriage_play_pick_their = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = { scope:player = { tag = CHI } }
	is_valid = {
		# [BT-Catherine fix 2026-07-22] TARGET may be ALREADY MARRIED — a marriage play can now
		# contest an existing union (e.g. Catherine II, who carries a placeholder spouse Orlov). The
		# play OVERRIDES it: MARRIAGE_PLAY_resolve divorces the target's current spouse before the
		# wedding. So the target filter no longer requires is_married=no (our OWN courtier still must
		# be single — we are not breaking our own marriages, only winning the target away).
		is_alive = yes
		scope:player = { has_variable = marriage_play_target }
		employer = scope:player.var:marriage_play_target
	}
	effect = {
		# [#356 row-highlight 2026-07-21] clear the per-character selection marker from the PREVIOUS
		# pick (if any) so only one row is highlighted, then set marriage_play_their_pick + mark THIS
		# character. The screen-3 row reads this marker in its `down=` to highlight the selected row
		# (Character.MakeScope.Var(...).IsSet idiom, imp19c_windows.gui:110). ROOT here = the clicked char.
		if = {
			limit = { scope:player = { has_variable = marriage_play_their_pick } }
			scope:player.var:marriage_play_their_pick = { remove_variable = qing_marriage_their_selected }
		}
		scope:player = { set_variable = { name = marriage_play_their_pick  value = prev } }
		set_variable = { name = qing_marriage_their_selected  value = 1 }
		LOG_line = { sys = MARRIAGE  msg = "marriage picker: their character chosen" }
	}
}

# ---- STEP 4: LAUNCH the proposal play --------------------------------------
marriage_play_launch = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		# both characters chosen, both still unmarried + alive, and a target realm set.
		has_variable = marriage_play_our_pick
		has_variable = marriage_play_their_pick
		var:marriage_play_our_pick = { is_alive = yes  is_married = no }
		# [BT-Catherine fix 2026-07-22] the TARGET (their_pick) may be a MARRIED RULER — the play
		# overrides the union (MARRIAGE_PLAY_resolve divorces her spouse before the wedding), so the
		# Propose button must NOT grey out for a wed sovereign like Catherine II. Married is allowed only
		# for a ruler; our own pick still must be single. Mirrors the screen-3 builder + picker gates.
		var:marriage_play_their_pick = { is_alive = yes  OR = { is_married = no  is_ruler = yes } }
		# [#356 review-fix] neither pick already betrothed. The builders now exclude the betrothed,
		# but a betrothal could form (AI, or a prior play resolving) AFTER the picker was populated;
		# without this a stale pick would let MARRIAGE_PLAY_betroth_pair orphan the existing betrothal.
		var:marriage_play_our_pick = { NOT = { has_variable = betrothed_to_char } }
		var:marriage_play_their_pick = { NOT = { has_variable = betrothed_to_char } }
		# [#356 review-fix, MAJOR] the pair MUST be opposite sex — marry_character requires it, and
		# the columns are clickable in any order, so a stale same-sex their_pick can survive a re-pick.
		# Without this guard a same-sex pair could launch, pay the begin price, and resolve to a silent
		# no-op (the engine rejects same-sex marry_character).
		OR = {
			AND = { var:marriage_play_our_pick = { is_female = yes }  var:marriage_play_their_pick = { is_male = yes } }
			AND = { var:marriage_play_our_pick = { is_male = yes }    var:marriage_play_their_pick = { is_female = yes } }
		}
	}
	effect = {
		var:marriage_play_our_pick   = { save_scope_as = marriage_play_our_char }
		var:marriage_play_their_pick = { save_scope_as = marriage_play_their_char }
		MARRIAGE_PLAY_begin = yes
		LOG_line = { sys = MARRIAGE  msg = "marriage picker: proposal play launched" }
	}
}

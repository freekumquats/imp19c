########################################################################
# QING_greatgame_panel.txt — scripted GUIs for the Great Game panel (#108).
#
# Read-only surfacing of the great-power rivalry state the mod already tracks:
# for each great power (Britain / France / Russia), show the qing_gp_tension_*
# level, that power's standing/influence toward the Qing, and (where tracked) their
# activity in the Qing's sphere/frontier. The 以夷制夷 dashboard.
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# The panel is opened by a button in the diplomatic view (see RETURN PAYLOAD for
# the splice). The GUI window itself is defined in gui/qing_greatgame.gui.
#
# Prefix = qing_greatgame_
########################################################################

# ---- OPEN THE GREAT GAME PANEL (button in the diplomatic view) --------------
qing_greatgame_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	# The effect opens the GUI window (the window is named qing_greatgame_panel in
	# the .gui file; the engine wires the button click to this effect).
	effect = {
		# No state to set — the window reads the GP-tension counters directly from
		# ROOT (the Qing player). The GUI is stateless + read-only.
	}
}

# ---- READ-ONLY INDICATORS for each great power (the GUI iterates these) -----
# Britain: tension + standing. The GUI reads var:qing_gp_tension_britain (0..100)
# and c:GBR's opinion of ROOT (if GBR exists).
qing_greatgame_britain_exists = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { exists = c:GBR }
}

qing_greatgame_france_exists = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { exists = c:FRA }
}

qing_greatgame_russia_exists = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { exists = c:RUS }
}

# ---- ZONGLI YAMEN (總理衙門) HOLDER INDICATOR (the GUI reads it) ------------
# The panel shows the Zongli holder (if present) and gates advice on his skill/loyalty.
qing_greatgame_zongli_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_zongli_holder
		var:qing_office_zongli_holder = { is_alive = yes  employer = ROOT }
	}
}

# ---- CLOSE THE GREAT GAME PANEL (button in the panel window) ----------------
qing_greatgame_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = {
		# The window closes via the GUI engine when the button is clicked; no
		# state to clear (the panel is read-only).
	}
}

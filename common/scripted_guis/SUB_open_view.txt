show_player_subjects_button = {
	scope = country
	ai_is_valid = {
		always = no
	}
	effect = {
		every_subject = {
			root = {
				add_to_variable_list = {
					name = player_subjects
					target = prev
				}
			}
		}
		# [#228] Seed a default right-pane selection so the summary panel always
		# has a valid subject to describe when the tab opens (row-click then
		# overrides it via qing_subjects_tab_select_button). Highest-loyalty
		# subject is the stable, deterministic default.
		if = {
			limit = { any_subject = { count >= 1 } }
			ordered_subject = {
				order_by = has_subject_loyalty
				max = 1
				root = { set_variable = { name = SUBJ_tab_selected  value = prev } }
			}
		}
	}
}

remove_player_subjects_variables_button = {
	scope = country
	effect = {
        every_in_list = {
            variable = player_subjects
            root = {
	            remove_list_variable = {
	                name = player_subjects
	                target = prev
	            }
	        }
        }
		# [#228] Clear the right-pane selection alongside the list rebuild.
		remove_variable = SUBJ_tab_selected
	}
}

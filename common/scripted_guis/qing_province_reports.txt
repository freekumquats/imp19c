########################################################################
# qing_province_reports.txt — scripted GUI definitions for three Qing
# province-query panels: #120 New World crop distribution, #109 province
# ethnic-tension report, and #160 national migration / demographic-pressure
# report. Built as an ALTERNATIVE to custom map modes (which are not
# scriptable in Imperator 2.0.5, per oracle verdict OVERNIGHT_DECISIONS.md).
# Each panel is a self-contained scripted_gui that opens a window + iterates
# provinces via every_governorships -> every_governorship_state ->
# every_state_province, rendering a scrollable list.
#
# Pattern: PROVEN via common/scripted_guis/sell_provinces.txt.
# Prefix: qing_report_
########################################################################

# ======================================================================
# COUNTRY-SCOPE: open the New World Crop Distribution panel (#120).
# Iterates all owned provinces, building a list of those producing New
# World crops (maize, sweet_potato, potato — as trade_goods, per
# se_QING_COLON.txt). The effect populates a variable list that the GUI
# window reads via datamodel. READ-ONLY reporting; no new diffusion mechanic.
# ======================================================================
qing_report_open_crops = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# [scan Tier-B #120] wired to a button in the Grand Council action strip.
		LOG_line = { sys = QING  msg = "crop report opened by" }
		clear_variable_list = qing_crop_report_provinces

		# [#120 review-fix] Descend state -> PROVINCE. trade_goods is a per-PROVINCE
		# property in this engine (a governorship_state is only a container), so the
		# old scope:this_state read of trade_goods matched nothing and the list came
		# up permanently empty. The mod's own production code always reads trade_goods
		# on any_state_province (se_GOODS.txt:975), never on the state — match it here.
		every_governorships = {
			every_governorship_state = {
				every_state_province = {
					save_scope_as = this_prov
					# only include provinces producing New World crops
					if = {
						limit = {
							scope:this_prov = {
								OR = {
									trade_goods = maize
									trade_goods = sweet_potato
									trade_goods = potato
								}
							}
						}
						root = {
							add_to_variable_list = {
								name = qing_crop_report_provinces
								target = scope:this_prov
							}
						}
					}
				}
			}
		}
	}
}

# ======================================================================
# COUNTRY-SCOPE: open the Province Ethnic-Tension panel (#109).
# Iterates all owned provinces, reading a per-province tension variable
# (built by feature #107, running in parallel). The variable name is
# qing_prov_ethnic_tension (a sensible default; must match #107's actual
# name). Lists provinces with tension above a threshold (>= 30), sorted
# by tension value. Degrades gracefully (empty list) if the variable is
# never set.
# ======================================================================
qing_report_open_tension = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# [scan Tier-B #109] wired to a button in the Grand Council action strip.
		LOG_line = { sys = QING  msg = "tension report opened by" }
		clear_variable_list = qing_tension_report_provinces

		# [#109 review-fix] Descend state -> PROVINCE. Feature #107 seeds
		# qing_prov_ethnic_tension on PROVINCES (its pulse runs every_owned_province,
		# this = the province — se_QING_ETHNIC_TENSION.txt:50). The old scope:this_state
		# read of that var matched nothing (states never carry it), so the list came up
		# permanently empty — exactly the failure this fix removes. The has_variable
		# guard still lets it degrade gracefully to an empty list if #107 never ran.
		every_governorships = {
			every_governorship_state = {
				every_state_province = {
					save_scope_as = this_prov
					# only include provinces with tension >= 30 (guard with has_variable)
					if = {
						limit = {
							scope:this_prov = {
								has_variable = qing_prov_ethnic_tension
								var:qing_prov_ethnic_tension >= 30
							}
						}
						root = {
							add_to_variable_list = {
								name = qing_tension_report_provinces
								target = scope:this_prov
							}
						}
					}
				}
			}
		}
	}
}

# ======================================================================
# COUNTRY-SCOPE: open the National Migration / Demographic-Pressure panel (#160).
#
# WHY AN AGGREGATE, NOT A LIVE MIGRANT TICKER: the engine's live migration data
# (per-pop migrant progress, GetActivitySpeed('migrant')) is exposed ONLY as C++
# datafunctions on the hardcoded vanilla pops window (view_pops_list.gui,
# ViewPopsWindow.GetMigrants) — it is NOT readable from script, so a scripted_gui
# cannot rebuild that list. What script CAN read on a province is its population,
# capacity, and the concrete migration/boom PROVINCE MODIFIERS the frontier-migration
# mechanic stamps (se_QING_COLON.txt / qing_frontier_migration_events.txt). So this
# panel surfaces the demographic pressure + in-migration hotspots that the mechanic
# actually creates, per the concrete-over-abstract house rule.
#
# A province is listed when EITHER:
#   • it carries an in-migration / boom modifier — qing_migr_hongkong_boom (the
#     Hong Kong harbour district) or qing_nwcrop_abundance (a New World crop province,
#     the population-boom driver); OR
#   • it is under demographic PRESSURE — total_population >= 20 (a crowded province,
#     the source side of internal migration as pops seek room on the frontier).
# Degrades gracefully to an empty list (has_province_modifier / total_population are
# always-safe reads; no variable deref).
# ======================================================================
qing_report_open_migration = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# [#160] wired to a button in the Grand Council action strip.
		LOG_line = { sys = QING  msg = "migration report opened by" }
		clear_variable_list = qing_migration_report_provinces

		# [BT-55 / boot #10 fix] Order the listed provinces by population, highest -> lowest.
		# PASS 1 marks each qualifying province with qing_migr_report_pick; PASS 2 emits them via
		# ordered_owned_province( order_by = total_population ) filtered on that mark, giving a global
		# high-pop-first order; then the marker is cleared.
		# BOOT #10 ROOT CAUSE: PASS 1 previously walked every_governorships > every_governorship_state
		# > every_state_province, which iterates ONLY governorship provinces and EXCLUDES the entire
		# CAPITAL DOMAIN — i.e. most of China proper, where nearly all the crowded/over-capacity
		# provinces actually sit. So the report surfaced almost nothing (the user saw only Nanchong,
		# which happens to fall under a governorship). PASS 2 already scans ordered_owned_province (ALL
		# owned provinces), so the membership walk must match it: mark over every_owned_province too.
		# every_owned_province is a proven iterator (00_tech_modifier_applicator_effects.txt).
		every_owned_province = {
			save_scope_as = this_prov
			# include in-migration / boom hotspots OR crowded (pressure) provinces
			if = {
				limit = {
					scope:this_prov = {
						OR = {
							has_province_modifier = qing_migr_hongkong_boom
							has_province_modifier = qing_nwcrop_abundance
							# [#49] a province the engine has flagged over its pop capacity (auto-applied
							# `overpopulation` modifier, 00_hardcoded.txt:251) is a crowded province by
							# definition — the report is meant to list exactly these, so include them
							# explicitly rather than only the raw total_population >= 20 floor (which
							# missed crowded-but-smaller provinces, so the report showed almost none).
							has_province_modifier = overpopulation
							total_population >= 20
						}
					}
				}
				scope:this_prov = { set_variable = { name = qing_migr_report_pick  value = 1 } }
			}
		}
		# PASS 2: emit in descending-population order (global sort across all marked provinces).
		ordered_owned_province = {
			limit = { has_variable = qing_migr_report_pick }
			order_by = total_population
			# [BT-55b fix] ordered_owned_province switches scope to PROVINCE, so the list must be built on
			# ROOT (the country CHI) — the GUI reads Player.MakeScope.GetList. The BT-55 rework dropped the
			# root wrapper, so the list was written per-province and the country list stayed empty (report
			# showed nothing). Mirrors the sinicization report's root-wrapped add just below.
			save_scope_as = migr_prov
			root = {
				add_to_variable_list = {
					name = qing_migration_report_provinces
					target = scope:migr_prov
				}
			}
		}
		# clear the transient marker so it never leaks to the next open.
		every_owned_province = {
			limit = { has_variable = qing_migr_report_pick }
			remove_variable = qing_migr_report_pick
		}
	}
}

# ======================================================================
# COUNTRY-SCOPE: open the Sinicization panel (#314 B26).
#
# Surfaces the spread of Han settlement/culture across the empire — the
# 漢化 process by which frontier and conquest-elite regions gradually take on a
# Han-majority character. A province is listed when its dominant culture belongs
# to the chinese_group (Han), which — read against the banner/frontier regions —
# is the concrete on-map footprint of sinicization the player can track. Reuses
# the same every_governorships -> state -> province descent and the proven
# dominant_province_culture read (move_capital_button.txt:16). Read-only.
# Degrades gracefully (empty list) if no owned province is Han-dominant.
# ======================================================================
qing_report_open_sinicization = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# [#314 B26] wired to the Reports hub window (qing_reports_window).
		LOG_line = { sys = QING  msg = "sinicization report opened by" }
		clear_variable_list = qing_sinicization_report_provinces
		# [#385] refresh the live Han-dominant count during THIS walk so the footer figure
		# agrees with the rows just built and is current even before the first quarterly
		# QING_sinic_pulse runs. Only the present count is recomputed here; the ANNUAL ROLL of
		# _prev is pulse-only (recompute_delta below may SEED _prev on first touch, never rolls it).
		set_variable = { name = qing_sinic_count  value = 0 }

		every_governorships = {
			every_governorship_state = {
				every_state_province = {
					save_scope_as = this_prov
					# [F6 boot-test fix] list provinces whose dominant culture is Han. The old form
					# `dominant_province_culture.culture_group = chinese_group` FAILED to parse at load (error.log:
					# 'Badly read script value chinese_group / Illegal use of operator =', qing_province_reports.txt:208)
					# — you cannot chain `.culture_group = <group>` off dominant_province_culture in a province limit.
					# PROVEN province idiom = keyword dominant_province_culture_group = <group> (oracle: Invictus
					# 00_steppe_inv.txt:247 / 00_persian.txt:57). Tested directly in every_state_province (no re-wrapper).
					# [logfix] the RIGHT side must be a culture-group SCOPE, not a bare key: every Invictus use
					# passes a scope (`= root.culture_group`), so a bare `= chinese_group` threw "Failed to compare
					# scopes" at runtime (609/session). Use the scope literal `culture_group:chinese_group` (the same
					# form the mod uses in 00_language_groups.txt: `culture.culture_group = culture_group:german`).
					if = {
						limit = {
							dominant_province_culture.culture_group = culture_group:chinese_group
						}
						root = {
							add_to_variable_list = {
								name = qing_sinicization_report_provinces
								target = scope:this_prov
							}
						change_variable = { name = qing_sinic_count  add = 1 }  # [#385] tally the Han-dominant count alongside the displayed list
						}
					}
				}
			}
		}
		# [#385] the count is now current for this open; recompute the year-over-year
		# delta so the footer trend line (rising / steady / receding) reads a fresh
		# figure immediately, without waiting for the next quarterly pulse. The pulse
		# still owns the annual _prev baseline roll; this only refreshes the delta.
		QING_sinic_recompute_delta = yes
	}
}

# ======================================================================
# EMPTY-STATE gates (BT-23/BT-24/BT-22). Each report's effect runs
# clear_variable_list then only add_to_variable_list when a province
# qualifies, so an empty report leaves the list variable UNSET. These
# is_shown triggers drive an empty-state textbox in the window (visible
# only when the list is empty), so a legitimately-empty report reads as
# informative ("no provinces qualify") instead of looking broken.
#   - New World crops: genuinely empty at a fresh 1763 start (no setup
#     province carries maize/sweet_potato/potato as trade_goods; the
#     crop-spread mechanic stamps them over time).
#   - Ethnic tension: qing_prov_ethnic_tension seeds at 0 and climbs, so
#     >= 30 matches nothing early game.
#   - Sinicization: China-proper provinces DO qualify, so this normally
#     shows the list; the gate is a safety net only.
# PROVEN idiom: has_variable_list negation (WAR_scripted_guis.txt:1220
# guards a report list the same way). A list built by add_to_variable_list
# and never added to is simply not present -> has_variable_list = no.
# ======================================================================
qing_report_crops_empty = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { NOT = { has_variable_list = qing_crop_report_provinces } }
}

qing_report_tension_empty = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { NOT = { has_variable_list = qing_tension_report_provinces } }
}

qing_report_sinicization_empty = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { NOT = { has_variable_list = qing_sinicization_report_provinces } }
}

# ======================================================================
# [#385] Sinicization TREND gates — drive the mutually-exclusive year-over-year
# trend line in the report footer (rising / receding / steady). Read the delta
# maintained by se_QING_SINICIZATION.txt (qing_sinic_delta = current count minus
# the year-old baseline; refreshed on report-open + every quarterly pulse). The
# "steady" gate also covers the first-year case where the delta var is not yet set
# (baseline seeded == count, so no meaningful trend yet). Same is_shown idiom as
# the empty-state gate above; read-only.
# ======================================================================
qing_report_sinic_rising = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_sinic_delta
		var:qing_sinic_delta > 0
	}
}

qing_report_sinic_receding = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_sinic_delta
		var:qing_sinic_delta < 0
	}
}

qing_report_sinic_steady = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		OR = {
			NOT = { has_variable = qing_sinic_delta }
			var:qing_sinic_delta = 0
		}
	}
}

# ======================================================================
# [BT-39] Per-province "newly sinicized this year" gate — drives the per-row
# indicator in the sinicization report. Reads the province var qing_prov_sinicized_new
# (set by se_QING_SINICIZATION.txt's annual roll: province is Han-dominant NOW but was
# NOT at the last annual snapshot). PROVINCE scope — evaluated per list row with the
# row's province as root. Read-only.
# ======================================================================
qing_report_prov_newly_sinicized = {
	scope = province
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_prov_sinicized_new
		var:qing_prov_sinicized_new = 1
	}
}

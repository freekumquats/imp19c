########################################################################
# QING_colonise_frontier.txt — [boot #18] COLONISE AN UNOWNED PROVINCE.
#
# Native ENGINE colonisation IS enabled in this mod (MINIMUM_COLONISATION_POP = 8, re-enabled by #304),
# so the province window's built-in ColonizeButton already lets the Qing colonise the CONTIGUOUS frontier
# (adjacent unowned land — the #16b-vacated interior, Xinjiang/Ili marches, etc.). What the engine CANNOT
# do is reach OVERSEAS land — Pacific islands and other coasts out of colonisation range from the Asian
# mainland. THIS button fills only that gap. (The mod separately deleted the migrant `settle` ARMY ability
# and keeps `military_colonies`, which settles already-owned land — a different mechanic, not this.)
# A province-scope player button cloned from the proven found_city_button: shown to CHI on an unowned,
# inhabitable COASTAL province that is NOT adjacent to Qing land (so the engine won't already offer it),
# while the throne holds a Pacific colonial foothold (the treasure-fleet / fur-coast / silver-road reach).
#
# On use: pay price_colonise_frontier, transfer the province to the Qing (set_owned_by = ROOT — the
# proven in-mod province-transfer idiom, se_QING_BURMA:149), and seed a starter pop
# (create_state_pop tribesmen — the se_QING_COLON settler idiom). CHI-only, ai_is_valid no.
# Wired into gui/province_window.gui on the UNOWNED-province path. Effect: se_QING_AMERICAS.txt.
########################################################################

QING_colonise_frontier_button = {
	scope = province
	confirm_title = "QING_colonise_frontier_confirm_t"
	confirm_text = "QING_colonise_frontier_confirm_desc"
	saved_scopes = { player }

	is_shown = {
		scope:player = { tag = CHI }
		# only ever offered on genuinely UNOWNED, settleable land
		has_owner = no
		is_uninhabitable = no
		NOT = { is_sea = yes }
	}

	ai_is_valid = { always = no }

	is_valid = {
		has_owner = no
		is_uninhabitable = no
		NOT = { is_sea = yes }
		# [boot #18 review-fix] The native ENGINE colonisation (MINIMUM_COLONISATION_POP=8, re-enabled
		# #304) already handles the ADJACENT frontier via the province window's built-in ColonizeButton,
		# so this scripted button must NOT duplicate it. It exists only for the reach the engine CANNOT
		# do: OVERSEAS / Pacific-island land, which is out of colonisation range from the Asian mainland.
		# Gate accordingly — a COASTAL province NOT adjacent to any Qing land (so the engine won't offer
		# it), while the throne holds a Pacific foothold (the treasure-fleet / fur-coast reach).
		is_coastal = yes
		NOT = { any_neighbor_province = { owner = scope:player } }
		# [B15] DROPPED the Pacific-foothold gate (needed one of qing_col_silver_road / _fur_coast /
		# _pacific_trade / _golden_shore OR an owned California/Cascadia/Alaska/Pacific_Mexico province).
		# At a fresh 1763 start the Qing holds NONE of those, so the button was ALWAYS greyed on the
		# islands the player selects — the reported bug. The whole point of this button is to plant an
		# OVERSEAS colony the engine's contiguous colonisation cannot reach; requiring a colony to already
		# exist before you can found your first one was circular. The remaining gates (CHI + unowned +
		# inhabitable + coastal + NOT adjacent to Qing land + affordable) already scope it correctly to
		# genuine overseas/island land without duplicating the engine's adjacent-frontier ColonizeButton.
		scope:player = { tag = CHI }
		# affordable (proven price-gate idiom from found_city_button).
		trigger_if = {
			limit = { scope:player = { NOT = { can_pay_price = price_colonise_frontier } } }
			hidden:scope:player = { can_pay_price = price_colonise_frontier }
		}
	}

	effect = {
		scope:player = { pay_price = price_colonise_frontier }
		QING_colonise_this_province = yes
		LOG_line = { sys = QING  msg = "colonisation: a colony is planted on the unowned frontier for" }
	}
}

########################################################################
# SUCCESSION_actions.txt — [#312] female-line succession-law reform buttons.
#
# The mod's "decision" idiom is a scripted_gui action wired to a GUI button. Each of
# these sets the realm's succession law via the SUCCESSION_* effects (which wrap the
# engine's change_law). Gated: monarchy only, one reform per 5 years (a succession law
# is not changed lightly), and each hidden while ALREADY on that law. ROOT = player.
#
# Prefix = succession_reform_
########################################################################

# ---- Adopt AGNATIC (male-line only) ----------------------------------
succession_reform_agnatic = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		is_monarchy = yes
		NOT = { has_law = agnatic_succession_law }
	}
	is_valid = {
		custom_tooltip = {
			text = succession_reform_cooldown_tt
			NOT = { has_variable = succession_reform_cooldown }
		}
	}
	effect = {
		set_variable = { name = succession_reform_cooldown  days = 1825  value = 1 }
		SUCCESSION_set_agnatic = yes
	}
}

# ---- Adopt AGNATIC-COGNATIC (males preferred, daughters if no male line) ----
succession_reform_agnatic_cognatic = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		is_monarchy = yes
		# shown unless we are already on plain cognatic WITHOUT the absolute dressing
		OR = {
			NOT = { has_law = cognatic_succession_law }
			has_country_modifier = absolute_cognatic_succession
		}
	}
	is_valid = {
		custom_tooltip = {
			text = succession_reform_cooldown_tt
			NOT = { has_variable = succession_reform_cooldown }
		}
	}
	effect = {
		set_variable = { name = succession_reform_cooldown  days = 1825  value = 1 }
		SUCCESSION_set_agnatic_cognatic = yes
	}
}

# ---- Adopt ABSOLUTE-COGNATIC (presented as gender-equal) -------------
succession_reform_absolute_cognatic = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		is_monarchy = yes
		NOT = { has_country_modifier = absolute_cognatic_succession }
	}
	is_valid = {

			text = succession_reform_cooldown_tt
			NOT = { has_variable = succession_reform_cooldown }
		}

	effect = {
		set_variable = { name = succession_reform_cooldown  days = 1825  value = 1 }

	}

# ---- read-only indicators (for the GUI to show the ACTIVE law) -------
succession_is_agnatic = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_law = agnatic_succession_law
	}
}
succession_is_agnatic_cognatic = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_law = cognatic_succession_law
		NOT = { has_country_modifier = absolute_cognatic_succession }
	}
}
succession_is_absolute_cognatic = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_country_modifier = absolute_cognatic_succession
	}
}

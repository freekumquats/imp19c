        }
    }

# Cash minting and recall rates

CURRENCY_increase_minting_rate = {
    # Scope: Country
    # Increases the minting rate of new currency by 1,000
    scope = country
    ai_is_valid = {
        always = no
    }
    effect = {
        if = {
            limit = {
                trigger_if = {
                    limit = {
                        has_variable = CURRENCY_minting_rate
                    }
                    var:CURRENCY_minting_rate < CURRENCY_minting_rate_cap
                }
                has_variable = CURRENCY_minting_rate
            }
            change_variable = {
                name = CURRENCY_minting_rate
                add = 1
            }
        }
    }
}

DIPLO_colonial_canton = {
    # Determines if this country gets a rectangular canton of its overlord in the top left corner
    scope = country
    is_shown = {
        is_subject = yes
        trigger_if = {
            limit = {
                is_subject = yes
            }
            NOT = {
                overlord = c:NED
                tag = FRG
            }
            OR = {
                NOT = {
                    overlord = c:SPA
                    overlord = c:POR
                }
                capital_scope.governorship.var:trade_center = {
                    OR = {
                        has_variable = is_southern_africa_tradezone
                        has_variable = is_east_africa_tradezone
                        has_variable = is_west_africa_tradezone
                    }
                }
            }
            OR = {
                is_subject_type = client_colony
                is_subject_type = autonomous_colony

                # Certain royal unions
                AND = {
                    OR = {
                        overlord = {
                            OR = {
                                religion = orthodox
                                religion = anglican
                                religion = lutheran
                                religion = evangelical
                            }
                        }
                    }
                    
                    NOT = {
                        has_variable = member_of_federation
                    }
                    is_subject_type = royal_union
                    DIPLOMACY_power < 200
                }
            }
        }
        
    }
}

DIPLO_begin_or_end_play = {
    # Scopes: player
    scope = province
    ai_is_valid = {
        always = no
    }

    is_valid = {
        NOT = { scope:player = scope:player.var:selected_province.owner }
        scope:player = {
            political_influence > 0
            stability > 10
        }
    }

    effect = {

        if = {
            limit = {
                scope:player = { has_variable = selected_province }
            }
            if = {
                limit = { # If player has a diplomatic play active in this area and country
                    scope:player = {
                        any_in_list = {
                            variable = local_area_diplomatic_plays

                            var:play_instigator = scope:player
                            var:play_target_country = scope:player.var:selected_province.owner
                        }
                    }
                }
                # End the diplomatic play
                # Costs political power
                scope:player = {
                    pay_price = end_diplomatic_play_price

                    random_in_list = {
                        variable = local_area_diplomatic_plays
                        limit = {
                            var:play_instigator = scope:player
                            var:play_target_country = scope:player.var:selected_province.owner
                        }
                        save_scope_as = diplomatic_play_to_remove
                    }

                    scope:diplomatic_play_to_remove = {
                        AI_remove_diplomatic_play = yes
                    }
                    
                }
            }

            else = { # Otherwise, begin a diplomatic play
                # TODO: Make the price of beginning a diplomatic play sensitive to your diplomatic influence in an area
                # TODO: Make it impossible to begin a diplomatic play if PI would go negative
                # [#210] Begin the goal the player SELECTED with the type buttons (var:pending_play_goal),
                # defaulting to get_territory if none chosen. play_instigator = scope:player is the
                # player COUNTRY — correct. (The old goal-button path began plays via
                # DIPLOMACY_player_begin_play, whose `play_instigator = this` resolved to the province
                # inside AI_begin's random_province block, so plays displayed as "Nobody"; that path is
                # now unused by the buttons.) Colonise passes target = instigator, which AI_begin
                # normalises to c:BAR (its unclaimed-land convention).
                scope:player = {
                    pay_price = begin_diplomatic_play_price
                    if = {
                        limit = { has_variable = pending_play_goal }
                        if = {
                            limit = { var:pending_play_goal = flag:colonise_area }
                            AI_begin_diplomatic_play = {
                                play_instigator = scope:player
                                play_target_area = scope:player.var:selected_province
                                play_target_country = scope:player
                                play_goal = scope:player.var:pending_play_goal
                                play_type = manual
                            }
                        }
                        else = {
                            AI_begin_diplomatic_play = {
                                play_instigator = scope:player
                                play_target_area = scope:player.var:selected_province
                                play_target_country = scope:player.var:selected_province.owner
                                play_goal = scope:player.var:pending_play_goal
                                play_type = manual
                            }
                        }
                    }
                    else = {
                        AI_begin_diplomatic_play = {
                            play_instigator = scope:player
                            play_target_area = scope:player.var:selected_province
                            play_target_country = scope:player.var:selected_province.owner
                            play_goal = flag:get_territory
                            play_type = manual
                        }
                    }
                    # TODO: Trigger an AI attitude assessment immediately
                }
            }
        }

    }
    
}

# ======================================================================
# PLAYER SUPPORT-PLAY GUI (#147). Backs an existing diplomatic play, nudging its
# success upward for a political-influence cost. Root scope = the play provobj
# (the plays-management list binds each item's Scope to the play province);
# scope:player = the backing country. Valid when the play still exists, the player
# holds influence, and the player is not already maxing an unrelated concern. The
# success bump routes through DIPLOMACY_modify_play_success (province scope), which
# scales the gain by the balance of power and caps at 100.
# ======================================================================
DIPLO_player_support_play = {
    scope = province
    ai_is_valid = { always = no }
    is_valid = {
        has_variable = is_diplomatic_play
        scope:player = { political_influence >= 10 }
        # [#147 review-fix 1] Backing a play always raises the INSTIGATOR's success, so
        # the player must never be the target of the play they are supporting — else the
        # (list-wide) Support button becomes silent self-harm. Only bar the target; an
        # uninvolved third party backing an ally's play is legitimate diplomacy.
        NOT = { var:play_target_country = scope:player }
        # no point paying influence once the play is already certain to succeed.
        var:diplomatic_play_success < 100
    }
    effect = {
        scope:player = {
            pay_price = support_diplomatic_play_price
        }
        # this = the play provobj (root scope of the GUI).
        DIPLOMACY_modify_play_success = { amt = 10 }
        LOG_line = { sys = DIPLO  msg = "player supports play, +10 success" }
    }
}

# [scan Tier-C] OPPOSE a diplomatic play — the mirror of DIPLO_player_support_play.
# Spend political influence to push the play's success DOWN (toward a fizzle). Root =
# the play provobj; 'player' scope added by the GUI. Same influence outlay as support.
DIPLO_player_oppose_play = {
    scope = province
    ai_is_valid = { always = no }
    is_valid = {
        has_variable = is_diplomatic_play
        scope:player = { political_influence >= 10 }
        # Opposing a play you INSTIGATED is nonsense — that is what Cancel is for. Bar
        # the instigator; anyone else (the target, or a concerned third party) may oppose.
        NOT = { var:play_instigator = scope:player }
        # no point paying influence once the play has already collapsed to nothing.
        var:diplomatic_play_success > 0
    }
    effect = {
        scope:player = {
            pay_price = oppose_diplomatic_play_price
        }
        # this = the play provobj (root scope of the GUI). Negative amt runs the
        # weak-instigator divide branch in DIPLOMACY_modify_play_success (floored, #79).
        DIPLOMACY_modify_play_success = { amt = -10 }
        LOG_line = { sys = DIPLO  msg = "player opposes play, -10 success" }
    }
}

# [scan Tier-C] CANCEL your own diplomatic play — withdraw a play you instigated before
# it resolves. Only the instigator of a MANUAL (player-started) play may cancel; the
# influence already spent to begin it is not refunded (matches end-play semantics). Reuses
# the proven AI_remove_diplomatic_play teardown, which strips is_diplomatic_play + every
# play_* var and de-indexes the provobj, so the next update pass finds nothing to resolve.
DIPLO_player_cancel_play = {
    scope = province

    is_valid = {

        has_variable = manual_play

    effect = {
        # this = the play provobj (root scope of the GUI).
        LOG_line = { sys = DIPLO  msg = "player cancels own play over" }
        AI_remove_diplomatic_play = yes
    }
}

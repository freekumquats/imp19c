# [#228] Subjects-tab right-panel selection.
# The Qing Subjects tab lists the player's own subjects on the left, but the
# right pane was hardwired to DiplomaticView.GetTargetCountry, which on the
# player's own tab resolves to Qing itself -> the summary always described the
# overlord, never the clicked subject. This scripted GUI records the row the
# player clicked into a player-country variable (SUBJ_tab_selected) so the
# right pane can read that subject's info instead. Row click = select; the
# flag click still opens the diplomatic view (handled in gui, unchanged).
qing_subjects_tab_select_button = {
	scope = country
	saved_scopes = { target }
	ai_is_valid = { always = no }
	is_shown = {
		tag = CHI
		exists = scope:target
		scope:target = { is_subject_of = ROOT }
	}
	is_valid = { always = yes }
	effect = {
		set_variable = { name = SUBJ_tab_selected  value = scope:target }
	}
}

########################################################################
# QING_war_ministry_panel.txt — scripted GUIs for the Ministry of War panel (#349).
#
# The Board of War (兵部) bureaucracy: a Grand-Council-clone L4 window headed by the
# Grand Minister of War (兵部尚書), surfacing the officer corps + garrisons he answers
# for, the ministry-performance meter qing_min_perf_war (which folds into his Grand
# Council standing — se_QING_MINISTRY.txt / QING_council_fold_ministry_perf), and the
# officer-corps ethnic balance. Read-mostly: the roster + meters are read directly from
# ROOT (the Qing player); the officer-death recommendation lifecycle (qing_office.32)
# already handles command turnover, so this panel is a dashboard, not an appointer.
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_war_ministry_open_panel_button (a button in government_view.gui).
# The window itself is gui/qing_war_ministry.gui. Prefix = qing_war_ministry_.
########################################################################

# ---- OPEN THE WAR MINISTRY PANEL (button in the government view) ------------
qing_war_ministry_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# The window reads the officer roster + qing_min_perf_war directly from ROOT.
		# Stateless dashboard; the createwidget console command opens it (see the .gui splice).
		LOG_line = { sys = QING  msg = "EVENT qing_war_ministry_open_panel_button fired; opening the Board of War panel for" }
	}
}

# ---- WAR MINISTER HOLDER INDICATOR (the GUI reveals his card when filled) ---
qing_war_ministry_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_war_holder
		var:qing_office_war_holder = { is_alive = yes  employer = ROOT }
	}
}

# ---- VACANT indicator (the GUI shows a "post vacant" note) ------------------
qing_war_ministry_vacant = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = {
				has_variable = qing_office_war_holder
				var:qing_office_war_holder = { is_alive = yes  employer = ROOT }
			}
		}
	}
}

# ---- OFFICER-CORPS present indicator (gates the roster list vs empty note) --
qing_war_ministry_has_officers = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_officer_corps_count
		var:qing_officer_corps_count > 0
	}
}

# ---- CLOSE (button in the panel window) ------------------------------------
qing_war_ministry_close_panel = {
	scope = country
	ai_is_valid = { always = no }
	is_valid = { always = yes }
	effect = { }
}

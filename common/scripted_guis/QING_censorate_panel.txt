# QING_censorate_panel.txt — scripted GUIs for the Censorate panel (#362).
#
# The Censorate (都察院) bureaucracy: a Grand-Council-clone L4 window headed by the
# Grand Inspector (左都御史), surfacing the corps of investigating censors (御史/言官)
# that reports to him, the ministry-performance meter qing_min_perf_censor (which folds
# into his Grand Council standing — a fold that had NEVER been computed before #362), the
# realm's corruption level, and two action levers: COMMISSION AN INSPECTOR (an 御史, via the
# shared court picker) and IMPEACH THE VENAL (the proven QING_censorate_find_corrupt +
# QING_censorate_impeach_uphold pair — find the most corrupt courtier and disgrace him).
#
# Scoped to the Qing player (ROOT = CHI). AI never evaluates these (ai_is_valid = no).
# Opened by qing_censorate_open_panel_button (a button in government_view.gui).
# Window: gui/qing_censorate.gui. Prefix = qing_censorate_. se_LOG sys = QING.
########################################################################

# ---- OPEN THE CENSORATE PANEL (button in the government view) --------------
qing_censorate_open_panel_button = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		# recompute so the meter + inspector roster are fresh the moment the panel opens.
		QING_ministry_recompute_perf_censor = yes
		LOG_line = { sys = QING  msg = "EVENT qing_censorate_open_panel_button fired; opening the Censorate panel for" }
	}
}

# ---- CENSOR-IN-CHIEF HOLDER INDICATOR --------------------------------------
qing_censorate_filled = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		has_variable = qing_office_censor_holder
		var:qing_office_censor_holder = { is_alive = yes  employer = ROOT }
	}
}

qing_censorate_vacant = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = {
		NOR = {
			AND = {
				has_variable = qing_office_censor_holder
				var:qing_office_censor_holder = { is_alive = yes  employer = ROOT }
			}
		}
	}
}

# ---- COMMISSION AN INSPECTOR — flag a courtier as a member of the inspecting-
# censor corps (御史). Row action on the court-candidate picker (reuses the shared
# court pool the Appoint window lists). Capped at 6 inspectors so the corps is a
# select service, not the whole court. ROOT = the player; the char comes via the
# picker datacontext (GuiScope char + player). Mirrors qing_zongli_appoint_diplomat.
# ----------------------------------------------------------------------
qing_censorate_appoint_inspector = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
	}
	is_valid = {
		# an eligible, unassigned courtier of the player, corps not yet full.
		is_alive = yes
		is_adult = yes
		employer = scope:player
		NOT = { has_variable = qing_is_censor_inspector }
		NOT = { has_variable = qing_office_held }
		scope:player = {
			OR = {
				NOT = { has_variable = qing_censor_inspector_count }
				var:qing_censor_inspector_count < 6
			}
		}
	}
	effect = {
		# stamp the inspector mark (read by QING_ministry_recompute_perf_censor, which
		# rebuilds the roster + count each pulse).
		set_variable = { name = qing_is_censor_inspector  value = 1 }
		LOG_line = { sys = QING  msg = "censorate: a courtier commissioned to the inspecting-censor corps (御史) for" }
		# recompute immediately so the panel + the count gate reflect the new appointee.
		scope:player = { QING_ministry_recompute_perf_censor = yes }
	}
}

# ---- RECALL AN INSPECTOR — remove a serving inspector from the corps. -------
qing_censorate_recall_inspector = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
		has_variable = qing_is_censor_inspector
	}
	is_valid = {
		has_variable = qing_is_censor_inspector
		employer = scope:player
	}
	effect = {
		remove_variable = qing_is_censor_inspector
		LOG_line = { sys = QING  msg = "censorate: an inspector recalled from the corps for" }
		scope:player = { QING_ministry_recompute_perf_censor = yes }
	}
}

# ---- IMPEACH THE VENAL — find the most corrupt courtier and disgrace him.
# [#443 boot-crash fix] The effect must NOT call QING_censorate_find_corrupt inline:
# a scripted_gui compile-inlines its button's effect chain at PARSE time, and that
# effect's body is an `ordered_character { order_by = corruption ... }` SORTING
# iterator — the sole scripted_gui in the mod inlining a sorting char-iterator, which
# access-violated the loader (same class as the guard-panel legion iterator + the
# office-vacate chain). So the effect ONLY trigger_events the hidden qing_censorate.5,
# which does the find + uphold + cooldown + recompute at RUNTIME (trigger_event is not
# compile-inlined). Gated on a present Grand Inspector AND a corrupt target (mirrors
# the trampoline's own guard, so no false affordance). Country scope.
# ----------------------------------------------------------------------
qing_censorate_impeach_venal = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = {
		# a present Grand Inspector must be in place to prosecute...
		has_variable = qing_office_censor_holder
		var:qing_office_censor_holder = { is_alive = yes  employer = ROOT }
		# [#362 review-fix R1] ...and NOT within the impeachment cooldown. Without this the
		# lever was spam-clickable: each click ran QING_censorate_impeach_uphold (corruption
		# -6 + a fresh disgrace) with no throttle, so a player could floor qing_corruption_level
		# to 0 in one day and re-hammer the same already-disgraced courtier (whom find_corrupt
		# keeps re-selecting, since disgrace keeps his loyalty < 40). A ~1-quarter cooldown paces
		# it to the intended quarterly-event cadence. Engine auto-decrements day-typed vars.
		NOT = { has_variable = qing_censorate_impeach_cooldown }
		# ...and there must BE a venal courtier to impeach (mirror find_corrupt's limit,
		# so the lever is dead when there is no one to prosecute — no no-op affordance).
		any_character = {
			employer = ROOT
			is_ruler = no
			is_adult = yes
			is_alive = yes
			NOT = { has_variable = qing_rebel_foil }
			OR = {
				corruption >= 30
				has_trait = corrupt
				loyalty < 40
			}
		}
	}
	effect = {
		# [#9 2026-07-17] NO LONGER auto-picks. This verb now survives only as the panel button's
		# ENABLE-GATE (its is_valid above): the button's onclick fires qing_censorate_refresh_venal_btn
		# + opens the qing_censorate_impeach_window picker, and a row-click there impeaches the CHOSEN
		# man (qing_censorate_impeach_selected) — mirroring the Accuse-an-Official flow, so the player
		# sees WHOM he disgraces sorted by corruption instead of a blind auto-pick. Left empty (not the
		# old trigger_event qing_censorate.5 auto-pick) so it never silently disgraces a man behind the
		# player's back if anything invokes it.
	}
}

# [#9 2026-07-17] IMPEACH-PICKER ROSTER refresh — the panel button's FIRST onclick. Fires the hidden
# qing_censorate.6 runtime event, which builds the corruption-sorted qing_censorate_venal list
# (routed off the compiled button chain because it is a sorting iterator — the #443 crash class).
qing_censorate_refresh_venal_btn = {
	scope = country
	ai_is_valid = { always = no }
	is_shown = { tag = CHI }
	is_valid = { always = yes }
	effect = {
		trigger_event = { id = qing_censorate.6 }
		LOG_line = { sys = QING  msg = "censorate: impeach-picker roster refreshed from the panel for" }
	}
}

# [#9 2026-07-17] IMPEACH A SELECTED OFFICIAL — row-click in the qing_censorate_impeach_window picker.
# The chosen courtier comes via datacontext (GuiScope char + player), exactly like the Accuse picker's
# row lever. Saves him as scope:qing_censorate_target, upholds the impeachment (QING_censorate_impeach_
# uphold: cleanse, disgrace, vacate office), stamps the ~1-quarter cooldown, and recomputes the meter.
# No trampoline needed here: this is not a sorting iterator (the roster was already built by .6), just a
# save-scope + the uphold effect, so it is safe inline in the row's scripted-gui effect.
qing_censorate_impeach_selected = {
	scope = character
	saved_scopes = { player }
	ai_is_valid = { always = no }
	is_shown = {
		scope:player = { tag = CHI }
		employer = scope:player
	}
	is_valid = {
		# a present Grand Inspector must be seated + not in the impeachment cooldown (mirror the
		# panel gate qing_censorate_impeach_venal).
		scope:player = {
			has_variable = qing_office_censor_holder
			var:qing_office_censor_holder = { is_alive = yes  employer = scope:player }
			NOT = { has_variable = qing_censorate_impeach_cooldown }
		}
		# the selected man must still be a valid impeachment target (mirror find_corrupt's limit).
		employer = scope:player
		is_ruler = no
		is_adult = yes
		is_alive = yes
		NOT = { has_variable = qing_rebel_foil }
		OR = {
			corruption >= 30
			has_trait = corrupt
			loyalty < 40
		}
	}
	effect = {
		# [#9 boot-crash fix] Route the impeachment through the hidden CHARACTER event qing_censorate.7
		# (a RUNTIME trigger_event, never compile-inlined) — DO NOT call QING_censorate_impeach_uphold
		# inline here. That effect reaches QING_office_vacate_dispatch -> QING_office_vacate, whose body
		# fires a nested trigger_event ( qing_office.40 ), plus QING_ministry_recompute_perf_censor (an
		# iterator-heavy recompute). A scripted_gui compile-inlines its button's entire effect chain at
		# PARSE time, so inlining that chain into this row button reproduced the #443 EXCEPTION_ACCESS_
		# VIOLATION load crash (same class as the guard legion iterator -> qing_guard.10 and the .5 auto-
		# pick trampoline). The #443 class is NOT limited to sorting iterators: any nested trigger_event /
		# office-vacate chain inlined into a button crashes the loader. The proven-booting sibling this was
		# cloned from ( qing_justice_accuse_selected ) already routes through qing_justice.8 for exactly this
		# reason; this now mirrors it. ROOT = the selected man; the saved player scope carries into the event.
		trigger_event = { id = qing_censorate.7 }
		LOG_line = { sys = QING  msg = "censorate: the venal impeached via the picker for" }
	}
}
